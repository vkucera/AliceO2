93  inline void addEntry(int isig, zdcBaseline_t val)
94  {
95  if (!mHisto[isig].mOverflow) {
96  int ibin = val - BaselineMin;
97  if (mHisto[isig].mData[ibin] < 0xffffffff) {
98  mHisto[isig].mData[ibin]++;
99  } else {
100  mHisto[isig].mOverflow = true;
101  mOverflow = true;
102  }
103  }
104  }
