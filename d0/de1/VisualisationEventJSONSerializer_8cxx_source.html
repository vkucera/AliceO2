47  fclose(file); // file exists

62 std::string VisualisationEventJSONSerializer::toJson(const VisualisationEvent& event) const
63 {
64  Document tree(kObjectType);
65  Document::AllocatorType& allocator = tree.GetAllocator();
66 
67  // compatibility verification
68  tree.AddMember("fileVersion", rapidjson::Value().SetInt(JSON_FILE_VERSION), allocator);
69  tree.AddMember("runNumber", rapidjson::Value().SetInt(event.mRunNumber), allocator);
70  tree.AddMember("clMask", rapidjson::Value().SetInt(event.mClMask), allocator);
71  tree.AddMember("trkMask", rapidjson::Value().SetInt(event.mTrkMask), allocator);
72  tree.AddMember("tfCounter", rapidjson::Value().SetInt(event.mTfCounter), allocator);
73  tree.AddMember("firstTForbit", rapidjson::Value().SetInt(event.mFirstTForbit), allocator);
74 
75  tree.AddMember("collisionTime", rapidjson::Value().SetString(event.mCollisionTime.c_str(), event.mCollisionTime.size()), allocator);
76  tree.AddMember("workflowVersion", rapidjson::Value().SetString(event.mWorkflowVersion.c_str(), event.mWorkflowVersion.size()), allocator);
77  tree.AddMember("workflowParameters", rapidjson::Value().SetString(event.mWorkflowParameters.c_str(), event.mWorkflowParameters.size()), allocator);
78  // Tracks
79  tree.AddMember("trackCount", rapidjson::Value().SetInt(event.getTrackCount()), allocator);
80 
81  Value jsonTracks(kArrayType);
82  for (auto track : event.getTracksSpan()) {
83  jsonTracks.PushBack(jsonTree(track, allocator), allocator);
84  }
85  tree.AddMember("mTracks", jsonTracks, allocator);
86 
87  // Clusters
88  rapidjson::Value clusterCount(rapidjson::kNumberType);
89  clusterCount.SetInt(event.getClusterCount());
90  tree.AddMember("clusterCount", clusterCount, allocator);
91  Value jsonClusters(kArrayType);
92  for (auto cluster : event.getClustersSpan()) {
93  jsonClusters.PushBack(jsonTree(cluster, allocator), allocator);
94  }
95  tree.AddMember("mClusters", jsonClusters, allocator);
96 
97  // Calorimeters
98  rapidjson::Value caloCount(rapidjson::kNumberType);
99  caloCount.SetInt(event.getCaloCount());
100  tree.AddMember("caloCount", caloCount, allocator);
101  Value jsonCalos(kArrayType);
102  for (auto calo : event.getCalorimetersSpan()) {
103  jsonCalos.PushBack(jsonTree(calo, allocator), allocator);
104  }
105  tree.AddMember("mCalo", jsonCalos, allocator);
106 
107  // stringify
108  rapidjson::StringBuffer buffer;
109  rapidjson::PrettyWriter<rapidjson::StringBuffer> writer(buffer);
110  tree.Accept(writer);
111  std::string json_str = std::string(buffer.GetString(), buffer.GetSize());
112  return json_str;
113 }

124 void VisualisationEventJSONSerializer::fromJson(VisualisationEvent& event, std::string json)
125 {
126  event.mTracks.clear();
127  event.mClusters.clear();
128  event.mCalo.clear();
129 
130  rapidjson::Document tree;
131  tree.Parse(json.c_str());
132 
133  auto version = 1;
134  if (tree.HasMember("fileVersion")) {
135  rapidjson::Value& jsonFileVersion = tree["fileVersion"];
136  version = jsonFileVersion.GetInt();
137  }
138 
139  o2::header::DataHeader::RunNumberType runNumber = 0;
140  if (tree.HasMember("runNumber")) {
141  rapidjson::Value& jsonRunNumber = tree["runNumber"];
142  runNumber = jsonRunNumber.GetInt();
143  }
144  event.setRunNumber(runNumber);
145 
146  event.setClMask(getIntOrDefault(tree, "clMask"));
147  event.setTrkMask(getIntOrDefault(tree, "trkMask"));
148  event.setTfCounter(getIntOrDefault(tree, "tfCounter"));
149  event.setFirstTForbit(getIntOrDefault(tree, "firstTForbit"));
150 
151  auto collisionTime = "not specified";
152  if (tree.HasMember("collisionTime")) {
153  rapidjson::Value& jsonCollisionTime = tree["collisionTime"];
154  collisionTime = jsonCollisionTime.GetString();
155  }
156  event.setCollisionTime(collisionTime);
157 
158  rapidjson::Value& trackCount = tree["trackCount"];
159  event.mTracks.reserve(trackCount.GetInt());
160  rapidjson::Value& jsonTracks = tree["mTracks"];
161  for (auto& v : jsonTracks.GetArray()) {
162  event.mTracks.emplace_back(trackFromJSON(v));
163  }
164 
165  if (tree.HasMember("caloCount")) {
166  rapidjson::Value& caloCount = tree["caloCount"];
167  event.mCalo.reserve(caloCount.GetInt());
168  rapidjson::Value& jsonCalo = tree["mCalo"];
169  for (auto& v : jsonCalo.GetArray()) {
170  event.mCalo.emplace_back(caloFromJSON(v));
171  }
172  }
173 
174  event.mMinTimeOfTracks = std::numeric_limits<float>::max();
175  event.mMaxTimeOfTracks = std::numeric_limits<float>::min();
176  for (auto& v : event.mTracks) {
177  event.mMinTimeOfTracks = std::min(event.mMinTimeOfTracks, v.getTime());
178  event.mMaxTimeOfTracks = std::max(event.mMaxTimeOfTracks, v.getTime());
179  }
180 
181  rapidjson::Value& clusterCount = tree["clusterCount"];
182  event.mClusters.reserve(clusterCount.GetInt());
183  rapidjson::Value& jsonClusters = tree["mClusters"];
184  for (auto& v : jsonClusters.GetArray()) {
185  event.mClusters.emplace_back(clusterFromJSON(v));
186  }
187 }

250 VisualisationTrack VisualisationEventJSONSerializer::trackFromJSON(rapidjson::Value& tree)

254  rapidjson::Value& jsonPolyX = tree["mPolyX"];
255  rapidjson::Value& jsonPolyY = tree["mPolyY"];
256  rapidjson::Value& jsonPolyZ = tree["mPolyZ"];
257  rapidjson::Value& count = tree["count"];
258  track.mCharge = 0;
259 
260  if (tree.HasMember("source")) {
261  track.mSource = (o2::dataformats::GlobalTrackID::Source)tree["source"].GetInt();
262  } else {
263  track.mSource = o2::dataformats::GlobalTrackID::TPC; // temporary
264  }
265  track.mPID = tree["source"].GetInt();
266  track.mTime = tree["time"].GetFloat();
267  if (tree.HasMember("gid")) {
268  track.mGID = tree["gid"].GetString();
269  } else {
270  track.mGID = "track";
271  }
272  track.mPolyX.reserve(count.GetInt());
273  track.mPolyY.reserve(count.GetInt());
274  track.mPolyZ.reserve(count.GetInt());
275  for (auto& v : jsonPolyX.GetArray()) {
276  track.mPolyX.push_back(v.GetDouble());
277  }
278  for (auto& v : jsonPolyY.GetArray()) {
279  track.mPolyY.push_back(v.GetDouble());
280  }
281  for (auto& v : jsonPolyZ.GetArray()) {
282  track.mPolyZ.push_back(v.GetDouble());
283  }
284  if (tree.HasMember("mClusters")) {
285  rapidjson::Value& jsonClusters = tree["mClusters"];
286  auto jsonArray = jsonClusters.GetArray();
287  track.mClusters.reserve(jsonArray.Size());
288  for (auto& v : jsonClusters.GetArray()) {
289  track.mClusters.emplace_back(clusterFromJSON(v));
290  }
291  }
292  return track;

296 {

306  gid.SetString(track.mGID.c_str(), track.mGID.size(), allocator);
307  tree.AddMember("gid", gid, allocator);
308  tree.AddMember("time", rapidjson::Value().SetFloat(std::isnan(track.mTime) ? 0 : track.mTime), allocator);
309  tree.AddMember("charge", rapidjson::Value().SetInt(track.mCharge), allocator);
310  tree.AddMember("theta", rapidjson::Value().SetFloat(std::isnan(track.mTheta) ? 0 : track.mTheta), allocator);
311  tree.AddMember("phi", rapidjson::Value().SetFloat(std::isnan(track.mPhi) ? 0 : track.mPhi), allocator);
312  tree.AddMember("eta", rapidjson::Value().SetFloat(std::isnan(track.mEta) ? 0 : track.mEta), allocator);
313  tree.AddMember("PID", rapidjson::Value().SetInt(track.mPID), allocator);
314 
315  jsonStartCoordinates.PushBack((float)track.mStartCoordinates[0], allocator);
316  jsonStartCoordinates.PushBack((float)track.mStartCoordinates[1], allocator);
317  jsonStartCoordinates.PushBack((float)track.mStartCoordinates[2], allocator);
318  tree.AddMember("jsonStartingXYZ", jsonStartCoordinates, allocator);
319 
320  for (size_t i = 0; i < track.getPointCount(); i++) {
321  jsonPolyX.PushBack((float)track.mPolyX[i], allocator);
322  jsonPolyY.PushBack((float)track.mPolyY[i], allocator);
323  jsonPolyZ.PushBack((float)track.mPolyZ[i], allocator);
324  }
325  tree.AddMember("mPolyX", jsonPolyX, allocator);
326  tree.AddMember("mPolyY", jsonPolyY, allocator);
327  tree.AddMember("mPolyZ", jsonPolyZ, allocator);
328 
329  rapidjson::Value jsonClusters(rapidjson::kArrayType);
330 
331  for (auto cluster : track.getClustersSpan()) {
332  jsonClusters.PushBack(jsonTree(cluster, allocator), allocator);
333  }
334  tree.AddMember("mClusters", jsonClusters, allocator);
335 
336  return tree;
337 }
