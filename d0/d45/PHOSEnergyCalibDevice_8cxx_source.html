31 void PHOSEnergyCalibDevice::init(o2::framework::InitContext& ic)
32 {
33  o2::base::GRPGeomHelper::instance().setRequest(mCCDBRequest);
34  mOutputDir = ic.options().get<std::string>("output-dir");
35  mOutputDir = o2::utils::Str::rectifyDirectory(mOutputDir);
36  mMetaFileDir = ic.options().get<std::string>("meta-output-dir");
37  mMetaFileDir = o2::utils::Str::rectifyDirectory(mMetaFileDir);
38  LOG(info) << "Meta dir=" << mMetaFileDir;
39 
40  mPtMin = ic.options().get<float>("ptminmgg");
41  mEminHGTime = ic.options().get<float>("eminhgtime");
42  mEminLGTime = ic.options().get<float>("eminlgtime");
43  mEDigMin = ic.options().get<float>("ecalibdigitmin");
44  mECluMin = ic.options().get<float>("ecalibclumin");
45 
46  LOG(info) << "Energy calibration options";
47  LOG(info) << " output-dif=" << mOutputDir;
48  LOG(info) << " meta-output-dir=" << mMetaFileDir;
49  LOG(info) << " mgg histo ptMin=" << mPtMin;
50  LOG(info) << " Emin for HG time=" << mEminHGTime;
51  LOG(info) << " Emin for LG time=" << mEminLGTime;
52  LOG(info) << " Emin for out digits=" << mEDigMin;
53  LOG(info) << " Cluster Emin for out digits=" << mECluMin;
54 
55  mCalibrator = std::make_unique<PHOSEnergyCalibrator>();
56 
57  // read calibration and bad map objects and send them to calibrator
58  if (!mHasCalib && !mUseCCDB) {
59  mCalibParams = std::make_unique<CalibParams>(1); // Create test calibration coefficients
60  mCalibrator->setCalibration(mCalibParams.get());
61  mBadMap = std::make_unique<BadChannelsMap>(); // Create empty bad map
62  mCalibrator->setBadMap(mBadMap.get());
63  LOG(info) << "No reading BadMap/Calibration from ccdb requested, set default";
64  mHasCalib = true;
65  }
66  mCalibrator->setCuts(mPtMin, mEminHGTime, mEminLGTime, mEminHGTime, mEminLGTime);
67 
68  // Create geometry instance (inclusing reading mis-alignement)
69  // instance will be pick up by Calibrator
70  Geometry::GetInstance("Run3");
71 }

84 void PHOSEnergyCalibDevice::run(o2::framework::ProcessingContext& pc)
85 {
86  updateTimeDependentParams(pc);
87 
88  // Do not use ccdb if already created
89  if (!mHasCalib) { // Default map and calibration was not set, use CCDB
90  LOG(info) << "Getting calib from CCDB";
91  // update BadMap and calibration if necessary
92  auto badMapPtr = pc.inputs().get<o2::phos::BadChannelsMap*>("bdmap");
93  mCalibrator->setBadMap(badMapPtr.get());
94 
95  auto calibPtr = pc.inputs().get<o2::phos::CalibParams*>("clb");
96  mCalibrator->setCalibration(calibPtr.get());
97  mHasCalib = true;
98  }
99  mOutputDigits.clear();
100 
101  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("clusters").header)->startTime; // is this the timestamp of the current TF?
102  const gsl::span<const Cluster>& clusters = pc.inputs().get<gsl::span<Cluster>>("clusters");
103  const gsl::span<const CluElement>& cluelements = pc.inputs().get<gsl::span<CluElement>>("cluelements");
104  const gsl::span<const TriggerRecord>& cluTR = pc.inputs().get<gsl::span<TriggerRecord>>("clusterTriggerRecords");
105 
106  LOG(debug) << "[PHOSEnergyCalibDevice - run] Received " << cluTR.size() << " TRs and " << clusters.size() << " clusters, running calibration";
107  if (mRunStartTime == 0) {
108  const auto ref = pc.inputs().getFirstValid(true);
109  mRunStartTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->creation; // approximate time in ms
110  mRunNumber = DataRefUtils::getHeader<o2::header::DataHeader*>(ref)->runNumber;
111 
112  auto LHCPeriodStr = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("LHCPeriod", "");
113  if (!(LHCPeriodStr.empty())) {
114  mLHCPeriod = LHCPeriodStr;
115  } else {
116  const char* months[12] = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN",
117  "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};
118  std::time_t tt = mRunStartTime / 1000; // ms->s
119  std::tm* ltm = std::gmtime(&tt);
120  mLHCPeriod = (std::string)months[ltm->tm_mon];
121  LOG(warning) << "LHCPeriod is not available, using current month " << mLHCPeriod;
122  }
123  }
124  mCalibrator->process(tfcounter, clusters, cluelements, cluTR, mOutputDigits);
125 
126  fillOutputTree();
127 }

142 void PHOSEnergyCalibDevice::fillOutputTree()
143 {
144 
145  if (mOutputDigits.size() == 0) { // nothing to fill
146  return;
147  }
148 
149  LOG(info) << "Filling tree with " << mOutputDigits.size() << " digits";
150  if (!mFileOut) { // create file and tree
151  mFileName = mOutputDir + fmt::format("PHOS_CalibDigits_{}.root", mRunNumber);
152  LOG(info) << "Creating new tree for PHOS calib digits, output file=" << mFileName;
153  mFileOut = std::make_unique<TFile>(mFileName.c_str(), "recreate");
154  mTreeOut = std::make_unique<TTree>("phosCalibDig", "O2 PHOS calib tree");
155  mFileMetaData = std::make_unique<o2::dataformats::FileMetaData>();
156  mHistoFileName = mOutputDir + fmt::format("PHOS_CalibHistos_{}.root", mRunNumber);
157  mHistoFileOut = std::make_unique<TFile>(mHistoFileName.c_str(), "recreate");
158  mHistoFileMetaData = std::make_unique<o2::dataformats::FileMetaData>();
159  }
160  auto* br = mTreeOut->GetBranch("PHOSCalib");
161  auto* pptr = &mOutputDigits;
162  if (br) {
163  LOG(info) << " Using existing branch";
164  br->SetAddress(&pptr);
165  } else {
166  LOG(info) << " Create new branch";
167  br = mTreeOut->Branch("PHOSCalib", &pptr);
168  }
169  mTreeOut->Fill();
170  br->ResetAddress();
171 }

173 void PHOSEnergyCalibDevice::writeOutFile()
174 {
175  // write collected vector and metadata
176  if (!mTreeOut) { // nothing to write,
177  return;
178  }
179  LOG(info) << "Writing calibration digits";
180  mFileOut->cd();
181  int nbits = mTreeOut->Write();
182  LOG(info) << "Wrote " << nbits << " bits";
183  mTreeOut.reset();
184  mFileOut->Close();
185  mFileOut.reset();
186 
187  // write metaFile data
188  mFileMetaData->fillFileData(mFileName);
189  mFileMetaData->setDataTakingContext(mDataTakingContext);
190  mFileMetaData->type = "calib";
191  mFileMetaData->priority = "high";
192 
193  std::string metaFileNameTmp = mMetaFileDir + fmt::format("PHOS_CalibDigits_{}.tmp", mRunNumber);
194  std::string metaFileName = mMetaFileDir + fmt::format("PHOS_CalibDigits_{}.done", mRunNumber);
195  try {
196  std::ofstream metaFileOut(metaFileNameTmp);
197  metaFileOut << *mFileMetaData.get();
198  metaFileOut.close();
199  std::filesystem::rename(metaFileNameTmp, metaFileName);
200  } catch (std::exception const& e) {
201  LOG(error) << "Failed to store PHOS meta data file " << metaFileName << ", reason: " << e.what();
202  }
203  LOG(info) << "Stored metadate file " << mFileName << ".done";
204  mFileMetaData.reset();
205 
206  LOG(info) << "Writing calibration histograms";
207  mHistoFileOut->cd();
208  mHistoFileOut->WriteObjectAny(mCalibrator->getCollectedHistos(), "o2::phos::ETCalibHistos", "histos");
209  mHistoFileOut->Close();
210  mHistoFileOut.reset();
211 
212  // write metaFile data
213  mHistoFileMetaData->fillFileData(mHistoFileName);
214  mHistoFileMetaData->setDataTakingContext(mDataTakingContext);
215  mHistoFileMetaData->type = "calib";
216  mHistoFileMetaData->priority = "high";
217 
218  metaFileNameTmp = mMetaFileDir + fmt::format("PHOS_CalibHistos_{}.tmp", mRunNumber);
219  metaFileName = mMetaFileDir + fmt::format("PHOS_CalibHistos_{}.done", mRunNumber);
220  try {
221  std::ofstream metaFileOut(metaFileNameTmp);
222  metaFileOut << *mHistoFileMetaData.get();
223  metaFileOut.close();
224  std::filesystem::rename(metaFileNameTmp, metaFileName);
225  } catch (std::exception const& e) {
226  LOG(error) << "Failed to store PHOS histos meta data file " << metaFileName << ", reason: " << e.what();
227  }
228  LOG(info) << "Stored histos metadate file " << mHistoFileName << ".done";
229  mHistoFileMetaData.reset();
230 }
231 o2::framework::DataProcessorSpec o2::phos::getPHOSEnergyCalibDeviceSpec(bool useCCDB)
232 {
233 
234  std::vector<InputSpec> inputs;
235  inputs.emplace_back("clusters", o2::header::gDataOriginPHS, "CLUSTERS", 0, o2::framework::Lifetime::Timeframe);
236  inputs.emplace_back("cluelements", o2::header::gDataOriginPHS, "CLUELEMENTS", 0, o2::framework::Lifetime::Timeframe);
237  inputs.emplace_back("clusterTriggerRecords", o2::header::gDataOriginPHS, "CLUSTERTRIGREC", 0, o2::framework::Lifetime::Timeframe);
238  if (useCCDB) {
239  inputs.emplace_back("bdmap", o2::header::gDataOriginPHS, "PHS_BM", 0, o2::framework::Lifetime::Condition, o2::framework::ccdbParamSpec("PHS/Calib/BadMap"));
240  inputs.emplace_back("clb", o2::header::gDataOriginPHS, "PHS_Calibr", 0, o2::framework::Lifetime::Condition, o2::framework::ccdbParamSpec("PHS/Calib/CalibParams"));
241  }
242  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
243  true, // GRPECS=true
244  false, // GRPLHCIF
245  false, // GRPMagField
246  false, // askMatLUT
247  o2::base::GRPGeomRequest::None, // geometry
248  inputs);
249  using clbUtils = o2::calibration::Utils;
250  std::vector<OutputSpec> outputs;
251 
252  return o2::framework::DataProcessorSpec{"PHOSEnergyCalibDevice",
253  inputs,
254  outputs,
255  o2::framework::adaptFromTask<PHOSEnergyCalibDevice>(useCCDB, ccdbRequest),
256  o2::framework::Options{
257  {"ptminmgg", o2::framework::VariantType::Float, 1.5f, {"minimal pt to fill mgg calib histos"}},
258  {"eminhgtime", o2::framework::VariantType::Float, 1.5f, {"minimal E (GeV) to fill HG time calib histos"}},
259  {"eminlgtime", o2::framework::VariantType::Float, 5.f, {"minimal E (GeV) to fill LG time calib histos"}},
260  {"ecalibdigitmin", o2::framework::VariantType::Float, 0.05f, {"minimal digtit E (GeV) to keep digit for calibration"}},
261  {"ecalibclumin", o2::framework::VariantType::Float, 0.4f, {"minimal cluster E (GeV) to keep digit for calibration"}},
262  {"output-dir", VariantType::String, "./", {"ROOT trees output directory"}},
263  {"meta-output-dir", VariantType::String, "./", {"metafile output directory"}}}};
264 }
