57 void MeanVertexCalibDevice::run(o2::framework::ProcessingContext& pc)
58 {
59  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
60  auto data = pc.inputs().get<gsl::span<o2::dataformats::PrimaryVertex>>("input");
61  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mCalibrator->getCurrentTFInfo());
62  LOG(info) << "Processing TF " << mCalibrator->getCurrentTFInfo().tfCounter << " with " << data.size() << " vertices";
63  mCalibrator->process(data);
64  sendOutput(pc.outputs());
65  const auto& infoVec = mCalibrator->getMeanVertexObjectInfoVector();
66  LOG(info) << "Created " << infoVec.size() << " objects for TF " << mCalibrator->getCurrentTFInfo().tfCounter;
67 }

87 void MeanVertexCalibDevice::sendOutput(DataAllocator& output)
88 {
89 
90  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
91  // TODO in principle, this routine is generic, can be moved to Utils.h
92 
93  using clbUtils = o2::calibration::Utils;
94  const auto& payloadVec = mCalibrator->getMeanVertexObjectVector();
95  auto& infoVec = mCalibrator->getMeanVertexObjectInfoVector(); // use non-const version as we update it
96  assert(payloadVec.size() == infoVec.size());
97 
98  for (uint32_t i = 0; i < payloadVec.size(); i++) {
99  auto& w = infoVec[i];
100  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
101  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
102  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
103 
104  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "MEANVERTEX", i}, *image.get()); // vector<char>
105  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "MEANVERTEX", i}, w); // root-serialized
106  }
107  if (payloadVec.size()) {
108  mCalibrator->initOutput(); // reset the outputs once they are already sent
109  }
110 }

136  "mean-vertex-calibration",
