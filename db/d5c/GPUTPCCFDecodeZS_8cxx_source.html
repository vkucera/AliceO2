37 GPUdii() void GPUTPCCFDecodeZS::decode(GPUTPCClusterFinder& clusterer, GPUSharedMemory& s, int nBlocks, int nThreads, int iBlock, int iThread, int firstHBF)
38 {
39  const unsigned int slice = clusterer.mISlice;
40 #ifdef GPUCA_GPUCODE
41  const unsigned int endpoint = clusterer.mPzsOffsets[iBlock].endpoint;
42 #else
43  const unsigned int endpoint = iBlock;
44 #endif
45  const GPUTrackingInOutZS::GPUTrackingInOutZSSlice& zs = clusterer.GetConstantMem()->ioPtrs.tpcZS->slice[slice];
46  if (zs.count[endpoint] == 0) {
47  return;
48  }
49  ChargePos* positions = clusterer.mPpositions;
50  Array2D<PackedCharge> chargeMap(reinterpret_cast<PackedCharge*>(clusterer.mPchargeMap));
51  const size_t nDigits = clusterer.mPzsOffsets[iBlock].offset;
52  if (iThread == 0) {
53  const int region = endpoint / 2;
54  s.nRowsRegion = clusterer.Param().tpcGeometry.GetRegionRows(region);
55  s.regionStartRow = clusterer.Param().tpcGeometry.GetRegionStart(region);
56  s.nThreadsPerRow = CAMath::Max(1u, nThreads / ((s.nRowsRegion + (endpoint & 1)) / 2));
57  s.rowStride = nThreads / s.nThreadsPerRow;
58  s.rowOffsetCounter = 0;
59  }
60  GPUbarrier();
61  const unsigned int myRow = iThread / s.nThreadsPerRow;
62  const unsigned int mySequence = iThread % s.nThreadsPerRow;
63 #ifdef GPUCA_GPUCODE
64  const unsigned int i = 0;
65  const unsigned int j = clusterer.mPzsOffsets[iBlock].num;
66  {
67  {
68 #else
69  for (unsigned int i = clusterer.mMinMaxCN[endpoint].minC; i < clusterer.mMinMaxCN[endpoint].maxC; i++) {
70  const unsigned int minJ = (i == clusterer.mMinMaxCN[endpoint].minC) ? clusterer.mMinMaxCN[endpoint].minN : 0;
71  const unsigned int maxJ = (i + 1 == clusterer.mMinMaxCN[endpoint].maxC) ? clusterer.mMinMaxCN[endpoint].maxN : zs.nZSPtr[endpoint][i];
72  for (unsigned int j = minJ; j < maxJ; j++) {
73 #endif
74  const unsigned int* pageSrc = (const unsigned int*)(((const unsigned char*)zs.zsPtr[endpoint][i]) + j * TPCZSHDR::TPC_ZS_PAGE_SIZE);
75  CA_SHARED_CACHE_REF(&s.ZSPage[0], pageSrc, TPCZSHDR::TPC_ZS_PAGE_SIZE, unsigned int, pageCache);
76  GPUbarrier();
77  const unsigned char* page = (const unsigned char*)pageCache;
78  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)page;
79  if (o2::raw::RDHUtils::getMemorySize(*rdh) == sizeof(o2::header::RAWDataHeader)) {
80 #ifdef GPUCA_GPUCODE
81  return;
82 #else
83  continue;
84 #endif
85  }
86  const unsigned char* pagePtr = page + sizeof(o2::header::RAWDataHeader);
87  const TPCZSHDR* hdr = reinterpret_cast<const TPCZSHDR*>(pagePtr);
88  pagePtr += sizeof(*hdr);
89  const bool decode12bit = hdr->version == 2;
90  const unsigned int decodeBits = decode12bit ? TPCZSHDR::TPC_ZS_NBITS_V2 : TPCZSHDR::TPC_ZS_NBITS_V1;
91  const float decodeBitsFactor = 1.f / (1 << (decodeBits - 10));
92  unsigned int mask = (1 << decodeBits) - 1;
93  int timeBin = (hdr->timeOffset + (o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstHBF) * o2::constants::lhc::LHCMaxBunches) / LHCBCPERTIMEBIN;
94  const int rowOffset = s.regionStartRow + ((endpoint & 1) ? (s.nRowsRegion / 2) : 0);
95  const int nRows = (endpoint & 1) ? (s.nRowsRegion - s.nRowsRegion / 2) : (s.nRowsRegion / 2);
96 
97  for (int l = 0; l < hdr->nTimeBins; l++) { // TODO: Parallelize over time bins
98  pagePtr += (pagePtr - page) & 1; // Ensure 16 bit alignment
99  const TPCZSTBHDR* tbHdr = reinterpret_cast<const TPCZSTBHDR*>(pagePtr);
100  if ((tbHdr->rowMask & 0x7FFF) == 0) {
101  pagePtr += 2;
102  continue;
103  }
104  const int nRowsUsed = CAMath::Popcount((unsigned int)(tbHdr->rowMask & 0x7FFF));
105  pagePtr += 2 * nRowsUsed;
106 
107  GPUbarrier();
108  for (int n = iThread; n < nRowsUsed; n += nThreads) {
109  const unsigned char* rowData = n == 0 ? pagePtr : (page + tbHdr->rowAddr1()[n - 1]);
110  s.RowClusterOffset[n] = CAMath::AtomicAddShared<unsigned int>(&s.rowOffsetCounter, rowData[2 * *rowData]);
111  }
112  /*if (iThread < GPUCA_WARP_SIZE) { // TODO: Seems to miscompile with HIP, CUDA performance doesn't really change, for now sticking to the AtomicAdd
113  GPUSharedMemory& smem = s;
114  int o;
115  if (iThread < nRowsUsed) {
116  const unsigned char* rowData = iThread == 0 ? pagePtr : (page + tbHdr->rowAddr1()[iThread - 1]);
117  o = rowData[2 * *rowData];
118  } else {
119  o = 0;
120  }
121  int x = warp_scan_inclusive_add(o);
122  if (iThread < nRowsUsed) {
123  s.RowClusterOffset[iThread] = s.rowOffsetCounter + x - o;
124  } else if (iThread == GPUCA_WARP_SIZE - 1) {
125  s.rowOffsetCounter += x;
126  }
127  }*/
128  GPUbarrier();
129 
130  if (myRow < s.rowStride) {
131  for (int m = myRow; m < nRows; m += s.rowStride) {
132  if ((tbHdr->rowMask & (1 << m)) == 0) {
133  continue;
134  }
135  const int rowPos = CAMath::Popcount((unsigned int)(tbHdr->rowMask & ((1 << m) - 1)));
136  size_t nDigitsTmp = nDigits + s.RowClusterOffset[rowPos];
137  const unsigned char* rowData = rowPos == 0 ? pagePtr : (page + tbHdr->rowAddr1()[rowPos - 1]);
138  const int nSeqRead = *rowData;
139  const int nSeqPerThread = (nSeqRead + s.nThreadsPerRow - 1) / s.nThreadsPerRow;
140  const int mySequenceStart = mySequence * nSeqPerThread;
141  const int mySequenceEnd = CAMath::Min(mySequenceStart + nSeqPerThread, nSeqRead);
142  if (mySequenceEnd > mySequenceStart) {
143  const unsigned char* adcData = rowData + 2 * nSeqRead + 1;
144  const unsigned int nSamplesStart = mySequenceStart ? rowData[2 * mySequenceStart] : 0;
145  nDigitsTmp += nSamplesStart;
146  unsigned int nADCStartBits = nSamplesStart * decodeBits;
147  const unsigned int nADCStart = (nADCStartBits + 7) / 8;
148  const int nADC = (rowData[2 * mySequenceEnd] * decodeBits + 7) / 8;
149  adcData += nADCStart;
150  nADCStartBits &= 0x7;
151  unsigned int byte = 0, bits = 0;
152  if (nADCStartBits) { // % 8 != 0
153  bits = 8 - nADCStartBits;
154  byte = ((*(adcData - 1) & (0xFF ^ ((1 << nADCStartBits) - 1)))) >> nADCStartBits;
155  }
156  int nSeq = mySequenceStart;
157  int seqLen = nSeq ? (rowData[(nSeq + 1) * 2] - rowData[nSeq * 2]) : rowData[2];
158  Pad pad = rowData[nSeq++ * 2 + 1];
159  for (int n = nADCStart; n < nADC; n++) {
160  byte |= *(adcData++) << bits;
161  bits += 8;
162  while (bits >= decodeBits) {
163  if (seqLen == 0) {
164  seqLen = rowData[(nSeq + 1) * 2] - rowData[nSeq * 2];
165  pad = rowData[nSeq++ * 2 + 1];
166  }
167  const CfFragment& fragment = clusterer.mPmemory->fragment;
168  TPCTime globalTime = timeBin + l;
169  bool inFragment = fragment.contains(globalTime);
170  Row row = rowOffset + m;
171  ChargePos pos(row, Pad(pad), inFragment ? fragment.toLocal(globalTime) : INVALID_TIME_BIN);
172  positions[nDigitsTmp++] = pos;
173 
174  if (inFragment) {
175  float q = float(byte & mask) * decodeBitsFactor;
176  q *= clusterer.GetConstantMem()->calibObjects.tpcPadGain->getGainCorrection(slice, row, pad);
177  chargeMap[pos] = PackedCharge(q);
178  }
179  pad++;
180  byte = byte >> decodeBits;
181  bits -= decodeBits;
182  seqLen--;
183  }
184  }
185  }
186  }
187  }
188  if (nRowsUsed > 1) {
189  pagePtr = page + tbHdr->rowAddr1()[nRowsUsed - 2];
190  }
191  pagePtr += 2 * *pagePtr; // Go to entry for last sequence length
192  pagePtr += 1 + (*pagePtr * decodeBits + 7) / 8; // Go to beginning of next time bin
193  }
194  }
195  }
196 }

300 GPUd() void GPUTPCCFDecodeZSLink::DecodeTBSingleThread(
301  processorType& clusterer,
302  const unsigned char* adcData,
303  unsigned int nAdc,
304  const uint32_t* channelMask,
305  int timeBin,
306  int cru,
307  int fecInPartition,
308  size_t pageDigitOffset)
309 {
310  const CfFragment& fragment = clusterer.mPmemory->fragment;
311 
312  if CONSTEXPR17 (TPCZSHDRV2::TIGHTLY_PACKED_V3) {
313 
314  unsigned int byte = 0, bits = 0, posXbits = 0, rawFECChannel = 0;
315 
316  // unpack adc values, assume tightly packed data
317  while (posXbits < nAdc) {
318  byte |= adcData[0] << bits;
319  adcData++;
320  bits += CHAR_BIT;
321  while (bits >= DECODE_BITS) {
322 
323  // Find next channel with data
324  for (; !ChannelIsActive(channelMask, rawFECChannel); rawFECChannel++) {
325  }
326 
327  // Unpack data for cluster finder
328  o2::tpc::PadPos padAndRow = GetPadAndRowFromFEC(clusterer, cru, rawFECChannel, fecInPartition);
329 
330  WriteCharge(clusterer, byte, padAndRow, fragment.toLocal(timeBin), pageDigitOffset + posXbits);
331 
332  byte = byte >> DECODE_BITS;
333  bits -= DECODE_BITS;
334  posXbits++;
335  rawFECChannel++; // Ensure we don't decode same channel twice
336  } // while (bits >= DECODE_BITS)
337  } // while (posXbits < nAdc)
338 
339  } else { // ! TPCZSHDRV2::TIGHTLY_PACKED_V3
340  unsigned int rawFECChannel = 0;
341  const unsigned long* adcData64 = (const unsigned long*)adcData;
342  for (unsigned int j = 0; j < nAdc; j++) {
343  for (; !ChannelIsActive(channelMask, rawFECChannel); rawFECChannel++) {
344  }
345 
346  unsigned int adc = (adcData64[j / TPCZSHDRV2::SAMPLESPER64BIT] >> ((j % TPCZSHDRV2::SAMPLESPER64BIT) * DECODE_BITS)) & DECODE_MASK;
347 
348  o2::tpc::PadPos padAndRow = GetPadAndRowFromFEC(clusterer, cru, rawFECChannel, fecInPartition);
349  WriteCharge(clusterer, adc, padAndRow, fragment.toLocal(timeBin), pageDigitOffset + j);
350  rawFECChannel++;
351  }
352  }
353 }

446 GPUd() void GPUTPCCFDecodeZSLink::WriteCharge(processorType& clusterer, unsigned int adc, PadPos padAndRow, TPCFragmentTime localTime, size_t positionOffset)

460 GPUd() o2::tpc::PadPos GPUTPCCFDecodeZSLink::GetPadAndRowFromFEC(processorType& clusterer, int cru, int rawFECChannel, int fecInPartition)

491 GPUd() bool GPUTPCCFDecodeZSLink::ChannelIsActive(const uint32_t* chan, unsigned char chanIndex)
492 {
493  if (chanIndex >= zerosupp_link_based::ChannelPerTBHeader) {
494  return false;
495  }
496  constexpr unsigned char N_BITS_PER_ENTRY = sizeof(*chan) * CHAR_BIT;
