106 int CTPConfiguration::loadConfigurationRun3(const std::string& ctpconfiguration)
107 {
108  LOG(info) << "Loading CTP configuration.";
109  mConfigString = ctpconfiguration;
110  std::istringstream iss(ctpconfiguration);
111  int ret = 0;
112  int level = MASKS;
113  std::string line;
114  while (std::getline(iss, line)) {
115  o2::utils::Str::trim(line);
116  if ((ret = processConfigurationLineRun3(line, level)) != 0) {
117  return ret;
118  }
119  }
120  return ret;
121  return 0;

123 int CTPConfiguration::processConfigurationLineRun3(std::string& line, int& level)
124 {
125  LOG(info) << "Processing line";
126  LOG(info) << "line:" << line << " lev:" << level;
127  if (line.size() == 0) {
128  return 0;
129  }
130  if (line.at(0) == '#') {
131  return 0;
132  }
133  //
134  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
135  size_t ntokens = tokens.size();
136  if (ntokens == 0) {
137  LOG(warning) << "# of tokens zero in line:" << line;
138  return 0;
139  }
140  size_t first;
141  if ((first = line.find("run")) != std::string::npos) {
142  level = RUN;
143  } else if (CTPGenerator::Generators.count(tokens[0])) {
144  if (level != CLASS) {
145  level = GENS;
146  }
147  level = GENS;
148  } else if ((first = line.find("bcm")) != std::string::npos) {
149  if (level == MASKS) {
150  level = MASKS;
151  }
152  } else if ((first = line.find("LTG")) != std::string::npos) {
153  level = LTG;
154  } else if ((first = line.find("cluster")) != std::string::npos) {
155  level = CLUSTER;
156  } else {
157  bool knownlevels = ((level == LTGitems) || (level == CLASS));
158  if (knownlevels == false) {
159  level = UNKNOWN;
160  }
161  }
162  LOG(info) << "Level:" << level;
163  switch (level) {
164  case RUN: {
165  mRunNumber = std::stoul(tokens[1]);
166  level = MASKS;
167  break;
168  }
169  case MASKS: {
170  BCMask bcmask;
171  if (tokens.size() < 3) {
172  LOG(error) << "Wrong bc mask:" << line;
173  break;
174  }
175  bcmask.name = tokens[1];
176  for (int i = 2; i < tokens.size(); i++) {
177  bcmask.mask = tokens[i] = " ";
178  }
179  bool coded = tokens[2].find("L") != std::string::npos;
180  coded |= tokens[2].find("H") != std::string::npos;
181  std::cout << "coded:" << coded << std::endl;
182  if (coded) {
183  // jusko notation
184  } else {
185  // list of integers
186  for (int i = 2; i < ntokens; i++) {
187  uint32_t bc;
188  try {
189  bc = std::stoull(tokens[i]);
190  } catch (...) {
191  LOG(info) << "mask syntax:" << tokens[i];
192  continue;
193  }
194  bcmask.BCmask.set(bc, 1);
195  }
196  }
197  mBCMasks.push_back(bcmask);
198  LOG(info) << "BC mask added:" << bcmask.name;
199  break;
200  }
201  case GENS: {
202  CTPGenerator gen;
203  gen.name = tokens[0];
204  gen.frequency = tokens[1];
205  LOG(info) << "Gen added:" << line;
206  break;
207  }
208  case LTG: {
209  CTPDetector ctpdet;
210  std::string detname = tokens[1];
211  capitaliseString(detname);
212  o2::detectors::DetID det(detname.c_str());
213  if (isDetector(det)) {
214  ctpdet.detID = det.getID();
215  LOG(info) << "Detector found:" << det.getID() << " " << detname;
216  } else {
217  LOG(info) << "Unknown detectors:" << line;
218  }
219  mDetectors.push_back(ctpdet);
220  level = LTGitems;
221  break;
222  }
223  case LTGitems: {
224  if (ntokens == 1) {
225  mDetectors.back().mode = tokens[0];
226  }
227  LOG(info) << "LTGitem:" << line;
228  break;
229  }
230  case CLUSTER: {
231  CTPCluster cluster;
232  try {
233  cluster.hwMask = std::stoull(tokens[0]);
234  } catch (...) {
235  LOG(info) << "Cluster syntax error:" << line;
236  return level;
237  }
238  LOG(info) << "Cluster:" << line;
239  cluster.name = tokens[2];
240  o2::detectors::DetID::mask_t mask;
241  for (int item = 3; item < ntokens; item++) {
242  std::string detname = tokens[item];
243  capitaliseString(detname);
244  // LOG(info) << "Detector:" << detname;
245  o2::detectors::DetID det(detname.c_str());
246  isDetector(det);
247  mask |= det.getMask();
248  }
249  cluster.maskCluster = mask;
250  mClusters.push_back(cluster);
251  level = CLASS;
252  // LOG(info) << "Cluster done:" << cluster.name << std::endl;
253  break;
254  }
255  case CLASS: {
256  // add to the last cluster
257  if (tokens.size() < 3) {
258  LOG(error) << "CTPClass: less than 3 items in class:" << line;
259  break;
260  }
261  uint64_t index;
262  try {
263  index = std::stoull(tokens[0]);
264  } catch (...) {
265  LOG(info) << "Class syntax error:" << line;
266  return level;
267  }
268  LOG(info) << "Class:" << line;
269  CTPClass cls;
270  cls.classMask = 1ull << index;
271  cls.name = tokens[1];
272  cls.clusterIndex = mClusters.size() - 1;
273  // LOG(info) << "point:" << cls.cluster << " " << &mClusters.front();
274  // descriptor
275  CTPDescriptor* desc = new CTPDescriptor;
276  desc->name = tokens[2];
277  mDescriptors.push_back(*desc);
278  cls.descriptorIndex = mDescriptors.size() - 1;
279  //
280  mCTPClasses.push_back(cls);
281  break;
282  }
283  default: {
284  LOG(info) << "unknown line:" << line;
285  }
286  }
287  for (auto& cls : mCTPClasses) {
288  cls.cluster = &mClusters[cls.clusterIndex];
289  }
290  return 0;
291 }

456 int CTPRunManager::addScalers(uint32_t irun, std::time_t time)
457 {
458  if (mActiveRuns[irun] == nullptr) {
459  LOG(error) << "No config for run index:" << irun;
460  return 1;
461  }
462  std::string orb = "extorb";
463  CTPScalerRecordRaw scalrec;
464  scalrec.epochTime = time;
465  std::vector<int> clslist = mActiveRuns[irun]->cfg.getTriggerClassList();
466  for (auto const& cls : clslist) {
467  std::string cmb = "clamb" + std::to_string(cls + 1);
468  std::string cma = "clama" + std::to_string(cls + 1);
469  std::string c0b = "cla0b" + std::to_string(cls + 1);
470  std::string c0a = "cla0a" + std::to_string(cls + 1);
471  std::string c1b = "cla1b" + std::to_string(cls + 1);
472  std::string c1a = "cla1a" + std::to_string(cls + 1);
473  CTPScalerRaw scalraw;
474  scalraw.classIndex = (uint32_t)cls;
475  // std::cout << "cls:" << cls << " " << scalraw.classIndex << std::endl;
476  scalraw.lmBefore = mCounters[mScalerName2Position[cmb]];
477  scalraw.lmAfter = mCounters[mScalerName2Position[cma]];
478  scalraw.l0Before = mCounters[mScalerName2Position[c0b]];
479  scalraw.l0After = mCounters[mScalerName2Position[c0a]];
480  scalraw.l1Before = mCounters[mScalerName2Position[c1b]];
481  scalraw.l1After = mCounters[mScalerName2Position[c1a]];
482  // std::cout << "positions:" << cmb << " " << mScalerName2Position[cmb] << std::endl;
483  // std::cout << "positions:" << cma << " " << mScalerName2Position[cma] << std::endl;
484  scalrec.scalers.push_back(scalraw);
485  }
486  // detectors
487  // std::vector<std::string> detlist = mActiveRuns[irun]->cfg.getDetectorList();
488  o2::detectors::DetID::mask_t detmask = mActiveRuns[irun]->cfg.getDetectorMask();
489  for (uint32_t i = 0; i < 32; i++) {
490  o2::detectors::DetID::mask_t deti = 1ul << i;
491  bool detin = (detmask & deti).count();
492  if (detin) {
493  std::string detname(o2::detectors::DetID::getName(i));
494  std::string countername = detname + CTPConfiguration::detName2LTG.at(detname) + "_PH";
495  uint32_t detcount = mCounters[mScalerName2Position[countername]];
496  scalrec.scalersDets.push_back(detcount);
497  LOG(info) << "Scaler for detector:" << countername << ":" << detcount;
498  }
499  }
500  //
501  scalrec.intRecord.orbit = mCounters[mScalerName2Position[orb]];
502  scalrec.intRecord.bc = 0;
503  mActiveRuns[irun]->scalers.addScalerRacordRaw(scalrec);
504  LOG(info) << "Adding scalers for orbit:" << scalrec.intRecord.orbit;
505  // scalrec.printStream(std::cout);
506  // printCounters();
507  return 0;
508 }
509 int CTPRunManager::processMessage(std::string& topic, const std::string& message)
510 {
511  if (mQC == 1) {
512  LOG(info) << "processMessage: skipping, QC=1";
513  return 1;
514  }
515  LOG(info) << "Processing message with topic:" << topic;
516  std::string firstcounters;
517  if (topic.find("ctpconfig") != std::string::npos) {
518  LOG(info) << "ctpcfg received";
519  startRun(message);
520  mCtpcfg = 1;
521  return 0;
522  }
523  if (topic.find("sox") != std::string::npos) {
524  // get config
525  size_t irun = message.find("run");
526  if (irun == std::string::npos) {
527  LOG(error) << "run keyword not found in SOX:\n"
528  << message;
529  return 1;
530  }
531  LOG(info) << "SOX received, Run keyword position:" << irun;
532  if (mCtpcfg == 0) {
533  std::string cfg = message.substr(irun, message.size() - irun);
534  LOG(info) << "Config:" << cfg;
535  startRun(cfg);
536  } else {
537  mCtpcfg = 0;
538  }
539  firstcounters = message.substr(0, irun);
540  }
541  if (topic.find("eox") != std::string::npos) {
542  LOG(info) << "EOX received";
543  mEOX = 1;
544  }
545  //
546  std::vector<std::string> tokens;
547  if (firstcounters.size() > 0) {
548  tokens = o2::utils::Str::tokenize(firstcounters, ' ');
549  } else {
550  tokens = o2::utils::Str::tokenize(message, ' ');
551  }
552  if (tokens.size() != (CTPRunScalers::NCOUNTERS + 1)) {
553  LOG(error) << "Scalers size wrong:" << tokens.size() << " expected:" << CTPRunScalers::NCOUNTERS + 1;
554  return 1;
555  }
556  double timeStamp = std::stold(tokens.at(0));
557  std::time_t tt = timeStamp;
558  LOG(info) << "Processing scalers, all good, time:" << tokens.at(0) << " " << std::asctime(std::localtime(&tt));
559  for (int i = 1; i < tokens.size(); i++) {
560  mCounters[i - 1] = std::stoull(tokens.at(i));
561  if (i < (NRUNS + 1)) {
562  std::cout << mCounters[i - 1] << " ";
563  }
564  }
565  std::cout << std::endl;
566  LOG(info) << "Counter size:" << tokens.size();
567  //
568  for (uint32_t i = 0; i < NRUNS; i++) {
569  if ((mCounters[i] == 0) && (mActiveRunNumbers[i] == 0)) {
570  // not active
571  } else if ((mCounters[i] != 0) && (mActiveRunNumbers[i] == mCounters[i])) {
572  // active , do scalers
573  LOG(info) << "Run continue:" << mCounters[i];
574  addScalers(i, tt);
575  } else if ((mCounters[i] != 0) && (mActiveRunNumbers[i] == 0)) {
576  LOG(info) << "Run started:" << mCounters[i];
577  mActiveRunNumbers[i] = mCounters[i];
578  if (mRunInStart == nullptr) {
579  LOG(error) << "Internal error in processMessage: nullptr != 0 expected";
580  }
581  mActiveRuns[i] = mRunInStart;
582  mRunInStart = nullptr;
583  addScalers(i, tt);
584  } else if ((mCounters[i] == 0) && (mActiveRunNumbers[i] != 0)) {
585  if (mEOX != 1) {
586  LOG(error) << "Internal error in processMessage: mEOX != 1 expected 0: mEOX:" << mEOX;
587  }
588  LOG(info) << "Run stopped:" << mActiveRunNumbers[i];
589  addScalers(i, tt);
590  mActiveRunNumbers[i] = 0;
591  mEOX = 0;
592  stopRun(i);
593  }
594  }
595  mEOX = 0;
596  printActiveRuns();
597  return 0;
598 }

607 int CTPRunManager::saveRunScalersToCCDB(int i)
608 {
609  // data base
610  CTPActiveRun* run = mActiveRuns[i];
