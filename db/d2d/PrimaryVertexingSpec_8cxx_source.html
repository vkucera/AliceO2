70 void PrimaryVertexingSpec::init(InitContext& ic)
71 {
72  o2::base::GRPGeomHelper::instance().setRequest(mGGCCDBRequest);
73  //-------- init geometry and field --------//
74  o2::base::GeometryManager::loadGeometry();
75  o2::base::Propagator::initFieldFromGRP();
76 
77  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
78  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
79  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
80  if (o2::utils::Str::pathExists(matLUTFile)) {
81  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
82  o2::base::Propagator::Instance()->setMatLUT(lut);
83  LOG(info) << "Loaded material LUT from " << matLUTFile;
84  } else {
85  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
86  }
87  mTimer.Stop();
88  mTimer.Reset();
89  mVertexer.setValidateWithIR(mValidateWithIR);
90 
91  // set bunch filling. Eventually, this should come from CCDB
92  const auto* digctx = o2::steer::DigitizationContext::loadFromFile();
93  const auto& bcfill = digctx->getBunchFilling();
94  mVertexer.setBunchFilling(bcfill);
95 }

97 void PrimaryVertexingSpec::run(ProcessingContext& pc)
98 {
99  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
100  mTimer.Start(false);
101  std::vector<PVertex> vertices;
102  std::vector<GIndex> vertexTrackIDs;
103  std::vector<V2TRef> v2tRefs;
104  std::vector<o2::MCEventLabel> lblVtx;
105 
106  if (!mSkip) {
107  o2::globaltracking::RecoContainer recoData;
108  recoData.collectData(pc, *mDataRequest.get()); // select tracks of needed type, with minimal cuts, the real selected will be done in the vertexer
109  updateTimeDependentParams(pc); // Make sure this is called after recoData.collectData, which may load some conditions
110 
111  std::vector<TrackWithTimeStamp> tracks;
112  std::vector<o2::MCCompLabel> tracksMCInfo;
113  std::vector<o2d::GlobalTrackID> gids;
114  auto maxTrackTimeError = PVertexerParams::Instance().maxTimeErrorMUS;
115  auto halfROFITS = 0.5 * mITSROFrameLengthMUS;
116  auto hw2ErrITS = 2.f / std::sqrt(12.f) * mITSROFrameLengthMUS; // conversion from half-width to error for ITS
117 
118  auto creator = [maxTrackTimeError, hw2ErrITS, halfROFITS, &tracks, &gids](auto& _tr, GTrackID _origID, float t0, float terr) {
119  if (!_origID.includesDet(DetID::ITS)) {
120  return true; // just in case this selection was not done on RecoContainer filling level
121  }
122  if constexpr (isITSTrack<decltype(_tr)>()) {
123  t0 += halfROFITS; // ITS time is supplied in \mus as beginning of ROF
124  terr *= hw2ErrITS; // error is supplied as a half-ROF duration, convert to \mus
125  }
126  // for all other tracks the time is in \mus with gaussian error
127  if constexpr (std::is_base_of_v<o2::track::TrackParCov, std::decay_t<decltype(_tr)>>) {
128  if (terr < maxTrackTimeError) {
129  tracks.emplace_back(TrackWithTimeStamp{_tr, {t0, terr}});
130  gids.emplace_back(_origID);
131  }
132  }
133  return true;
134  };
135 
136  recoData.createTracksVariadic(creator); // create track sample considered for vertexing
137 
138  if (mUseMC) {
139  recoData.fillTrackMCLabels(gids, tracksMCInfo);
140  }
141  mVertexer.setStartIR(recoData.startIR);
142  std::vector<o2::InteractionRecord> ft0Data;
143  if (mValidateWithIR) { // select BCs for validation
144  const o2::ft0::InteractionTag& ft0Params = o2::ft0::InteractionTag::Instance();
145  auto ft0all = recoData.getFT0RecPoints();
146  for (const auto& ftRP : ft0all) {
147  if (ft0Params.isSelected(ftRP)) {
148  ft0Data.push_back(ftRP.getInteractionRecord());
149  }
150  }
151  }
152  mVertexer.process(tracks, gids, ft0Data, vertices, vertexTrackIDs, v2tRefs, tracksMCInfo, lblVtx);
153  }
154 
155  pc.outputs().snapshot(Output{"GLO", "PVTX", 0, Lifetime::Timeframe}, vertices);
156  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe}, v2tRefs);
157  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTID", 0, Lifetime::Timeframe}, vertexTrackIDs);
158 
159  if (mUseMC) {
160  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0, Lifetime::Timeframe}, lblVtx);
161  }
162 
163  mTimer.Stop();
164  LOG(info) << "Found " << vertices.size() << " primary vertices, timing: CPU: "
165  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
166 }

187 void PrimaryVertexingSpec::updateTimeDependentParams(ProcessingContext& pc)
188 {
189  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
190  static bool initOnceDone = false;
191  if (!initOnceDone) { // this params need to be queried only once
192  initOnceDone = true;
193  // Note: reading of the ITS AlpideParam needed for ITS timing is done by the RecoContainer
194  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
195  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
196  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
197  mITSROFrameLengthMUS = alpParams.roFrameLengthTrig / 1.e3; // ITS ROFrame duration in \mus
198  } else {
199  mITSROFrameLengthMUS = alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS ROFrame duration in \mus
200  }
201  mVertexer.setITSROFrameLength(mITSROFrameLengthMUS);
202  mVertexer.init();
203  }
204  // we may have other params which need to be queried regularly
205 }

207 DataProcessorSpec getPrimaryVertexingSpec(GTrackID::mask_t src, bool skip, bool validateWithFT0, bool useMC)
208 {
209  std::vector<OutputSpec> outputs;
210  auto dataRequest = std::make_shared<DataRequest>();
211 
212  dataRequest->requestTracks(src, useMC);
213  if (validateWithFT0 && src[GTrackID::FT0]) {
214  dataRequest->requestFT0RecPoints(false);
215  }
216 
217  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
218  outputs.emplace_back("GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
219  outputs.emplace_back("GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
220 
221  if (useMC) {
222  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
223  }
224 
225  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
226  true, // GRPECS=true
227  false, // GRPLHCIF
228  false, // GRPMagField
229  false, // askMatLUT
230  o2::base::GRPGeomRequest::None, // geometry
231  dataRequest->inputs);
232 
233  return DataProcessorSpec{
234  "primary-vertexing",
235  dataRequest->inputs,
236  outputs,
237  AlgorithmSpec{adaptFromTask<PrimaryVertexingSpec>(dataRequest, ggRequest, skip, validateWithFT0, useMC)},
238  Options{{"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
239 }
