36 template <TrackletMode Mode>
37 void trackleterKernelSerial(
38  const gsl::span<const Cluster>& clustersNextLayer, // 0 2
39  const gsl::span<const Cluster>& clustersCurrentLayer, // 1 1
40  int* indexTableNext,
41  const float phiCut,
42  std::vector<Tracklet>& Tracklets,
43  gsl::span<int> foundTracklets,
44  const IndexTableUtils& utils,
45  const int maxTrackletsPerCluster = static_cast<int>(2e3))
46 {
47  const int PhiBins{utils.getNphiBins()};
48  const int ZBins{utils.getNzBins()};
49  // loop on layer1 clusters
50  for (unsigned int iCurrentLayerClusterIndex{0}; iCurrentLayerClusterIndex < clustersCurrentLayer.size(); ++iCurrentLayerClusterIndex) {
51  int storedTracklets{0};
52  const Cluster& currentCluster{clustersCurrentLayer[iCurrentLayerClusterIndex]};
53  const int4 selectedBinsRect{VertexerTraits::getBinsRect(currentCluster, (int)Mode, 0.f, 50.f, phiCut / 2, utils)};
54  if (selectedBinsRect.x != 0 || selectedBinsRect.y != 0 || selectedBinsRect.z != 0 || selectedBinsRect.w != 0) {
55  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
56  if (phiBinsNum < 0) {
57  phiBinsNum += PhiBins;
58  }
59  // loop on phi bins next layer
60  for (int iPhiBin{selectedBinsRect.y}, iPhiCount{0}; iPhiCount < phiBinsNum; iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
61  const int firstBinIndex{utils.getBinIndex(selectedBinsRect.x, iPhiBin)};
62  const int firstRowClusterIndex{indexTableNext[firstBinIndex]};
63  const int maxRowClusterIndex{indexTableNext[firstBinIndex + ZBins]};
64  // loop on clusters next layer
65  for (int iNextLayerClusterIndex{firstRowClusterIndex}; iNextLayerClusterIndex < maxRowClusterIndex && iNextLayerClusterIndex < static_cast<int>(clustersNextLayer.size()); ++iNextLayerClusterIndex) {
66  const Cluster& nextCluster{clustersNextLayer[iNextLayerClusterIndex]};
67  if (o2::gpu::GPUCommonMath::Abs(currentCluster.phi - nextCluster.phi) < phiCut) {
68  if (storedTracklets < maxTrackletsPerCluster) {
69  if constexpr (Mode == TrackletMode::Layer0Layer1) {
70  Tracklets.emplace_back(iNextLayerClusterIndex, iCurrentLayerClusterIndex, nextCluster, currentCluster);
71  } else {
72  Tracklets.emplace_back(iCurrentLayerClusterIndex, iNextLayerClusterIndex, currentCluster, nextCluster);
73  }
74  ++storedTracklets;
75  }
76  }
77  }
78  }
79  }
80  foundTracklets[iCurrentLayerClusterIndex] = storedTracklets;
81  }
82 }

84 void trackletSelectionKernelSerial(
85  const gsl::span<const Cluster> clusters0, // 0
86  const gsl::span<const Cluster> clusters1, // 1
87  const gsl::span<const Tracklet>& tracklets01,
88  const gsl::span<const Tracklet>& tracklets12,
89  const gsl::span<int> foundTracklets01,
90  const gsl::span<int> foundTracklets12,
91  std::vector<Line>& destTracklets,
92  const float tanLambdaCut = 0.025f,
93  const float phiCut = 0.005f,
94  const int maxTracklets = static_cast<int>(1e2))
95 {
96  int offset01{0}, offset12{0};
97  std::vector<bool> usedTracklets(tracklets01.size(), false);
98  for (unsigned int iCurrentLayerClusterIndex{0}; iCurrentLayerClusterIndex < clusters1.size(); ++iCurrentLayerClusterIndex) {
99  int validTracklets{0};
100  for (int iTracklet12{offset12}; iTracklet12 < offset12 + foundTracklets12[iCurrentLayerClusterIndex]; ++iTracklet12) {
101  for (int iTracklet01{offset01}; iTracklet01 < offset01 + foundTracklets01[iCurrentLayerClusterIndex]; ++iTracklet01) {
102  const float deltaTanLambda{o2::gpu::GPUCommonMath::Abs(tracklets01[iTracklet01].tanLambda - tracklets12[iTracklet12].tanLambda)};
103  const float deltaPhi{o2::gpu::GPUCommonMath::Abs(tracklets01[iTracklet01].phi - tracklets12[iTracklet12].phi)};
104  if (!usedTracklets[iTracklet01] && deltaTanLambda < tanLambdaCut && deltaPhi < phiCut && validTracklets != maxTracklets) {
105  usedTracklets[iTracklet01] = true;
106  destTracklets.emplace_back(tracklets01[iTracklet01], clusters0.data(), clusters1.data());
107  ++validTracklets;
108  }
109  }
110  }
111  offset01 += foundTracklets01[iCurrentLayerClusterIndex];
112  offset12 += foundTracklets12[iCurrentLayerClusterIndex];
113  }

116 const std::vector<std::pair<int, int>> VertexerTraits::selectClusters(const int* indexTable,
117  const std::array<int, 4>& selectedBinsRect,
118  const IndexTableUtils& utils)
119 {
120  std::vector<std::pair<int, int>> filteredBins{};
121  int phiBinsNum{selectedBinsRect[3] - selectedBinsRect[1] + 1};
122  if (phiBinsNum < 0) {
123  phiBinsNum += utils.getNphiBins();
124  }
125  filteredBins.reserve(phiBinsNum);
126  for (int iPhiBin{selectedBinsRect[1]}, iPhiCount{0}; iPhiCount < phiBinsNum;
127  iPhiBin = ++iPhiBin == utils.getNphiBins() ? 0 : iPhiBin, iPhiCount++) {
128  const int firstBinIndex{utils.getBinIndex(selectedBinsRect[0], iPhiBin)};
129  filteredBins.emplace_back(
130  indexTable[firstBinIndex],
131  utils.countRowSelectedBins(indexTable, iPhiBin, selectedBinsRect[0], selectedBinsRect[2]));
132  }
133  return filteredBins;
134 }

244 void VertexerTraits::computeVertices()
245 {
246 #ifdef VTX_DEBUG
247  std::vector<std::vector<ClusterLines>> dbg_clusLines(mTimeFrame->getNrof());
248 #endif
249  std::vector<int> noClustersVec(mTimeFrame->getNrof(), 0);
250  for (int rofId{0}; rofId < mTimeFrame->getNrof(); ++rofId) {
251  const int numTracklets{static_cast<int>(mTimeFrame->getLines(rofId).size())};
252  std::vector<bool> usedTracklets(numTracklets, false);
253  for (int tracklet1{0}; tracklet1 < numTracklets; ++tracklet1) {
254  if (usedTracklets[tracklet1]) {
255  continue;
256  }
257  for (int tracklet2{tracklet1 + 1}; tracklet2 < numTracklets; ++tracklet2) {
258  if (usedTracklets[tracklet2]) {
259  continue;
260  }
261  if (Line::getDCA(mTimeFrame->getLines(rofId)[tracklet1], mTimeFrame->getLines(rofId)[tracklet2]) < mVrtParams.pairCut) {
262  mTimeFrame->getTrackletClusters(rofId).emplace_back(tracklet1, mTimeFrame->getLines(rofId)[tracklet1], tracklet2, mTimeFrame->getLines(rofId)[tracklet2]);
263  std::array<float, 3> tmpVertex{mTimeFrame->getTrackletClusters(rofId).back().getVertex()};
264  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
265  mTimeFrame->getTrackletClusters(rofId).pop_back();
266  break;
267  }
268  usedTracklets[tracklet1] = true;
269  usedTracklets[tracklet2] = true;
270  for (int tracklet3{0}; tracklet3 < numTracklets; ++tracklet3) {
271  if (usedTracklets[tracklet3]) {
272  continue;
273  }
274  if (Line::getDistanceFromPoint(mTimeFrame->getLines(rofId)[tracklet3], tmpVertex) < mVrtParams.pairCut) {
275  mTimeFrame->getTrackletClusters(rofId).back().add(tracklet3, mTimeFrame->getLines(rofId)[tracklet3]);
276  usedTracklets[tracklet3] = true;
277  tmpVertex = mTimeFrame->getTrackletClusters(rofId).back().getVertex();
278  }
279  }
280  break;
281  }
282  }
283  }
284  std::sort(mTimeFrame->getTrackletClusters(rofId).begin(), mTimeFrame->getTrackletClusters(rofId).end(),
285  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
286  noClustersVec[rofId] = static_cast<int>(mTimeFrame->getTrackletClusters(rofId).size());
287  for (int iCluster1{0}; iCluster1 < noClustersVec[rofId]; ++iCluster1) {
288  std::array<float, 3> vertex1{mTimeFrame->getTrackletClusters(rofId)[iCluster1].getVertex()};
289  std::array<float, 3> vertex2{};
290  for (int iCluster2{iCluster1 + 1}; iCluster2 < noClustersVec[rofId]; ++iCluster2) {
291  vertex2 = mTimeFrame->getTrackletClusters(rofId)[iCluster2].getVertex();
292  if (std::abs(vertex1[2] - vertex2[2]) < mVrtParams.clusterCut) {
293  float distance{(vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
294  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
295  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2])};
296  if (distance < mVrtParams.pairCut * mVrtParams.pairCut) {
297  for (auto label : mTimeFrame->getTrackletClusters(rofId)[iCluster2].getLabels()) {
298  mTimeFrame->getTrackletClusters(rofId)[iCluster1].add(label, mTimeFrame->getLines(rofId)[label]);
299  vertex1 = mTimeFrame->getTrackletClusters(rofId)[iCluster1].getVertex();
300  }
301  }
302  mTimeFrame->getTrackletClusters(rofId).erase(mTimeFrame->getTrackletClusters(rofId).begin() + iCluster2);
303  --iCluster2;
304  --noClustersVec[rofId];
305  }
306  }
307  }
308  }
309 
310  for (int rofId{0}; rofId < mTimeFrame->getNrof(); ++rofId) {
311 #ifdef VTX_DEBUG
312  for (auto& cl : mTimeFrame->getTrackletClusters(rofId)) {
313  dbg_clusLines[rofId].push_back(cl);
314  }
315 #endif
316  for (int iCluster{0}; iCluster < noClustersVec[rofId]; ++iCluster) {
317  if (mTimeFrame->getTrackletClusters(rofId)[iCluster].getSize() < mVrtParams.clusterContributorsCut && noClustersVec[rofId] > 1) {
318  mTimeFrame->getTrackletClusters(rofId).erase(mTimeFrame->getTrackletClusters(rofId).begin() + iCluster);
319  noClustersVec[rofId]--;
320  continue;
321  }
322  if (mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[0] * mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[0] +
323  mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[1] * mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[1] <
324  1.98 * 1.98) {
325  mVertices.emplace_back(mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[0],
326  mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[1],
327  mTimeFrame->getTrackletClusters(rofId)[iCluster].getVertex()[2],
328  mTimeFrame->getTrackletClusters(rofId)[iCluster].getRMS2(), // Symm matrix. Diagonal: RMS2 components,
329  // off-diagonal: square mean of projections on planes.
330  mTimeFrame->getTrackletClusters(rofId)[iCluster].getSize(), // Contributors
331  mTimeFrame->getTrackletClusters(rofId)[iCluster].getAvgDistance2(), // In place of chi2
332  rofId);
333  }
334  }
335  std::vector<Vertex> vertices;
336  for (auto& vertex : mVertices) {
337  vertices.emplace_back(o2::math_utils::Point3D<float>(vertex.mX, vertex.mY, vertex.mZ), vertex.mRMS2, vertex.mContributors, vertex.mAvgDistance2);
338  vertices.back().setTimeStamp(vertex.mTimeStamp);
339  }
340  mTimeFrame->addPrimaryVertices(vertices);
341  mVertices.clear();
342  }
343 #ifdef VTX_DEBUG
344  TFile* dbg_file = TFile::Open("artefacts_tf.root", "update");
345  TTree* ln_clus_lines_tree = new TTree("clusterlines", "tf");
346  std::vector<o2::its::ClusterLines> cl_lines_vec_pre(0);
347  std::vector<o2::its::ClusterLines> cl_lines_vec_post(0);
348  ln_clus_lines_tree->Branch("cllines_pre", &cl_lines_vec_pre);
349  ln_clus_lines_tree->Branch("cllines_post", &cl_lines_vec_post);
350  for (auto rofId{0}; rofId < mTimeFrame->getNrof(); ++rofId) {
351  cl_lines_vec_pre.clear();
352  cl_lines_vec_post.clear();
353  for (auto& clln : mTimeFrame->getTrackletClusters(rofId)) {
354  cl_lines_vec_post.push_back(clln);
355  }
356  for (auto& cl : dbg_clusLines[rofId]) {
357  cl_lines_vec_pre.push_back(cl);
358  }
359  ln_clus_lines_tree->Fill();
360  }
361  dbg_file->cd();
362  ln_clus_lines_tree->Write();
363  dbg_file->Close();
364 #endif
365 }
