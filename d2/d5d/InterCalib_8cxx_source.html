63 int InterCalib::process(const gsl::span<const o2::zdc::BCRecData>& RecBC,
64  const gsl::span<const o2::zdc::ZDCEnergy>& Energy,
65  const gsl::span<const o2::zdc::ZDCTDCData>& TDCData,
66  const gsl::span<const uint16_t>& Info)
67 {
68  if (!mInitDone) {
69  init();
70  }
71  LOG(info) << "o2::zdc::InterCalib processing " << RecBC.size() << " b.c.";
72  o2::zdc::RecEventFlat ev;
73  ev.init(RecBC, Energy, TDCData, Info);
74  while (ev.next()) {
75  if (ev.getNInfo() > 0) {
76  auto& decodedInfo = ev.getDecodedInfo();
77  for (uint16_t info : decodedInfo) {
78  uint8_t ch = (info >> 10) & 0x1f;
79  uint16_t code = info & 0x03ff;
80  // hmsg->Fill(ch, code);
81  }
82  if (mVerbosity > DbgMinimal) {
83  ev.print();
84  }
85  // Need clean data (no messages)
86  // We are sure there is no pile-up in any channel (too restrictive?)
87  continue;
88  }
89  if (ev.getNEnergy() > 0 && ev.mCurB.triggers == 0) {
90  LOGF(info, "%9u.%04u Untriggered bunch", ev.mCurB.ir.orbit, ev.mCurB.ir.bc);
91  // Skip!
92  continue;
93  }
94  if ((ev.ezdcDecoded & MaskZNA) == MaskZNA) {
95  cumulate(HidZNA, ev.EZDC(IdZNAC), ev.EZDC(IdZNA1), ev.EZDC(IdZNA2), ev.EZDC(IdZNA3), ev.EZDC(IdZNA4), 1.);
96  }
97  if ((ev.ezdcDecoded & MaskZPA) == MaskZPA) {
98  cumulate(HidZPA, ev.EZDC(IdZPAC), ev.EZDC(IdZPA1), ev.EZDC(IdZPA2), ev.EZDC(IdZPA3), ev.EZDC(IdZPA4), 1.);
99  }
100  if ((ev.ezdcDecoded & MaskZNC) == MaskZNC) {
101  cumulate(HidZNC, ev.EZDC(IdZNCC), ev.EZDC(IdZNC1), ev.EZDC(IdZNC2), ev.EZDC(IdZNC3), ev.EZDC(IdZNC4), 1.);
102  }
103  if ((ev.ezdcDecoded & MaskZPC) == MaskZPC) {
104  cumulate(HidZPC, ev.EZDC(IdZPCC), ev.EZDC(IdZPC1), ev.EZDC(IdZPC2), ev.EZDC(IdZPC3), ev.EZDC(IdZPC4), 1.);
105  }
106  if ((ev.ezdcDecoded & MaskZEM) == MaskZEM) {
107  cumulate(HidZEM, ev.EZDC(IdZEM1), ev.EZDC(IdZEM2), 0., 0., 0., 1.);
108  }
109  }
110  return 0;
111 }

115 int InterCalib::endOfRun()
116 {
117  if (mVerbosity > DbgZero) {
118  LOGF(info, "Computing intercalibration coefficients");
119  }
120  for (int ih = 0; ih < NH; ih++) {
121  LOGF(info, "%s %g events and cuts (%g:%g)", InterCalibData::DN[ih], mData.mSum[ih][5][5], mInterCalibConfig->cutLow[ih], mInterCalibConfig->cutHigh[ih]);
122  if (!mInterCalibConfig->enabled[ih]) {
123  LOGF(info, "DISABLED processing of RUN3 data for ih = %d: %s", ih, InterCalibData::DN[ih]);
124  assign(ih, false);
125  } else if (mData.mSum[ih][5][5] >= mInterCalibConfig->min_e[ih]) {
126  int ierr = mini(ih);
127  if (ierr) {
128  LOGF(error, "FAILED processing RUN3 data for ih = %d: %s", ih, InterCalibData::DN[ih]);
129  assign(ih, false);
130  } else {
131  LOGF(info, "Processed RUN3 data for ih = %d: %s", ih, InterCalibData::DN[ih]);
132  assign(ih, true);
133  }
134  } else {
135  LOGF(info, "FAILED processing RUN3 data for ih = %d: %s: TOO FEW EVENTS: %g", ih, InterCalibData::DN[ih], mData.mSum[ih][5][5]);
136  assign(ih, false);
137  }
138  }
139 
140  auto clName = o2::utils::MemFileHelper::getClassName(mTowerParamUpd);
141  mInfo.setObjectType(clName);
142  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
143  mInfo.setFileName(flName);
144  mInfo.setPath(CCDBPathTowerCalib);
145  std::map<std::string, std::string> md;
146  md["config"] = mInterCalibConfig->desc;
147  mInfo.setMetaData(md);
148  uint64_t starting = mData.mCTimeBeg;
149  if (starting >= 10000) {
150  starting = starting - 10000; // start 10 seconds before
151  }
152  uint64_t stopping = mData.mCTimeEnd + 10000; // stop 10 seconds after
153  mInfo.setStartValidityTimestamp(starting);
154  mInfo.setEndValidityTimestamp(stopping);
155  mInfo.setAdjustableEOV();
156 
157  if (mSaveDebugHistos) {
158  write();
159  }
160  return 0;
161 }

164 // Update calibration object for the five detectors
165 // ismod=false if it was not possible to update the calibration coefficients
166 // due to low statistics or minimization error
167 // ismod=true if the calibration was updated
168 void InterCalib::assign(int ih, bool ismod)
169 {
170  int id_0[4] = {IdZNA1, IdZNA2, IdZNA3, IdZNA4};
171  int id_1[4] = {IdZPA1, IdZPA2, IdZPA3, IdZPA4};
172  int id_2[4] = {IdZNC1, IdZNC2, IdZNC3, IdZNC4};
173  int id_3[4] = {IdZPC1, IdZPC2, IdZPC3, IdZPC4};
174  int id_4[1] = {IdZEM2};
175  int nid = 0;
176  int* id = nullptr;
177  if (ih == 0) {
178  nid = 4;
179  id = id_0;
180  } else if (ih == 1) {
181  nid = 4;
182  id = id_1;
183  } else if (ih == 2) {
184  nid = 4;
185  id = id_2;
186  } else if (ih == 3) {
187  nid = 4;
188  id = id_3;
189  } else if (ih == 4) {
190  nid = 1;
191  id = id_4;
192  } else {
193  LOG(fatal) << "InterCalib::assign accessing not existing ih = " << ih;
194  }
195  for (int iid = 0; iid < nid; iid++) {
196  auto ich = id[iid];
197  auto oldval = mTowerParam->getTowerCalib(ich);
198  if (ismod == true) {
199  auto val = oldval;
200  if (oldval > 0) {
201  val = val * mPar[ih][iid + 1];
202  }
203  if (mVerbosity > DbgZero) {
204  LOGF(info, "%s updated %8.6f -> %8.6f", ChannelNames[ich].data(), oldval, val);
205  }
206  mTowerParamUpd.setTowerCalib(ich, val, true);
207  } else {
208  if (mVerbosity > DbgZero) {
209  LOGF(info, "%s NOT CHANGED %8.6f", ChannelNames[ich].data(), oldval);
210  }
211  mTowerParamUpd.setTowerCalib(ich, oldval, false);
212  }
213  }

216 int InterCalib::process(const char* hname, int ic)
217 {
218  // Run 2 ZDC calibration is based on multi-dimensional histograms
219  // with dimensions:
220  // TC, T1, T2, T3, T4, Trigger
221  // ic is the number of the selected trigger class
222  THnSparse* hs = (THnSparse*)gROOT->FindObject(hname);
223  if (hs == nullptr) {
224  LOGF(error, "Not found: %s\n", hname);
225  return -1;
226  }
227  if (!hs->InheritsFrom(THnSparse::Class())) {
228  LOGF(error, "Not a THnSparse: %s\n", hname);
229  hs->IsA()->Print();
230  return -1;
231  }
232  if (!mInitDone) {
233  init();
234  }
235  TString hn = hname;
236  int ih = -1;
237  if (hn.EqualTo("hZNA")) {
238  ih = HidZNA;
239  } else if (hn.EqualTo("hZPA")) {
240  ih = HidZPA;
241  } else if (hn.EqualTo("hZNC")) {
242  ih = HidZNC;
243  } else if (hn.EqualTo("hZPC")) {
244  ih = HidZPC;
245  } else if (hn.EqualTo("hZEM")) {
246  ih = HidZEM;
247  } else {
248  LOGF(error, "Not recognized histogram name: %s\n", hname);
249  return -1;
250  }
251  clear(ih);
252  const int32_t dim = 6;
253  double x[dim];
254  int32_t bins[dim];
255  int64_t nb = hs->GetNbins();
256  int64_t nn = 0;
257  LOGF(info, "Histogram %s has %ld bins\n", hname, nb);
258  double cutl = mInterCalibConfig->cutLow[ih];
259  double cuth = mInterCalibConfig->cutHigh[ih];
260  double contt = 0;
261  for (int64_t i = 0; i < nb; i++) {
262  double cont = hs->GetBinContent(i, bins);
263  if (cont <= 0) {
264  continue;
265  }
266  for (int32_t d = 0; d < dim; ++d) {
267  x[d] = hs->GetAxis(d)->GetBinCenter(bins[d]);
268  }
269  if (TMath::Nint(x[5] - ic) == 0 && x[0] > cutl && x[0] < cuth) {
270  nn++;
271  contt += cont;
272  cumulate(ih, x[0], x[1], x[2], x[3], x[4], cont);
273  }
274  }
275  int ierr = mini(ih);
276  if (ierr) {
277  LOGF(error, "FAILED processing RUN2 data for %s ih = %d\n", hname, ih);
278  } else {
279  LOGF(info, "Processed RUN2 data for %s ih = %d\n", hname, ih);
280  replay(ih, hs, ic);
281  }
282  LOGF(info, "Trigger class selection %d and %d bins %g events and cuts (%g:%g): %ld\n", ic, nn, contt, cutl, cuth);
283  return 0;
284 }

286 void InterCalib::replay(int ih, THnSparse* hs, int ic)
287 {
288  auto* cfg = mInterCalibConfig;
289  // clang-format off
290  if(ih == 0){ mHCorr[ih] = std::make_unique<TH1F>("hZNASc","ZNA sum corr",cfg->nb1[ih],cfg->amin1[ih],cfg->amax1[ih]); }
291  if(ih == 1){ mHCorr[ih] = std::make_unique<TH1F>("hZPASc","ZPA sum corr",cfg->nb1[ih],cfg->amin1[ih],cfg->amax1[ih]); }
292  if(ih == 2){ mHCorr[ih] = std::make_unique<TH1F>("hZNCSc","ZNC sum corr",cfg->nb1[ih],cfg->amin1[ih],cfg->amax1[ih]); }
293  if(ih == 3){ mHCorr[ih] = std::make_unique<TH1F>("hZPCSc","ZPC sum corr",cfg->nb1[ih],cfg->amin1[ih],cfg->amax1[ih]); }
294  if(ih == 4){ mHCorr[ih] = std::make_unique<TH1F>("hZEM2c","ZEM2" ,cfg->nb1[ih],cfg->amin1[ih],cfg->amax1[ih]); }
295  if(ih == 0){ mCCorr[ih] = std::make_unique<TH2F>("cZNAc","ZNA;TC;SUM corr",cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih],cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih]); }
296  if(ih == 1){ mCCorr[ih] = std::make_unique<TH2F>("cZPAc","ZPA;TC;SUM corr",cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih],cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih]); }
297  if(ih == 2){ mCCorr[ih] = std::make_unique<TH2F>("cZNCc","ZNC;TC;SUM corr",cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih],cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih]); }
298  if(ih == 3){ mCCorr[ih] = std::make_unique<TH2F>("cZPCc","ZPC;TC;SUM corr",cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih],cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih]); }
299  if(ih == 4){ mCCorr[ih] = std::make_unique<TH2F>("cZEMc","ZEM;ZEM1;ZEM2 corr",cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih],cfg->nb2[ih],cfg->amin2[ih],cfg->amax2[ih]); }
300  // clang-format on
301  const int32_t dim = 6;
302  double x[dim];
303  int32_t bins[dim];
304  int64_t nb = hs->GetNbins();
305  double cutl = cfg->cutLow[ih];
306  double cuth = cfg->cutHigh[ih];
307  double c1 = mPar[ih][1];
308  double c2 = mPar[ih][2];
309  double c3 = mPar[ih][3];
310  double c4 = mPar[ih][4];
311  double of = mPar[ih][5];
312  for (int64_t i = 0; i < nb; i++) {
313  double cont = hs->GetBinContent(i, bins);
314  if (cont <= 0) {
315  continue;
316  }
317  for (int32_t d = 0; d < dim; ++d) {
318  x[d] = hs->GetAxis(d)->GetBinCenter(bins[d]);
319  }
320  if (TMath::Nint(x[5] - ic) == 0 && x[0] > cutl && x[0] < cuth) {
321  double sumquad = c1 * x[1] + c2 * x[2] + c3 * x[3] + c4 * x[4] + of;
322  mHCorr[ih]->Fill(sumquad, cont);
323  mCCorr[ih]->Fill(x[0], sumquad, cont);
324  }
325  }
326 }

385 void InterCalib::cumulate(int ih, double tc, double t1, double t2, double t3, double t4, double w = 1)
386 {
387  if (tc < mInterCalibConfig->cutLow[ih] || tc > mInterCalibConfig->cutHigh[ih]) {
388  return;
389  }
390  double val[NPAR] = {0, 0, 0, 0, 0, 1};
391  val[0] = tc;
392  val[1] = t1;
393  val[2] = t2;
394  val[3] = t3;
395  val[4] = t4;
396  for (int32_t i = 0; i < NPAR; i++) {
397  for (int32_t j = i; j < NPAR; j++) {
398  mData.mSum[ih][i][j] += val[i] * val[j] * w;
399  }
400  }
401  // mData.mSum[ih][5][5] contains the number of analyzed events
402  double sumquad = val[1] + val[2] + val[3] + val[4];
403  mHUnc[ih]->fill(sumquad, w);
404  mHUnc[ih + NH]->fill(val[0]);
405  mCUnc[ih]->fill(val[0], sumquad, w);
406 }

409 // Compute chi2 for minimization (static function)
410 void InterCalib::fcn(int& npar, double* gin, double& chi, double* par, int iflag)
411 {
412  chi = 0;
413  for (int32_t i = 0; i < NPAR; i++) {
414  for (int32_t j = 0; j < NPAR; j++) {
415  chi += (i == 0 ? par[i] : -par[i]) * (j == 0 ? par[j] : -par[j]) * mAdd[i][j];
416  }
417  }
418 }

420 int InterCalib::mini(int ih)
421 {
422  // Copy to static object and symmetrize matrix
423  // We use a static function and therefore we can do only one minimization at a time
424  mMtx.lock();
425  for (int32_t i = 0; i < NPAR; i++) {
426  for (int32_t j = 0; j < NPAR; j++) {
427  if (j < i) {
428  mAdd[i][j] = mData.mSum[ih][j][i];
429  } else {
430  mAdd[i][j] = mData.mSum[ih][i][j];
431  }
432  }
433  }
434  double arglist[10];
435  int ierflg = 0;
436  double l_bnd = mInterCalibConfig->l_bnd[ih];
437  double u_bnd = mInterCalibConfig->u_bnd[ih];
438  double start = 1.0;
439  double step = 0.1;
440  mMn[ih] = std::make_unique<TMinuit>(NPAR);
441  mMn[ih]->SetFCN(fcn);
442  mMn[ih]->mnparm(0, "c0", 1., 0., 1., 1., ierflg);
443  mMn[ih]->mnparm(1, "c1", start, step, l_bnd, u_bnd, ierflg);
444  if (ih == 4) {
445  // Only two ZEM calorimeters: equalize response
446  l_bnd = 0;
447  u_bnd = 0;
448  start = 0;
449  step = 0;
450  }
451  mMn[ih]->mnparm(2, "c2", start, step, l_bnd, u_bnd, ierflg);
452  mMn[ih]->mnparm(3, "c3", start, step, l_bnd, u_bnd, ierflg);
453  mMn[ih]->mnparm(4, "c4", start, step, l_bnd, u_bnd, ierflg);
454  l_bnd = mInterCalibConfig->l_bnd_o[ih];
455  u_bnd = mInterCalibConfig->u_bnd_o[ih];
456  start = 0;
457  step = mInterCalibConfig->step_o[ih];
458  mMn[ih]->mnparm(5, "offset", start, step, l_bnd, u_bnd, ierflg);
459  mMn[ih]->mnexcm("MIGRAD", arglist, 0, ierflg);
460  for (Int_t i = 0; i < NPAR; i++) {
461  mMn[ih]->GetParameter(i, mPar[ih][i], mErr[ih][i]);
462  }
463  mMtx.unlock();
464  return ierflg;
