59 struct TripletHash : std::unary_function<Triplet_t, std::size_t> {
60  std::size_t operator()(Triplet_t const& e) const
61  {
62  std::size_t seed = 0;
63  boost::hash_combine(seed, e.get<0>());
64  boost::hash_combine(seed, e.get<1>());
65  boost::hash_combine(seed, e.get<2>());
66  return seed;
67  }
68 };

70 struct TripletEqualTo : std::binary_function<Triplet_t, Triplet_t, bool> {
71  bool operator()(Triplet_t const& x, Triplet_t const& y) const
72  {
73  return (x.get<0>() == y.get<0>() &&
74  x.get<1>() == y.get<1>() &&
75  x.get<2>() == y.get<2>());
76  }
77 };
