83 void InterCalibEPNSpec::run(ProcessingContext& pc)
84 {
85  if (!mInitialized) {
86  mInitialized = true;
87  updateTimeDependentParams(pc);
88  mTimer.Stop();
89  mTimer.Reset();
90  mTimer.Start(false);
91  }
92 
93  const auto ref = pc.inputs().getFirstValid(true);
94  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->creation; // approximate time in ms
95  mWorker.getData().setCreationTime(creationTime);
96 
97  auto bcrec = pc.inputs().get<gsl::span<o2::zdc::BCRecData>>("bcrec");
98  auto energy = pc.inputs().get<gsl::span<o2::zdc::ZDCEnergy>>("energy");
99  auto tdc = pc.inputs().get<gsl::span<o2::zdc::ZDCTDCData>>("tdc");
100  auto info = pc.inputs().get<gsl::span<uint16_t>>("info");
101 
102  // Process reconstructed data
103  mWorker.process(bcrec, energy, tdc, info);
104 
105  // Send intermediate calibration data and debug histograms
106  o2::framework::Output output("ZDC", "INTERCALIBDATA", 0, Lifetime::Timeframe);
107  pc.outputs().snapshot(output, mWorker.mData);
108  char outputd[o2::header::gSizeDataDescriptionString];
109  for (int ih = 0; ih < (2 * InterCalibData::NH); ih++) {
110  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_1DH%d", ih);
111  o2::framework::Output output("ZDC", outputd, 0, Lifetime::Timeframe);
112  pc.outputs().snapshot(output, mWorker.mH[ih]->getBase());
113  }
114  for (int ih = 0; ih < InterCalibData::NH; ih++) {
115  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_2DH%d", ih);
116  o2::framework::Output output("ZDC", outputd, 0, Lifetime::Timeframe);
117  pc.outputs().snapshot(output, mWorker.mC[ih]->getBase());
118  }
119 }

128 framework::DataProcessorSpec getInterCalibEPNSpec()
129 {
130  std::vector<InputSpec> inputs;
131  inputs.emplace_back("bcrec", "ZDC", "BCREC", 0, Lifetime::Timeframe);
132  inputs.emplace_back("energy", "ZDC", "ENERGY", 0, Lifetime::Timeframe);
133  inputs.emplace_back("tdc", "ZDC", "TDCDATA", 0, Lifetime::Timeframe);
134  inputs.emplace_back("info", "ZDC", "INFO", 0, Lifetime::Timeframe);
135  inputs.emplace_back("intercalibconfig", "ZDC", "INTERCALIBCONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathInterCalibConfig.data()));
136 
137  std::vector<OutputSpec> outputs;
138  outputs.emplace_back("ZDC", "INTERCALIBDATA", 0, Lifetime::Timeframe);
139  char outputd[o2::header::gSizeDataDescriptionString];
140  for (int ih = 0; ih < (2 * InterCalibData::NH); ih++) {
141  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_1DH%d", ih);
142  outputs.emplace_back("ZDC", outputd, 0, Lifetime::Timeframe);
143  }
144  for (int ih = 0; ih < InterCalibData::NH; ih++) {
145  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_2DH%d", ih);
146  outputs.emplace_back("ZDC", outputd, 0, Lifetime::Timeframe);
147  }
148  return DataProcessorSpec{
149  "zdc-intercalib-epn",
150  inputs,
151  outputs,
152  AlgorithmSpec{adaptFromTask<InterCalibEPNSpec>()},
153  o2::framework::Options{{"verbosity-level", o2::framework::VariantType::Int, 0, {"Verbosity level"}}}};
154 }
