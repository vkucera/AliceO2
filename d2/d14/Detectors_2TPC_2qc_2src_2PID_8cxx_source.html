104 bool PID::processTrack(const o2::tpc::TrackTPC& track)
105 {
106  // ===| variables required for cutting and filling |===
107  const auto& dEdx = track.getdEdx();
108  const auto pTPC = track.getP();
109  const auto tgl = track.getTgl();
110  const auto snp = track.getSnp();
111  const auto phi = track.getPhi();
112  const auto ncl = uint8_t(track.getNClusters());
113 
114  const auto eta = track.getEta();
115  mMapHist["hdEdxTotVspBeforeCuts"][0]->Fill(phi, dEdx.dEdxTotTPC);
116  mMapHist["hdEdxMaxVspBeforeCuts"][0]->Fill(pTPC, dEdx.dEdxMaxTPC);
117 
118  if (pTPC < mCutMinpTPC || pTPC > mCutMaxpTPC || ncl < mCutMinnCls) {
119  return true;
120  }
121  const std::vector<float> dEdxTot{dEdx.dEdxTotTPC, dEdx.dEdxTotIROC, dEdx.dEdxTotOROC1, dEdx.dEdxTotOROC2, dEdx.dEdxTotOROC3};
122  const std::vector<float> dEdxMax{dEdx.dEdxMaxTPC, dEdx.dEdxMaxIROC, dEdx.dEdxMaxOROC1, dEdx.dEdxMaxOROC2, dEdx.dEdxMaxOROC3};
123  const std::vector<uint8_t> dEdxNcl{static_cast<uint8_t>(dEdx.NHitsIROC + dEdx.NHitsOROC1 + dEdx.NHitsOROC2 + dEdx.NHitsOROC3), dEdx.NHitsIROC, dEdx.NHitsOROC1, dEdx.NHitsOROC2, dEdx.NHitsOROC3};
124  const std::vector<uint8_t> dEdxNclSub{static_cast<uint8_t>(dEdx.NHitsSubThresholdIROC + dEdx.NHitsSubThresholdOROC1 + dEdx.NHitsSubThresholdOROC2 + dEdx.NHitsSubThresholdOROC3), dEdx.NHitsSubThresholdIROC, dEdx.NHitsSubThresholdOROC1, dEdx.NHitsSubThresholdOROC2, dEdx.NHitsSubThresholdOROC3};
125  mMapHist["hdEdxVsTgl"][0]->Fill(tgl, dEdxTot[0]);
126 
127  if (std::abs(tgl) < mCutAbsTgl) {
128 
129  mMapHist["hdEdxVsPhi"][0]->Fill(phi, dEdxTot[0]);
130  mMapHist["hdEdxVsncls"][0]->Fill(ncl, dEdxTot[0]);
131  mMapHist["hNClsPID"][0]->Fill(dEdxNcl[0]);
132  mMapHist["hNClsSubPID"][0]->Fill(dEdxNclSub[0]);
133 
134  if (track.getCharge() > 0) {
135  mMapHist["hdEdxTotVspPos"][0]->Fill(pTPC, dEdxTot[0]);
136  } else {
137  mMapHist["hdEdxTotVspNeg"][0]->Fill(pTPC, dEdxTot[0]);
138  }
139  }
140  for (size_t idEdxType = 0; idEdxType < rocNames.size(); ++idEdxType) {
141  bool ok = false;
142  float sec = 18.f * o2::math_utils::to02PiGen(track.getXYZGloAt(xks[idEdxType], 2, ok).Phi()) / o2::constants::math::TwoPI;
143  if (track.hasCSideClusters()) {
144  sec += 18.f;
145  }
146  if (!ok) {
147  sec = -1;
148  }
149 
150  if (dEdxTot[idEdxType] < mCutMindEdxTot || dEdxTot[idEdxType] > binsdEdxTot.max || dEdxNcl[idEdxType] < nclCuts[idEdxType]) {
151  continue;
152  }
153  if (std::abs(tgl) < mCutAbsTgl) {
154  mMapHist["hdEdxTotVsp"][idEdxType]->Fill(pTPC, dEdxTot[idEdxType]);
155  mMapHist["hdEdxMaxVsp"][idEdxType]->Fill(pTPC, dEdxMax[idEdxType]);
156  }
157  // ===| cuts and histogram filling for MIPs |===
158  if (pTPC > mCutMinpTPCMIPs && pTPC < mCutMaxpTPCMIPs) {
159  mMapHist["hdEdxTotMIPVsTgl"][idEdxType]->Fill(tgl, dEdxTot[idEdxType]);
160  mMapHist["hdEdxMaxMIPVsTgl"][idEdxType]->Fill(tgl, dEdxMax[idEdxType]);
161 
162  if (dEdxTot[idEdxType] < mCutMaxdEdxTot) {
163  mMapHist["hMIPNclVsTgl"][idEdxType]->Fill(tgl, dEdxNcl[idEdxType]);
164  mMapHist["hMIPNclVsTglSub"][idEdxType]->Fill(tgl, dEdxNclSub[idEdxType]);
165  }
166 
167  if (std::abs(tgl) < mCutAbsTgl) {
168  if (track.hasASideClustersOnly()) {
169  mMapHist["hdEdxVsPhiMipsAside"][0]->Fill(phi, dEdxTot[0]);
170  } else if (track.hasCSideClustersOnly()) {
171  mMapHist["hdEdxVsPhiMipsCside"][0]->Fill(phi, dEdxTot[0]);
172  }
173 
174  mMapHist["hdEdxTotMIP"][idEdxType]->Fill(dEdxTot[idEdxType]);
175  mMapHist["hdEdxMaxMIP"][idEdxType]->Fill(dEdxMax[idEdxType]);
176 
177  mMapHist["hdEdxTotMIPVsNcl"][idEdxType]->Fill(dEdxNcl[idEdxType], dEdxTot[idEdxType]);
178  mMapHist["hdEdxMaxMIPVsNcl"][idEdxType]->Fill(dEdxNcl[idEdxType], dEdxMax[idEdxType]);
179 
180  mMapHist["hdEdxTotMIPVsSec"][idEdxType]->Fill(sec, dEdxTot[idEdxType]);
181  mMapHist["hdEdxMaxMIPVsSec"][idEdxType]->Fill(sec, dEdxMax[idEdxType]);
182 
183  mMapHist["hdEdxTotMIPVsSnp"][idEdxType]->Fill(snp, dEdxTot[idEdxType]);
184  mMapHist["hdEdxMaxMIPVsSnp"][idEdxType]->Fill(snp, dEdxMax[idEdxType]);
185  }
186  }
187  }
188 
189  return true;
190 }
