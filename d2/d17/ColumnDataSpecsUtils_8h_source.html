80 {
81  std::array<gsl::span<const T>, 3> data;

83  auto const* dh = framework::DataRefUtils::getHeader<o2::header::DataHeader*>(inputRef);

86  data[subSpecIdx] = pc.inputs().get<gsl::span<T>>(inputRef);
