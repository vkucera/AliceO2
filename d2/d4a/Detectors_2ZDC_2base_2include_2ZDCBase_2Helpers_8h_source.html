32 std::string removeNamespace(const std::string& strin)
33 {
34  std::string str = strin;
35  for (auto pos = str.find(":"); pos != std::string::npos; pos = str.find(":")) {
36  str = str.substr(pos + 1);
37  }
38  return str;
39 }
