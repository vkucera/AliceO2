37 int DCSProcessor::process(const gsl::span<const DPCOM> dps)
38 {
39 
40  // first we check which DPs are missing - if some are, it means that
41  // the delta map was sent
42  if (mVerbosity > 0) {
43  LOG(info) << "\n\n\nProcessing new TF\n-----------------";
44  }
45 
46  if (mVerbosity > 1) {
47  std::unordered_map<DPID, DPVAL> mapin;
48  for (auto& it : dps) {
49  mapin[it.id] = it.data;
50  }
51 
52  for (auto& it : mPids) {
53  const auto& el = mapin.find(it.first);
54  if (el == mapin.end()) {
55  LOG(info) << "DP " << it.first << " not found in map";
56  } else {
57  LOG(info) << "DP " << it.first << " found in map";
58  }
59  }
60  }
61 
62  // now we process all DPs, one by one
63  for (const auto& it : dps) {
64  // we process only the DPs defined in the configuration
65  const auto& el = mPids.find(it.id);
66  if (el == mPids.end()) {
67  if (mVerbosity > 1) {
68  LOG(info) << "DP " << it.id << " not found in DCSProcessor, we will not process it";
69  }
70  continue;
71  }
72  processDP(it);
73  mPids[it.id] = true;
74  }
75 
76  return 0;
77 }

81 int DCSProcessor::processDP(const DPCOM& dpcom)
82 {
83 
84  // processing single DP
85 
86  auto& dpid = dpcom.id;
87  const auto& type = dpid.get_type();
88  if (mVerbosity > 1) {
89  if (type == DPVAL_DOUBLE) {
90  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<double>(dpcom);
91  } else if (type == DPVAL_INT) {
92  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<int32_t>(dpcom);
93  } else if (type == DPVAL_STRING) {
94  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<string>(dpcom);
95  }
96  }
97  auto flags = dpcom.data.get_flags();
98  if (processFlags(flags, dpid.get_alias()) == 0) {
99  if (type == DPVAL_DOUBLE) {
100  auto etime = dpcom.data.get_epoch_time();
101 
102  // check if DP is one of the gas values
103  if (std::strstr(dpid.get_alias(), "trd_gas") != nullptr) {
104  if (!mGasStartTSset) {
105  mGasStartTS = mCurrentTS;
106  mGasStartTSset = true;
107  }
108  auto& dpInfoGas = mTRDDCSGas[dpid];
109  if (dpInfoGas.nPoints == 0 || etime != mLastDPTimeStamps[dpid]) {
110  // only add data point in case it was not already read before
111  dpInfoGas.addPoint(o2::dcs::getValue<double>(dpcom), etime);
112  mLastDPTimeStamps[dpid] = etime;
113  }
114  }
115 
116  // check if DP is HV current value
117  if (std::strstr(dpid.get_alias(), "Imon") != nullptr) {
118  if (!mCurrentsStartTSSet) {
119  mCurrentsStartTS = mCurrentTS;
120  mCurrentsStartTSSet = true;
121  }
122  auto& dpInfoCurrents = mTRDDCSCurrents[dpid];
123  if (dpInfoCurrents.nPoints == 0 || etime != mLastDPTimeStamps[dpid]) {
124  // only add data point in case it was not already read before
125  dpInfoCurrents.addPoint(o2::dcs::getValue<double>(dpcom), etime);
126  mLastDPTimeStamps[dpid] = etime;
127  }
128  }
129 
130  // check if DP is HV voltage value
131  if (std::strstr(dpid.get_alias(), "Umon") != nullptr) {
132  if (!mVoltagesStartTSSet) {
133  mVoltagesStartTS = mCurrentTS;
134  mVoltagesStartTSSet = true;
135  }
136  auto& dpInfoVoltages = mTRDDCSVoltages[dpid];
137  if (etime != mLastDPTimeStamps[dpid]) {
138  int chamberId = getChamberIdFromAlias(dpid.get_alias());
139  if (mVoltageSet.test(chamberId)) {
140  if (std::fabs(dpInfoVoltages - o2::dcs::getValue<double>(dpcom)) > 1.f) {
141  // trigger update of voltage CCDB object
142  mShouldUpdateVoltages = true;
143  // OS: this will still overwrite the current voltage value of the object going into the CCDB
144  // Should instead the old value be kept until a new obect has been stored in the CCDB?
145  }
146  }
147  dpInfoVoltages = o2::dcs::getValue<double>(dpcom);
148  mLastDPTimeStamps[dpid] = etime;
149  mVoltageSet.set(chamberId);
150  }
151  }
152 
153  // check if DP is env value
154  if (std::strstr(dpid.get_alias(), "trd_aliEnv") != nullptr) {
155  if (!mEnvStartTSSet) {
156  mEnvStartTS = mCurrentTS;
157  mEnvStartTSSet = true;
158  }
159  auto& dpInfoEnv = mTRDDCSEnv[dpid];
160  if (dpInfoEnv.nPoints == 0 || etime != mLastDPTimeStamps[dpid]) {
161  // only add data point in case it was not already read before
162  dpInfoEnv.addPoint(o2::dcs::getValue<double>(dpcom), etime);
163  mLastDPTimeStamps[dpid] = etime;
164  }
165  }
166  }
167  if (type == DPVAL_INT) {
168  if (std::strstr(dpid.get_alias(), "trd_runNo") != nullptr) { // DP is trd_runNo
169  if (!mRunStartTSSet) {
170  mRunStartTS = mCurrentTS;
171  mRunStartTSSet = true;
172  }
173  auto& runNumber = mTRDDCSRun[dpid];
174  if (mPids[dpid] && runNumber != o2::dcs::getValue<int32_t>(dpcom)) {
175  LOGF(info, "Run number has already been processed and the new one %i differs from the old one %i", runNumber, o2::dcs::getValue<int32_t>(dpcom));
176  mShouldUpdateRun = true;
177  mRunEndTS = mCurrentTS;
178  } else {
179  runNumber = o2::dcs::getValue<int32_t>(dpcom);
180  }
181  } else if (std::strstr(dpid.get_alias(), "trd_runType") != nullptr) { // DP is trd_runType
182  if (!mRunStartTSSet) {
183  mRunStartTS = mCurrentTS;
184  mRunStartTSSet = true;
185  }
186  auto& runType = mTRDDCSRun[dpid];
187  if (mPids[dpid] && runType != o2::dcs::getValue<int32_t>(dpcom)) {
188  LOGF(info, "Run type has already been processed and the new one %i differs from the old one %i", runType, o2::dcs::getValue<int32_t>(dpcom));
189  mShouldUpdateRun = true;
190  mRunEndTS = mCurrentTS;
191  } else {
192  runType = o2::dcs::getValue<int32_t>(dpcom);
193  }
194  }
195  }
196 
197  if (type == DPVAL_STRING) {
198  if (std::strstr(dpid.get_alias(), "trd_fedCFGtag") != nullptr) { // DP is trd_fedCFGtag
199  auto cfgTag = o2::dcs::getValue<std::string>(dpcom);
200  if (mVerbosity > 1) {
201  LOG(info) << "CFG tag " << dpid.get_alias() << " is " << cfgTag;
202  }
203  }
204  }
205  }
206  return 0;
207 }

209 int DCSProcessor::getChamberIdFromAlias(const char* alias) const
210 {
211  // chamber ID is the last three characaters from the alias
212  auto length = strlen(alias);
213  std::string id(alias + length - 3, alias + length);
214  return std::stoi(id);

281 bool DCSProcessor::updateGasDPsCCDB()
282 {
283  // here we create the object containing the gas data points to then be sent to CCDB
284  LOG(info) << "Preparing CCDB object for TRD gas DPs";
285 
286  bool retVal = false; // set to 'true' in case at least one DP for gas has been processed
287 
288  for (const auto& it : mPids) {
289  const auto& type = it.first.get_type();
290  if (type == o2::dcs::DPVAL_DOUBLE) {
291  if (std::strstr(it.first.get_alias(), "trd_gas") != nullptr) {
292  if (it.second == true) { // we processed the DP at least 1x
293  retVal = true;
294  }
295  if (mVerbosity > 0) {
296  LOG(info) << "PID = " << it.first.get_alias();
297  mTRDDCSGas[it.first].print();
298  }
299  }
300  }
301  }
302  std::map<std::string, std::string> md;
303  md["responsible"] = "Ole Schmidt";
304  o2::calibration::Utils::prepareCCDBobjectInfo(mTRDDCSGas, mCcdbGasDPsInfo, "TRD/Calib/DCSDPsGas", md, mGasStartTS, mGasStartTS + 3 * o2::ccdb::CcdbObjectInfo::DAY);
305 
306  return retVal;
307 }
