57 void PHOSBadMapCalibDevice::run(o2::framework::ProcessingContext& ctx)
58 {
59  if (mRunStartTime == 0) {
60  const auto ref = ctx.inputs().getFirstValid(true);
61  mRunStartTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->creation; // approximate time in ms
62  mValidityTime = mRunStartTime + 31622400000; // one year validity range
63  }
64 
65  // Read previous bad map if not read yet
66  if (!mOldBadMap) {
67  mOldBadMap = ctx.inputs().get<o2::phos::BadChannelsMap*>("prevbdmap").get();
68  }
69 
70  // scan Cells stream, collect occupancy
71  if (mMode == 0) { // Cell occupancy and time
72  auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
73  LOG(info) << "[PHOSBadMapCalibDevice - run] Received " << cells.size() << " cells, running calibration ...";
74  for (const auto& c : cells) {
75  float e = c.getEnergy();
76  if (e > mElowMin && e < mElowMax) {
77  mMeanLow->Fill(c.getAbsId() - 1792);
78  }
79  if (e > mEhighMin && e < mEhighMax) {
80  mMeanHigh->Fill(c.getAbsId() - 1792);
81  mMeanTime->Fill(c.getAbsId() - 1792, c.getTime());
82  }
83  }
84  } // mMode 0: cell occupancy
85 
86  if (mMode == 1) { // Raw data analysis, chi2
87  auto chi2list = ctx.inputs().get<gsl::span<short>>("fitqa");
88  LOG(info) << "[PHOSBadMapCalibDevice - run] Received " << chi2list.size() << " Chi2, running calibration ...";
89  auto b = chi2list.begin();
90  while (b != chi2list.end()) {
91  short absId = *b;
92  ++b;
93  float c = 0.2 * (*b);
94  ++b;
95  mChi2norm->Fill(absId - 1792);
96  mChi2->Fill(absId - 1792, c);
97  }
98  }
99 
100  if (mMode == 2) { // Pedestals
101  if (mStatistics <= 0) { // skip the rest of the run
102  return;
103  }
104  if (mStatistics % 100 == 0) {
105  LOG(info) << mStatistics << " left to produce pedestal BadMap";
106  }
107 
108  auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
109  LOG(debug) << "[PHOSBadMapCalibDevice - run] Received " << cells.size() << " cells, running calibration ...";
110  for (const auto& c : cells) {
111  if (c.getHighGain()) {
112  mHGMean->Fill(c.getAbsId() - 1792, c.getEnergy());
113  mHGRMS->Fill(c.getAbsId() - 1792, 1.e+7 * c.getTime());
114  mHGNorm->Fill(c.getAbsId() - 1792, 1.);
115  } else {
116  mLGMean->Fill(c.getAbsId() - 1792);
117  mLGRMS->Fill(c.getAbsId() - 1792, 1.e+7 * c.getTime());
118  mLGNorm->Fill(c.getAbsId() - 1792, 1.);
119  }
120  }
121  --mStatistics;
122  if (mStatistics <= 0) {
123  LOG(info) << "Start calculating bad map";
124  calculateBadMap();
125  checkBadMap();
126  sendOutput(ctx.outputs());
127  }
128  }
129 }

184 bool PHOSBadMapCalibDevice::calculateBadMap()
185 {
186  mBadMap = std::make_unique<BadChannelsMap>();
187 
188  if (mMode == 0) { // occupancy
189  // --check if statistics is sufficient
190  // --prepare map of pairs (number of entries in cell, number of such cells)
191  // --sort according number of entries
192  // --remove pairs in the beginning and the end of the list till variation of mean becomes small
193  if (mMeanLow->Integral() < 1.e+4) {
194  LOG(error) << "Insufficient statistics: " << mMeanLow->Integral() << " entries in lowE histo, do nothing";
195  return false;
196  }
197 
198  float nMean, nRMS;
199  calculateLimits(mMeanLow.get(), nMean, nRMS); // for low E occupamcy
200  float nMinLow = std::max(float(0.), nMean - 6 * nRMS);
201  float nMaxLow = nMean + 6 * nRMS;
202  LOG(info) << "Limits for low E histo: " << nMinLow << "<n<" << nMaxLow;
203  for (int i = 1; i <= mMeanLow->GetNbinsX(); i++) {
204  int c = mMeanLow->GetBinContent(i);
205  if (c < nMinLow || c > nMaxLow) {
206  mBadMap->addBadChannel(i + 1792);
207  }
208  }
209  calculateLimits(mMeanHigh.get(), nMean, nRMS); // for high
210  float nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
211  float nMaxHigh = nMean + 6 * nRMS;
212  LOG(info) << "Limits for high E histo: " << nMinHigh << "<n<" << nMaxHigh;
213  for (int i = 1; i <= mMeanHigh->GetNbinsX(); i++) {
214  int c = mMeanHigh->GetBinContent(i);
215  if (c < nMinHigh || c > nMaxHigh) {
216  mBadMap->addBadChannel(i + 1792);
217  }
218  }
219  }
220 
221  if (mMode == 1) { // chi2 distribution
222  if (mChi2norm->Integral() < 1.e+4) {
223  LOG(important) << "Insufficient statistics: " << mChi2norm->Integral() << " entries in chi2Norm histo, do nothing";
224  return false;
225  }
226  mChi2->Divide(mChi2norm.get());
227  float nMean, nRMS;
228  calculateLimits(mChi2.get(), nMean, nRMS); // for low E occupamcy
229  float nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
230  float nMaxHigh = nMean + 6 * nRMS;
231  for (int i = 1; i <= mChi2->GetNbinsX(); i++) {
232  int c = mChi2->GetBinContent(i);
233  if (c < nMinHigh || c > nMaxHigh) {
234  mBadMap->addBadChannel(i + 1792);
235  }
236  }
237  } // Chi2
238 
239  if (mMode == 2) { // Pedestals
240  if (mHGNorm->Integral() < 2.e+4) {
241  LOG(important) << "Insufficient statistics: " << mHGNorm->Integral() << " entries in mHGNorm histo, do nothing";
242  return false;
243  }
244  float nMean, nRMS;
245  mHGMean->Divide(mHGNorm.get());
246  calculateLimits(mHGMean.get(), nMean, nRMS); // mean HG pedestals
247  float nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
248  float nMaxHigh = nMean + 6 * nRMS;
249  LOG(info) << "Limits for HG mean: " << nMinHigh << " < meanHG < " << nMaxHigh;
250  int nBad = 0;
251  for (int i = 1; i <= mHGMean->GetNbinsX(); i++) {
252  int c = mHGMean->GetBinContent(i);
253  if (c < nMinHigh || c > nMaxHigh) {
254  mBadMap->addBadChannel(i + 1792);
255  nBad++;
256  }
257  }
258  LOG(info) << "HG mean: removed " << nBad << " channels";
259  mLGMean->Divide(mLGNorm.get());
260  calculateLimits(mLGMean.get(), nMean, nRMS); // for low E occupamcy
261  nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
262  nMaxHigh = nMean + 6 * nRMS;
263  LOG(info) << "Limits for LG mean: " << nMinHigh << " < meanHG < " << nMaxHigh;
264  nBad = 0;
265  for (int i = 1; i <= mLGMean->GetNbinsX(); i++) {
266  int c = mLGMean->GetBinContent(i);
267  if (c < nMinHigh || c > nMaxHigh) {
268  mBadMap->addBadChannel(i + 1792);
269  nBad++;
270  }
271  }
272  LOG(info) << " LG mean: removed " << nBad << " channels";
273  mHGRMS->Divide(mHGNorm.get());
274  calculateLimits(mHGRMS.get(), nMean, nRMS); // mean HG pedestals
275  nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
276  nMaxHigh = nMean + 6 * nRMS;
277  LOG(info) << "Limits for HG RMS: " << nMinHigh << " < HG rms < " << nMaxHigh;
278  nBad = 0;
279  for (int i = 1; i <= mHGRMS->GetNbinsX(); i++) {
280  int c = mHGRMS->GetBinContent(i);
281  if (c < nMinHigh || c > nMaxHigh) {
282  mBadMap->addBadChannel(i + 1792);
283  nBad++;
284  }
285  }
286  LOG(info) << " HG RMS: removed " << nBad << " channels";
287  mLGRMS->Divide(mHGNorm.get());
288  calculateLimits(mLGRMS.get(), nMean, nRMS); // for low E occupamcy
289  nMinHigh = std::max(float(0.), nMean - 6 * nRMS);
290  nMaxHigh = nMean + 6 * nRMS;
291  LOG(info) << "Limits for LG RMS: " << nMinHigh << " < LG rms < " << nMaxHigh;
292  nBad = 0;
293  for (int i = 1; i <= mLGRMS->GetNbinsX(); i++) {
294  int c = mLGRMS->GetBinContent(i);
295  if (c < nMinHigh || c > nMaxHigh) {
296  mBadMap->addBadChannel(i + 1792);
297  nBad++;
298  }
299  }
300  LOG(info) << " LG RMS: removed " << nBad << " channels";
301 
302  } // Pedestals
303 
304  return true;

306 void PHOSBadMapCalibDevice::calculateLimits(TH1F* tmp, float& nMean, float& nRMS)
307 {
308 
309  // --prepare map of pairs (number of entries in cell, number of such cells)
310  // --sort according number of entries
311  // --remove pairs in the beginning and the end of the list till variation of mean becomes small
312 
313  std::map<short, int> histo{};
314  for (int i = 1; i <= tmp->GetNbinsX(); i++) {
315  int c = tmp->GetBinContent(i);
316  if (c > 0) {
317  histo[c] = histo[c] + 1;
318  }
319  }
320  float mean = 0, rms = 0, nTot = 0, maxN = 0.;
321  for (const auto& [n, nCells] : histo) {
322  mean += n * nCells;
323  rms += n * n * nCells;
324  nTot += nCells;
325  if (n > maxN) {
326  maxN = n;
327  }
328  }
329  const float eps = 0.05; // maximal variation of mean if remove outlyers
330  // now remove largest entries till mean remains almost the same
331  for (std::map<short, int>::reverse_iterator iter = histo.rbegin(); iter != histo.rend(); ++iter) {
332  float nextMean = mean - iter->first * iter->second;
333  float nextRMS = rms - iter->first * iter->first * iter->second;
334  float nextTot = nTot - iter->second;
335  if (nTot == 0. || nextTot == 0.) {
336  break;
337  }
338  if (mean / nTot - nextMean / nextTot > eps * mean / nTot) {
339  mean = nextMean;
340  rms = nextRMS;
341  nTot = nextTot;
342  } else { // converged
343  break;
344  }
345  }
346  // now remove smallest entries till mean remains almost the same
347  for (std::map<short, int>::iterator iter = histo.begin(); iter != histo.end(); ++iter) {
348  float nextMean = mean - iter->first * iter->second;
349  float nextRMS = rms - iter->first * iter->first * iter->second;
350  float nextTot = nTot - iter->second;
351  if (nTot == 0. || nextTot == 0.) {
352  break;
353  }
354  if (mean / nTot - nextMean / nextTot > eps * mean / nTot) {
355  mean = nextMean;
356  rms = nextRMS;
357  nTot = nextTot;
358  } else { // converged
359  break;
360  }
361  }
362  // Now we have stable mean. calculate limits
363  if (nTot > 0) {
364  nMean = mean / nTot;
365  rms /= nTot;
366  nRMS = rms - nMean * nMean;
367  if (rms > 0) {
368  nRMS = sqrt(nRMS);
369  } else {
370  nRMS = 0.;
371  }
372  } else {
373  nMean = 0.5 * maxN;
374  nRMS = 0.5 * maxN;
375  }
376 }

378 void PHOSBadMapCalibDevice::checkBadMap()
379 {
380  if (!mOldBadMap) {
381  return;
382  }
383 
384  // Compare old to current
385  int nNewBad = 0;
386  int nNewGood = 0;
387  for (short i = o2::phos::Mapping::NCHANNELS; i > 1792; i--) {
388  mBadMapDiff[i] = mBadMap->isChannelGood(i) - mOldBadMap->isChannelGood(i);
389  if (mBadMapDiff[i] > 0) { // new good channel
390  nNewGood++;
391  }
392  if (mBadMapDiff[i] < 0) { // new bad channel
393  nNewBad++;
394  }
395  }
396  LOG(info) << nNewBad + nNewGood << " channels changed: " << nNewGood << " new good ch., " << nNewBad << " new bad ch.";
397  if (nNewBad + nNewGood > kMinorChange) { // serious change, do not update CCDB automatically, use "force" option to overwrite
398  LOG(important) << "too many channels changed: " << nNewGood << " new good ch., " << nNewBad << " new bad ch.";
399  }
