36 void NoiseCalibratorSpec::init(InitContext& ic)
37 {
38  o2::base::GRPGeomHelper::instance().setRequest(mCCDBRequest);
39  auto onepix = ic.options().get<bool>("1pix-only");
40  LOG(info) << "Fast 1=pixel calibration: " << onepix;
41  auto probT = ic.options().get<float>("prob-threshold");
42  auto probTRelErr = ic.options().get<float>("prob-rel-err");
43  LOGP(info, "Setting the probability threshold to {} with relative error {}", probT, probTRelErr);
44  mStopMeOnly = ic.options().get<bool>("stop-me-only");
45  mCalibrator = std::make_unique<CALIBRATOR>(onepix, probT, probTRelErr);
46  mCalibrator->setNThreads(ic.options().get<int>("nthreads"));
47 
48  mValidityDays = ic.options().get<int>("validity-days");
49  if (mValidityDays < 1) {
50  mValidityDays = 1;
51  }
52 }

54 void NoiseCalibratorSpec::run(ProcessingContext& pc)
55 {
56  updateTimeDependentParams(pc);
57  mTimer.Start(false);
58  static bool firstCall = true;
59  if (firstCall) {
60  firstCall = false;
61  mCalibrator->setInstanceID(pc.services().get<const o2::framework::DeviceSpec>().inputTimesliceId);
62  mCalibrator->setNInstances(pc.services().get<const o2::framework::DeviceSpec>().maxInputTimeslices);
63  }
64  bool done = false;
65  if (mUseClusters) {
66  const auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
67  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
68  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
69  mNClustersProc += compClusters.size();
70  mDataSizeStat +=
71  rofs.size() * sizeof(o2::itsmft::ROFRecord) + patterns.size() +
72  compClusters.size() * sizeof(o2::itsmft::CompClusterExt);
73  done = mCalibrator->processTimeFrameClusters(compClusters, patterns, rofs);
74  } else {
75  const auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
76  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
77  mNClustersProc += digits.size();
78  mDataSizeStat += digits.size() * sizeof(o2::itsmft::Digit) + rofs.size() * sizeof(o2::itsmft::ROFRecord);
79  done = mCalibrator->processTimeFrameDigits(digits, rofs);
80  }
81  if (done) {
82  LOG(info) << "Minimum number of noise counts has been reached !";
83  sendOutput(pc.outputs());
84  pc.services().get<ControlService>().readyToQuit(mStopMeOnly ? QuitRequest::Me : QuitRequest::All);
85  }
86 
87  mTimer.Stop();
88 }

90 void NoiseCalibratorSpec::sendOutput(DataAllocator& output)
91 {
92  static bool done = false;
93  if (done) {
94  return;
95  }
96  done = true;
97  mCalibrator->finalize();
98 
99  long tstart = o2::ccdb::getCurrentTimestamp();
100  long tend = o2::ccdb::getFutureTimestamp(3600 * 24 * mValidityDays);
101 #ifdef TIME_SLOT_CALIBRATION
102  const auto& payload = mCalibrator->getNoiseMap(tstart, tend);
103 #else
104  const auto& payload = mCalibrator->getNoiseMap();
105 #endif
106  std::map<std::string, std::string> md;
107  o2::ccdb::CcdbObjectInfo info("ITS/Calib/NoiseMap", "NoiseMap", "noise.root", md, tstart, tend);
108 
109  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
110  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
111  << " of size " << image->size()
112  << " bytes, valid for " << info.getStartValidityTimestamp()
113  << " : " << info.getEndValidityTimestamp();
114 
115  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "ITS_NOISE", 0}, *image.get());
116  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "ITS_NOISE", 0}, info);
117  LOG(info) << "sending done";
118  LOGP(info, "Timing: {:.2f} CPU / {:.2f} Real s. in {} TFs for {} {} / {:.2f} GB",
119  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1,
120  mUseClusters ? "clusters" : "digits",
121  mNClustersProc, double(mDataSizeStat) / (1024L * 1024L * 1024L));
122 }

148 DataProcessorSpec getNoiseCalibratorSpec(bool useClusters)
149 {
150  std::vector<InputSpec> inputs;
151  if (useClusters) {
152  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
153  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
154  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
155  inputs.emplace_back("cldict", "ITS", "CLUSDICT", 0, Lifetime::Condition, ccdbParamSpec("ITS/Calib/ClusterDictionary"));
156  } else {
157  inputs.emplace_back("digits", "ITS", "DIGITS", 0, Lifetime::Timeframe);
158  inputs.emplace_back("ROframes", "ITS", "DIGITSROF", 0, Lifetime::Timeframe);
159  }
160  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
161  false, // GRPECS=true
162  false, // GRPLHCIF
163  false, // GRPMagField
164  false, // askMatLUT
165  o2::base::GRPGeomRequest::None, // geometry
166  inputs);
167  std::vector<OutputSpec> outputs;
168  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ITS_NOISE"}, Lifetime::Sporadic);
169  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ITS_NOISE"}, Lifetime::Sporadic);
170 
171  return DataProcessorSpec{
172  "its-noise-calibrator",
173  inputs,
174  outputs,
175  AlgorithmSpec{adaptFromTask<NoiseCalibratorSpec>(useClusters, ccdbRequest)},
176  Options{
177  {"1pix-only", VariantType::Bool, false, {"Fast 1-pixel calibration only (cluster input only)"}},
178  {"prob-threshold", VariantType::Float, 3.e-6f, {"Probability threshold for noisy pixels"}},
179  {"prob-rel-err", VariantType::Float, 0.2f, {"Relative error on channel noise to apply the threshold"}},
180  {"nthreads", VariantType::Int, 1, {"Number of map-filling threads"}},
181  {"validity-days", VariantType::Int, 3, {"Validity on days from upload time"}},
182  {"stop-me-only", VariantType::Bool, false, {"At sufficient statistics stop only this device, otherwise whole workflow"}}}};
183 }
