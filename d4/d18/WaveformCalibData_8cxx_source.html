58 WaveformCalibChData& WaveformCalibChData::operator+=(const WaveformCalibChData& other)
59 {
60  if (other.mEntries > 0) {
61  if (other.mFirstValid > mFirstValid) {
62  mFirstValid = other.mFirstValid;
63  }
64  if (other.mLastValid < mLastValid) {
65  mLastValid = other.mLastValid;
66  }
67  mEntries = mEntries + other.mEntries;
68  for (int32_t i = mFirstValid; i <= mLastValid; i++) {
69  mData[i] = mData[i] + other.mData[i];
70  }
71  }
72  return *this;
73 }

154 int WaveformCalibData::saveDebugHistos(const std::string fn)
155 {
156  TDirectory* cwd = gDirectory;
157  TFile* f = new TFile(fn.data(), "recreate");
158  if (f->IsZombie()) {
159  LOG(error) << "Cannot create file: " << fn;
160  return 1;
161  }
162  for (int32_t is = 0; is < NChannels; is++) {
163  if (mWave[is].mEntries > 0) {
164  TString n = TString::Format("h%d", is);
165  TString t = TString::Format("Waveform %d %s", is, ChannelNames[is].data());
166  int nbx = mWave[is].mLastValid - mWave[is].mFirstValid + 1;
167  float min = mWave[is].mFirstValid - mPeak - 0.5;
168  float max = mWave[is].mLastValid - mPeak + 0.5;
169  TH1F h(n, t, nbx, min, max);
170  for (int ibx = 0; ibx < nbx; ibx++) {
171  h.SetBinContent(ibx + 1, mWave[is].mData[mWave[is].mFirstValid + ibx]);
172  }
173  h.SetEntries(mWave[is].mEntries);
174  h.Write("", TObject::kOverwrite);
175  }
176  }
177  f->Close();
178  cwd->cd();
179  return 0;
180 }
