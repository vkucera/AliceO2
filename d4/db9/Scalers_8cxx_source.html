31 void CTPScalerO2::createCTPScalerO2FromRaw(const CTPScalerRaw& raw, const std::array<uint32_t, 6>& overflow)
32 {
33  classIndex = raw.classIndex;
34  lmBefore = (uint64_t)(raw.lmBefore) + 0xffffffffull * (uint64_t)(overflow[0]);
35  lmAfter = (uint64_t)(raw.lmAfter) + 0xffffffffull * (uint64_t)(overflow[1]);
36  l0Before = (uint64_t)(raw.l0Before) + 0xffffffffull * (uint64_t)(overflow[2]);
37  l0After = (uint64_t)(raw.l0After) + 0xffffffffull * (uint64_t)(overflow[3]);
38  l1Before = (uint64_t)(raw.l1Before) + 0xffffffffull * (uint64_t)(overflow[4]);
39  l1After = (uint64_t)(raw.l1After) + 0xffffffffull * (uint64_t)(overflow[5]);
40  //std::cout << "lmb overflow:" << overflow[0] << " lmb:" << lmBefore << " raw:" << raw.lmBefore << std::endl;

109 int CTPRunScalers::readScalers(const std::string& rawscalers)
110 {
111  LOG(info) << "Loading CTP scalers.";
112  std::istringstream iss(rawscalers);
113  int ret = 0;
114  int level = 0;
115  std::string line;
116  int nclasses = 0;
117  int nlines = 0;
118  while (std::getline(iss, line)) {
119  o2::utils::Str::trim(line);
120  if ((ret = processScalerLine(line, level, nclasses)) != 0) {
121  return ret;
122  }
123  nlines++;
124  }
125  if (nlines < 4) {
126  LOG(error) << "Input string seems too small:\n"
127  << rawscalers;
128  return 6;
129  }
130  if (nclasses != 0) {
131  LOG(error) << "Wrong number of classes in final record";
132  return 6;
133  }
134  return 0;
135 }
136 int CTPRunScalers::processScalerLine(const std::string& line, int& level, int& nclasses)
137 {
138  //std::cout << "Processing line" << std::endl;
139  if (line.size() == 0) {
140  return 0;
141  }
142  if (line.at(0) == '#') {
143  return 0;
144  }
145  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
146  size_t ntokens = tokens.size();
147  if (ntokens == 0) {
148  return 0;
149  }
150  //std::cout << line << " level in::" << level << std::endl;
151  // Version
152  if (level == 0) {
153  if (ntokens != 1) {
154  LOG(error) << "Expected version in the first line";
155  return 1;
156  } else {
157  mVersion = std::stoi(tokens[0]);
158  level = 1;
159  return 0;
160  }
161  }
162  // Run Number N Classes [class indexes]
163  if (level == 1) {
164  if (ntokens < 3) {
165  LOG(error) << "Wrong syntax of second line in CTP scalers";
166  return 2;
167  } else {
168  mRunNumber = std::stol(tokens[0]);
169  int numofclasses = std::stoi(tokens[1]);
170  if ((numofclasses + 2) != ntokens) {
171  LOG(error) << "Wrong syntax of second line in CTP scalers";
172  return 3;
173  }
174  mClassMask.reset();
175  for (int i = 0; i < numofclasses; i++) {
176  int index = std::stoi(tokens[i + 2]);
177  mClassMask[index] = 1;
178  }
179  level = 2;
180  nclasses = 0;
181  return 0;
182  }
183  }
184  // Time stamp: orbit bcid linuxtime
185  if (level == 2) {
186  if (ntokens != 4) {
187  LOG(error) << "Wrong syntax of time stamp line in CTP scalers";
188  return 4;
189  } else {
190  CTPScalerRecordRaw rec;
191  rec.intRecord.orbit = std::stol(tokens[0]);
192  rec.intRecord.bc = std::stol(tokens[1]);
193  // rec.seconds = std::stol(tokens[2]);
194  rec.epochTime = std::stoll(tokens[2]);
195  // rec.microSeconds = std::stol(tokens[3]);
196  mScalerRecordRaw.push_back(rec);
197  level = 3;
198  return 0;
199  }
200  }
201  if (level == 3) {
202  if (ntokens != 6) {
203  LOG(error) << "Wrong syntax of counters line in CTP scalers";
204  return 5;
205  } else {
206  //std::cout << "nclasses:" << nclasses << std::endl;
207  CTPScalerRaw scaler;
208  scaler.classIndex = getClassIndexes()[nclasses];
209  scaler.lmBefore = std::stol(tokens[0]);
210  scaler.lmAfter = std::stol(tokens[1]);
211  scaler.l0Before = std::stol(tokens[2]);
212  scaler.l0After = std::stol(tokens[3]);
213  scaler.l1Before = std::stol(tokens[4]);
214  scaler.l1After = std::stol(tokens[5]);
215  (mScalerRecordRaw.back()).scalers.push_back(scaler);
216  nclasses++;
217  if (nclasses >= mClassMask.count()) {
218  level = 2;
219  nclasses = 0;
220  }
221  return 0;
222  }
223  }
224  return 0;

228 int CTPRunScalers::convertRawToO2()
229 {
230  //struct classScalersOverflows {uint32_t overflows[6];};
231  overflows_t overflows;
232  for (uint32_t i = 0; i < mClassMask.size(); i++) {
233  if (mClassMask[i]) {
234  overflows[i] = {0, 0, 0, 0, 0, 0};
235  }
236  }
237  // 1st o2 rec is just copy
238  CTPScalerRecordO2 o2rec;
239  copyRawToO2ScalerRecord(mScalerRecordRaw[0], o2rec, overflows);
240  mScalerRecordO2.push_back(o2rec);
241  for (uint32_t i = 1; i < mScalerRecordRaw.size(); i++) {
242  //update overflows
243  updateOverflows(mScalerRecordRaw[i - 1], mScalerRecordRaw[i], overflows);
244  //
245  CTPScalerRecordO2 o2rec;
246  copyRawToO2ScalerRecord(mScalerRecordRaw[i], o2rec, overflows);
247  mScalerRecordO2.push_back(o2rec);
248  // Check consistency
249  checkConsistency(mScalerRecordO2[i - 1], mScalerRecordO2[i]);
250  }
251  return 0;
252 }
253 int CTPRunScalers::copyRawToO2ScalerRecord(const CTPScalerRecordRaw& rawrec, CTPScalerRecordO2& o2rec, overflows_t& classesoverflows)
254 {
255  if (rawrec.scalers.size() != (mClassMask.count())) {
256  LOG(error) << "Inconsistent scaler record size:" << rawrec.scalers.size() << " Expected:" << mClassMask.count();
257  return 1;
258  }
259  o2rec.scalers.clear();
260  o2rec.intRecord = rawrec.intRecord;
261  // o2rec.seconds = rawrec.seconds;
262  // o2rec.microSeconds = rawrec.microSeconds;
263  o2rec.epochTime = rawrec.epochTime;
264  for (uint32_t i = 0; i < rawrec.scalers.size(); i++) {
265  CTPScalerRaw rawscal = rawrec.scalers[i];
266  CTPScalerO2 o2scal;
267  int k = (getClassIndexes())[i];
268  o2scal.createCTPScalerO2FromRaw(rawscal, classesoverflows[k]);
269  o2rec.scalers.push_back(o2scal);
270  }
271  return 0;
272 }
273 int CTPRunScalers::checkConsistency(const CTPScalerO2& scal0, const CTPScalerO2& scal1) const
274 {
275  int ret = 0;
276  // Scaler should never decrease
277  if (scal0.lmBefore > scal1.lmBefore) {
278  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.lmBefore << " lmBefore :" << scal1.lmBefore;
279  ret++;
280  }
281  if (scal0.l0Before > scal1.l0Before) {
282  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.l0Before << " lmBefore :" << scal1.l0Before;
283  ret++;
284  }
285  if (scal0.l1Before > scal1.l1Before) {
286  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.l1Before << " lmBefore :" << scal1.l1Before;
287  ret++;
288  }
289  if (scal0.lmAfter > scal1.lmAfter) {
290  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.lmAfter << " lmAfter :" << scal1.lmAfter;
291  ret++;
292  }
293  if (scal0.l0After > scal1.l0After) {
294  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.l0After << " lmAfter :" << scal1.l0After;
295  ret++;
296  }
297  if (scal0.l1After > scal1.l1After) {
298  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.l1After << " lmAfter :" << scal1.l1After;
299  ret++;
300  }
301  //
302  // LMB >= LMA >= L0B >= L0A >= L1B >= L1A: 5 relations
303  // broken for classes started at L0
304  //
305  if ((scal1.lmAfter - scal0.lmAfter) > (scal1.lmBefore - scal0.lmBefore)) {
306  LOG(error) << "LMA > LMB eerror";
307  ret++;
308  }
309  if ((scal1.l0After - scal0.l0After) > (scal1.l0Before - scal0.l0Before)) {
310  LOG(error) << "L0A > L0B error";
311  ret++;
312  }
313  if ((scal1.l1After - scal0.l1After) > (scal1.l1Before - scal0.l1Before)) {
314  LOG(error) << "L1A > L1B error";
315  ret++;
316  }
317  if ((scal1.l0Before - scal0.l0Before) > (scal1.lmAfter - scal0.lmAfter)) {
318  LOG(warning) << "L0B > LMA ok if L0 class.";
319  ret++;
320  }
321  if ((scal1.l1Before - scal0.l1Before) > (scal1.l0After - scal0.l0After)) {
322  LOG(error) << "L1B > L0A Before error.";
323  ret++;
324  }
325  //
326  if (ret) {
327  scal0.printStream(std::cout);
328  scal1.printStream(std::cout);
329  }
330  return ret;
331 }
332 int CTPRunScalers::updateOverflows(const CTPScalerRecordRaw& rec0, const CTPScalerRecordRaw& rec1, overflows_t& classesoverflows) const
333 {
334  if (rec1.scalers.size() != mClassMask.count()) {
335  LOG(error) << "Inconsistent scaler record size:" << rec1.scalers.size() << " Expected:" << mClassMask.count();
336  return 1;
337  }
338  for (int i = 0; i < rec0.scalers.size(); i++) {
339  int k = (getClassIndexes())[i];
340  updateOverflows(rec0.scalers[i], rec1.scalers[i], classesoverflows[k]);
341  }
342  return 0;
343 }

351 int CTPRunScalers::updateOverflows(const CTPScalerRaw& scal0, const CTPScalerRaw& scal1, std::array<uint32_t, 6>& overflow) const
352 {

354  overflow[0] += 1;
355  }
356  if (scal0.lmAfter > scal1.lmAfter) {

360  overflow[2] += 1;
361  }
362  if (scal0.l0After > scal1.l0After) {
363  overflow[3] += 1;
364  }
365  if (scal0.l1Before > scal1.l1Before) {
366  overflow[4] += 1;
367  }
368  if (scal0.l1After > scal1.l1After) {
369  overflow[5] += 1;
370  }
371  //std::cout << "lmB0:" << scal0.lmBefore << " lmB1:" << scal1.lmBefore << " over:" << overflow[0] << std::endl;
372  //for(int i = 0; i < 6; i++)std::cout << overflow[i] << " ";
373  //std::cout << std::endl;
374  return 0;

376 int CTPRunScalers::printRates()
377 {
378  if (mScalerRecordO2.size() == 0) {
379  LOG(info) << "ScalerRecord is empty, doing nothing";
380  return 0;
381  }
382  LOG(info) << "Scaler rates for run:" << mRunNumber;
383  CTPScalerRecordO2* scalrec0 = &mScalerRecordO2[0];
384  for (int i = 1; i < mScalerRecordO2.size(); i++) {
385  CTPScalerRecordO2* scalrec1 = &mScalerRecordO2[i];
386  double_t tt = (double_t)(scalrec1->intRecord.orbit - scalrec0->intRecord.orbit);
387  tt = tt * 88e-6;
388  for (int j = 0; j < scalrec1->scalers.size(); j++) {
389  CTPScalerO2* s0 = &(scalrec0->scalers[j]);
390  CTPScalerO2* s1 = &(scalrec1->scalers[j]);
391  double_t rMB = (s1->lmBefore - s0->lmBefore) / tt;
392  double_t rMA = (s1->lmAfter - s0->lmAfter) / tt;
393  double_t r0B = (s1->l0Before - s0->l0Before) / tt;
394  double_t r0A = (s1->l0After - s0->l0After) / tt;
395  double_t r1B = (s1->l1Before - s0->l1Before) / tt;
396  double_t r1A = (s1->l1After - s0->l1After) / tt;
397  std::cout << "Class:" << s0->classIndex << ": ";
398  std::cout << rMB << " " << rMA << " ";
399  std::cout << r0B << " " << r0A << " ";
400  std::cout << r1B << " " << r1A;
401  std::cout << std::endl;
402  }
403  scalrec0 = scalrec1;
404  }
405  return 0;
406 }
