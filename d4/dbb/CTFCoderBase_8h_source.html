70  template <typename S>
71  void createCoder(OpType op, const o2::rans::RenormedFrequencyTable& renormedFrequencyTable, int slot)
72  {
73  if (renormedFrequencyTable.empty()) {
74  LOG(warning) << "Empty dictionary provided for slot " << slot << ", " << (op == OpType::Encoder ? "encoding" : "decoding") << " will assume literal symbols only";
75  }
76 
77  switch (op) {
78  case OpType::Encoder:
79  mCoders[slot].reset(new o2::rans::LiteralEncoder64<S>(renormedFrequencyTable));
80  break;
81  case OpType::Decoder:
82  mCoders[slot].reset(new o2::rans::LiteralDecoder64<S>(renormedFrequencyTable));
83  break;
84  }
85  }

87  void clear()
88  {
89  for (auto c : mCoders) {

91  }
92  }

132 template <typename T>
133 bool CTFCoderBase::readFromTree(TTree& tree, const std::string brname, T& dest, int ev)
134 {
135  auto* br = tree.GetBranch(brname.c_str());
136  if (br && br->GetEntries() > ev) {
137  auto* ptr = &dest;
138  br->SetAddress(&ptr);
139  br->GetEntry(ev);
140  br->ResetAddress();
141  return true;
142  }
143  return false;
144 }

161 template <typename CTF>
162 std::vector<char> CTFCoderBase::readDictionaryFromFile(const std::string& dictPath, bool mayFail)
163 {
164  std::vector<char> bufVec;
165  std::unique_ptr<TFile> fileDict;
166  if (std::filesystem::exists(dictPath)) {
167  fileDict.reset(TFile::Open(dictPath.c_str()));
168  }
169  if (!fileDict || fileDict->IsZombie()) {
170  std::string errstr = fmt::format("CTF dictionary file {} for detector {} is absent", dictPath, mDet.getName());
171  if (mayFail) {
172  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
173  } else {
174  throw std::runtime_error(errstr);
175  }
176  return bufVec;
177  }
178  std::unique_ptr<TTree> tree((TTree*)fileDict->Get(std::string(o2::base::NameConf::CTFDICT).c_str()));
179  std::unique_ptr<std::vector<char>> bv((std::vector<char>*)fileDict->GetObjectChecked(o2::base::NameConf::CCDBOBJECT.data(), "std::vector<char>"));
180  if (tree) {
181  CTFHeader ctfHeader;
182  if (!readFromTree(*tree.get(), "CTFHeader", ctfHeader) || !ctfHeader.detectors[mDet]) {
183  std::string errstr = fmt::format("CTF dictionary file for detector {} is absent in the tree from file {}", mDet.getName(), dictPath);
184  if (mayFail) {
185  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
186  } else {
187  throw std::runtime_error(errstr);
188  }
189  return bufVec;
190  }
191  CTF::readFromTree(bufVec, *tree.get(), mDet.getName());
192  } else if (bv) {
193  bufVec.swap(*bv);
194  if (bufVec.size()) {
195  auto dictHeader = static_cast<const o2::ctf::CTFDictHeader&>(CTF::get(bufVec.data())->getHeader());
196  if (dictHeader.det != mDet) {
197  throw std::runtime_error(fmt::format("{} contains dictionary vector for {}, expected {}", dictPath, dictHeader.det.getName(), mDet.getName()));
198  }
199  }
200  }
201  if (bufVec.size()) {
202  mExtHeader = static_cast<CTFDictHeader&>(CTF::get(bufVec.data())->getHeader());
203  LOGP(debug, "Found {} in {}", mExtHeader.asString(), dictPath);
204  } else {
205  std::string errstr = fmt::format("CTF dictionary file for detector {} is empty", mDet.getName());
206  if (mayFail) {
207  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
208  } else {
209  throw std::runtime_error(errstr);
210  }
211  }
212  return bufVec;
213 }

216 template <typename CTF>
217 void CTFCoderBase::init(o2::framework::InitContext& ic)
218 {
219  if (ic.options().hasOption("mem-factor")) {
220  setMemMarginFactor(ic.options().get<float>("mem-factor"));
221  }
222  auto dict = ic.options().get<std::string>("ctf-dict");
223  if (dict.empty() || dict == "ccdb") { // load from CCDB
224  mLoadDictFromCCDB = true;
225  } else {
226  if (dict != "none") { // none means per-CTF dictionary will created on the fly
227  createCodersFromFile<CTF>(dict, mOpType);
228  LOGP(info, "Loaded {} from {}", mExtHeader.asString(), dict);
229  } else {
230  LOGP(info, "Internal per-TF CTF Dict will be created");
231  }
232  mLoadDictFromCCDB = false; // don't try to load from CCDB
233  }
234 }

248 template <typename CTF>
249 bool CTFCoderBase::finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj)
250 {
251  bool match = false;
252  if (mLoadDictFromCCDB && (match = (matcher == o2::framework::ConcreteDataMatcher(mDet.getDataOrigin(), "CTFDICT", 0)))) {
253  const auto* dict = (std::vector<char>*)obj;
254  if (dict->empty()) {
255  LOGP(info, "Empty dictionary object fetched from CCDB, internal per-TF CTF Dict will be created");
256  } else {
257  createCoders(*dict, mOpType);
258  mExtHeader = static_cast<const CTFDictHeader&>(CTF::get(dict->data())->getHeader());
259  LOGP(info, "Loaded {} from CCDB", mExtHeader.asString());
260  }
261  mLoadDictFromCCDB = false; // we read the dictionary at most once!
262  }
263  return match;
264 }
