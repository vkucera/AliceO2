51 
52  std::vector<char> result(size() * CRUWordSize);
53  char* out = result.data();
54  if (!tcm) {
55  std::memcpy(out, &mEventHeader, sPayloadSize);
56  out += sPayloadSize;
57  LOG(debug) << "write PM header words " << (int)mEventHeader.nGBTWords << " orbit: " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
58  printHexEventHeader();
59  out += CRUWordSize - sPayloadSize;
60 
61  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
62  std::memcpy(out, &mEventData[2 * i], sPayloadSizeFirstWord);
63  LOG(debug) << " 1st word " << mEventData[2 * i].channelID << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time << " out " << result.size();
64  out += sPayloadSizeFirstWord;
65  std::memcpy(out, &mEventData[2 * i + 1], sPayloadSizeSecondWord);
66  out += sPayloadSizeSecondWord;
67  LOG(debug) << " 2nd word " << mEventData[2 * i + 1].channelID << " charge " << mEventData[2 * i + 1].charge << " time " << mEventData[2 * i + 1].time << " out " << result.size();
68  out += CRUWordSize - sPayloadSizeSecondWord - sPayloadSizeFirstWord;
69  printHexEventData(i);
70  }
71  } else {
72  // TCM data
73  std::memcpy(out, &mEventHeader, sPayloadSize);
74  out += sPayloadSize;

79  }
80
