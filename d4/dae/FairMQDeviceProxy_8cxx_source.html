59 ChannelIndex FairMQDeviceProxy::getOutputChannelIndex(OutputSpec const& query, size_t timeslice) const
60 {
61  assert(mOutputRoutes.size() == mOutputs.size());
62  for (size_t ri = 0; ri < mOutputs.size(); ++ri) {
63  auto& route = mOutputs[ri];
64 
65  LOG(debug) << "matching: " << DataSpecUtils::describe(query) << " to route " << DataSpecUtils::describe(route.matcher);
66  if (DataSpecUtils::match(route.matcher, query) && ((timeslice % route.maxTimeslices) == route.timeslice)) {
67  return mOutputRoutes[ri].channel;
68  }
69  }
70  return ChannelIndex{ChannelIndex::INVALID};
71 };

127 void FairMQDeviceProxy::bind(std::vector<OutputRoute> const& outputs, std::vector<InputRoute> const& inputs, fair::mq::Device& device)
128 {
129  mOutputs.clear();
130  mOutputRoutes.clear();
131  mOutputChannels.clear();
132  mOutputChannelNames.clear();
133  mInputs.clear();
134  mInputRoutes.clear();
135  mInputChannels.clear();
136  mInputChannelNames.clear();
137  {
138  mOutputs = outputs;
139  mOutputRoutes.reserve(outputs.size());
140  size_t ri = 0;
141  std::unordered_map<std::string, ChannelIndex> channelNameToChannel;
142  for (auto& route : outputs) {
143  // If the channel is not yet registered, register it.
144  // If the channel is already registered, use the existing index.
145  auto channelPos = channelNameToChannel.find(route.channel);
146  ChannelIndex channelIndex;
147 
148  if (channelPos == channelNameToChannel.end()) {
149  channelIndex = ChannelIndex{(int)mOutputChannels.size()};
150  mOutputChannels.push_back(&device.fChannels.at(route.channel).at(0));
151  mOutputChannelNames.push_back(route.channel);
152  channelNameToChannel[route.channel] = channelIndex;
153  LOGP(debug, "Binding channel {} to channel index {}", route.channel, channelIndex.value);
154  } else {
155  LOGP(debug, "Using index {} for channel {}", channelPos->second.value, route.channel);
156  channelIndex = channelPos->second;
157  }
158  LOGP(debug, "Binding route {}@{}%{} to index {} and channelIndex {}", route.matcher, route.timeslice, route.maxTimeslices, ri, channelIndex.value);
159  mOutputRoutes.emplace_back(RouteState{channelIndex, false});
160  ri++;
161  }
162  for (auto& route : mOutputRoutes) {
163  assert(route.channel.value != -1);
164  assert(route.channel.value < mOutputChannels.size());
165  }
166  LOGP(debug, "Total channels found {}, total routes {}", mOutputChannels.size(), mOutputRoutes.size());
167  assert(mOutputRoutes.size() == outputs.size());
168  }
169 
170  {
171  auto maxLanes = InputRouteHelpers::maxLanes(inputs);
172  mInputs = inputs;
173  mInputRoutes.reserve(inputs.size());
174  size_t ri = 0;
175  std::unordered_map<std::string, ChannelIndex> channelNameToChannel;
176  for (auto& route : inputs) {
177  // If the channel is not yet registered, register it.
178  // If the channel is already registered, use the existing index.
179  auto channelPos = channelNameToChannel.find(route.sourceChannel);
180  ChannelIndex channelIndex;
181 
182  if (channelPos == channelNameToChannel.end()) {
183  channelIndex = ChannelIndex{(int)mInputChannels.size()};
184  mInputChannels.push_back(&device.fChannels.at(route.sourceChannel).at(0));
185  mInputChannelNames.push_back(route.sourceChannel);
186  channelNameToChannel[route.sourceChannel] = channelIndex;
187  LOGP(debug, "Binding channel {} to channel index {}", route.sourceChannel, channelIndex.value);
188  } else {
189  LOGP(debug, "Using index {} for channel {}", channelPos->second.value, route.sourceChannel);
190  channelIndex = channelPos->second;
191  }
192  LOGP(debug, "Binding route {}@{}%{} to index {} and channelIndex {}", route.matcher, route.timeslice, maxLanes, ri, channelIndex.value);
193  mInputRoutes.emplace_back(RouteState{channelIndex, false});
194  ri++;
195  }
196  for (auto& route : mInputRoutes) {
197  assert(route.channel.value != -1);
198  assert(route.channel.value < mInputChannels.size());
199  }
200  LOGP(debug, "Total input channels found {}, total routes {}", mInputChannels.size(), mInputRoutes.size());
201  assert(mInputRoutes.size() == inputs.size());
202  }
203  mStateChangeCallback = [&device]() -> bool { return device.NewStatePending(); };
204 }
