72  void run(framework::ProcessingContext& pc)
73  {
74 
75  if (mFinished) {
76  return;
77  }
78 
79  // read collision context from input
80  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
81  context->initSimChains(o2::detectors::DetID::FT0, mSimChains);
82  const bool withQED = context->isQEDProvided() && !mDisableQED;
83  auto& timesview = context->getEventRecords(withQED);
84  //set CCDB for miscalibration
85  if (mUseCCDB) {
86  auto caliboffsets = pc.inputs().get<o2::ft0::FT0ChannelTimeCalibrationObject*>("ft0offsets");
87  mDigitizer.SetChannelOffset(caliboffsets.get());
88  }
89  // if there is nothing to do ... return
90  if (timesview.size() == 0) {
91  return;
92  }
93 
94  TStopwatch timer;
95  timer.Start();
96 
97  LOG(info) << "CALLING FT0 DIGITIZATION";
98 
99  static std::vector<o2::ft0::HitType> hits;
100  // o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labelAccum;
101  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labels;
102 
103  // mDigitizer.setMCLabels(&labels);
104  auto& eventParts = context->getEventParts(withQED);
105  // loop over all composite collisions given from context
106  // (aka loop over all the interaction records)
107  for (int collID = 0; collID < timesview.size(); ++collID) {
108  mDigitizer.setInteractionRecord(timesview[collID]);
109  LOG(debug) << " setInteractionRecord " << timesview[collID] << " bc " << mDigitizer.getBC() << " orbit " << mDigitizer.getOrbit();
110  // for each collision, loop over the constituents event and source IDs
111  // (background signal merging is basically taking place here)
112  for (auto& part : eventParts[collID]) {
113  // get the hits for this event and this source
114  hits.clear();
115  context->retrieveHits(mSimChains, "FT0Hit", part.sourceID, part.entryID, &hits);
116  LOG(debug) << "For collision " << collID << " eventID " << part.entryID << " source ID " << part.sourceID << " found " << hits.size() << " hits ";
117  if (hits.size() > 0) {
118  // call actual digitization procedure
119  mDigitizer.setEventID(part.entryID);
120  mDigitizer.setSrcID(part.sourceID);
121  mDigitizer.process(&hits, mDigitsBC, mDigitsCh, mDigitsTrig, labels);
122  }
123  }
124  }
125  mDigitizer.flush_all(mDigitsBC, mDigitsCh, mDigitsTrig, labels);
126 
127  // send out to next stage
128  pc.outputs().snapshot(Output{"FT0", "DIGITSBC", 0, Lifetime::Timeframe}, mDigitsBC);
129  pc.outputs().snapshot(Output{"FT0", "DIGITSCH", 0, Lifetime::Timeframe}, mDigitsCh);
130  pc.outputs().snapshot(Output{"FT0", "TRIGGERINPUT", 0, Lifetime::Timeframe}, mDigitsTrig);
131  if (pc.outputs().isAllowed({"FT0", "DIGITSMCTR", 0})) {
132  pc.outputs().snapshot(Output{"FT0", "DIGITSMCTR", 0, Lifetime::Timeframe}, labels);
133  }
134  LOG(info) << "FT0: Sending ROMode= " << mROMode << " to GRPUpdater";
135  pc.outputs().snapshot(Output{"FT0", "ROMode", 0, Lifetime::Timeframe}, mROMode);
136 
137  timer.Stop();
138  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
139 
140  // we should be only called once; tell DPL that this process is ready to exit
141  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
142  mFinished = true;
143  }
144 
145  protected:
146  bool mFinished = false;
147  std::vector<o2::ft0::ChannelData> mDigitsCh;
148  std::vector<o2::ft0::Digit> mDigitsBC;
149  std::vector<o2::ft0::DetTrigInput> mDigitsTrig;
150 
151  Bool_t mContinuous = kFALSE;
152  double mFairTimeUnitInNS = 1;
153  o2::ft0::Digitizer mDigitizer;
154 
155  // RS: at the moment using hardcoded flag for continuos readout
156  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
157 
158  //
159  bool mDisableQED = false;
160  bool mUseCCDB = true;
161  bool mUpdateCCDB = true;
162  std::vector<TChain*> mSimChains;
163 };

165 o2::framework::DataProcessorSpec getFT0DigitizerSpec(int channel, bool mctruth, bool useCCDB)
166 {
167  // create the full data processor spec using
168  // a name identifier
169  // input description
170  // algorithmic description (here a lambda getting called once to setup the actual processing function)
171  // options that can be used for this processor (here: input file names where to take the hits)
172  std::vector<OutputSpec> outputs;
173  outputs.emplace_back("FT0", "DIGITSBC", 0, Lifetime::Timeframe);
174  outputs.emplace_back("FT0", "DIGITSCH", 0, Lifetime::Timeframe);
175  outputs.emplace_back("FT0", "TRIGGERINPUT", 0, Lifetime::Timeframe);
176  if (mctruth) {
177  outputs.emplace_back("FT0", "DIGITSMCTR", 0, Lifetime::Timeframe);
178  }
179  outputs.emplace_back("FT0", "ROMode", 0, Lifetime::Timeframe);
180  std::vector<InputSpec> inputs;
181  inputs.emplace_back("collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
182  if (useCCDB) {
183  inputs.emplace_back("ft0offsets", "FT0", "TimeOffset", 0,
184  Lifetime::Condition,
185  ccdbParamSpec("FT0/Calib/ChannelTimeOffset"));
186  }
187  return DataProcessorSpec{
188  "FT0Digitizer",
189  inputs,
190  outputs,
191  AlgorithmSpec{adaptFromTask<FT0DPLDigitizerTask>(useCCDB)},
192  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}},
193  {"disable-qed", o2::framework::VariantType::Bool, false, {"disable QED handling"}}}};
194 }
