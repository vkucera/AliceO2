72  void run(of::ProcessingContext& pc)
73  {
74  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
75  auto noiseRof = pc.inputs().get<gsl::span<ROFRecord>>("mid_noise_rof");
76  unsigned long nEvents = noiseRof.size();
77  if (nEvents == 0) {
78  return;
79  }
80 
81  auto deadRof = pc.inputs().get<gsl::span<ROFRecord>>("mid_dead_rof");
82  auto noise = pc.inputs().get<gsl::span<ColumnData>>("mid_noise");
83  auto dead = pc.inputs().get<gsl::span<ColumnData>>("mid_dead");
84 
85  std::vector<ColumnData> calibData;
86  calibData.insert(calibData.end(), noise.begin(), noise.end());
87  calibData.insert(calibData.end(), dead.begin(), dead.end());
88  mCalibrator.addEvents(nEvents);
89  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mCalibrator.getCurrentTFInfo());
90  mCalibrator.process(calibData);
91  }
92 
93  void endOfStream(of::EndOfStreamContext& ec)
94  {
95  mCalibrator.checkSlotsToFinalize(ChannelCalibrator::INFINITE_TF);
96  sendOutput(ec.outputs());
97  }

117  template <typename T>
118  void sendOutput(of::DataAllocator& output, T& payload, const char* path, header::DataHeader::SubSpecificationType subSpec)
119  {
120  o2::ccdb::CcdbObjectInfo info;
121  std::map<std::string, std::string> md;
122  o2::calibration::Utils::prepareCCDBobjectInfo(payload, info, path, md, mCalibrator.getCurrentTFInfo().creation, mCalibrator.getCurrentTFInfo().creation + 5 * o2::ccdb::CcdbObjectInfo::DAY);
123  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
124  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
125  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
126  output.snapshot(of::Output{o2::calibration::Utils::gDataOriginCDBPayload, "MID_BAD_CHANNELS", subSpec}, *image.get()); // vector<char>
127  output.snapshot(of::Output{o2::calibration::Utils::gDataOriginCDBWrapper, "MID_BAD_CHANNELS", subSpec}, info); // root-serialized
128  }

131 of::DataProcessorSpec getChannelCalibratorSpec(const FEEIdConfig& feeIdConfig, const CrateMasks& crateMasks)
132 {
133  std::vector<of::InputSpec> inputSpecs;
134  inputSpecs.emplace_back("mid_noise", header::gDataOriginMID, "NOISE");
135  inputSpecs.emplace_back("mid_noise_rof", header::gDataOriginMID, "NOISEROF");
136  inputSpecs.emplace_back("mid_dead", header::gDataOriginMID, "DEAD");
137  inputSpecs.emplace_back("mid_dead_rof", header::gDataOriginMID, "DEADROF");
138  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
139  true, // GRPECS=true
140  false, // GRPLHCIF
141  false, // GRPMagField
142  false, // askMatLUT
143  o2::base::GRPGeomRequest::None, // geometry
144  inputSpecs);
145  std::vector<of::OutputSpec> outputSpecs;
146  outputSpecs.emplace_back(o2::calibration::Utils::gDataOriginCDBPayload, "MID_BAD_CHANNELS", 0, of::Lifetime::Sporadic);
147  outputSpecs.emplace_back(o2::calibration::Utils::gDataOriginCDBWrapper, "MID_BAD_CHANNELS", 0, of::Lifetime::Sporadic);
148  outputSpecs.emplace_back(o2::calibration::Utils::gDataOriginCDBPayload, "MID_BAD_CHANNELS", 1, of::Lifetime::Sporadic);
149  outputSpecs.emplace_back(o2::calibration::Utils::gDataOriginCDBWrapper, "MID_BAD_CHANNELS", 1, of::Lifetime::Sporadic);
150 
151  return of::DataProcessorSpec{
152  "MIDChannelCalibrator",
153  {inputSpecs},
154  {outputSpecs},
155  of::AlgorithmSpec{of::adaptFromTask<o2::mid::ChannelCalibratorDeviceDPL>(feeIdConfig, crateMasks, ccdbRequest)},
156  of::Options{{"mid-mask-threshold", of::VariantType::Double, 0.9, {"Tolerated occupancy before producing a map"}}}};
157 }
