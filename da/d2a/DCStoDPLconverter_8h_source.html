54 o2f::InjectorFunction dcs2dpl(std::unordered_map<DPID, o2h::DataDescription>& dpid2group, bool fbiFirst, bool verbose = false)
55 {
56 
57  return [dpid2group, fbiFirst, verbose](o2::framework::TimingInfo& tinfo, fair::mq::Device& device, fair::mq::Parts& parts, o2f::ChannelRetriever channelRetriever) {
58  static std::unordered_map<DPID, DPCOM> cache; // will keep only the latest measurement in the 1-second wide window for each DPID
59  static auto timer = std::chrono::high_resolution_clock::now();
60  static auto timer0 = std::chrono::high_resolution_clock::now();
61  static bool seenFBI = false;
62  static uint32_t localTFCounter = 0;
63  static size_t nInp = 0, nInpFBI = 0;
64  static size_t szInp = 0, szInpFBI = 0;
65  LOG(debug) << "In lambda function: ********* Size of unordered_map (--> number of defined groups) = " << dpid2group.size();
66  // check if we got FBI (Master) or delta (MasterDelta)
67  if (!parts.Size()) {
68  LOGP(warn, "Empty input recieved at timeslice {}", tinfo.timeslice);
69  return;
70  }
71  std::string firstName = std::string((char*)&(reinterpret_cast<const DPCOM*>(parts.At(0)->GetData()))->id);
72 
73  bool isFBI = false;
74  nInp++;
75  if (o2::utils::Str::endsWith(firstName, "Master")) {
76  isFBI = true;
77  nInpFBI++;
78  seenFBI = true;
79  } else if (o2::utils::Str::endsWith(firstName, "MasterDelta")) {
80  isFBI = false;
81  } else {
82  LOGP(error, "Cannot determine if the map is FBI or Delta, 1st DP name is {}", firstName);
83  }
84  if (verbose) {
85  LOGP(info, "New input of {} parts received, map type: {}, timeslice {}", parts.Size(), isFBI ? "FBI" : "Delta", tinfo.timeslice);
86  }
87 
88  // We first iterate over the parts of the received message
89  for (size_t i = 0; i < parts.Size(); ++i) { // DCS sends only 1 part, but we should be able to receive more
90  auto sz = parts.At(i)->GetSize();
91  szInp += sz;
92  if (isFBI) {
93  szInpFBI += sz;
94  }
95  auto nDPCOM = sz / sizeof(DPCOM); // number of DPCOM in current part
96  for (size_t j = 0; j < nDPCOM; j++) {
97  const auto& src = *(reinterpret_cast<const DPCOM*>(parts.At(i)->GetData()) + j);
98  // do we want to check if this DP was requested ?
99  auto mapEl = dpid2group.find(src.id);
100  if (verbose) {
101  LOG(info) << "Received DP " << src.id << " (data = " << src.data << "), matched to output-> " << (mapEl == dpid2group.end() ? "none " : mapEl->second.as<std::string>());
102  }
103  if (mapEl != dpid2group.end()) {
104  cache[src.id] = src; // this is needed in case in the 1s window we get a new value for the same DP
105  }
106  }
107  }
108  auto timerNow = std::chrono::high_resolution_clock::now();
109  if (fbiFirst && nInpFBI < 2) { // 1st FBI might be obsolete
110  seenFBI = false;
111  static int prevDelay = 0;
112  std::chrono::duration<double, std::ratio<1>> duration = timerNow - timer0;
113  int delay = duration.count();
114  if (delay > prevDelay) {
115  LOGP(info, "Waiting for requested 1st FBI since {} s", delay);
116  prevDelay = delay;
117  }
118  }
119 
120  std::chrono::duration<double, std::ratio<1>> duration = timerNow - timer;
121  if (duration.count() > 1 && (seenFBI || !fbiFirst)) { // did we accumulate for 1 sec and have we seen FBI if it was requested?
122  std::unordered_map<o2h::DataDescription, pmr::vector<DPCOM>, std::hash<o2h::DataDescription>> outputs;
123  // in the cache we have the final values of the DPs that we should put in the output
124  // distribute DPs over the vectors for each requested output
125  for (auto& it : cache) {
126  auto mapEl = dpid2group.find(it.first);
127  if (mapEl != dpid2group.end()) {
128  outputs[mapEl->second].push_back(it.second);
129  }
130  }
131  std::uint64_t creation = std::chrono::time_point_cast<std::chrono::milliseconds>(timerNow).time_since_epoch().count();
132  std::unordered_map<std::string, std::unique_ptr<fair::mq::Parts>> messagesPerRoute;
133  // create and send output messages
134  for (auto& it : outputs) { // distribute messages per routes
135  o2h::DataHeader hdr(it.first, "DCS", 0);
136  o2f::OutputSpec outsp{hdr.dataOrigin, hdr.dataDescription, hdr.subSpecification};
137  if (it.second.empty()) {
138  LOG(warning) << "No data for OutputSpec " << outsp;
139  continue;
140  }
141  auto channel = channelRetriever(outsp, tinfo.timeslice);
142  if (channel.empty()) {
143  LOG(warning) << "No output channel found for OutputSpec " << outsp << ", discarding its data";
144  it.second.clear();
145  continue;
146  }
147 
148  hdr.tfCounter = localTFCounter; // this also
149  hdr.payloadSerializationMethod = o2h::gSerializationMethodNone;
150  hdr.splitPayloadParts = 1;
151  hdr.splitPayloadIndex = 1;
152  hdr.payloadSize = it.second.size() * sizeof(DPCOM);
153  hdr.firstTForbit = 0; // this should be irrelevant for DCS
154  o2h::Stack headerStack{hdr, o2::framework::DataProcessingHeader{tinfo.timeslice, 1, creation}};
155  auto fmqFactory = device.GetChannel(channel).Transport();
156  auto hdMessage = fmqFactory->CreateMessage(headerStack.size(), fair::mq::Alignment{64});
157  auto plMessage = fmqFactory->CreateMessage(hdr.payloadSize, fair::mq::Alignment{64});
158  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
159  memcpy(plMessage->GetData(), it.second.data(), hdr.payloadSize);
160 
161  fair::mq::Parts* parts2send = messagesPerRoute[channel].get(); // fair::mq::Parts*
162  if (!parts2send) {
163  messagesPerRoute[channel] = std::make_unique<fair::mq::Parts>();
164  parts2send = messagesPerRoute[channel].get();
165  }
166  parts2send->AddPart(std::move(hdMessage));
167  parts2send->AddPart(std::move(plMessage));
168  if (verbose) {
169  LOGP(info, "Pushing {} DPs to {} for TimeSlice {} at {}", it.second.size(), o2f::DataSpecUtils::describe(outsp), tinfo.timeslice, creation);
170  }
171  it.second.clear();
172  }
173  // push output of every route
174  for (auto& msgIt : messagesPerRoute) {
175  LOG(info) << "Sending " << msgIt.second->Size() / 2 << " parts to channel " << msgIt.first;
176  o2f::sendOnChannel(device, *msgIt.second.get(), msgIt.first, tinfo.timeslice);
177  }
178  timer = timerNow;
179  cache.clear();
180  if (!messagesPerRoute.empty()) {
181  localTFCounter++;
182  }
183  }
184  if (isFBI) {
185  float runtime = 1e-3 * std::chrono::duration_cast<std::chrono::milliseconds>(timerNow - timer0).count();
186  LOGP(info, "{} inputs ({} bytes) of which {} FBI ({} bytes) seen in {:.3f} s", nInp, fmt::group_digits(szInp), nInpFBI, fmt::group_digits(szInpFBI), runtime);
187  }
188  };
189 }
