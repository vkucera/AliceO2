43 void DataProcessingHelpers::sendOldestPossibleTimeframe(fair::mq::Channel& channel, size_t timeslice)
44 {
45  fair::mq::Parts parts;
46  fair::mq::MessagePtr payload(channel.Transport()->CreateMessage());
47  o2::framework::DomainInfoHeader dih;
48  dih.oldestPossibleTimeslice = timeslice;
49  auto channelAlloc = o2::pmr::getTransportAllocator(channel.Transport());
50  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dih});
51  // sigh... See if we can avoid having it const by not
52  // exposing it to the user in the first place.
53  parts.AddPart(std::move(header));
54  parts.AddPart(std::move(payload));
55 
56  auto timeout = 1000;
57  auto res = channel.Send(parts, timeout);
58  if (res == (size_t)fair::mq::TransferCode::timeout) {
59  LOGP(warning, "Timed out sending oldest possible timeslice after {}s. Downstream backpressure detected on {}.", timeout / 1000, channel.GetName());
60  channel.Send(parts);
61  LOGP(info, "Downstream backpressure on {} recovered.", channel.GetName());
62  } else if (res == (size_t)fair::mq::TransferCode::error) {
63  LOGP(fatal, "Error while sending on channel {}", channel.GetName());
64  }
65 }
