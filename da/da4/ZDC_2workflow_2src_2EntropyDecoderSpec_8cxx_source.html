47 void EntropyDecoderSpec::run(ProcessingContext& pc)
48 {
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51  o2::ctf::CTFIOSize iosize;
52 
53  mCTFCoder.updateTimeDependentParams(pc);
54  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
55 
56  auto& bcdata = pc.outputs().make<std::vector<o2::zdc::BCData>>(OutputRef{"trig"});
57  auto& chans = pc.outputs().make<std::vector<o2::zdc::ChannelData>>(OutputRef{"chan"});
58  auto& peds = pc.outputs().make<std::vector<o2::zdc::OrbitData>>(OutputRef{"peds"});
59 
60  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
61  if (buff.size()) {
62  const auto ctfImage = o2::zdc::CTF::getImage(buff.data());
63  iosize = mCTFCoder.decode(ctfImage, bcdata, chans, peds);
64  }
65  pc.outputs().snapshot({"ctfrep", 0}, iosize);
66  mTimer.Stop();
67  LOG(info) << "Decoded " << chans.size() << " ZDC channels in " << bcdata.size() << " triggers and " << peds.size() << " pedestals, (" << iosize.asString() << ") in " << mTimer.CpuTime() - cput << " s";
68 }

92  AlgorithmSpec{adaptFromTask<EntropyDecoderSpec>(verbosity)},
