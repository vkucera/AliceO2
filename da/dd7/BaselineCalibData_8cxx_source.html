35 BaselineCalibData& BaselineCalibData::operator+=(const BaselineCalibData& other)
36 {
37  if (other.mOverflow) {
38  // Refusing to add an overflow
39  LOG(warn) << "BaselineCalibData" << __func__ << " Refusing to add an overflow. BREAK!";
40  return *this;
41  }
42  for (int32_t is = 0; is < NChannels; is++) {
43  if (other.mHisto[is].mOverflow) {
44  // Refusing to add an overflow histogram
45  LOG(warn) << "BaselineCalibData::" << __func__ << " Refusing to add an overflow histogram for ch " << is << " BREAK!";
46  return *this;
47  }
48  // Check if sum will result into an overflow
49  for (int32_t i = 0; i < BaselineCalibChData::NW; i++) {
50  uint64_t sum = mHisto[is].mData[i] + other.mHisto[is].mData[i];
51  if (sum > 0xffffffff) {
52  LOG(warn) << "BaselineCalibData::" << __func__ << " Addition would result in an overflow for ch " << is << " BREAK!";
53  return *this;
54  }
55  }
56  }
57  // No problems with overflow
58  for (int32_t is = 0; is < NChannels; is++) {
59  for (int32_t i = 0; i < BaselineCalibChData::NW; i++) {
60  mHisto[is].mData[i] = mHisto[is].mData[i] + other.mHisto[is].mData[i];
61  }
62  }
63  if (mCTimeBeg == 0 || other.mCTimeBeg < mCTimeBeg) {
64  mCTimeBeg = other.mCTimeBeg;
65  }
66  if (other.mCTimeEnd > mCTimeEnd) {
67  mCTimeEnd = other.mCTimeEnd;
68  }
69 #ifdef O2_ZDC_DEBUG
70  LOG(info) << __func__;
71  print();
72 #endif
73  return *this;
74 }

84  mHisto[is].clear();

88  LOG(warn) << "Overflow bit set on signal " << is;

93  mHisto[bin.id].mData[bin.ibin] = bin.cont;

178 int BaselineCalibChData::getStat(uint64_t& en, double& mean, double& var) const
179 {
180  en = 0;
181  uint64_t sum = 0;
182  for (uint64_t i = 0; i < NW; i++) {
183  en += mData[i];
184  sum += i * mData[i];
185  }
186  if (en == 0) {
187  return 1;
188  }
189  mean = double(sum) / double(en);
190  if (en > 1) {
191  double sums = 0;
192  for (int32_t i = 0; i < NW; i++) {
193  double diff = i - mean;
194  sums += (mData[i] * diff * diff);
195  }
196  var = sums / (en - 1);
197  } else {
198  var = 0;
199  }
200  // Convert mean to correct range
201  mean = mean + BaselineMin;
202  return 0;
203 }

206 int BaselineCalibData::saveDebugHistos(const std::string fn, float factor = 1)
207 {
208  TDirectory* cwd = gDirectory;
209  TFile* f = new TFile(fn.data(), "recreate");
210  if (f->IsZombie()) {
211  LOG(error) << "Cannot create file: " << fn;
212  return 1;
213  }
214  for (int32_t is = 0; is < NChannels; is++) {
215  uint64_t nen = mHisto[is].getEntries();
216  if (nen > 0) {
217  TString n = TString::Format("h%d", is);
218  TString t = TString::Format("Baseline %d %s", is, ChannelNames[is].data());
219  TH1F h(n, t, BaselineRange, (BaselineMin - 0.5) * factor, (BaselineMax + 0.5) * factor);
220  for (int ibx = 0; ibx < BaselineRange; ibx++) {
221  h.SetBinContent(ibx + 1, mHisto[is].mData[ibx]);
222  }
223  h.SetEntries(nen);
224  h.Write("", TObject::kOverwrite);
225  }
226  }
227  f->Close();
228  cwd->cd();
229  return 0;
230 }

262 BaselineCalibSummaryData& BaselineCalibData::getSummary()
263 {
264  mSummary.clear();
265  mSummary.mCTimeBeg = mCTimeBeg;
266  mSummary.mCTimeEnd = mCTimeEnd;
267  mSummary.mOverflow = mOverflow;
268  for (int32_t is = 0; is < NChannels; is++) {
269  mSummary.mOverflowCh[is] = mHisto[is].mOverflow;
270  for (int32_t ib = 0; ib < BaselineRange; ib++) {
271  uint32_t cont = mHisto[is].mData[ib];

273  mSummary.mData.emplace_back(is, ib, cont);
