75 int BaselineCalib::endOfRun()
76 {
77  if (mVerbosity > DbgZero) {
78  LOGF(info, "Finalizing BaselineCalibData object");
79  }
80 
81  // Compute average baseline
82  float factor = mModuleConfig->baselineFactor;
83  for (int ic = 0; ic < NChannels; ic++) {
84  double sum = 0;
85  double nsum = 0;
86  double bmin = mConfig->cutLow[ic];
87  double bmax = mConfig->cutHigh[ic];
88  for (int ib = 0; ib < BaselineRange; ib++) {
89  double bval = (BaselineMin + ib) * factor;
90  if (bval >= bmin && bval <= bmax) {
91  nsum += mData.mHisto[ic].mData[ib];
92  sum += bval * mData.mHisto[ic].mData[ib];
93  }
94  }
95  if (nsum > 0 && mConfig->min_e[ic]) {
96  float ave = sum / nsum;
97  LOGF(info, "Baseline %s %g events and cuts (%g:%g): %f", ChannelNames[ic].data(), nsum, bmin, bmax, ave);
98  mParamUpd.setCalib(ic, ave, true);
99  } else {
100  if (mParam == nullptr) {
101  LOGF(error, "Baseline %s %g events and cuts (%g:%g): CANNOT UPDATE AND MISSING OLD VALUE", ChannelNames[ic].data(), nsum, bmin, bmax);
102  mParamUpd.setCalib(ic, -std::numeric_limits<float>::infinity(), false);
103  } else {
104  float val = mParam->getCalib(ic);
105  LOGF(info, "Baseline %s %g events and cuts (%g:%g): %f NOT UPDATED", ChannelNames[ic].data(), nsum, bmin, bmax, val);
106  mParamUpd.setCalib(ic, val, false);
107  }
108  }
109  }
110 
111  // Creating calibration object and info
112  auto clName = o2::utils::MemFileHelper::getClassName(mParamUpd);
113  mInfo.setObjectType(clName);
114  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
115  mInfo.setFileName(flName);
116  mInfo.setPath(CCDBPathBaselineCalib);
117  std::map<std::string, std::string> md;
118  md["config"] = mConfig->desc;
119  mInfo.setMetaData(md);
120  uint64_t starting = mData.mCTimeBeg;
121  if (starting >= 10000) {
122  starting = starting - 10000; // start 10 seconds before
123  }
124  uint64_t stopping = mData.mCTimeEnd + 10000; // stop 10 seconds after
125  mInfo.setStartValidityTimestamp(starting);
126  mInfo.setEndValidityTimestamp(stopping);
127  mInfo.setAdjustableEOV();
128  LOGF(info, "Validity: %llu:%llu", starting, stopping);
129 
130  if (mSaveDebugHistos) {
131  LOG(info) << "Saving debug histograms";
132  saveDebugHistos();
133  }
134  return 0;
135 }
