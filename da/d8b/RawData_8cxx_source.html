186 uint32_t getChargeFromRawHeaders(const o2::trd::TrackletHCHeader& hcheader, const o2::trd::TrackletMCMHeader* header, const std::array<o2::trd::TrackletMCMData, 3>& data, int pidindex, int trackletindex)
187 {
188  uint32_t pid = 0;
189  uint32_t highPID, lowPID; // highPID holds the 8 bits from the mcmheader, and lowPID holds the 12 bits from mcmdata
190  uint32_t datatype = (hcheader.format) >> 2;
191  switch (datatype) {
192  case 0: //Cosmic
193  // LOG(warn) << "This is a problem cosmic format tracklets ";
194  //break;
195  case 1: //TPT
196  //LOG(warn) << "This is a problem TPT format tracklets ";
197  //break;
198  case 2: //DIS
199  //LOG(warn) << "This is a problem DIS format tracklets ";
200  //break;
201  LOG(warn) << "This could be a problem format tracklets is 0x" << std::hex << datatype << " thcheader:0x" << std::hex << hcheader.word << " tmcmheader:0x" << header->word;
202  case 3:
203  //PID VERSION 1
204  //PID is 20 bits, 8 bits in mcmheader and 12 bits in mcmdata word
205  //frist part of pid (highPID) is in the TrackletMCMHeader
206  //highPID is 7 bits Q2, 1 bit Q1 OR ... 2 bit offset, 6 bits Q2.
207  switch (trackletindex) {
208  case 0:
209  highPID = header->pid0;
210  break;
211  case 1:
212  highPID = header->pid1;
213  break;
214  case 2:
215  highPID = header->pid2;
216  break;
217  default:
218  LOG(warn) << " unknown trackletindex of " << trackletindex << " to " << __func__ << " : " << pidindex;
219  break;
220  }
221  lowPID = data[trackletindex].pid;
222  //lowPID is 6 bits Q0 and 6 bits of Q1
223  int pidword = (highPID << 12) + lowPID; // the entire original 20 bit pid in the trap chips
224  int dynamicq = hcheader.format & 0x1; // last bit of format (lsb) defines the version of tracklet charge calculation
225  uint32_t pidoffset = ((pidword >> 18) & 0x3); //<<6; // used for dynamic ranged charge windows, may or may not be used below.
226  //pidword is here to make this code more readible and less error prone.
227  switch (pidindex) {
228  case 2: //Q2
229  if (!dynamicq) {
230  pid = (pidword >> 13) & 0x7f; // 7 bits at the top of all of pid (MSB)
231  } else {
232  pid = (pidword >> 12) & 0x3f; // 6 bits of Q2 and a shift
233  pid |= pidoffset << 6;
234  // LOG(info) << "Q2 pid : " << std::hex << pid << " pidoffset: " << pidoffset;
235  }
236  break;
237  case 1: //Q1
238  if (!dynamicq) {
239  pid = (pidword >> 6) & 0x7f; // 7 bits Q1 above the 6 bits of Q0
240  } else {
241  pid = (pidword >> 6) & 0x3f; // 6 bits of Q1 and a shift
242  pid |= pidoffset << 6;
243  //LOG(info) << "Q1 pid : " << std::hex << pid << " pidoffset: " << pidoffset;;
244  }
245  break;
246  case 0: //Q0
247  if (!dynamicq) {
248  pid = pidword & 0x3f; // 6 least significant bits
249  } else {
250  pid = pidword & 0x3f; // 6 bits of Q0
251  pid |= pidoffset << 6;
252  // LOG(info) << "Q0 pid : " << std::hex << pid << " pidoffset: " << pidoffset;
253  }
254  break;
255  default:
256  LOG(warn) << " unknown pid index of : " << pidindex;
257  break;
258  }
259  } // end of case of various formats.
260  return pid;
261 }

277 void constructTrackletMCMData(TrackletMCMData& trackletword, const int format, const uint slope, const uint pos, const uint q0, const uint q1, const uint q2)
278 {
279  trackletword.word = 0;
280  // create a tracklet word as it would be sent from the FEE
281  // slope and position have the 8-th bit flipped each
282  trackletword.word = 0;
283  trackletword.slope = slope ^ 0x80;
284  trackletword.pos = pos ^ 0x80;
285  trackletword.checkbit = 0;
286  if (format & 0x1) {
287  //length of q1 and q2 are 6 with a 2 bit offset.
288  //What happens when q2 and q1 dont share the same offset (upper 2 bits ?) ?
289  LOG(error) << "This tracklet format has not been tested yet";
290  trackletword.pid = (q0 & 0x3f) & ((q1 & 0x1) << 6);
291  //TODO check the 2 bit offset is still valid ... ??
292  } else {
293  trackletword.pid = (q0 & 0x3f) & ((q1 & 0x1) << 6);
294  }
295  //q2 sits with upper 2 bits of q1 in the header pid word, hence the 0x1f so 5 bits are used here.
296 }

677 int getNextMCMADCfromBP(uint32_t& bp, int channel)
678 {
679  //given a bitpattern (adcmask) find next channel with in the mask starting from the current channel.
680  //channels are read from right to left, lsb to msb. channel zero is position 0 in the bit pattern.
681  if (bp == 0) {
682  return 22;
683  }
684  int position = channel;
685  int m = 1 << channel;
686  while (!(bp & m)) {
687  m = m << 1;
688  position++;
689  if (position > 21) {
690  break;
691  }
692  }
693  bp &= ~(1UL << (position));
694  return position;
