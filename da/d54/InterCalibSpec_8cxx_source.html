74 void InterCalibSpec::finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj)
75 {
76  if (matcher == ConcreteDataMatcher("ZDC", "INTERCALIBCONFIG", 0)) {
77  // InterCalib configuration
78  auto* config = (const o2::zdc::InterCalibConfig*)obj;
79  if (mVerbosity > DbgZero) {
80  config->print();
81  }
82  mWorker.setInterCalibConfig(config);
83  }
84  if (matcher == ConcreteDataMatcher("ZDC", "ENERGYCALIB", 0)) {
85  // Current energy calibration
86  auto* config = (const o2::zdc::ZDCEnergyParam*)obj;
87  if (mVerbosity > DbgZero) {
88  config->print();
89  }
90  mWorker.setEnergyParam(config);
91  }
92  if (matcher == ConcreteDataMatcher("ZDC", "TOWERCALIB", 0)) {
93  // Tower intercalibration
94  auto* config = (const o2::zdc::ZDCTowerParam*)obj;
95  if (mVerbosity > DbgZero) {
96  config->print();
97  }
98  mWorker.setTowerParam(config);
99  }

102 void InterCalibSpec::run(ProcessingContext& pc)
103 {
104  if (!mInitialized) {
105  mInitialized = true;
106  updateTimeDependentParams(pc);
107  mTimer.Stop();
108  mTimer.Reset();
109  mTimer.Start(false);
110  }
111 
112  auto data = pc.inputs().get<InterCalibData>("intercalibdata");
113  mWorker.process(data);
114  for (int ih = 0; ih < (2 * InterCalibData::NH); ih++) {
115  o2::dataformats::FlatHisto1D<float> histoView(pc.inputs().get<gsl::span<float>>(fmt::format("inter_1dh{}", ih).data()));
116  mWorker.add(ih, histoView);
117  }
118  for (int ih = 0; ih < InterCalibData::NH; ih++) {
119  o2::dataformats::FlatHisto2D<float> histoView(pc.inputs().get<gsl::span<float>>(fmt::format("inter_2dh{}", ih).data()));
120  mWorker.add(ih, histoView);
121  }
122 }

152 framework::DataProcessorSpec getInterCalibSpec()
153 {
154  using device = o2::zdc::InterCalibSpec;
155  using clbUtils = o2::calibration::Utils;
156 
157  std::vector<InputSpec> inputs;
158  inputs.emplace_back("intercalibconfig", "ZDC", "INTERCALIBCONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathInterCalibConfig.data()));
159  inputs.emplace_back("energycalib", "ZDC", "ENERGYCALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathEnergyCalib.data()));
160  inputs.emplace_back("towercalib", "ZDC", "TOWERCALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathTowerCalib.data()));
161  inputs.emplace_back("intercalibdata", "ZDC", "INTERCALIBDATA", 0, Lifetime::Timeframe);
162 
163  char outputa[o2::header::gSizeDataDescriptionString];
164  char outputd[o2::header::gSizeDataDescriptionString];
165  for (int ih = 0; ih < (2 * InterCalibData::NH); ih++) {
166  snprintf(outputa, o2::header::gSizeDataDescriptionString, "inter_1dh%d", ih);
167  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_1DH%d", ih);
168  inputs.emplace_back(outputa, "ZDC", outputd, 0, Lifetime::Timeframe);
169  }
170  for (int ih = 0; ih < InterCalibData::NH; ih++) {
171  snprintf(outputa, o2::header::gSizeDataDescriptionString, "inter_2dh%d", ih);
172  snprintf(outputd, o2::header::gSizeDataDescriptionString, "INTER_2DH%d", ih);
173  inputs.emplace_back(outputa, "ZDC", outputd, 0, Lifetime::Timeframe);
174  }
175 
176  std::vector<OutputSpec> outputs;
177  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ZDC_Intercalib"}, Lifetime::Sporadic);
178  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ZDC_Intercalib"}, Lifetime::Sporadic);
179 
180  return DataProcessorSpec{
181  "zdc-calib-towers",
182  inputs,
183  outputs,
184  AlgorithmSpec{adaptFromTask<device>()},
185  Options{{"verbosity-level", o2::framework::VariantType::Int, 1, {"Verbosity level"}}}};
186 }
