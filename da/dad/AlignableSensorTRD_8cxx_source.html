77  // calculate point position X,Y,Z derivatives wrt calibration parameter calibID of given parent
78  // parent=0 means top detector object calibration
79  //
80  deriv[0] = deriv[1] = deriv[2] = 0;
81  //
82  if (!parent) { // TRD detector global calibration
83  //
84  switch (calibID) {
85  case AlignableDetectorTRD::kCalibNRCCorrDzDtgl: { // correction for Non-Crossing tracklets Z,Y shift: Z -> Z + calib*tgl, Y -> Y + calib*tgl*tilt*sign(tilt);
86  double sgYZ = pnt->getYZErrTracking()[1]; // makes sense only for nonRC tracklets
87  if (Abs(sgYZ) > 0.01) {
88  const double kTilt = 2. * TMath::DegToRad();

96  // error in VdriftT equivalent to shift in X at which Y measurement is evaluated

98  double snp = pnt->getTrParamWSA(AlignmentPoint::kParSnp), slpY = snp / Sqrt((1 - snp) * (1 + snp));
