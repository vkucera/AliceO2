34 auto AsyncQueueHelpers::run(AsyncQueue& queue) -> void
35 {
36  std::vector<int> order;
37  order.resize(queue.tasks.size());
38  std::iota(order.begin(), order.end(), 0);
39 
40  // Sort by priority and debounce
41  std::sort(order.begin(), order.end(), [&queue](int a, int b) {
42  if (queue.tasks[a].id.value == -1 || queue.tasks[b].id.value == -1) {
43  return false;
44  }
45  if (queue.tasks[a].id.value == queue.tasks[b].id.value) {
46  return queue.tasks[a].debounce > queue.tasks[b].debounce;
47  }
48  return queue.prototypes[queue.tasks[a].id.value].score > queue.prototypes[queue.tasks[b].id.value].score;
49  });
50  // Keep only the tasks with the highest debounce value for a given id
51  auto newEnd = std::unique(order.begin(), order.end(), [&queue](int a, int b) {
52  return queue.tasks[a].id.value == queue.tasks[b].id.value;
53  });
54  order.erase(newEnd, order.end());
55 
56  for (auto i : order) {
57  queue.tasks[i].task();
58  }
59  queue.tasks.clear();
60 }
