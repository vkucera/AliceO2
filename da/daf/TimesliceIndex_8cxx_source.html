38 TimesliceSlot TimesliceIndex::findOldestSlot(TimesliceId timestamp) const
39 {
40  size_t lane = timestamp.value % mMaxLanes;
41  TimesliceSlot oldest{lane};
42  auto oldPVal = std::get_if<uint64_t>(&mVariables[oldest.index].get(0));
43  if (oldPVal == nullptr) {
44  return oldest;
45  }
46  uint64_t oldTimestamp = *oldPVal;
47 
48  for (size_t i = lane + mMaxLanes; i < mVariables.size(); i += mMaxLanes) {
49  auto newPVal = std::get_if<uint64_t>(&mVariables[i].get(0));
50  if (newPVal == nullptr) {
51  return TimesliceSlot{i};
52  }
53  uint64_t newTimestamp = *newPVal;
54 
55  if (oldTimestamp > newTimestamp) {
56  oldest = TimesliceSlot{i};
57  oldTimestamp = newTimestamp;
58  }
59  }
60  return oldest;
61 }

63 std::tuple<TimesliceIndex::ActionTaken, TimesliceSlot> TimesliceIndex::replaceLRUWith(data_matcher::VariableContext& newContext, TimesliceId timestamp)
64 {
65  auto oldestSlot = findOldestSlot(timestamp);
66  if (TimesliceIndex::isValid(oldestSlot) == false) {
67  mVariables[oldestSlot.index] = newContext;
68  return std::make_tuple(ActionTaken::ReplaceUnused, oldestSlot);
69  }
70  auto oldTimestamp = std::get_if<uint64_t>(&mVariables[oldestSlot.index].get(0));
71  if (oldTimestamp == nullptr) {
72  mVariables[oldestSlot.index] = newContext;
73  return std::make_tuple(ActionTaken::ReplaceUnused, oldestSlot);
74  }
75 
76  auto newTimestamp = std::get_if<uint64_t>(&newContext.get(0));
77  if (newTimestamp == nullptr) {
78  return std::make_tuple(ActionTaken::DropInvalid, TimesliceSlot{TimesliceSlot::INVALID});
79  }
80 
81  if (*newTimestamp > *oldTimestamp) {
82  switch (mBackpressurePolicy) {
83  case BackpressureOp::DropAncient:
84  mVariables[oldestSlot.index] = newContext;
85  return std::make_tuple(ActionTaken::ReplaceObsolete, oldestSlot);
86  case BackpressureOp::DropRecent:
87  return std::make_tuple(ActionTaken::DropObsolete, TimesliceSlot{TimesliceSlot::INVALID});
88  case BackpressureOp::Wait:
89  return std::make_tuple(ActionTaken::Wait, TimesliceSlot{TimesliceSlot::INVALID});
90  }
91  } else {
92  switch (mBackpressurePolicy) {
93  case BackpressureOp::DropRecent:
94  mVariables[oldestSlot.index] = newContext;
95  return std::make_tuple(ActionTaken::ReplaceObsolete, oldestSlot);
96  case BackpressureOp::DropAncient:
97  return std::make_tuple(ActionTaken::DropObsolete, TimesliceSlot{TimesliceSlot::INVALID});
98  case BackpressureOp::Wait:
99  return std::make_tuple(ActionTaken::Wait, TimesliceSlot{TimesliceSlot::INVALID});
100  }
101  }
102  O2_BUILTIN_UNREACHABLE();
103 }

105 TimesliceIndex::OldestInputInfo TimesliceIndex::setOldestPossibleInput(TimesliceId timestamp, ChannelIndex channel)
106 {
107  // Each channel oldest possible input must be monotoically increasing.
108  assert(mChannels[channel.value].oldestForChannel.value <= timestamp.value);
109  mChannels[channel.value].oldestForChannel = timestamp;
110  OldestInputInfo result{timestamp, channel};
111  bool changed = false;
112  for (int ci = 0; ci < mChannels.size(); ci++) {
113  auto& a = mChannels[ci].oldestForChannel;
114  if (a.value < result.timeslice.value) {
115  changed = true;
116  result = {a, ChannelIndex{ci}};
117  }
118  }
119  mOldestPossibleInput = result;
120  if (changed) {
121  LOG(debug) << "Success: Oldest possible input is " << mOldestPossibleInput.timeslice.value << " due to channel " << mOldestPossibleInput.channel.value;
122  }
123  return mOldestPossibleInput;
124 }

126 bool TimesliceIndex::validateSlot(TimesliceSlot slot, TimesliceId currentOldest)
127 {
128  if (mDirty[slot.index] == true) {
129  return true;
130  }
131 
132  auto timestamp = std::get_if<uint64_t>(&mVariables[slot.index].get(0));
133  if (timestamp != nullptr && *timestamp < mOldestPossibleInput.timeslice.value) {
134  markAsInvalid(slot);
135  return false;
136  }
137  return true;
138 }

140 TimesliceIndex::OldestOutputInfo TimesliceIndex::updateOldestPossibleOutput()
141 {
142  auto oldestInput = getOldestPossibleInput();
143  OldestOutputInfo result{oldestInput.timeslice, oldestInput.channel};
144 
145  bool changed = false;
146  for (size_t i = 0; i < mVariables.size(); i++) {
147  // We do not check invalid slots.
148  if (isValid(TimesliceSlot{i}) == false) {
149  continue;
150  }
151  auto timestamp = std::get_if<uint64_t>(&mVariables[i].get(0));
152  if (timestamp != nullptr && *timestamp < result.timeslice.value) {
153  changed = true;
154  result.timeslice = TimesliceId{*timestamp};
155  result.slot = {i};
156  result.channel = {(int)-1};
157  }
158  }
159  mOldestPossibleOutput = result;
160  if (changed) {
161  LOGP(debug, "Oldest possible output {} due to {} {}",
162  mOldestPossibleOutput.timeslice.value,
163  result.channel.value == -1 ? "slot" : "channel",
164  result.channel.value == -1 ? mOldestPossibleOutput.slot.index : mOldestPossibleOutput.channel.value);
165  }
166  return result;
167 }
