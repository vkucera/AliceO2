29 void WaveformCalibConfig::restrictRange(int ib, int ie)
30 {
31  ibeg = -WaveformCalib_NBB;
32  iend = WaveformCalib_NBA;
33  if (ib >= ibeg && ib <= 0) {
34  ibeg = ib;
35  } else {
36  LOG(fatal) << "WaveformCalibConfig::restrictRange wrong setting for ibeg = " << ib;
37  }
38  if (ie <= iend && ie >= 0) {
39  iend = ie;
