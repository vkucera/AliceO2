1130 template <typename DataT>
1131 template <typename Fields>
1132 void SpaceCharge<DataT>::integrateEFieldsSimpsonIterative(const DataT p1r, const DataT p2r, const DataT p1phi, const DataT p2phi, const DataT p1z, const DataT p2z, DataT& localIntErOverEz, DataT& localIntEPhiOverEz, DataT& localIntDeltaEz, const Fields& formulaStruct) const
1133 {
1134  //==========simpsons rule see: https://en.wikipedia.org/wiki/Simpson%27s_rule =============================
1135  const Side side = formulaStruct.getSide();
1136  const DataT ezField = getEzField(side);
1137  const DataT p2phiSave = regulatePhi(p2phi, side);
1138 
1139  const DataT fielder0 = formulaStruct.evalEr(p1z, p1r, p1phi);
1140  const DataT fieldez0 = formulaStruct.evalEz(p1z, p1r, p1phi);
1141  const DataT fieldephi0 = formulaStruct.evalEphi(p1z, p1r, p1phi);
1142 
1143  const DataT fielder1 = formulaStruct.evalEr(p2z, p2r, p2phiSave);
1144  const DataT fieldez1 = formulaStruct.evalEz(p2z, p2r, p2phiSave);
1145  const DataT fieldephi1 = formulaStruct.evalEphi(p2z, p2r, p2phiSave);
1146 
1147  const DataT eZ0Inv = isCloseToZero(ezField, fieldez0) ? 0 : 1. / (ezField + fieldez0);
1148  const DataT eZ1Inv = isCloseToZero(ezField, fieldez1) ? 0 : 1. / (ezField + fieldez1);
1149 
1150  const DataT pHalfZ = 0.5 * (p1z + p2z); // dont needs to be regulated since p1z and p2z are already regulated
1151  const DataT pHalfPhiSave = regulatePhi(0.5 * (p1phi + p2phi), side); // needs to be regulated since p2phi is not regulated
1152  const DataT pHalfR = 0.5 * (p1r + p2r);

1155  const DataT eZHalfInv = (isCloseToZero(ezField, ezField2) | isCloseToZero(ezField, fieldez0) | isCloseToZero(ezField, fieldez1)) ? 0 : 1. / (ezField + ezField2);
1156  const DataT fieldSum2ErOverEz = formulaStruct.evalEr(pHalfZ, pHalfR, pHalfPhiSave);
1157  const DataT fieldSum2EphiOverEz = formulaStruct.evalEphi(pHalfZ, pHalfR, pHalfPhiSave);
1158 
1159  const DataT deltaXSimpsonSixth = (p2z - p1z) / 6;
1160  localIntErOverEz = deltaXSimpsonSixth * (4 * fieldSum2ErOverEz * eZHalfInv + fielder0 * eZ0Inv + fielder1 * eZ1Inv);
1161  localIntEPhiOverEz = deltaXSimpsonSixth * (4 * fieldSum2EphiOverEz * eZHalfInv + fieldephi0 * eZ0Inv + fieldephi1 * eZ1Inv);
1162  localIntDeltaEz = getSign(side) * deltaXSimpsonSixth * (4 * ezField2 + fieldez0 + fieldez1);
1163 }

1165 template <typename DataT>
1166 template <typename Fields>
1167 void SpaceCharge<DataT>::calcDistCorr(const DataT p1r, const DataT p1phi, const DataT p1z, const DataT p2z, DataT& ddR, DataT& ddPhi, DataT& ddZ, const Fields& formulaStruct, const bool localDistCorr) const
1168 {
1169  // see: https://edms.cern.ch/ui/file/1108138/1/ALICE-INT-2010-016.pdf
1170  // needed for calculation of distortions/corrections
1171  DataT localIntErOverEz = 0; // integral_p1z^p2z Er/Ez dz
1172  DataT localIntEPhiOverEz = 0; // integral_p1z^p2z Ephi/Ez dz
1173  DataT localIntDeltaEz = 0; // integral_p1z^p2z Ez dz
1174 
1175  // there are differentnumerical integration strategys implements. for details see each function.
1176  switch (sNumericalIntegrationStrategy) {
1177  case IntegrationStrategy::SimpsonIterative: // iterative simpson integration (should be more precise at least for the analytical E-Field case but takes alot more time than normal simpson integration)
1178  for (int i = 0; i < sSimpsonNIteratives; ++i) { // TODO define a convergence criterion to abort the algorithm earlier for speed up.
1179  const DataT tmpZ = localDistCorr ? (p2z + ddZ) : regulateZ(p2z + ddZ, formulaStruct.getSide()); // dont regulate for local distortions/corrections! (to get same result as using electric field at last/first bin)
1180  integrateEFieldsSimpsonIterative(p1r, p1r + ddR, p1phi, p1phi + ddPhi, p1z, tmpZ, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
1181  langevinCylindrical(ddR, ddPhi, ddZ, (p1r + 0.5 * ddR), localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz); // using the mean radius '(p1r + 0.5 * ddR)' for calculation of distortions/corections
1182  }
1183  break;
1184  case IntegrationStrategy::Simpson: // simpson integration
1185  integrateEFieldsSimpson(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
1186  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
1187  break;
1188  case IntegrationStrategy::Trapezoidal: // trapezoidal integration (fastest)
1189  integrateEFieldsTrapezoidal(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
1190  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
1191  break;
1192  case IntegrationStrategy::Root: // using integration implemented in ROOT (slow)
1193  integrateEFieldsRoot(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
1194  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
1195  break;
1196  default:
1197  integrateEFieldsSimpson(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
1198  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
1199  }
1200 }
