82 void BaselineCalibEPNSpec::run(ProcessingContext& pc)
83 {
84  if (!mInitialized) {
85  mInitialized = true;
86  updateTimeDependentParams(pc);
87  mTimer.Stop();
88  mTimer.Reset();
89  mTimer.Start(false);
90  }
91 
92  const auto ref = pc.inputs().getFirstValid(true);
93  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->creation; // approximate time in ms
94  BaselineCalibData& data = mWorker.getData();
95  data.setCreationTime(creationTime);
96 
97  auto peds = pc.inputs().get<gsl::span<o2::zdc::OrbitData>>("peds");
98 
99  // Process reconstructed data
100  mWorker.process(peds);
101 
102  // Send intermediate calibration data
103  auto& summary = mWorker.mData.getSummary();
104  o2::framework::Output outputData("ZDC", "BASECALIBDATA", 0, Lifetime::Timeframe);
105  pc.outputs().snapshot(outputData, summary);
106 }

108 void BaselineCalibEPNSpec::endOfStream(EndOfStreamContext& ec)
