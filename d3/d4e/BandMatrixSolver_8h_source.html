124 template <int BandWidthT>
125 template <int nRows>
126 inline void BandMatrixSolver<BandWidthT>::triangulateBlock(double AA[], double bb[])
127 {
128  {
129  int m = BandWidthT;
130  double* A = AA;
131  for (int rows = 0; rows < nRows; rows++) {
132  double c = 1. / A[0];
133  A[0] = c; // store 1/a[0][0]
134  double* rowi = A + BandWidthT - 1;
135  for (int i = 1; i < m; i++) { // row 0+i
136  double ai = c * A[i]; // A[0][i]
137  for (int j = i; j < m; j++) {
138  rowi[j] -= ai * A[j]; // A[i][j] -= A[0][j]/A[0][0]*A[i][0]
139  }
140  A[i] = ai; // A[0][i] /= A[0][0]
141  rowi += BandWidthT - 1;
142  }
143  m--;
144  A += BandWidthT;
145  }
146  }
147 
148  for (int k = 0; k < mBdim; k++) {
149  int m = BandWidthT;
150  double* A = AA;
151  double* b = bb;
152  for (int rows = 0; rows < nRows; rows++) {
153  double bk = b[k];
154  for (int i = 1; i < m; i++) {
155  b[mBdim * i + k] -= A[i] * bk;
156  }
157  b[k] *= A[0];
158  m--;
159  A += BandWidthT;
160  b += mBdim;
161  }
162  }
163 }

165 template <int BandWidthT>
166 template <int nCols>
167 inline void BandMatrixSolver<BandWidthT>::dioganalizeBlock(double AA[], double bb[])
168 {
169  for (int k = 0; k < mBdim; k++) {
170  int rows = BandWidthT;
171  double* A = AA;
172  double* b = bb;
173  for (int col = 0; col < nCols; col++) {
174  double bk = b[k];
175  for (int i = 1; i < rows; i++) {
176  b[-i * mBdim + k] -= A[BandWidthT * (-i) + i] * bk;
177  }
178  A -= BandWidthT;
179  b -= mBdim;
180  rows--;
181  }
182  }

185 template <int BandWidthT>
186 inline void BandMatrixSolver<BandWidthT>::solve()
187 {
189 
190  const int stepA = BandWidthT;
191  const int stepB = mBdim;
192  // Upper Triangulization
193  {
194  int k = 0;
195  double* Ak = &mA[0];
196  double* bk = &mB[0];
197  for (; k < mN - BandWidthT; k += 1, Ak += stepA, bk += stepB) { // for each row k
198  triangulateBlock<1>(Ak, bk);
199  }
200  // last m rows
201  triangulateBlock<BandWidthT>(Ak, bk);
202  }
203 
204  // Diagonalization
205  {
206  int k = mN - 1;
207  double* Ak = &mA[BandWidthT * k];
208  double* bk = &mB[mBdim * k];
209  for (; k > BandWidthT - 1; k -= 1, Ak -= stepA, bk -= stepB) { // for each row k
210  dioganalizeBlock<1>(Ak, bk);
211  }
212  // first m rows
213  dioganalizeBlock<BandWidthT>(Ak, bk);
214  }
215 }
