182 template <unsigned int Dim, unsigned int Degree, bool InteractionOnly>
183 void MultivariatePolynomial<Dim, Degree, InteractionOnly>::setFromContainer(const MultivariatePolynomialContainer& container)
184 {
185  if constexpr (Dim > 0 && Degree > 0) {
186  if (this->getDim() != container.mDim) {
187 #ifndef GPUCA_ALIROOT_LIB
188  LOGP(info, fmt::format("wrong number of dimensions! this {} container {}", this->getDim(), container.mDim));
189 #endif
190  return;
191  }
192  if (this->getDegree() != container.mDegree) {
193 #ifndef GPUCA_ALIROOT_LIB
194  LOGP(info, fmt::format("wrong number of degrees! this {} container {}", this->getDegree(), container.mDegree));
195 #endif
196  return;
197  }
198  if (this->isInteractionOnly() != container.mInteractionOnly) {
199 #ifndef GPUCA_ALIROOT_LIB
200  LOGP(info, fmt::format("InteractionOnly is set for this object to {}, but stored as {} in the container", this->isInteractionOnly(), container.mInteractionOnly));
201 #endif
202  return;
203  }
204  setParams(container.mParams.data());
205  } else {
206  MultivariatePolynomial polTmp(container.mDim, container.mDegree);
207  polTmp.setParams(container.mParams.data());
208  this->cloneFromObject(polTmp, nullptr);
209  }
210 }

219 template <unsigned int Dim, unsigned int Degree, bool InteractionOnly>
220 void MultivariatePolynomial<Dim, Degree, InteractionOnly>::cloneFromObject(const MultivariatePolynomial<Dim, Degree, InteractionOnly>& obj, char* newFlatBufferPtr)
221 {
222  const char* oldFlatBufferPtr = obj.mFlatBufferPtr;
223  FlatObject::cloneFromObject(obj, newFlatBufferPtr);
224  mNParams = obj.mNParams;
225  if constexpr (Dim == 0 && Degree == 0) {
226  this->mDim = obj.mDim;
227  this->mDegree = obj.mDegree;
228  this->mInteractionOnly = obj.mInteractionOnly;
229  }
230  if (obj.mParams) {
231  mParams = FlatObject::relocatePointer(oldFlatBufferPtr, mFlatBufferPtr, obj.mParams);
232  }
233 }
