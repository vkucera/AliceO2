66  void printPayload(gsl::span<const uint8_t> payload, bool isBare)
67  {
68  std::stringstream ss;
69  size_t wordLength = isBare ? 16 : 32;
70  ss << "\n";
71  bool show = false;
72  for (size_t iword = 0; iword < payload.size(); iword += wordLength) {
73  auto word = payload.subspan(iword, wordLength);
74  if (isBare) {
75  for (auto it = word.rbegin(); it != word.rend(); ++it) {
76  auto ibInWord = word.rend() - it;
77  if (ibInWord == 4 || ibInWord == 9) {
78  ss << " ";
79  }
80  if (ibInWord == 5 || ibInWord == 10) {
81  ss << " ";
82  }
83  ss << fmt::format("{:02x}", static_cast<int>(*it));
84  }
85  } else {
86  for (auto it = word.begin(); it != word.end(); ++it) {
87  if (iword == 0 && std::distance(word.begin(), it) == 1 && static_cast<int>(*it) != 1) {
88  show = true;
89  }
90  ss << fmt::format("{:02x}", static_cast<int>(*it));
91  }
92  }
93  ss << "\n";
94  }
95  // show = o2::raw::RDHUtils::getTriggerType(rdhPtr) & 0xA00;
96  if (show) {
97  LOG(info) << ss.str();
98  }
99  }

101  void run(of::ProcessingContext& pc)
102  {
103  if (isDroppedTF(pc, header::gDataOriginMID)) {
104  std::vector<ROBoard> data;
105  std::vector<ROFRecord> rofs;
106  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODED", mSubSpec}, data);
107  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODEDROF", mSubSpec}, rofs);
108  return;
109  }
110 
111  std::vector<of::InputSpec> filter{of::InputSpec{"filter", of::ConcreteDataTypeMatcher{header::gDataOriginMID, header::gDataDescriptionRawData}, of::Lifetime::Timeframe}};
112 
113  of::DPLRawParser parser(pc.inputs(), filter);
114 
115  if (!mDecoder) {
116  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(parser.begin().raw());
117  mDecoder = createDecoder(*rdhPtr, mIsDebugMode, mElectronicsDelay, mCrateMasks, mFeeIdConfig);
118  }
119 
120  mDecoder->clear();
121  size_t firstRof = 0;
122  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
123  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(it.raw());
124  gsl::span<const uint8_t> payload(it.data(), it.size());
125  bool isBare = o2::mid::raw::isBare(*rdhPtr);
126  o2::raw::RDHUtils::printRDH(rdhPtr);
127  if (mDumpPayload) {
128  printPayload(payload, isBare);
129  }
130 
131  if (mDumpDecoded) {
132  mDecoder->process(payload, *rdhPtr);
133  std::stringstream ss;
134  for (auto rofIt = mDecoder->getROFRecords().begin() + firstRof, end = mDecoder->getROFRecords().end(); rofIt != end; ++rofIt) {
135  ss << fmt::format("BCid: 0x{:x} Orbit: 0x{:x} EvtType: {:d}", rofIt->interactionRecord.bc, rofIt->interactionRecord.orbit, static_cast<int>(rofIt->eventType)) << std::endl;
136  for (auto colIt = mDecoder->getData().begin() + rofIt->firstEntry, end = mDecoder->getData().begin() + rofIt->getEndIndex(); colIt != end; ++colIt) {
137  ss << *colIt << std::endl;
138  }
139  }
140  LOG(info) << ss.str();
141  firstRof = mDecoder->getROFRecords().size();
142  }
143  }
144  }
