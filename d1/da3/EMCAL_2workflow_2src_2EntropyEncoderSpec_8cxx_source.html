47 void EntropyEncoderSpec::run(ProcessingContext& pc)
48 {
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51  mCTFCoder.updateTimeDependentParams(pc);
52  auto triggers = pc.inputs().get<gsl::span<TriggerRecord>>("triggers");
53  auto cells = pc.inputs().get<gsl::span<Cell>>("cells");
54 
55  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"EMC", "CTFDATA", 0, Lifetime::Timeframe});
56  auto iosize = mCTFCoder.encode(buffer, triggers, cells);
57  pc.outputs().snapshot({"ctfrep", 0}, iosize);
58  mTimer.Stop();
59  LOG(info) << iosize.asString() << " in " << mTimer.CpuTime() - cput << " s";
60 }

74 
75  return DataProcessorSpec{

77  inputs,
