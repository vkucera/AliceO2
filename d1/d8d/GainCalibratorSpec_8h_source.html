54  LOG(info) << "tf-per-slot = 0 (this calibrator works only in single infinite slot mode)";
55  LOG(info) << "max-delay = 1000 (inconfigurable for this calibrator)";

66  void run(o2::framework::ProcessingContext& pc) final
67  {
68  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
69  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mCalibrator->getCurrentTFInfo());
70  TFType tfcounter = mCalibrator->getCurrentTFInfo().startTime;
71 
72  // update config
73  static bool isConfigFetched = false;
74  if (!isConfigFetched) {
75  LOG(info) << "GainCalibratorSpec::run() : fetching o2::cpv::CPVCalibParams from CCDB";
76  pc.inputs().get<o2::cpv::CPVCalibParams*>("calibparams");
77  LOG(info) << "GainCalibratorSpec::run() : o2::cpv::CPVCalibParams::Instance() now is following:";
78  o2::cpv::CPVCalibParams::Instance().printKeyValues();
79  mCalibrator->configParameters();
80  isConfigFetched = true;
81  }
82 
83  // previous gains
84  if (!mCalibrator->isSettedPreviousGains()) {
85  const auto previousGains = pc.inputs().get<o2::cpv::CalibParams*>("gains");
86  if (previousGains) {
87  mCalibrator->setPreviousGains(new o2::cpv::CalibParams(*previousGains));
88  LOG(info) << "GainCalibratorSpec()::run() : I got previous gains";
89  }
90  }
91 
92  // previous calib data
93  if (!mCalibrator->isSettedPreviousGainCalibData()) {
94  // const auto previousGainCalibData = o2::framework::DataRefUtils::as<CCDBSerialized<o2::cpv::GainCalibData>>(pc.inputs().get("calibdata"));
95  const auto previousGainCalibData = pc.inputs().get<o2::cpv::GainCalibData*>("calibdata");
96  if (previousGainCalibData) {
97  mCalibrator->setPreviousGainCalibData(new o2::cpv::GainCalibData(*previousGainCalibData));
98  LOG(info) << "GainCalibratorSpec()::run() : I got previous GainCalibData: ";
99  previousGainCalibData->print();
100  }
101  }
102 
103  // process calib input
104  auto&& digits = pc.inputs().get<gsl::span<o2::cpv::Digit>>("calibdigs");
105 
106  // fill statistics
107  LOG(info) << "Processing TF " << tfcounter << " with " << digits.size() << " digits";
108  mCalibrator->process(digits); // fill TimeSlot with digits from 1 event and check slots for finalization
109 
110  // inform about results and send output to ccdb
111  auto gainsInfoVecSize = mCalibrator->getCcdbInfoGainsVector().size();
112  if (gainsInfoVecSize) {
113  LOG(info) << "Created " << gainsInfoVecSize << " o2::cpv::CalibParams objects.";
114  sendOutput(pc.outputs());
115  }
116 
117  // and send results to CCDB (if any)
118  if (gainsInfoVecSize) {
119  sendOutput(pc.outputs());
120  }
121  }
122  void endOfStream(o2::framework::EndOfStreamContext& ec) final
123  {
124  // prepare GainCalibData outputs and send it to ccdb
125  mCalibrator->prepareForEnding();
126  auto gainCDInfoVecSize = mCalibrator->getCcdbInfoGainCalibDataVector().size();
127  if (gainCDInfoVecSize) {
128  LOG(info) << "Created " << gainCDInfoVecSize << " o2::cpv::GainCalibData objects.";
129  sendOutput(ec.outputs());
130  }
131  }

137  void sendOutput(DataAllocator& output)
138  {
139  bool isSomethingSent = false;
140  isSomethingSent = sendOutputWhat<o2::cpv::CalibParams>(mCalibrator->getGainsVector(), mCalibrator->getCcdbInfoGainsVector(), "CPV_Gains", output);
141  isSomethingSent += sendOutputWhat<o2::cpv::GainCalibData>(mCalibrator->getGainCalibDataVector(), mCalibrator->getCcdbInfoGainCalibDataVector(), "CPV_GainCD", output);
142 
143  if (isSomethingSent) {
144  mCalibrator->initOutput(); // reset the outputs once they are already sent
145  }
146  }
147 
148  template <class Payload>
149  bool sendOutputWhat(const std::vector<Payload>& payloadVec, std::vector<o2::ccdb::CcdbObjectInfo>& infoVec, header::DataDescription what, DataAllocator& output)
150  {
151  assert(payloadVec.size() == infoVec.size());
152  if (!payloadVec.size()) {
153  return false;
154  }
155 
156  for (uint32_t i = 0; i < payloadVec.size(); i++) {
157  auto& w = infoVec[i];
158  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
159  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
160  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
161  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, what, i}, *image.get()); // vector<char>
162  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, what, i}, w); // root-serialized
163  }
164 
165  return true;
166  }

173 DataProcessorSpec getCPVGainCalibratorSpec()
174 {
175  using device = o2::calibration::CPVGainCalibratorSpec;
176 
177  std::vector<InputSpec> inputs;
178  inputs.emplace_back("calibdigs", "CPV", "CALIBDIGITS", 0, Lifetime::Timeframe);
179  inputs.emplace_back("calibdata", "CPV", "CPV_GainCD", 0, Lifetime::Condition, ccdbParamSpec("CPV/PhysicsRun/GainCalibData"));
180  inputs.emplace_back("gains", "CPV", "CPV_Gains", 0, Lifetime::Condition, ccdbParamSpec("CPV/Calib/Gains"));
181  inputs.emplace_back("calibparams", "CPV", "CPV_CalibPars", 0, Lifetime::Condition, ccdbParamSpec("CPV/Config/CPVCalibParams"));
182 
183  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
184  true, // GRPECS=true
185  false, // GRPLHCIF
186  false, // GRPMagField
187  false, // askMatLUT
188  o2::base::GRPGeomRequest::None, // geometry
189  inputs);
190  std::vector<OutputSpec> outputs;
191  // Length of data description ("CPV_Pedestals") must be < 16 characters.
192  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "CPV_Gains"}, Lifetime::Sporadic);
193  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "CPV_Gains"}, Lifetime::Sporadic);
194  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "CPV_GainCD"}, Lifetime::Sporadic);
195  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "CPV_GainCD"}, Lifetime::Sporadic);
196 
197  return DataProcessorSpec{
198  "cpv-gain-calibration",
199  inputs,
200  outputs,
201  AlgorithmSpec{adaptFromTask<device>(ccdbRequest)},
202  Options{
203  {"updateAtTheEndOfRunOnly", VariantType::Bool, false, {"finalize the slots and prepare the CCDB entries only at the end of the run."}},
204  {"updateInterval", VariantType::UInt32, 100u, {"try to finalize the slot (and produce calibration) when the updateInterval has passed."}}}};
205 }
