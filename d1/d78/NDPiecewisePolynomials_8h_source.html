433 template <unsigned int Dim, unsigned int Degree, bool InteractionOnly>
434 template <unsigned int DimTmp>
435 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::clamp(float x[/* Dim */], int index[/* Dim */]) const
436 {
437  if (index[DimTmp] <= 0) {
438  index[DimTmp] = 0;
439 
440  if (x[DimTmp] < mMin[DimTmp]) {
441  x[DimTmp] = mMin[DimTmp];
442  }
443 
444  } else {
445  if (index[DimTmp] >= int(mN[DimTmp] - 1)) {
446  index[DimTmp] = mN[DimTmp] - 2;
447  x[DimTmp] = mMax[DimTmp];
448  }
449  }
450 
451  if constexpr (DimTmp > 0) {
452  return clamp<DimTmp - 1>(x, index);
453  }
454 }

496 template <unsigned int Dim, unsigned int Degree, bool InteractionOnly>
497 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::performFits(const std::function<double(const double x[/* Dim */])>& func, const unsigned int nAuxiliaryPoints[/* Dim */])
498 {
499  const int nTotalFits = getNPolynomials();
500 #ifndef GPUCA_ALIROOT_LIB
501  LOGP(info, "Perform fitting of {}D-Polynomials of degree {} for a total of {} fits.", Dim, Degree, nTotalFits);
502 #endif
503 
504  MultivariatePolynomialHelper<0, 0, false> pol(Dim, Degree, InteractionOnly);
505  TLinearFitter fitter = pol.getTLinearFitter();
506 
507  unsigned int nPoints = 1;
508  for (unsigned int i = 0; i < Dim; ++i) {
509  nPoints *= nAuxiliaryPoints[i];
510  }
511 
512  std::vector<double> xCords;
513  std::vector<double> response;
514  xCords.reserve(Dim * nPoints);
515  response.reserve(nPoints);
516 
517  unsigned int nPolynomials[Dim]{0};
518  for (unsigned int i = 0; i < Dim; ++i) {
519  nPolynomials[i] = getNPolynomials(i);
520  }
521 
522  int pos[Dim + 1]{0};
523  unsigned int counter = 0;
524  const int printDebugForNFits = int(nTotalFits / 20) + 1;
525 
526  for (;;) {
527  const bool debug = !(++counter % printDebugForNFits);
528  if (debug) {
529 #ifndef GPUCA_ALIROOT_LIB
530  LOGP(info, "Peforming fit {} out of {}", counter, nTotalFits);
531 #endif
532  }
533 
534  checkPos(nPolynomials, pos);
535 
536  if (pos[Dim] == 1) {
537  break;
538  }
539 
540  xCords.clear();
541  response.clear();
542  fitInnerGrid(func, nAuxiliaryPoints, pos, fitter, xCords, response);
543  ++pos[0];
544  }
545 }

547 template <unsigned int Dim, unsigned int Degree, bool InteractionOnly>
548 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::fitInnerGrid(const std::function<double(const double x[/* Dim */])>& func, const unsigned int nAuxiliaryPoints[/* Dim */], const int currentIndex[/* Dim */], TLinearFitter& fitter, std::vector<double>& xCords, std::vector<double>& response)
549 {
550  int pos[Dim + 1]{0};
551 
552  // add points which will be used for the fit
553  for (;;) {
554  checkPos(nAuxiliaryPoints, pos);
555 
556  if (pos[Dim] == 1) {
557  break;
558  }
559 
560  for (unsigned int iDim = 0; iDim < Dim; ++iDim) {
561  const double stepWidth = getStepWidth(iDim, nAuxiliaryPoints[iDim]);
562  const double vertexPos = getVertexPosition(currentIndex[iDim], iDim);
563  const double realPosTmp = vertexPos + pos[iDim] * stepWidth;
564  xCords.emplace_back(realPosTmp);
565  }
566 
567  // get response for last added points
568  const double responseTmp = func(&xCords[xCords.size() - Dim]);
569  response.emplace_back(responseTmp);
570  ++pos[0];
571  }
572 
573  // perform the fit on the points TODO make errors configurable
574  std::vector<double> error;
575  const auto params = MultivariatePolynomialHelper<0, 0, false>::fit(fitter, xCords, response, error, true);
576 
577  // store parameters
578  const unsigned int index = getDataIndex(currentIndex);
579  std::copy(params.begin(), params.end(), &mParams[index]);
580 }
