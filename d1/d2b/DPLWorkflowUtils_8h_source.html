37 // Finding out if the current process is the master DPL driver process,
38 // first setting up the topology. Might be important to know when we write
39 // files (to prevent that multiple processes write the same file)
40 bool isMasterWorkflowDefinition(ConfigContext const& configcontext)
41 {
42  int argc = configcontext.argc();
43  auto argv = configcontext.argv();
44  bool ismaster = true;
45  for (int argi = 0; argi < argc; ++argi) {
46  // when channel-config is present it means that this is started as
47  // as FairMQDevice which means it is already a forked process
48  if (strcmp(argv[argi], "--channel-config") == 0) {
49  ismaster = false;
50  break;
51  }
52  }
53  return ismaster;
54 }

102 // a utility combining multiple specs into one
103 // (with some checking that it makes sense)
104 // spits out the combined spec (merged inputs/outputs and AlgorithmSpec)
105 // Can put policies later whether to multi-thread or serialize internally etc.
106 // fills a "remaining" spec container for things it can't simply merge
107 // (later on we could do a full topological sort / spec minimization approach)
108 o2::framework::DataProcessorSpec specCombiner(std::string const& name, std::vector<DataProcessorSpec> const& speccollection,
109  std::vector<DataProcessorSpec>& remaining)
110 {
111  std::vector<OutputSpec> combinedOutputSpec;
112  std::vector<InputSpec> combinedInputSpec;
113  std::vector<ConfigParamSpec> combinedOptions;
114 
115  // this maps an option key to one of multiple specs where this is used
116  // we will need to introduce unambiguous options in case an option is present multiple times
117  std::unordered_map<std::string, std::vector<std::pair<std::string, ConfigParamSpec>>> optionMerger;
118 
119  // maps a process name to a map holder conversion of "namespaced-keys" to original "keys"
120  std::unordered_map<std::string, std::unordered_map<std::string, std::string>> optionsRemap;
121 
122  // keep track of which input bindings are already used
123  // (should not have duplicates ... since devices may fetch data just based on the binding)
124  std::unordered_map<std::string, bool> inputBindings;
125 
126  // we collect all outputs once ---> this is to check that none of the inputs matches
127  // an output
128  std::vector<OutputSpec> allOutputs;
129  std::vector<DataProcessorSpec> mergableSpecs;
130  for (auto& spec : speccollection) {
131  // merge output specs
132  for (auto& os : spec.outputs) {
133  allOutputs.push_back(os);
134  }
135  }
136 
137  for (auto& spec : speccollection) {
138  auto& procname = spec.name;
139  optionsRemap[procname] = std::unordered_map<std::string, std::string>();
140 
141  // merge input specs ... but only after we have verified that the spec does
142  // not depend in internal outputs
143  bool inputCheckOk = true;
144  for (auto& is : spec.inputs) {
145  // let's see if input is part of outputs
146  // ... in which case we can't easily merge the spec here
147  // ... and just neglect it for the moment
148  for (auto& o : allOutputs) {
149  if (DataSpecUtils::match(is, o)) {
150  std::cerr << "Found internal connection " << is << " ... will take out spec " << procname << " .. from merging process \n";
151  inputCheckOk = false;
152  break;
153  }
154  }
155  }
156  if (!inputCheckOk) {
157  remaining.push_back(spec);
158  // directly to next task
159  continue;
160  }
161  for (auto& is : spec.inputs) {
162  if (inputBindings.find(is.binding) != inputBindings.end()) {
163  LOG(error) << "Found duplicate input binding " << is.binding;
164  }
165  combinedInputSpec.push_back(is);
166  inputBindings[is.binding] = 1;
167  }
168  mergableSpecs.push_back(spec);
169 
170  // merge output specs
171  for (auto& os : spec.outputs) {
172  combinedOutputSpec.push_back(os);
173  }
174  // merge options (part 1)
175  for (auto& opt : spec.options) {
176  auto optkey = opt.name;
177  auto iter = optionMerger.find(optkey);
178  auto procconfigpair = std::pair<std::string, ConfigParamSpec>(procname, opt);
179  if (iter == optionMerger.end()) {
180  optionMerger[optkey] = std::vector<std::pair<std::string, ConfigParamSpec>>();
181  }
182  optionMerger[optkey].push_back(procconfigpair);
183  }
184  }
185  // merge options (part 2)
186  for (auto& iter : optionMerger) {
187  if (iter.second.size() > 1) {
188  // std::cerr << "Option " << iter.first << " duplicated in multiple procs --> applying namespacing \n";
189  for (auto& part : iter.second) {
190  auto procname = part.first;
191  auto originalSpec = part.second;
192  auto namespaced_name = procname + "." + originalSpec.name;
193  combinedOptions.push_back(ConfigParamSpec{namespaced_name,
194  originalSpec.type, originalSpec.defaultValue, originalSpec.help, originalSpec.kind});
195  optionsRemap[procname][namespaced_name] = originalSpec.name;
196  // we need to back-apply
197  }
198  } else {
199  // we can stay with original option
200  for (auto& part : iter.second) {
201  combinedOptions.push_back(part.second);
202  }
203  }
204  }

206  // logic for combined task processing function --> target is to run one only
207  class CombinedTask
208  {
209  public:
210  CombinedTask(std::vector<DataProcessorSpec> const& s, std::unordered_map<std::string, std::unordered_map<std::string, std::string>> optionsRemap) : tasks(s), optionsRemap(optionsRemap) {}
211 
212  void init(o2::framework::InitContext& ic)
213  {
214  std::cerr << "Init Combined\n";
215  for (auto& t : tasks) {
216  // the init function actually creates the onProcess function
217  // which we have to do here (maybe some more stuff needed)
218  auto& configRegistry = ic.mOptions;
219  // we can get hold of the store because the store is the only data member of configRegistry
220  static_assert(sizeof(o2::framework::ConfigParamRegistry) == sizeof(std::unique_ptr<o2::framework::ConfigParamStore>));
221  auto store = reinterpret_cast<std::unique_ptr<ConfigParamStore>*>(&(configRegistry));
222  auto& boost_store = (*store)->store(); // std::unique_ptr<boost::property_tree::ptree>
223  auto& originalDeviceName = t.name;
224  auto optionsiter = optionsRemap.find(originalDeviceName);
225  if (optionsiter != optionsRemap.end()) {
226  // we have options to remap
227  for (auto& key : optionsiter->second) {
228  //
229  // LOG(info) << "Applying value " << boost_store.get<std::string>(key.first) << " to original key " << key.second;
230  boost_store.put(key.second, boost_store.get<std::string>(key.first));
231  }
232  }
233  t.algorithm.onProcess = t.algorithm.onInit(ic);
234  }
235  }
236 
237  void run(o2::framework::ProcessingContext& pc)
238  {
239  std::cerr << "Processing Combined\n";
240  for (auto& t : tasks) {
241  std::cerr << " Executing sub-device " << t.name << "\n";
242  t.algorithm.onProcess(pc);
243  }
244  }
245 
246  private:
247  std::vector<DataProcessorSpec> tasks;
248  std::unordered_map<std::string, std::unordered_map<std::string, std::string>> optionsRemap;
249  };
250 
251  return DataProcessorSpec{
252  name,
253  combinedInputSpec,
254  combinedOutputSpec,
255  AlgorithmSpec{adaptFromTask<CombinedTask>(mergableSpecs, optionsRemap)},
256  combinedOptions
257  /* a couple of other fields can be set ... */
258  };
259 };
