51  void init(o2::framework::InitContext& ic) final
52  {
53  // setting up the histogram ranges
54  const auto nBins = ic.options().get<int>("nBins");
55  const auto reldEdxMin = ic.options().get<float>("reldEdxMin");
56  const auto reldEdxMax = ic.options().get<float>("reldEdxMax");
57  const auto underflowBin = ic.options().get<bool>("underflowBin");
58  const auto overflowBin = ic.options().get<bool>("overflowBin");
59  mPadGainTracks.init(nBins, reldEdxMin, reldEdxMax, underflowBin, overflowBin);
60 
61  const auto propagateTrack = ic.options().get<bool>("propagateTrack");
62  mPadGainTracks.setPropagateTrack(propagateTrack);
63 
64  const auto dedxRegionType = ic.options().get<int>("dedxRegionType");
65  mPadGainTracks.setdEdxRegion(static_cast<CalibPadGainTracks::DEdxRegion>(dedxRegionType));
66 
67  const auto dedxType = ic.options().get<int>("dedxType");
68  mPadGainTracks.setMode(static_cast<CalibPadGainTracks::DEdxType>(dedxType));
69 
70  const auto chargeType = ic.options().get<int>("chargeType");
71  assert(chargeType == 0 || chargeType == 1);
72  mPadGainTracks.setChargeType(static_cast<ChargeType>(chargeType));
73 
74  mUseEveryNthTF = ic.options().get<int>("useEveryNthTF");
75  if (mUseEveryNthTF <= 0) {
76  mUseEveryNthTF = 1;
77  }
78 
79  mMaxTracksPerTF = ic.options().get<int>("maxTracksPerTF");
80 
81  const std::string gainMapFile = ic.options().get<std::string>("gainMapFile");
82  if (!gainMapFile.empty()) {
83  LOGP(info, "Loading GainMap from file {}", gainMapFile);
84  mPadGainTracks.setRefGainMap(gainMapFile.data(), "GainMap");
85  }
86 
87  float field = ic.options().get<float>("field");
88  if (field <= -10.f) {
89  const auto inputGRP = o2::base::NameConf::getGRPFileName();
90  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
91  if (grp != nullptr) {
92  field = 5.00668f * grp->getL3Current() / 30000.f;
93  LOGP(info, "Using GRP file to set the magnetic field to {} kG", field);
94  }
95  }
96 
97  LOGP(info, "Setting magnetic field to {} kG", field);
98  mPadGainTracks.setField(field);
99 
100  const auto etaMax = ic.options().get<float>("etaMax");
101  mPadGainTracks.setMaxEta(etaMax);
102 
103  const auto minClusters = ic.options().get<int>("minClusters");
104  mPadGainTracks.setMinNClusters(minClusters);
105 
106  const auto momMin = ic.options().get<float>("momMin");
107  const auto momMax = ic.options().get<float>("momMax");
108  LOGP(info, "Using particle tracks with {} GeV/c < p < {} GeV/c ", momMin, momMax);
109  mPadGainTracks.setMomentumRange(momMin, momMax);
110  }

112  void finaliseCCDB(ConcreteDataMatcher& matcher, void* obj) final
113  {
114  LOGP(info, "finaliseCCDB");
115  if (matcher == ConcreteDataMatcher(gDataOriginTPC, "RESIDUALGAINMAP", 0)) {
116  if (!mUsingDefaultGainMapForFirstIter) {
117  LOGP(info, "Updating reference gain map from previous iteration from CCDB");
118  const auto* gainMapResidual = static_cast<std::unordered_map<string, o2::tpc::CalDet<float>>*>(obj);
119  mPadGainTracks.setRefGainMap(gainMapResidual->at("GainMap"));
120  } else {
121  // just skip for the first time asking for an object -> not gain map will be used as reference
122  LOGP(info, "Skipping loading reference gain map for first iteration from CCDB");
123  mUsingDefaultGainMapForFirstIter = false;
124  }
125  } else if (matcher == ConcreteDataMatcher(gDataOriginTPC, "TOPOLOGYGAIN", 0)) {
126  LOGP(info, "Updating Q topology correction from CCDB");
127  const auto* topologyCorr = static_cast<o2::tpc::CalibdEdxTrackTopologyPolContainer*>(obj);
128  mPadGainTracks.setPolTopologyCorrectionFromContainer(*topologyCorr);
129  }
130  }

132  void run(o2::framework::ProcessingContext& pc) final
133  {
134  const auto currentTF = processing_helpers::getCurrentTF(pc);
135  if (mTFCounter++ % mUseEveryNthTF) {
136  LOGP(info, "Skipping TF {}", currentTF);
137  return;
138  }
139 
140  auto tracks = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
141  auto clRefs = pc.inputs().get<gsl::span<o2::tpc::TPCClRefElem>>("trackTPCClRefs");
142  const auto& clusters = getWorkflowTPCInput(pc);
143  const auto nTracks = tracks.size();
144  if (nTracks == 0) {
145  return;
146  }
147  LOGP(info, "Processing TF {} with {} tracks by considering {} tracks", currentTF, nTracks, mMaxTracksPerTF);
148 
149  if (!mDisablePolynomialsCCDB) {
150  pc.inputs().get<o2::tpc::CalibdEdxTrackTopologyPolContainer*>("tpctopologygain");
151  }
152 
153  if (mUseLastExtractedMapAsReference) {
154  LOGP(info, "fetching residual gain map");
155  pc.inputs().get<std::unordered_map<std::string, o2::tpc::CalDet<float>>*>("tpcresidualgainmap");
156  }
157 
158  mPadGainTracks.setMembers(&tracks, &clRefs, clusters->clusterIndex);
159  mPadGainTracks.processTracks(mMaxTracksPerTF);
160  ++mProcessedTFs;
161  if ((mPublishAfter && (mProcessedTFs % mPublishAfter) == 0)) {
162  LOGP(info, "Publishing after {} TFs", mProcessedTFs);
163  mProcessedTFs = 0;
164  if (mDebug) {
165  mPadGainTracks.dumpToFile(fmt::format("calPadGain_TF{}.root", currentTF).data());
166  }
167  sendOutput(pc.outputs());
168  }
169  }
170 
171  private:
172  const uint32_t mPublishAfter{0};
173  const bool mDebug{false};
174  const bool mUseLastExtractedMapAsReference{false};
175  bool mDisablePolynomialsCCDB{false};
176  uint32_t mProcessedTFs{0};
177  uint32_t mTFCounter{0};
178  CalibPadGainTracks mPadGainTracks{false};
179  bool mUsingDefaultGainMapForFirstIter{true};
180  unsigned int mUseEveryNthTF{1};
181  int mMaxTracksPerTF{-1};

183  void sendOutput(DataAllocator& output)
184  {
185  output.snapshot(Output{"TPC", "TRACKGAINHISTOS", 0}, *mPadGainTracks.getHistos().get());
186  mPadGainTracks.resetHistos();
187  }

190 DataProcessorSpec getTPCCalibPadGainTracksSpec(const uint32_t publishAfterTFs, const bool debug, const bool useLastExtractedMapAsReference, const std::string polynomialsFile, bool disablePolynomialsCCDB)
191 {
192  std::vector<InputSpec> inputs;
193  inputs.emplace_back("trackTPC", gDataOriginTPC, "TRACKS", 0, Lifetime::Timeframe);
194  inputs.emplace_back("trackTPCClRefs", gDataOriginTPC, "CLUSREFS", 0, Lifetime::Timeframe);
195  inputs.emplace_back("clusTPC", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERNATIVE"}, Lifetime::Timeframe);
196 
197  if (!polynomialsFile.empty()) {
198  disablePolynomialsCCDB = true;
199  }
200 
201  if (!disablePolynomialsCCDB) {
202  inputs.emplace_back("tpctopologygain", gDataOriginTPC, "TOPOLOGYGAIN", 0, Lifetime::Condition, ccdbParamSpec(CDBTypeMap.at(CDBType::CalTopologyGain)));
203  }
204 
205  if (useLastExtractedMapAsReference) {
206  inputs.emplace_back("tpcresidualgainmap", gDataOriginTPC, "RESIDUALGAINMAP", 0, Lifetime::Condition, ccdbParamSpec(CDBTypeMap.at(CDBType::CalPadGainResidual)));
207  }
208 
209  std::vector<OutputSpec> outputs;
210  outputs.emplace_back(gDataOriginTPC, "TRACKGAINHISTOS", 0, o2::framework::Lifetime::Sporadic);
211 
212  return DataProcessorSpec{
213  "calib-tpc-gainmap-tracks",
214  inputs,
215  outputs,
216  AlgorithmSpec{adaptFromTask<TPCCalibPadGainTracksDevice>(publishAfterTFs, debug, useLastExtractedMapAsReference, polynomialsFile, disablePolynomialsCCDB)},
217  Options{
218  {"nBins", VariantType::Int, 20, {"Number of bins per histogram"}},
219  {"reldEdxMin", VariantType::Int, 0, {"Minimum x coordinate of the histogram for Q/(dE/dx)"}},
220  {"reldEdxMax", VariantType::Int, 3, {"Maximum x coordinate of the histogram for Q/(dE/dx)"}},
221  {"underflowBin", VariantType::Bool, false, {"Using under flow bin"}},
222  {"overflowBin", VariantType::Bool, true, {"Using under flow bin"}},
223  {"field", VariantType::Float, -100.f, {"Magnetic field in kG, need for track propagations, this value will be overwritten if a grp file is present"}},
224  {"momMin", VariantType::Float, 0.3f, {"minimum momentum of the tracks which are used for the pad-by-pad gain map"}},
225  {"momMax", VariantType::Float, 1.f, {"maximum momentum of the tracks which are used for the pad-by-pad gain map"}},
226  {"etaMax", VariantType::Float, 1.f, {"maximum eta of the tracks which are used for the pad-by-pad gain map"}},
227  {"minClusters", VariantType::Int, 50, {"minimum number of clusters of tracks which are used for the pad-by-pad gain map"}},
228  {"gainMapFile", VariantType::String, "", {"file to reference gain map, which will be used for correcting the cluster charge"}},
229  {"dedxRegionType", VariantType::Int, 2, {"using the dE/dx per chamber (0), stack (1) or per sector (2)"}},
230  {"dedxType", VariantType::Int, 0, {"recalculating the dE/dx (0), using it from tracking (1)"}},
231  {"chargeType", VariantType::Int, 0, {"Using qMax (0) or qTot (1) for the dE/dx and the pad-by-pad histograms"}},
232  {"propagateTrack", VariantType::Bool, false, {"Propagating the track instead of performing a refit for obtaining track parameters."}},
233  {"useEveryNthTF", VariantType::Int, 10, {"Using only a fraction of the data: 1: Use every TF, 10: Use only every tenth TF."}},
234  {"maxTracksPerTF", VariantType::Int, 10000, {"Maximum number of processed tracks per TF (-1 for processing all tracks)"}},
235  }}; // end DataProcessorSpec
236 }
