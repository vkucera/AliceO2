66  // this is the virtual interface for context objects
67  class ContextObject
68  {
69  public:
70  ContextObject() = delete;
71  ContextObject(fair::mq::MessagePtr&& headerMsg, fair::mq::MessagePtr&& payloadMsg, RouteIndex routeIndex)
72  : mParts{}, mRouteIndex{routeIndex}
73  {
74  mParts.AddPart(std::move(headerMsg));
75  mParts.AddPart(std::move(payloadMsg));
76  }
77 
78  ContextObject(fair::mq::MessagePtr&& headerMsg, RouteIndex routeIndex)
79  : mParts{}, mRouteIndex{routeIndex}
80  {
81  mParts.AddPart(std::move(headerMsg));
82  }
83 
84  virtual ~ContextObject() = default;
85 
89  virtual fair::mq::Parts finalize()
90  {
91  fair::mq::Parts parts = std::move(mParts);
92  assert(parts.Size() == 2);
93  auto* header = o2::header::get<o2::header::DataHeader*>(parts.At(0)->GetData());
94  if (header == nullptr) {
95  throw std::logic_error("No valid header message found");
96  } else {
97  // o2::header::get returns const pointer, but here we can change the message
98  const_cast<o2::header::DataHeader*>(header)->payloadSize = parts.At(1)->GetSize();
99  }
100  // return value optimization returns by move
101  return parts;
102  }
103 
105  RouteIndex route() const
106  {
107  return mRouteIndex;
108  }
109 
110  bool empty() const
111  {
112  return mParts.Size() == 0;
113  }
114 
115  o2::header::DataHeader const* header()
116  {
117  // we would expect this function to be const but the fair::mq::Parts API does not allow this
118  if (empty() || mParts.At(0) == nullptr) {
119  return nullptr;
120  }
121  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData());
122  }
123 
124  o2::framework::DataProcessingHeader const* dataProcessingHeader()
125  {
126  if (empty() || mParts.At(0) == nullptr) {
127  return nullptr;
128  }
129  return o2::header::get<o2::framework::DataProcessingHeader*>(mParts.At(0)->GetData());
130  }
131 
132  o2::header::Stack const* headerStack()
133  {
134  // we would expect this function to be const but the fair::mq::Parts API does not allow this
135  if (empty() || mParts.At(0) == nullptr) {
136  return nullptr;
137  }
138  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData()) ? reinterpret_cast<o2::header::Stack*>(mParts.At(0)->GetData()) : nullptr;
139  }

250  : ContextObject(std::forward<fair::mq::MessagePtr>(headerMsg), routeIndex),
251  // the transport factory
252  mFactory{context->proxy().getOutputTransport(routeIndex)},
253  // the memory resource takes ownership of the message
254  mResource{mFactory ? AlignedMemoryResource(mFactory->GetMemoryResource()) : AlignedMemoryResource(nullptr)},
255  // create the vector with apropriate underlying memory resource for the message
256  mData{std::forward<Args>(args)..., pmr::polymorphic_allocator<value_type>(&mResource)}
257  {
258  // FIXME: drop this repeated check and make sure at initial setup of devices that everything is fine
259  // introduce error policy
260  if (mFactory == nullptr) {
261  throw runtime_error_f("failed to get transport factory for route %d", routeIndex);
262  }
263  if (mResource.isValid() == false) {
264  throw runtime_error_f("no memory resource for channel %d", routeIndex);
265  }
266  }

486  void schedule(Messages::value_type&& message)
487  {
488  auto const* header = message->header();
489  if (header == nullptr) {
490  throw std::logic_error("No valid header message found");
491  }
492  mScheduledMessages.emplace_back(std::move(message));
493  if (mDispatchControl.dispatch != nullptr) {
494  // send all scheduled messages if there is no trigger callback or its result is true
495  if (mDispatchControl.trigger == nullptr || mDispatchControl.trigger(*header)) {
496  std::vector<fair::mq::Parts> outputsPerChannel;
497  outputsPerChannel.resize(mProxy.getNumOutputChannels());
498  for (auto& message : mScheduledMessages) {
499  fair::mq::Parts parts = message->finalize();
500  assert(message->empty());
501  assert(parts.Size() == 2);
502  for (auto& part : parts) {
503  outputsPerChannel[mProxy.getOutputChannelIndex(message->route()).value].AddPart(std::move(part));
504  }
505  }
506  for (int ci = 0; ci < mProxy.getNumOutputChannels(); ++ci) {
507  auto& parts = outputsPerChannel[ci];
508  if (parts.Size() == 0) {
509  continue;
510  }
511  mDispatchControl.dispatch(std::move(parts), ChannelIndex{ci}, DefaultChannelIndex);
512  }
513  mDidDispatch = mScheduledMessages.empty() == false;
514  mScheduledMessages.clear();
515  }
516  }
517  }
518 
519  Messages getMessagesForSending()
520  {

522  for (auto& message : mScheduledMessages) {
523  mMessages.emplace_back(std::move(message));
