52  void run(o2::framework::ProcessingContext& pc)
53  {
54 
55  auto inRofs = specs::getRofs(pc, "mid_timing_in");
56  std::array<std::vector<ROFRecord>, NEvTypes> outRofs;
57 
58  for (size_t idx = 0; idx < NEvTypes; ++idx) {
59  if (idx == static_cast<size_t>(EventType::Calib)) {
60  // Delays do not apply to triggered events
61  outRofs[idx].insert(outRofs[idx].end(), inRofs[idx].begin(), inRofs[idx].end());
62  } else {
63  for (auto rof : inRofs[idx]) {
64  applyElectronicsDelay(rof.interactionRecord.orbit, rof.interactionRecord.bc, mLocalToBC, mMaxBunches);
65  outRofs[idx].emplace_back(rof);
66  }
67  }
68  pc.outputs().snapshot(mOutputs[idx], outRofs[idx]);
69  }
70  }
71 
72  private:
73  int16_t mLocalToBC = 0;
74  uint16_t mMaxBunches = constants::lhc::LHCMaxBunches;
75  std::vector<of::Output> mOutputs;
76 };
