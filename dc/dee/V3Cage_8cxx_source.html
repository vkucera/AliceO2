75 void V3Cage::createAndPlaceCage(TGeoVolume* mother, const TGeoManager* mgr)
76 {
77  //
78  // Creates the Cage elements and place them into the mother volume
79  //
80  // Input:
81  // mother : the mother volume hosting the whole Cage
82  // mgr : the GeoManager (used only to get the proper material)
83  //
84  // Output:
85  //
86  // Return:
87  //
88  // Created: 22 Apr 2022 Mario Sitta
89  //
90 
91  // Local variables
92  Double_t zunit, zpos;
93 
94  // Create the cover elements
95  TGeoVolume* cageCover = createCageCover(mgr);
96  TGeoVolume* cageCoverRib = createCageCoverRib(mgr);
97 
98  // Now place all elements
99  mother->AddNode(cageCover, 1, new TGeoTranslation(0, sCageYInBarrel, 0));
100  mother->AddNode(cageCover, 2, new TGeoCombiTrans(0, sCageYInBarrel, 0, new TGeoRotation("", 180, 0, 0)));
101 
102  zunit = (sCageCoverZLength + sCageCoverRibZLength) / 2.;
103 
104  zpos = zunit;
105  mother->AddNode(cageCoverRib, 1, new TGeoTranslation(0, sCageYInBarrel, zpos));
106  mother->AddNode(cageCoverRib, 2, new TGeoCombiTrans(0, sCageYInBarrel, zpos, new TGeoRotation("", 180, 0, 0)));
107  mother->AddNode(cageCoverRib, 3, new TGeoTranslation(0, sCageYInBarrel, -zpos));
108  mother->AddNode(cageCoverRib, 4, new TGeoCombiTrans(0, sCageYInBarrel, -zpos, new TGeoRotation("", 180, 0, 0)));
109 
110  zpos = 2 * zunit;
111  mother->AddNode(cageCover, 3, new TGeoTranslation(0, sCageYInBarrel, zpos));
112  mother->AddNode(cageCover, 4, new TGeoCombiTrans(0, sCageYInBarrel, zpos, new TGeoRotation("", 180, 0, 0)));
113  mother->AddNode(cageCover, 5, new TGeoTranslation(0, sCageYInBarrel, -zpos));
114  mother->AddNode(cageCover, 6, new TGeoCombiTrans(0, sCageYInBarrel, -zpos, new TGeoRotation("", 180, 0, 0)));
115 
116  // Third ribs are only on A side
117  zpos = 3 * zunit;
118  mother->AddNode(cageCoverRib, 5, new TGeoTranslation(0, sCageYInBarrel, zpos));
119  mother->AddNode(cageCoverRib, 6, new TGeoCombiTrans(0, sCageYInBarrel, zpos, new TGeoRotation("", 180, 0, 0)));
120 
121  return;
122 }

124 TGeoVolume* V3Cage::createCageCover(const TGeoManager* mgr)
125 {
126  //
127  // Creates a Cage cover element (from drawings ALIITSUP0226, ALIITSUP0225)
128  //
129  // Input:
130  // mgr : the GeoManager (used only to get the proper material)
131  //
132  // Output:
133  //
134  // Return:
135  // The cover as a TGeoVolume
136  //
137  // Created: 22 Apr 2022 Mario Sitta
138  //
139 
140  // Local variables
141  static const Int_t coverFoldNVert = 6;
142 
143  Double_t xvert[coverFoldNVert], yvert[coverFoldNVert];
144  Double_t halfBase, zlen, alpha;
145 
146  zlen = 0.5 * sCageCoverZLength;
147 
148  // The cover core: a TGeoTubeSeg
149  halfBase = 0.5 * sCageCoverXBaseInt - sCageCoverSheetThick;
150  alpha = TMath::ACos(halfBase / sCageCoverCoreRext) * TMath::RadToDeg();
151  TGeoTubeSeg* coreSh = new TGeoTubeSeg("coverCore", sCageCoverCoreRint, sCageCoverCoreRext, zlen, alpha, 180. - alpha);
152 
153  // The upper and lower sheets: a TGeoTubeSeg
154  // (as a whole volume - will become a sheet when the core is inserted inside)
155  halfBase = 0.5 * sCageCoverXBaseInt;
156  alpha = TMath::ACos(halfBase / sCageCoverRint) * TMath::RadToDeg();
157  TGeoTubeSeg* sheet = new TGeoTubeSeg("coverSheet", sCageCoverRint, sCageCoverCoreRext, zlen, alpha, 180. - alpha);
158 
159  // The lateral fold: a Xtru
160  xvert[0] = sCageCoverXBaseExt / 2.;
161  yvert[0] = sCageCoverYBaseHeight + sCageCoverSheetThick;
162  xvert[1] = sCageCoverXWidth / 2.;
163  yvert[1] = yvert[0];
164  xvert[2] = xvert[1];
165  yvert[2] = sCageCoverYBaseHeight;
166  xvert[3] = sCageCoverXBaseInt / 2.;
167  yvert[3] = yvert[2];
168  xvert[4] = xvert[3];
169  yvert[4] = TMath::Sqrt(sCageCoverRint * sCageCoverRint - sCageCoverXBaseInt * sCageCoverXBaseInt / 4) + sCageCoverSheetThick;
170  xvert[5] = xvert[0];
171  yvert[5] = yvert[4];
172 
173  TGeoXtru* fold = new TGeoXtru(2);
174  fold->SetName("coverFold");
175  fold->DefinePolygon(coverFoldNVert, xvert, yvert);
176  fold->DefineSection(0, -zlen);
177  fold->DefineSection(1, zlen);
178 
179  // A BBox to cut away the curved portion above the fold
180  TGeoBBox* cutfold = new TGeoBBox("cutFold", xvert[1] - xvert[0], yvert[5] - yvert[1], 1.05 * zlen);
181 
182  // Some matrices to create the composite shape
183  TGeoRotation* rotfold = new TGeoRotation("rotFold", 180, 180, 0);
184  rotfold->RegisterYourself();
185 
186  TGeoTranslation* cutbox1 = new TGeoTranslation("cutBox1", xvert[1], yvert[5], 0);
187  cutbox1->RegisterYourself();
188 
189  TGeoTranslation* cutbox2 = new TGeoTranslation("cutBox2", -xvert[1], yvert[5], 0);
190  cutbox2->RegisterYourself();
191 
192  // The cover shape: a CompositeShape
193  TGeoCompositeShape* coverSh = new TGeoCompositeShape("coverSheet-cutFold:cutBox1-cutFold:cutBox2+coverFold+coverFold:rotFold");
194 
195  // We have all shapes: now create the real volumes
196  TGeoMedium* medRohacell = mgr->GetMedium("ITS_ROHACELL$");
197  TGeoMedium* medPrepreg = mgr->GetMedium("ITS_M46J6K$");
198 
199  TGeoVolume* coverVol = new TGeoVolume("CageCover", coverSh, medPrepreg);
200  coverVol->SetFillColor(kBlue);
201  coverVol->SetLineColor(kBlue);
202 
203  TGeoVolume* coverCoreVol = new TGeoVolume("CageCoverCore", coreSh, medRohacell);
204  coverCoreVol->SetFillColor(kYellow);
205  coverCoreVol->SetLineColor(kYellow);
206 
207  coverVol->AddNode(coverCoreVol, 1, nullptr);
208 
209  // Finally return the cover volume
210  return coverVol;
211 }

213 TGeoVolume* V3Cage::createCageCoverRib(const TGeoManager* mgr)
214 {
215  //
216  // Creates a Cage cover rib element (from drawing ALIITSUP0228)
217  //
218  // Input:
219  // mgr : the GeoManager (used only to get the proper material)
220  //
221  // Output:
222  //
223  // Return:
224  // The cover rib as a TGeoVolume
225  //
226  // Created: 22 Apr 2022 Mario Sitta
227  //
228 
229  // Local variables
230  static const Int_t ribFoldNVert = 3;
231 
232  Double_t xvert[ribFoldNVert], yvert[ribFoldNVert];
233  Double_t halfBase, zlen, alpha;
234 
235  zlen = 0.5 * sCageCoverRibZLength;
236 
237  // The rib main segment: a TGeoTubeSeg
238  halfBase = 0.5 * sCageCoverRibXBaseInt;
239  alpha = TMath::ACos(halfBase / sCageCoverRibRint) * TMath::RadToDeg();
240  TGeoTubeSeg* mainSh = new TGeoTubeSeg("coverRibMain", sCageCoverRibRint, sCageCoverRibRext, zlen, alpha, 180. - alpha);
241 
242  // The lateral fold: a Xtru
243  xvert[0] = sCageCoverRibXWidth / 2.;
244  yvert[0] = sCageCoverRibYBaseHi;
245  xvert[1] = sCageCoverRibXBaseInt / 2.;
246  yvert[1] = yvert[0];
247  xvert[2] = xvert[1];
248  yvert[2] = yvert[1] + sCageCoverRibFoldHi;
249 
250  TGeoXtru* fold = new TGeoXtru(2);
251  fold->SetName("coverRibFold");
252  fold->DefinePolygon(ribFoldNVert, xvert, yvert);
253  fold->DefineSection(0, -zlen);
254  fold->DefineSection(1, zlen);
255 
256  // Some matrices to create the composite shape
257  TGeoRotation* rotfold = new TGeoRotation("rotRibFold", 180, 180, 0);
258  rotfold->RegisterYourself();
259 
260  // The cover rib shape: a CompositeShape
261  TGeoCompositeShape* ribSh = new TGeoCompositeShape("coverRibMain+coverRibFold+coverRibFold:rotRibFold");
262 
263  // We have all shapes: now create the real volume
264  TGeoMedium* medAl = mgr->GetMedium("ITS_ALUMINUM$");
265 
266  TGeoVolume* ribVol = new TGeoVolume("CageCoverRib", ribSh, medAl);
267  ribVol->SetFillColor(kGray);
268  ribVol->SetLineColor(kGray);
269 
270  // Finally return the cover rib volume
271  return ribVol;
272 }
