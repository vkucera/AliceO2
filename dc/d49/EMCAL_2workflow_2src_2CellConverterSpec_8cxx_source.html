50 void CellConverterSpec::run(framework::ProcessingContext& ctx)
51 {
52  LOG(debug) << "[EMCALCellConverter - run] called";
53  const double CONVADCGEV = 0.016; // Conversion from ADC counts to energy: E = 16 MeV / ADC
54  constexpr double timeshift = 600.; // subtract 600 ns in order to center the time peak around the nominal delay
55 
56  mOutputCells.clear();
57  mOutputLabels.clear();
58  mOutputTriggers.clear();
59  auto digitsAll = ctx.inputs().get<gsl::span<o2::emcal::Digit>>("digits");
60  auto triggers = ctx.inputs().get<gsl::span<o2::emcal::TriggerRecord>>("triggers");
61  auto truthcont = ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::emcal::MCLabel>*>("digitsmctr");
62 
63  LOG(debug) << "[EMCALCellConverter - run] Received " << digitsAll.size() << " digits from " << triggers.size() << " trigger ...";
64  int currentstart = mOutputCells.size(), ncellsTrigger = 0;
65  for (const auto& trg : triggers) {
66  if (!trg.getNumberOfObjects()) {
67  mOutputTriggers.emplace_back(trg.getBCData(), trg.getTriggerBits(), currentstart, ncellsTrigger);
68  continue;
69  }
70 
71  gsl::span<const o2::emcal::Digit> digits(digitsAll.data() + trg.getFirstEntry(), trg.getNumberOfObjects());
72  std::vector<gsl::span<const o2::emcal::MCLabel>> mcLabels;
73 
74  if (mPropagateMC) {
75  for (int digitIndex = trg.getFirstEntry(); digitIndex < (trg.getFirstEntry() + trg.getNumberOfObjects()); digitIndex++) {
76  mcLabels.push_back(truthcont->getLabels(digitIndex));
77  }
78  }
79 
80  for (const auto& srucont : digitsToBunches(digits, mcLabels)) {
81 
82  if (srucont.mSRUid == 21 || srucont.mSRUid == 22 || srucont.mSRUid == 36 || srucont.mSRUid == 39) {
83  continue;
84  }
85 
86  for (const auto& [tower, channelData] : srucont.mChannelsData) {
87 
88  // define the conatiner for the fit results, and perform the raw fitting using the stadnard raw fitter
89  ChannelType_t channelType = channelData.mChanType;
90  CaloFitResults fitResults;
91  try {
92  fitResults = mRawFitter->evaluate(channelData.mChannelsBunchesHG);
93 
94  // If the high gain bunch is saturated then fit the low gain
95  if (fitResults.getAmp() > o2::emcal::constants::OVERFLOWCUT) {
96  fitResults = mRawFitter->evaluate(channelData.mChannelsBunchesLG);
97  fitResults.setAmp(fitResults.getAmp() * o2::emcal::constants::EMCAL_HGLGFACTOR);
98  channelType = ChannelType_t::LOW_GAIN;
99  } else {
100  channelType = ChannelType_t::HIGH_GAIN;
101  }
102 
103  if (fitResults.getAmp() < 0) {
104  fitResults.setAmp(0.);
105  }
106  if (fitResults.getTime() < 0) {
107  fitResults.setTime(0.);
108  }
109  mOutputCells.emplace_back(tower, fitResults.getAmp() * CONVADCGEV, fitResults.getTime() - timeshift, channelType);
110 
111  if (mPropagateMC) {
112  Int_t LabelIndex = mOutputLabels.getIndexedSize();
113  if (channelType == ChannelType_t::HIGH_GAIN) {
114  // if this channel has no bunches, then fill an empty label
115  if (channelData.mChannelLabelsHG.size() == 0) {
116  const o2::emcal::MCLabel label = o2::emcal::MCLabel(false, 1.);
117  mOutputLabels.addElementRandomAccess(LabelIndex, label);
118  } else {
119  // Fill only labels that corresponds to bunches with maximum ADC
120  const int bunchindex = selectMaximumBunch(channelData.mChannelsBunchesHG);
121  for (const auto& label : channelData.mChannelLabelsHG[bunchindex]) {
122  mOutputLabels.addElementRandomAccess(LabelIndex, label);
123  }
124  }
125  } else {
126  // if this channel has no bunches, then fill an empty label
127  if (channelData.mChannelLabelsLG.size() == 0) {
128  const o2::emcal::MCLabel label = o2::emcal::MCLabel(false, 1.);
129  mOutputLabels.addElementRandomAccess(LabelIndex, label);
130  } else {
131  // Fill only labels that corresponds to bunches with maximum ADC
132  const int bunchindex = selectMaximumBunch(channelData.mChannelsBunchesLG);
133  for (const auto& label : channelData.mChannelLabelsLG[bunchindex]) {
134  mOutputLabels.addElementRandomAccess(LabelIndex, label);
135  }
136  }
137  }
138  }
139  ncellsTrigger++;
140 
141  } catch (CaloRawFitter::RawFitterError_t& fiterror) {
142  if (fiterror != CaloRawFitter::RawFitterError_t::BUNCH_NOT_OK) {
143  LOG(error) << "Failure in raw fitting: " << CaloRawFitter::createErrorMessage(fiterror);
144  }
145  }
146  }
147  }
148  mOutputTriggers.emplace_back(trg.getBCData(), trg.getTriggerBits(), currentstart, ncellsTrigger);
149  currentstart = mOutputCells.size();
150  ncellsTrigger = 0;
151  }
152  LOG(debug) << "[EMCALCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
153  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
154  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggers);
155  if (mPropagateMC) {
156  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe}, mOutputLabels);
157  }
158 }

160 std::vector<o2::emcal::SRUBunchContainer> CellConverterSpec::digitsToBunches(gsl::span<const o2::emcal::Digit> digits, std::vector<gsl::span<const o2::emcal::MCLabel>>& mcLabels)
161 {
162 
163  std::vector<o2::emcal::SRUBunchContainer> sruBunchContainer;
164  std::vector<o2::emcal::DigitContainerPerSRU> sruDigitContainer;
165 
166  for (auto iddl = 0; iddl < 40; iddl++) {
167  o2::emcal::SRUBunchContainer srucontBunch;
168  srucontBunch.mSRUid = iddl;
169  sruBunchContainer.push_back(srucontBunch);
170 
171  o2::emcal::DigitContainerPerSRU srucontDigits;
172  srucontDigits.mSRUid = iddl;
173  sruDigitContainer.push_back(srucontDigits);
174  }
175 
176  std::vector<const o2::emcal::Digit*>* bunchDigits;
177  std::vector<gsl::span<const o2::emcal::MCLabel>>* bunchLabels;
178  int lasttower = -1;
179  // for (auto& dig : digits) {
180  for (const auto& [dig, labels] : boost::combine(digits, mcLabels)) {
181  auto tower = dig.getTower();
182  if (tower != lasttower) {
183  lasttower = tower;
184  if (tower > 20000) {
185  std::cout << "Wrong cell ID " << tower << std::endl;
186  }
187 
188  auto onlineindices = mGeometry->getOnlineID(tower);
189  int sruID = std::get<0>(onlineindices);
190 
191  auto towerdata = sruDigitContainer[sruID].mChannelsDigits.find(tower);
192  if (towerdata == sruDigitContainer[sruID].mChannelsDigits.end()) {
193  sruDigitContainer[sruID].mChannelsDigits[tower] = {dig.getType(), std::vector<const o2::emcal::Digit*>(o2::emcal::constants::EMCAL_MAXTIMEBINS), mPropagateMC ? std::vector<gsl::span<const o2::emcal::MCLabel>>(o2::emcal::constants::EMCAL_MAXTIMEBINS) : std::vector<gsl::span<const o2::emcal::MCLabel>>()};
194  bunchDigits = &(sruDigitContainer[sruID].mChannelsDigits[tower].mChannelDigits);
195  memset(bunchDigits->data(), 0, sizeof(o2::emcal::Digit*) * o2::emcal::constants::EMCAL_MAXTIMEBINS);
196  if (mPropagateMC) {
197  bunchLabels = &(sruDigitContainer[sruID].mChannelsDigits[tower].mChannelLabels);
198  memset(bunchLabels->data(), 0, sizeof(gsl::span<const o2::emcal::MCLabel>) * o2::emcal::constants::EMCAL_MAXTIMEBINS);
199  }
200  } else {
201  bunchDigits = &(towerdata->second.mChannelDigits);
202  if (mPropagateMC) {
203  bunchLabels = &(towerdata->second.mChannelLabels);
204  }
205  }
206  }
207 
208  // Get time sample of the digit:
209  // Digitizer stores the time sample in ns, needs to be converted to time sample dividing
210  // by the length of the time sample
211  auto timesample = int(dig.getTimeStamp() / emcal::constants::EMCAL_TIMESAMPLE);
212  if (timesample >= o2::emcal::constants::EMCAL_MAXTIMEBINS) {
213  LOG(error) << "Digit time sample " << timesample << " outside range [0," << o2::emcal::constants::EMCAL_MAXTIMEBINS << "]";
214  continue;
215  }
216  (*bunchDigits)[timesample] = &dig;
217  if (mPropagateMC) {
218  (*bunchLabels)[timesample] = labels;
219  }
220  }
221 
222  for (auto srucont : sruDigitContainer) {
223 
224  if (srucont.mSRUid == 21 || srucont.mSRUid == 22 || srucont.mSRUid == 36 || srucont.mSRUid == 39) {
225  continue;
226  }
227 
228  for (const auto& [tower, channelDigits] : srucont.mChannelsDigits) {
229 
230  std::vector<o2::emcal::Bunch> rawbunchesHG;
231  std::vector<o2::emcal::Bunch> rawbunchesLG;
232  std::vector<std::vector<o2::emcal::MCLabel>> rawLabelsHG;
233  std::vector<std::vector<o2::emcal::MCLabel>> rawLabelsLG;
234 
235  bool saturatedBunch = false;
236 
237  // Creating the high gain bunch with labels
238  for (auto& bunch : findBunches(channelDigits.mChannelDigits, channelDigits.mChannelLabels, ChannelType_t::HIGH_GAIN)) {
239  rawbunchesHG.emplace_back(bunch.mADCs.size(), bunch.mStarttime);
240  for (auto adc : bunch.mADCs) {
241  rawbunchesHG.back().addADC(adc);
242  if (adc > o2::emcal::constants::LG_SUPPRESSION_CUT) {
243  saturatedBunch = true;
244  }
245  }
246  if (mPropagateMC) {
247  rawLabelsHG.push_back(bunch.mLabels);
248  }
249  }
250 
251  // Creating the low gain bunch with labels if the HG bunch is saturated
252  if (saturatedBunch) {
253  for (auto& bunch : findBunches(channelDigits.mChannelDigits, channelDigits.mChannelLabels, ChannelType_t::LOW_GAIN)) {
254  rawbunchesLG.emplace_back(bunch.mADCs.size(), bunch.mStarttime);
255  for (auto adc : bunch.mADCs) {
256  rawbunchesLG.back().addADC(adc);
257  }
258  if (mPropagateMC) {
259  rawLabelsLG.push_back(bunch.mLabels);
260  }
261  }
262  }
263 
264  sruBunchContainer[srucont.mSRUid].mChannelsData[tower] = {channelDigits.mChanType, rawbunchesHG, rawbunchesLG, rawLabelsHG, rawLabelsLG};
265  }
266  }
267 
268  return sruBunchContainer;
269 }

271 std::vector<o2::emcal::AltroBunch> CellConverterSpec::findBunches(const std::vector<const o2::emcal::Digit*>& channelDigits, const std::vector<gsl::span<const o2::emcal::MCLabel>>& mcLabels, ChannelType_t channelType)
272 {
273  std::vector<AltroBunch> result;
274  AltroBunch currentBunch;
275  bool bunchStarted = false;
276  // Digits in ALTRO bunch in time-reversed order
277  int itime;
278  for (itime = channelDigits.size() - 1; itime >= 0; itime--) {
279  auto dig = channelDigits[itime];
280  auto labels = mcLabels[itime];
281  if (!dig) {
282  if (bunchStarted) {
283  // we have a bunch which is started and needs to be closed
284  // check if the ALTRO bunch has a minimum amount of 3 ADCs
285  if (currentBunch.mADCs.size() >= 3) {
286  // Bunch selected, set start time and push to bunches
287  result.push_back(currentBunch);
288  currentBunch = AltroBunch();
289  bunchStarted = false;
290  }
291  }
292  continue;
293  }
294  int adc = dig->getAmplitudeADC(channelType);
295  if (adc < 3) {
296  // ADC value below threshold
297  // in case we have an open bunch it needs to be stopped bunch
298  // Set the start time to the time sample of previous (passing) digit
299  if (bunchStarted) {
300  // check if the ALTRO bunch has a minimum amount of ADCs
301  if (currentBunch.mADCs.size() >= 3) {
302  // Bunch selected, set start time and push to bunches
303  result.push_back(currentBunch);
304  currentBunch = AltroBunch();
305  bunchStarted = false;
306  }
307  }
308  }
309  // Valid ADC value, if the bunch is closed we start a new bunch
310  if (!bunchStarted) {
311  bunchStarted = true;
312  currentBunch.mStarttime = itime;
313  }
314  currentBunch.mADCs.emplace_back(adc);
315  if (mPropagateMC) {
316  std::vector<o2::emcal::MCLabel> vectorLabels;
317  for (auto& label : labels) {
318  vectorLabels.push_back(label);
319  }
320  currentBunch.mEnergyLabels.insert(std::pair<int, std::vector<o2::emcal::MCLabel>>(adc, vectorLabels));
321  }
322  }
323  // if we have a last bunch set time start time to the time bin of teh previous digit
324  if (bunchStarted) {
325  if (currentBunch.mADCs.size() >= 3) {
326  result.push_back(currentBunch);
327  }
328  }
329 
330  if (mPropagateMC) {
331  mergeLabels(result);
332  }
333  return result;
334 }

336 void CellConverterSpec::mergeLabels(std::vector<o2::emcal::AltroBunch>& channelBunches)
337 {
338  for (auto& altroBunch : channelBunches) {
339  std::vector<o2::emcal::MCLabel> mcLabels;
340  std::map<uint64_t, std::vector<o2::emcal::MCLabel>> LabelsPerSource;
341  double totalEnergy = 0;
342  for (auto& [energy, Labels] : altroBunch.mEnergyLabels) {
343  for (auto& trueLabel : Labels) {
344  o2::emcal::MCLabel newLabel = trueLabel;
345  newLabel.setAmplitudeFraction(trueLabel.getAmplitudeFraction() * energy);
346  LabelsPerSource[newLabel.getRawValue()].push_back(newLabel);
347  }
348  totalEnergy += energy;
349  }
350 
351  for (auto& [trackID, labels] : LabelsPerSource) {
352  o2::emcal::MCLabel newLabel = labels[0];
353  for (int ilabel = 0; ilabel < labels.size(); ilabel++) {
354  auto& trueLabel = labels[ilabel];
355  if (ilabel == 0) {
356  continue;
357  }
358  newLabel.setAmplitudeFraction(newLabel.getAmplitudeFraction() + trueLabel.getAmplitudeFraction());
359  }
360  newLabel.setAmplitudeFraction(newLabel.getAmplitudeFraction() / totalEnergy);
361  mcLabels.push_back(newLabel);
362  }
363 
364  std::sort(mcLabels.begin(), mcLabels.end(), [](o2::emcal::MCLabel label1, o2::emcal::MCLabel label2) { return label1.getAmplitudeFraction() > label2.getAmplitudeFraction(); });
365 
366  altroBunch.mLabels = mcLabels;
367  // std::copy(mcLabels.begin(), mcLabels.end(), std::back_inserter(altroBunch.mLabels));
368  // altroBunch.mLabels = gsl::span<o2::emcal::MCLabel>(&mcLabels[0], mcLabels.size());
369  altroBunch.mEnergyLabels.clear();
370  }
371 }
