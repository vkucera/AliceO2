134  // | | | | |------ 4-6 layer
135  // | | | |-------- 7-11 sector

148  // 0 baseline PID 3 time slices, 7 bit each
149  // 1 DO NOT USE ! reserved for tracklet end marker disambiguation

151  // 15 Reserved for testing
152  } __attribute__((__packed__));
