125  // from https://stackoverflow.com/questions/87372/check-if-a-class-has-a-member-function-of-a-given-signature
126  // Primary template with a static assertion
127  // for a meaningful error message
128  // if it ever gets instantiated.
129  // We could leave it undefined if we didn't care.
130  template <typename, typename T>
131  struct has_fill_method {
132  static_assert(
133  std::integral_constant<T, false>::value,
134  "Second template parameter needs to be of function type.");
135  };
136 
137  // specialization that does the checking
138 
139  template <typename C, typename Ret, typename... Args>
140  struct has_fill_method<C, Ret(Args...)> {
141  private:
142  template <typename T>
143  static constexpr auto check(T*)
144  -> typename std::is_same<decltype(std::declval<T>().fill(std::declval<Args>()...)), Ret>::type; // attempt to call it and see if the return type is correct
145  template <typename>
146  static constexpr std::false_type check(...);
147  typedef decltype(check<C>(nullptr)) type;
148 
149  public:
150  static constexpr bool value = type::value;
151  };
152 
153  protected:
154  auto& getSlots() { return mSlots; }
155  int getRunStartOrbit() const { return mCurrentTFInfo.firstTForbit - o2::base::GRPGeomHelper::getNHBFPerTF() * mCurrentTFInfo.tfCounter; }

184 template <typename Input, typename Container>
185 template <typename DATA>
186 bool TimeSlotCalibration<Input, Container>::process(const DATA& data)
187 {
188 
189  // process current TF
190  TFType tf = mCurrentTFInfo.tfCounter;
191  uint64_t maxDelay64 = uint64_t(mSlotLength) * mMaxSlotsDelay;
192  TFType maxDelay = maxDelay64 > o2::calibration::INFINITE_TF ? o2::calibration::INFINITE_TF : TFType(maxDelay64);
193  if (!mUpdateAtTheEndOfRunOnly) { // if you update at the end of run only, then you accept everything
194  if (tf < mLastClosedTF || (!mSlots.empty() && getLastSlot().getTFStart() > tf + maxDelay64)) { // ignore TF; note that if you have only 1 timeslot
195  // which is INFINITE_TF wide, then maxDelay
196  // does not matter: you won't accept TFs from the past,
197  // so the first condition will be used
198  LOG(info) << "Ignoring TF " << tf << ", mLastClosedTF = " << mLastClosedTF;
199  return false;
200  }
201  }
202  auto& slotTF = getSlotForTF(tf);
203  using Cont_t = typename std::remove_pointer<decltype(slotTF.getContainer())>::type;
204  if constexpr (has_fill_method<Cont_t, void(const o2::dataformats::TFIDInfo&, const DATA&)>::value) {
205  slotTF.getContainer()->fill(mCurrentTFInfo, data);
206  } else {
207  slotTF.getContainer()->fill(data);
208  }
209  if (tf > mMaxSeenTF) {
210  mMaxSeenTF = tf; // keep track of the most recent TF processed
211  }
212  if (!mUpdateAtTheEndOfRunOnly) { // if you update at the end of run only, you don't check at every TF which slots can be closed
213  // check if some slots are done
214  checkSlotsToFinalize(tf, maxDelay);
215  }
216 
217  return true;
218 }

259 template <typename Input, typename Container>
260 void TimeSlotCalibration<Input, Container>::checkSlotsToFinalize(TFType tf, int maxDelay)
261 {
262  // Check which slots can be finalized, provided the newly arrived TF is tf
263 
264  // if slot finalization is asked as soon as the slot is ready, we need to check if we got enough statistics, and if so, redefine the slot
265  if (mSlots.size() == 1 && mFinalizeWhenReady) {
266  TFType checkInterval = mCheckIntervalInfiniteSlot + mLastCheckedTFInfiniteSlot;
267  if (mWasCheckedInfiniteSlot) {
268  checkInterval = mCheckDeltaIntervalInfiniteSlot + mLastCheckedTFInfiniteSlot;
269  }
270  if (tf >= checkInterval || tf == INFINITE_TF) {
271  LOG(debug) << "mMaxSeenTF = " << mMaxSeenTF << ", mLastCheckedTFInfiniteSlot = " << mLastCheckedTFInfiniteSlot << ", checkInterval = " << checkInterval << ", mSlots[0].getTFStart() = " << mSlots[0].getTFStart();
272  if (tf == INFINITE_TF) {
273  LOG(info) << "End of run reached, trying to calibrate what we have, if we have enough statistics";
274  } else {
275  LOG(info) << "Calibrating as soon as we have enough statistics:";
276  LOG(info) << "Update interval passed (" << checkInterval << "), checking slot for " << mSlots[0].getTFStart() << " <= TF <= " << INFINITE_TF;
277  }
278  mLastCheckedTFInfiniteSlot = tf;
279  if (hasEnoughData(mSlots[0])) {
280  mWasCheckedInfiniteSlot = false;
281  mSlots[0].setTFStart(mLastClosedTF);
282  mSlots[0].setTFEnd(mMaxSeenTF);
283  LOG(info) << "Finalizing slot for " << mSlots[0].getTFStart() << " <= TF <= " << mSlots[0].getTFEnd();
284  finalizeSlot(mSlots[0]); // will be removed after finalization
285  mLastClosedTF = mSlots[0].getTFEnd() < INFINITE_TF ? (mSlots[0].getTFEnd() + 1) : mSlots[0].getTFEnd() < INFINITE_TF; // will not accept any TF below this
286  mSlots.erase(mSlots.begin());
287  // creating a new slot if we are not at the end of run
288  if (tf != INFINITE_TF) {
289  LOG(info) << "Creating new slot for " << mLastClosedTF << " <= TF <= " << INFINITE_TF;
290  auto& sl = emplaceNewSlot(true, mLastClosedTF, INFINITE_TF);
291  sl.setRunStartOrbit(getRunStartOrbit());
292  }
293  } else {
294  LOG(info) << "Not enough data to calibrate";
295  mWasCheckedInfiniteSlot = true;
296  }
297  } else {
298  LOG(debug) << "Not trying to calibrate: either not at EoS, or update interval not passed";
299  }
300  } else {
301  // check if some slots are done
302  for (auto slot = mSlots.begin(); slot != mSlots.end();) {
303  uint64_t lim64 = uint64_t(maxDelay) + slot->getTFEnd();
304  TFType tfLim = lim64 < INFINITE_TF ? TFType(lim64) : INFINITE_TF;
305  if (tfLim < tf) {
306  if (hasEnoughData(*slot)) {
307  LOG(debug) << "Finalizing slot for " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
308  finalizeSlot(*slot); // will be removed after finalization
309  } else if ((slot + 1) != mSlots.end()) {
310  LOG(info) << "Merging underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd()
311  << " to slot " << (slot + 1)->getTFStart() << " <= TF <= " << (slot + 1)->getTFEnd();
312  (slot + 1)->mergeToPrevious(*slot);
313  } else {
314  LOG(info) << "Discard underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
315  break; // slot has no enough stat. and there is no other slot to merge it to
316  }
317  mLastClosedTF = slot->getTFEnd() + 1; // will not accept any TF below this
318  LOG(info) << "closing slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
319  slot = mSlots.erase(slot);
320  } else {
321  break; // all following slots will be even closer to the new TF
322  }
323  }
324  }
325 }

328 template <typename Input, typename Container>
329 void TimeSlotCalibration<Input, Container>::finalizeOldestSlot()
330 {
331  // Enforce finalization and removal of the oldest slot
332  if (mSlots.empty()) {
333  LOG(warning) << "There are no slots defined";
334  return;
335  }
336  finalizeSlot(mSlots.front());
337  mLastClosedTF = mSlots.front().getTFEnd() + 1; // do not accept any TF below this
338  mSlots.erase(mSlots.begin());
339 }

342 template <typename Input, typename Container>
343 inline TFType TimeSlotCalibration<Input, Container>::tf2SlotMin(TFType tf) const
344 {
345 
346  // returns the min TF of the slot to which "tf" belongs
347 
348  if (tf < mFirstTF) {
349  throw std::runtime_error("invalid TF");
350  }
351  if (mUpdateAtTheEndOfRunOnly) {
352  return mFirstTF;
353  }
354  uint64_t tft = int64_t(((tf - mFirstTF) / mSlotLength) * mSlotLength) + mFirstTF;
355  return tft < o2::calibration::INFINITE_TF ? TFType(tft) : INFINITE_TF;
356 }

359 template <typename Input, typename Container>
360 TimeSlot<Container>& TimeSlotCalibration<Input, Container>::getSlotForTF(TFType tf)
361 {
362 
363  LOG(debug) << "Getting slot for TF " << tf;
364  checkSlotLength();
365  if (mUpdateAtTheEndOfRunOnly) {
366  if (!mSlots.empty() && mSlots.back().getTFEnd() < tf) {
367  mSlots.back().setTFEnd(tf);
368  } else if (mSlots.empty()) {
369  auto& sl = emplaceNewSlot(true, mFirstTF, tf);
370  sl.setRunStartOrbit(getRunStartOrbit());
371  }
372  return mSlots.back();
373  }
374 
375  if (!mSlots.empty() && mSlots.front().getTFStart() > tf) { // we need to add a slot to the beginning
376  auto tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1); // min TF of the slot corresponding to a TF smaller than the first seen
377  auto tftgt = tf2SlotMin(tf); // min TF of the slot to which the TF "tf" would belong
378  while (tfmn >= tftgt) {
379  uint64_t tft = uint64_t(tfmn) + mSlotLength - 1;
380  TFType tfmx = tft < o2::calibration::INFINITE_TF ? TFType(tft) : o2::calibration::INFINITE_TF;
381  LOG(info) << "Adding new slot for " << tfmn << " <= TF <= " << tfmx;
382  auto& sl = emplaceNewSlot(true, tfmn, tfmx);
383  sl.setRunStartOrbit(getRunStartOrbit());
384  if (!tfmn) {
385  break;
386  }
387  tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1);
388  }
389  return mSlots[0];
390  }
391  for (auto it = mSlots.begin(); it != mSlots.end(); it++) {
392  auto rel = (*it).relateToTF(tf);
393  if (rel == 0) {
394  return (*it);
395  }
396  }
397  // need to add in the end
398  auto tfmn = mSlots.empty() ? tf2SlotMin(tf) : tf2SlotMin(mSlots.back().getTFEnd() + 1);
399  do {
400  uint64_t tft = uint64_t(tfmn) + mSlotLength - 1;
401  TFType tfmx = tft < o2::calibration::INFINITE_TF ? TFType(tft) : o2::calibration::INFINITE_TF;
402  LOG(info) << "Adding new slot for " << tfmn << " <= TF <= " << tfmx;
403  auto& sl = emplaceNewSlot(false, tfmn, tfmx);
404  sl.setRunStartOrbit(getRunStartOrbit());
405  tfmn = tf2SlotMin(mSlots.back().getTFEnd() + 1);
406  } while (tf > mSlots.back().getTFEnd());
407 
408  return mSlots.back();
409 }
