62 void NoiseCalibratorSpec::run(ProcessingContext& pc)
63 {
64  updateTimeDependentParams(pc);
65  if (mDigits) {
66  const auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
67  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("digitsROF");
68  const auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("digitsROF").header)->startTime;
69 
70  if (mCalibrator->processTimeFrame(tfcounter, digits, rofs)) {
71  LOG(info) << "Minimum number of noise counts has been reached !";
72  if (mOutputType.compare("CCDB") == 0) {
73  LOG(info) << "Sending an object to Production-CCDB";
74  sendOutputCcdb(pc.outputs());
75  } else if (mOutputType.compare("DCS") == 0) {
76  LOG(info) << "Sending an object to DCS-CCDB";
77  sendOutputDcs(pc.outputs());
78  } else {
79  LOG(info) << "Sending an object to Production-CCDB and DCS-CCDB";
80  sendOutputCcdbDcs(pc.outputs());
81  }
82  pc.services().get<ControlService>().readyToQuit(mStopMeOnly ? QuitRequest::Me : QuitRequest::All);
83  }
84  } else {
85  const auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
86  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
87  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
88  const auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("ROframes").header)->startTime;
89 
90  if (mCalibrator->processTimeFrame(tfcounter, compClusters, patterns, rofs)) {
91  LOG(info) << "Minimum number of noise counts has been reached !";
92  if (mOutputType.compare("CCDB") == 0) {
93  LOG(info) << "Sending an object to Production-CCDB";
94  sendOutputCcdb(pc.outputs());
95  } else if (mOutputType.compare("DCS") == 0) {
96  LOG(info) << "Sending an object to DCS-CCDB";
97  sendOutputDcs(pc.outputs());
98  } else {
99  LOG(info) << "Sending an object to Production-CCDB and DCS-CCDB";
100  sendOutputCcdbDcs(pc.outputs());
101  }
102  pc.services().get<ControlService>().readyToQuit(mStopMeOnly ? QuitRequest::Me : QuitRequest::All);
103  }
104  }

107 void NoiseCalibratorSpec::setOutputDcs(const o2::itsmft::NoiseMap& payload)
108 {
109  for (int iChip = 0; iChip < 936; ++iChip) {
110  for (int iRow = 0; iRow < 512; ++iRow) {
111  for (int iCol = 0; iCol < 1024; ++iCol) {
112 
113  if (!payload.isNoisy(iChip, iRow, iCol)) {
114  continue;
115  }
116  std::array<int, 4> noise = {iChip, iRow, iCol, 0};
117  mNoiseMapForDcs.emplace_back(noise);
118  }
119  }
120  }

123 void NoiseCalibratorSpec::sendOutputCcdbDcs(DataAllocator& output)
124 {
125 
126  LOG(info) << "CCDB-DCS mode";
127 
128  static bool done = false;
129  if (done) {
130  return;
131  }
132  done = true;
133 
134  mCalibrator->finalize();
135 
136  long tstart = mStart;
137  if (tstart == -1) {
138  tstart = o2::ccdb::getCurrentTimestamp();
139  }
140  long tend = mEnd;
141  if (tend == -1) {
142  constexpr long SECONDSPERYEAR = 365 * 24 * 60 * 60;
143  tend = o2::ccdb::getFutureTimestamp(SECONDSPERYEAR);
144  }
145 
146  std::map<std::string, std::string> meta;
147  auto toKeyValPairs = [&meta](std::vector<std::string> const& tokens) {
148  for (auto& token : tokens) {
149  auto keyval = Str::tokenize(token, '=', false);
150  if (keyval.size() != 2) {
151  LOG(error) << "Illegal command-line key/value string: " << token;
152  continue;
153  }
154  Str::trim(keyval[1]);
155  meta[keyval[0]] = keyval[1];
156  }
157  };
158  toKeyValPairs(Str::tokenize(mMeta, ';', true));
159 
160  long startTF, endTF;
161 
162  const auto& payload = mCalibrator->getNoiseMap();
163  // const auto& payload = mCalibrator->getNoiseMap(starTF, endTF); //For TimeSlot calibration
164 
165  o2::ccdb::CcdbObjectInfo info(mPath, "NoiseMap", "noise.root", meta, tstart, tend);
166  auto flName = o2::ccdb::CcdbApi::generateFileName("noise");
167  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
168  info.setFileName(flName);
169  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
170  << " of size " << image->size()
171  << " bytes, valid for " << info.getStartValidityTimestamp()
172  << " : " << info.getEndValidityTimestamp();
173 
174  using clbUtils = o2::calibration::Utils;
175  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "MFT_NoiseMap", 0}, *image.get());
176  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "MFT_NoiseMap", 0}, info);
177 
178  setOutputDcs(payload);
179 
180  o2::ccdb::CcdbObjectInfo infoDcs(mPathDcs, "NoiseMap", "noise.root", meta, tstart, tend);
181  auto flNameDcs = o2::ccdb::CcdbApi::generateFileName("noise");
182  auto imageDcs = o2::ccdb::CcdbApi::createObjectImage(&mNoiseMapForDcs, &infoDcs);
183  infoDcs.setFileName(flNameDcs);
184  LOG(info) << "Sending object " << infoDcs.getPath() << "/" << infoDcs.getFileName()
185  << " of size " << imageDcs->size()
186  << " bytes, valid for " << infoDcs.getStartValidityTimestamp()
187  << " : " << infoDcs.getEndValidityTimestamp();
188 
189  using clbUtilsDcs = o2::calibration::Utils;
190  output.snapshot(Output{clbUtilsDcs::gDataOriginCDBPayload, "MFT_NoiseMap", 1}, *imageDcs.get());
191  output.snapshot(Output{clbUtilsDcs::gDataOriginCDBWrapper, "MFT_NoiseMap", 1}, infoDcs);
192 }

341 DataProcessorSpec getNoiseCalibratorSpec(bool useDigits)
342 {
343  o2::header::DataOrigin detOrig = o2::header::gDataOriginMFT;
344  std::vector<InputSpec> inputs;
345  if (useDigits) {
346  inputs.emplace_back("digits", detOrig, "DIGITS", 0, Lifetime::Timeframe);
347  inputs.emplace_back("digitsROF", detOrig, "DIGITSROF", 0, Lifetime::Timeframe);
348  } else {
349  inputs.emplace_back("compClusters", detOrig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
350  inputs.emplace_back("patterns", detOrig, "PATTERNS", 0, Lifetime::Timeframe);
351  inputs.emplace_back("ROframes", detOrig, "CLUSTERSROF", 0, Lifetime::Timeframe);
352  inputs.emplace_back("cldict", "MFT", "CLUSDICT", 0, Lifetime::Condition, ccdbParamSpec("MFT/Calib/ClusterDictionary"));
353  }
354  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
355  false, // GRPECS=true
356  false, // GRPLHCIF
357  false, // GRPMagField
358  false, // askMatLUT
359  o2::base::GRPGeomRequest::None, // geometry
360  inputs);
361  using clbUtils = o2::calibration::Utils;
362  std::vector<OutputSpec> outputs;
363  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCDBPayload, "MFT_NoiseMap"}, Lifetime::Sporadic);
364  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCDBWrapper, "MFT_NoiseMap"}, Lifetime::Sporadic);
365 
366  return DataProcessorSpec{
367  "mft-noise-calibrator",
368  inputs,
369  outputs,
370  AlgorithmSpec{adaptFromTask<NoiseCalibratorSpec>(useDigits, ccdbRequest)},
371  Options{
372  {"prob-threshold", VariantType::Float, 1.e-6f, {"Probability threshold for noisy pixels"}},
373  {"prob-rel-err", VariantType::Float, 0.2f, {"Relative error on channel noise to apply the threshold"}},
374  {"tstart", VariantType::Int64, -1ll, {"Start of validity timestamp"}},
375  {"tend", VariantType::Int64, -1ll, {"End of validity timestamp"}},
376  {"path-CCDB", VariantType::String, "/MFT/Calib/NoiseMap", {"Path to write to in CCDB"}},
377  {"path-DCS", VariantType::String, "/MFT/Config/NoiseMap", {"Path to write to in CCDB"}},
378  {"meta", VariantType::String, "", {"meta data to write in CCDB"}},
379  {"send-to-server", VariantType::String, "CCDB-DCS", {"meta data to write in DCS-CCDB"}},
380  {"stop-me-only", VariantType::Bool, false, {"At sufficient statistics stop only this device, otherwise whole workflow"}}}};
381 }
