114 #endif

117  return new GPUDisplayFrontendWindows;
118  } else
119 #elif defined(GPUCA_BUILD_EVENT_DISPLAY_X11)
120  if (strcmp(type, "x11") == 0 || strcmp(type, "auto") == 0) {
121  return new GPUDisplayFrontendX11;
122  } else
123 #endif
124 #if defined(GPUCA_STANDALONE) && defined(GPUCA_BUILD_EVENT_DISPLAY_GLFW)

126  return new GPUDisplayFrontendGlfw;
127  } else
128 #endif
129 #ifdef GPUCA_BUILD_EVENT_DISPLAY_WAYLAND
130  if (strcmp(type, "wayland") == 0 || (strcmp(type, "auto") == 0 && getenv("XDG_SESSION_TYPE") && strcmp(getenv("XDG_SESSION_TYPE"), "wayland") == 0)) {
131  return new GPUDisplayFrontendWayland;
132  } else
133 #endif
134 #ifdef GPUCA_BUILD_EVENT_DISPLAY_GLUT
135  if (strcmp(type, "glut") == 0 || strcmp(type, "auto") == 0) {
136  return new GPUDisplayFrontendGlut;
137  } else
138 #endif
139  {
140  GPUError("Requested frontend not available");
141  }
142  return nullptr;
143 }
