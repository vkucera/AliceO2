278 inline void AlignmentPoint::diagonalizeResiduals(double rY, double rZ, double& resU, double& resV) const
279 {
280  // rotate residuals to frame where their error matrix is diagonal
281  resU = mCosDiagErr * rY - mSinDiagErr * rZ;
282  resV = mSinDiagErr * rY + mCosDiagErr * rZ;
283  //
284 }

287 inline void AlignmentPoint::getResidualsDiag(const double* pos, double& resU, double& resV) const
288 {
289  // calculate residuals in the frame where the errors are diagonal, given the position
290  // of the track in the standard tracking frame
291  diagonalizeResiduals(pos[0] - mXYZTracking[1], pos[1] - mXYZTracking[2], resU, resV);
292  //
293 }
