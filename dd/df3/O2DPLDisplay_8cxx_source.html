79 void O2DPLDisplaySpec::run(ProcessingContext& pc)
80 {
81  if (!this->mEveHostNameMatch) {
82  return;
83  }
84  if (this->mOnlyNthEvent && this->mEventCounter++ % this->mOnlyNthEvent != 0) {
85  return;
86  }
87  LOG(info) << "------------------------ O2DPLDisplay::run version " << o2_eve_version << " ------------------------------------";
88  // filtering out any run which occur before reaching next time interval
89  auto currentTime = std::chrono::high_resolution_clock::now();
90  std::chrono::duration<double> elapsed = currentTime - this->mTimeStamp;
91  if (elapsed < this->mTimeInterval) {
92  return; // skip this run - it is too often
93  }
94  this->mTimeStamp = currentTime;
95  updateTimeDependentParams(pc);
96 
97  EveWorkflowHelper::FilterSet enabledFilters;
98 
99  enabledFilters.set(EveWorkflowHelper::Filter::ITSROF, this->mFilterITSROF);
100  enabledFilters.set(EveWorkflowHelper::Filter::TimeBracket, this->mFilterTime);
101  enabledFilters.set(EveWorkflowHelper::Filter::EtaBracket, this->mRemoveTPCEta);
102  enabledFilters.set(EveWorkflowHelper::Filter::TotalNTracks, this->mNumberOfTracks != -1);
103 
104  EveWorkflowHelper helper(enabledFilters, this->mNumberOfTracks, this->mTimeBracket, this->mEtaBracket, this->mPrimaryVertexMode);
105 
106  helper.getRecoContainer().collectData(pc, *mDataRequest);
107  helper.selectTracks(&(mData.mConfig->configCalib), mClMask, mTrkMask, mTrkMask);
108 
109  helper.prepareITSClusters(mData.mITSDict);
110  helper.prepareMFTClusters(mData.mMFTDict);
111 
112  const auto& ref = pc.inputs().getFirstValid(true);
113  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
114  const auto* dph = DataRefUtils::getHeader<DataProcessingHeader*>(ref);
115 
116  std::unordered_map<o2::dataformats::GlobalTrackID, std::size_t> savedTracks;
117 
118  for (int i = 0; i < GID::Source::NSources; i++) {
119  savedTracks[i] = 0;
120  }
121 
122  bool anythingSaved = false;
123 
124  for (std::size_t pv = 0; pv < helper.mTotalPrimaryVertices; ++pv) {
125  helper.draw(pv, mTrackSorting);
126 
127  bool save = true;
128 
129  if (this->mMinITSTracks != -1 && helper.mEvent.getDetectorTrackCount(detectors::DetID::ITS) < this->mMinITSTracks) {
130  save = false;
131  }
132 
133  if (this->mMinTracks != -1 && helper.mEvent.getTrackCount() < this->mMinTracks) {
134  save = false;
135  }
136 
137  if (save) {
138  helper.mEvent.setClMask(this->mClMask.to_ulong());
139  helper.mEvent.setTrkMask(this->mTrkMask.to_ulong());
140  helper.mEvent.setRunNumber(dh->runNumber);
141  helper.mEvent.setTfCounter(dh->tfCounter);
142  helper.mEvent.setFirstTForbit(dh->firstTForbit);
143  helper.save(this->mJsonPath, this->mNumberOfFiles, this->mTrkMask, this->mClMask, dh->runNumber, dph->creation);
144  anythingSaved = true;
145  }
146 
147  helper.clear();
148  }
149 
150  for (const auto& gid : helper.mTotalAcceptedTracks) {
151  savedTracks[gid.getSource()] += 1;
152  }
153 
154  auto endTime = std::chrono::high_resolution_clock::now();
155  LOGP(info, "Visualization of TF:{} at orbit {} took {} s.", dh->tfCounter, dh->firstTForbit, std::chrono::duration_cast<std::chrono::microseconds>(endTime - currentTime).count() * 1e-6);
156 
157  if (mPrimaryVertexMode) {
158  LOGP(info, "Primary vertices: {}", helper.mTotalPrimaryVertices);
159  }
160 
161  LOGP(info, "JSON saved: {}", anythingSaved ? "YES" : "NO");
162 
163  std::vector<std::string> sourceStats;
164  sourceStats.reserve(GID::Source::NSources);
165 
166  for (int i = 0; i < GID::Source::NSources; i++) {
167  if (mTrkMask[i]) {
168  sourceStats.emplace_back(fmt::format("{}/{} {}", savedTracks.at(i), helper.mTotalTracks.at(i), GID::getSourceName(i)));
169  }
170  }
171 
172  LOGP(info, "Tracks: {}", fmt::join(sourceStats, ", "));
173 }

197 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
198 {
199  LOG(info) << "------------------------ defineDataProcessing " << o2_eve_version << " ------------------------------------";
200 
201  WorkflowSpec specs;
202 
203  std::string jsonFolder = cfgc.options().get<std::string>("jsons-folder");
204  std::string eveHostName = cfgc.options().get<std::string>("eve-hostname");
205  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
206  bool useMC = !cfgc.options().get<bool>("disable-mc");
207 
208  char hostname[_POSIX_HOST_NAME_MAX];
209  gethostname(hostname, _POSIX_HOST_NAME_MAX);
210  bool eveHostNameMatch = eveHostName.empty() || eveHostName == hostname;
211 
212  int eveDDSColIdx = cfgc.options().get<int>("eve-dds-collection-index");
213  if (eveDDSColIdx != -1) {
214  char* colIdx = getenv("DDS_COLLECTION_INDEX");
215  int myIdx = colIdx ? atoi(colIdx) : -1;
216  if (myIdx == eveDDSColIdx) {
217  LOG(important) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
218  } else {
219  LOG(info) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
220  }
221  eveHostNameMatch &= myIdx == eveDDSColIdx;
222  }
223 
224  std::chrono::milliseconds timeInterval(cfgc.options().get<int>("time-interval"));
225  int numberOfFiles = cfgc.options().get<int>("number-of_files");
226  int numberOfTracks = cfgc.options().get<int>("number-of_tracks");
227 
228  GID::mask_t allowedTracks = GID::getSourcesMask(O2DPLDisplaySpec::allowedTracks);
229  GID::mask_t allowedClusters = GID::getSourcesMask(O2DPLDisplaySpec::allowedClusters);
230 
231  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks")) & allowedTracks;
232  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters")) & allowedClusters;
233  if (!srcTrk.any() && !srcCl.any()) {
234  if (cfgc.options().get<bool>("skipOnEmptyInput")) {
235  LOG(info) << "No valid inputs for event display, disabling event display";
236  return std::move(specs);
237  }
238  throw std::runtime_error("No input configured");
239  }
240 
241  auto isRangeEnabled = [&opts = cfgc.options()](const char* min_name, const char* max_name) {
242  EveWorkflowHelper::Bracket bracket{opts.get<float>(min_name), opts.get<float>(max_name)};
243  bool optEnabled = false;
244 
245  if (bracket.getMin() < 0 && bracket.getMax() < 0) {
246  optEnabled = false;
247  } else if (bracket.getMin() >= 0 && bracket.getMax() >= 0) {
248  optEnabled = true;
249 
250  if (bracket.isInvalid()) {
251  throw std::runtime_error(fmt::format("{}, {} bracket is invalid", min_name, max_name));
252  }
253  } else {
254  throw std::runtime_error(fmt::format("Both boundaries, {} and {}, have to be specified at the same time", min_name, max_name));
255  }
256 
257  return std::make_tuple(optEnabled, bracket);
258  };
259 
260  const auto [filterTime, timeBracket] = isRangeEnabled("filter-time-min", "filter-time-max");
261 
262  const auto etaRange = cfgc.options().get<float>("remove-tpc-abs-eta");
263 
264  bool removeTPCEta = false;
265  EveWorkflowHelper::Bracket etaBracket;
266 
267  if (etaRange != 0.f) {
268  etaBracket = EveWorkflowHelper::Bracket{-etaRange, etaRange};
269  removeTPCEta = true;
270  }
271 
272  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
273  dataRequest->requestTracks(srcTrk, useMC);
274  dataRequest->requestClusters(srcCl, useMC);
275 
276  auto filterITSROF = cfgc.options().get<bool>("filter-its-rof");
277 
278  if (filterITSROF) {
279  dataRequest->requestIRFramesITS();
280  }
281 
282  auto primaryVertexMode = cfgc.options().get<bool>("primary-vertex-mode");
283 
284  if (primaryVertexMode) {
285  dataRequest->requestPrimaryVertertices(useMC);
286  }
287 
288  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
289 
290  auto minITSTracks = cfgc.options().get<int>("min-its-tracks");
291  auto minTracks = cfgc.options().get<int>("min-tracks");
292  auto onlyNthEvent = cfgc.options().get<int>("only-nth-event");
293  auto tracksSorting = cfgc.options().get<bool>("track-sorting");
294 
295  if (numberOfTracks == -1) {
296  tracksSorting = false; // do not sort if all tracks are allowed
297  }
298 
299  specs.emplace_back(DataProcessorSpec{
300  "o2-eve-export",
301  dataRequest->inputs,
302  {},
303  AlgorithmSpec{adaptFromTask<O2DPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest, jsonFolder, timeInterval, numberOfFiles, numberOfTracks, eveHostNameMatch, minITSTracks, minTracks, filterITSROF, filterTime, timeBracket, removeTPCEta, etaBracket, tracksSorting, onlyNthEvent, primaryVertexMode)}});
304 
305  return std::move(specs);
306 }
