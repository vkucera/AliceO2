61 void AlignmentTrack::defineDOFs()
62 {
63  // define varied DOF's (local parameters) for the track:
64  // 1) kinematic params (5 or 4 depending on Bfield)
65  // 2) mult. scattering angles (2)
66  // 3) if requested by point: energy loss
67  //
68  mNLocPar = mNLocExtPar = getFieldON() ? kNKinParBON : kNKinParBOFF;
69  int np = getNPoints();
70  //
71  // the points are sorted in order opposite to track direction -> outer points come 1st,
72  // but for the 2-leg cosmic track the innermost points are in the middle (1st lower leg, then upper one)
73  //
74  // start along track direction, i.e. last point in the ordered array
75  int minPar = mNLocPar;
76  for (int ip = getInnerPointID() + 1; ip--;) { // collision track or cosmic lower leg
77  AlignmentPoint* pnt = getPoint(ip);
78  pnt->setMinLocVarID(minPar);
79  if (pnt->containsMaterial()) {
80  mNLocPar += pnt->getNMatPar();
81  }
82  pnt->setMaxLocVarID(mNLocPar); // flag up to which parameted ID this points depends on
83  }
84  //
85  if (isCosmic()) {
86  minPar = mNLocPar;
87  for (int ip = getInnerPointID() + 1; ip < np; ip++) { // collision track or cosmic lower leg
88  AlignmentPoint* pnt = getPoint(ip);
89  pnt->setMinLocVarID(minPar);
90  if (pnt->containsMaterial()) {
91  mNLocPar += pnt->getNMatPar();
92  }
93  pnt->setMaxLocVarID(mNLocPar); // flag up to which parameted ID this points depends on
94  }
95  }
96  mLocPar.clear();
97  mLocPar.resize(mNLocPar);
98  for (int i = 0; i < 2; i++) {
99  mResid[i].clear();
100  mDResDLoc[i].clear();
101  mResid[i].resize(np);
102  mDResDLoc[i].resize(mNLocPar * np);
103  }
104 }

110  // The 1st 4 or 5 elements of params vector should be the reference trackParam_t
111  // Then parameters of material corrections for each point
112  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
113  // They correspond to kink parameters
114  // (trackParam_t_after_material - trackParam_t_before_material)
115  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
116  // increment will be done locally in the applyMatCorr routine.
117  //
118  // If params are not provided, use internal params array
119  //
120  if (!params) {
121  params = mLocPar.data();
122  }
123  //
124  if (!getResidDone()) {
125  calcResiduals(params);
126  }
127  //

144  }
145  //
146  setDerivDone();

151 bool AlignmentTrack::calcResidDeriv(double* extendedParams, bool invert, int pFrom, int pTo)
152 {
153  // Calculate derivatives of residuals vs params for points pFrom to pT. For cosmic upper leg
154  // track parameter may require inversion.
155  // The 1st 4 or 5 elements of params vector should be the reference trackParam_t
156  // Then parameters of material corrections for each point
157  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
158  // They correspond to kink parameters
159  // (trackParam_t_after_material - trackParam_t_before_material)
160  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
161  // increment will be done locally in the applyMatCorr routine.
162  //
163  // The derivatives are calculated using Richardson extrapolation
164  // (like http://root.cern.ch/root/html/ROOT__Math__RichardsonDerivator.html)
165  //
166  trackParam_t probD[kNRDClones]; // use this to vary supplied param for derivative calculation
167  double varDelta[kRichardsonN];
168  const int kInvElem[kNKinParBON] = {-1, 1, 1, -1, -1};
169  //
170  const double kDelta[kNKinParBON] = {0.02, 0.02, 0.001, 0.001, 0.01}; // variations for ExtTrackParam and material effects
171  //
172  double delta[kNKinParBON]; // variations of curvature term are relative
173  for (int i = kNKinParBOFF; i--;) {
174  delta[i] = kDelta[i];
175  }
176  if (getFieldON()) {
177  delta[kParQ2Pt] = kDelta[kParQ2Pt] * Abs(getQ2Pt());
178  }
179  //
180  int pinc;
181  if (pTo > pFrom) { // fit in points decreasing order: cosmics upper leg
182  pTo++;
183  pinc = 1;
184  } else { // fit in points increasing order: collision track or cosmics lower leg
185  pTo--;
186  pinc = -1;
187  }
188 
189  // 1) derivative wrt trackParam_t parameters
190  for (int ipar = mNLocExtPar; ipar--;) {
191 
192  setParams(probD, kNRDClones, getX(), getAlpha(), extendedParams, true);
193  if (invert) {
194  for (int ic = kNRDClones; ic--;) {
195  probD[ic].invert();
196  }
197  }
198  double del = delta[ipar];
199  //
200  for (int icl = 0; icl < kRichardsonN; icl++) { // calculate kRichardsonN variations with del, del/2, del/4...
201  varDelta[icl] = del;
202  modParam(probD[(icl << 1) + 0], ipar, del);
203  modParam(probD[(icl << 1) + 1], ipar, -del);
204  del *= 0.5;
205  }
206  // propagate varied tracks to each point
207  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
208  AlignmentPoint* pnt = getPoint(ip);
209  if (!propagateParamToPoint(probD, kNRDClones, pnt)) {
210  return false;
211  }
212  // if (pnt->containsMaterial()) { // apply material corrections
213  if (!applyMatCorr(probD, kNRDClones, extendedParams, pnt)) {
214  return false;
215  }
216  // }
217  //
218  if (pnt->containsMeasurement()) {
219  int offsDer = ip * mNLocPar + ipar;
220  richardsonDeriv(probD, varDelta, pnt, mDResDLoc[0][offsDer], mDResDLoc[1][offsDer]); // calculate derivatives
221  if (invert && kInvElem[ipar] < 0) {
222  mDResDLoc[0][offsDer] = -mDResDLoc[0][offsDer];
223  mDResDLoc[1][offsDer] = -mDResDLoc[1][offsDer];
224  }
225  }
226  } // loop over points
227  } // loop over ExtTrackParam parameters
228  //
229  // 2) now vary material effect related parameters: MS and eventually ELoss
230  //
231  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
232  AlignmentPoint* pnt = getPoint(ip);
233  //
234  // global derivatives at this point
235  if (pnt->containsMeasurement() && !calcResidDerivGlo(pnt)) {
236 #if DEBUG > 3
237  AliWarningF("Failed on global derivatives calculation at point %d", ip);
238  pnt->print("meas");
239 #endif
240  return false;
241  }
242  //
243  if (!pnt->containsMaterial()) {
244  continue;
245  }
246  //
247  int nParFreeI = pnt->getNMatPar();
248  //
249  // array delta gives desired variation of parameters in trackParam_t definition,
250  // while the variation should be done for parameters in the frame where the vector
251  // of material corrections has diagonal cov. matrix -> rotate the delta to this frame
252  double deltaMatD[kNKinParBON];
253  pnt->diagMatCorr(delta, deltaMatD);
254  //
255  // printf("Vary %d [%+.3e %+.3e %+.3e %+.3e] ",ip,deltaMatD[0],deltaMatD[1],deltaMatD[2],deltaMatD[3]); pnt->print();
256 
257  int offsI = pnt->getMaxLocVarID() - nParFreeI; // the parameters for this point start with this offset
258  // they are irrelevant for the points upstream
259  for (int ipar = 0; ipar < nParFreeI; ipar++) { // loop over DOFs related to MS and ELoss are point ip
260  double del = deltaMatD[ipar];
261  //
262  // We will vary the tracks starting from the original parameters propagated to given point
263  // and stored there (before applying material corrections for this point)
264  //
265  setParams(probD, kNRDClones, pnt->getXPoint(), pnt->getAlphaSens(), pnt->getTrParamWSB(), false);
266  // no need for eventual track inversion here: if needed, this is already done in ParamWSB
267  //
268  int offsIP = offsI + ipar; // parameter entry in the extendedParams array
269  // printf(" Var:%d (%d) %e\n",ipar,offsIP, del);
270 
271  for (int icl = 0; icl < kRichardsonN; icl++) { // calculate kRichardsonN variations with del, del/2, del/4...
272  varDelta[icl] = del;
273  double parOrig = extendedParams[offsIP];
274  extendedParams[offsIP] += del;
275  //
276  // apply varied material effects : incremented by delta
277  if (!applyMatCorr(probD[(icl << 1) + 0], extendedParams, pnt)) {
278  return false;
279  }
280  //
281  // apply varied material effects : decremented by delta
282  extendedParams[offsIP] = parOrig - del;
283  if (!applyMatCorr(probD[(icl << 1) + 1], extendedParams, pnt)) {
284  return false;
285  }
286  //
287  extendedParams[offsIP] = parOrig;
288  del *= 0.5;
289  }
290  if (pnt->containsMeasurement()) { // calculate derivatives at the scattering point itself
291  int offsDerIP = ip * mNLocPar + offsIP;
292  richardsonDeriv(probD, varDelta, pnt, mDResDLoc[0][offsDerIP], mDResDLoc[1][offsDerIP]); // calculate derivatives for ip
293  // printf("DR SELF: %e %e at %d (%d)\n",mDResDLoc[0][offsDerIP], mDResDLoc[1][offsDerIP],offsI, offsDerIP);
294  }
295  //
296  // loop over points whose residuals can be affected by the material effects on point ip
297  for (int jp = ip + pinc; jp != pTo; jp += pinc) {
298  AlignmentPoint* pntJ = getPoint(jp);
299 
300  // printf(" DerFor:%d ",jp); pntJ->print();
301 
302  if (!propagateParamToPoint(probD, kNRDClones, pntJ)) {
303  return false;
304  }
305  //
306  if (pntJ->containsMaterial()) { // apply material corrections
307  if (!applyMatCorr(probD, kNRDClones, extendedParams, pntJ)) {
308  return false;
309  }
310  }
311  //
312  if (pntJ->containsMeasurement()) {
313  int offsDerJ = jp * mNLocPar + offsIP;
314  // calculate derivatives
315  richardsonDeriv(probD, varDelta, pntJ, mDResDLoc[0][offsDerJ], mDResDLoc[1][offsDerJ]);
316  }
317  //
318  } // << loop over points whose residuals can be affected by the material effects on point ip
319  } // << loop over DOFs related to MS and ELoss are point ip
320  } // << loop over all points of the track
321  //
322  return true;
323 }

326 bool AlignmentTrack::calcResidDerivGlo(AlignmentPoint* pnt)
327 {
328  // calculate residuals derivatives over point's sensor and its parents global params
329  double deriv[AlignableVolume::kNDOFGeom * 3];
330  //
331  const AlignableSensor* sens = pnt->getSensor();
332  const AlignableVolume* vol = sens;
333  // precalculated track parameters
334  double snp = pnt->getTrParamWSA(kParSnp), tgl = pnt->getTrParamWSA(kParTgl);
335  // precalculate track slopes to account tracking X veriation
336  // these are coeffs to translate deltaX of the point to deltaY and deltaZ of track
337  double cspi = 1. / Sqrt((1 - snp) * (1 + snp)), slpY = snp * cspi, slpZ = tgl * cspi;
338  //
339  pnt->setDGloOffs(mNGloPar); // mark 1st entry of derivatives
340  do {
341  // measurement residuals
342  int nfree = vol->getNDOFsFree();
343  if (!nfree) {
344  continue;
345  } // no free parameters?
346  sens->dPosTraDParGeom(pnt, deriv, vol == sens ? nullptr : vol);
347  //
348  checkExpandDerGloBuffer(mNGloPar + nfree); // if needed, expand derivatives buffer
349  //
350  for (int ip = 0; ip < AlignableVolume::kNDOFGeom; ip++) { // we need only free parameters
351  if (!vol->isFreeDOF(ip)) {
352  continue;
353  }
354  double* dXYZ = &deriv[ip * 3]; // tracking XYZ derivatives over this parameter
355  // residual is defined as diagonalized track_estimate - measured Y,Z in tracking frame
356  // where the track is evaluated at measured X!
357  // -> take into account modified X using track parameterization at the point (paramWSA)
358  // Attention: small simplifications(to be checked if it is ok!!!):
359  // effect of changing X is accounted neglecting track curvature to preserve linearity
360  //
361  // store diagonalized residuals in track buffer
362  pnt->diagonalizeResiduals((dXYZ[AlignmentPoint::kX] * slpY - dXYZ[AlignmentPoint::kY]),
363  (dXYZ[AlignmentPoint::kX] * slpZ - dXYZ[AlignmentPoint::kZ]),
364  mDResDGlo[0][mNGloPar], mDResDGlo[1][mNGloPar]);
365  // and register global ID of varied parameter
366  mGloParID[mNGloPar] = vol->getParGloID(ip);
367  mNGloPar++;
368  }
369  //
370  } while ((vol = vol->getParent()));
371  //
372  // eventual detector calibration parameters
373  const AlignableDetector* det = sens->getDetector();
374  int ndof = 0;
375  if (det && (ndof = det->getNCalibDOFs())) {
376  // if needed, expand derivatives buffer
377  checkExpandDerGloBuffer(mNGloPar + det->getNCalibDOFsFree());
378  for (int idf = 0; idf < ndof; idf++) {
379  if (!det->isFreeDOF(idf)) {
380  continue;
381  }
382  sens->dPosTraDParCalib(pnt, deriv, idf, nullptr);
383  pnt->diagonalizeResiduals((deriv[AlignmentPoint::kX] * slpY - deriv[AlignmentPoint::kY]),
384  (deriv[AlignmentPoint::kX] * slpZ - deriv[AlignmentPoint::kZ]),
385  mDResDGlo[0][mNGloPar], mDResDGlo[1][mNGloPar]);
386  // and register global ID of varied parameter
387  mGloParID[mNGloPar] = det->getParGloID(idf);
388  mNGloPar++;
389  }
390  }
391  //
392  pnt->setNGloDOFs(mNGloPar - pnt->getDGloOffs()); // mark number of global derivatives filled
393  //
394  return true;
395 }

441 bool AlignmentTrack::calcResiduals(const double* extendedParams, bool invert, int pFrom, int pTo)
442 {
443  // Calculate residuals for the single leg from points pFrom to pT
444  // The 1st 4 or 5 elements of extendedParams vector should be corrections to
445  // the reference trackParam_t
446  // Then parameters of material corrections for each point
447  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
448  // They correspond to kink parameters
449  // (trackParam_t_after_material - trackParam_t_before_material)
450  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
451  // increment will be done locally in the applyMatCorr routine.
452  //
453  trackParam_t probe;
454  setParams(probe, getX(), getAlpha(), extendedParams, true);
455  if (invert) {
456  probe.invert();
457  }
458  int pinc;
459  if (pTo > pFrom) { // fit in points decreasing order: cosmics upper leg
460  pTo++;
461  pinc = 1;
462  } else { // fit in points increasing order: collision track or cosmics lower leg
463  pTo--;
464  pinc = -1;
465  }
466  //
467  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
468  AlignmentPoint* pnt = getPoint(ip);
469  if (!propagateParamToPoint(probe, pnt)) {
470  return false;
471  }
472  //
473  // store the current track kinematics at the point BEFORE applying eventual material
474  // corrections. This kinematics will be later varied around supplied parameters (in the calcResidDeriv)
475  pnt->setTrParamWSB(probe.getParams());
476  //
477  // account for materials
478  // if (pnt->ContainsMaterial()) { // apply material corrections
479  if (!applyMatCorr(probe, extendedParams, pnt)) {
480  return false;
481  }
482  // }
483  pnt->setTrParamWSA(probe.getParams());
484  //
485  if (pnt->containsMeasurement()) { // need to calculate residuals in the frame where errors are orthogonal
486  pnt->getResidualsDiag(probe.getParams(), mResid[0][ip], mResid[1][ip]);
487  mChi2 += mResid[0][ip] * mResid[0][ip] / pnt->getErrDiag(0);
488  mChi2 += mResid[1][ip] * mResid[1][ip] / pnt->getErrDiag(1);
489  mNDF += 2;
490  }
491  //
492  if (pnt->containsMaterial()) {
493  // material degrees of freedom do not contribute to NDF since they are constrained by 0 expectation
494  int nCorrPar = pnt->getNMatPar();
495  float* corCov = pnt->getMatCorrCov(); // correction diagonalized covariance
496  auto offs = pnt->getMaxLocVarID() - nCorrPar;
497  for (int i = 0; i < nCorrPar; i++) {
498  mChi2 += mLocPar[offs + i] * mLocPar[offs + i] / corCov[i];
499  }
500  }
501  }
502  return true;
503 }

539 bool AlignmentTrack::propagate(trackParam_t& track, const AlignmentPoint* pnt, double maxStep, double maxSnp, MatCorrType mt, track::TrackLTIntegral* tLT)
540 {

548  }

558  return -dir;

560  }();

566 bool AlignmentTrack::ApplyMS(trackParam_t& trPar, double tms,double pms)
567 {
568  //------------------------------------------------------------------------------
569  // Modify track par (e.g. trackParam_t) in the tracking frame
570  // (dip angle lam, az. angle phi)
571  // by multiple scattering defined by polar and azumuthal scattering angles in
572  // the track collinear frame (tms and pms resp).
573  // The updated direction vector in the tracking frame becomes
574  //
575  // | Cos[lam]*Cos[phi] Cos[phi]*Sin[lam] -Sin[phi] | | Cos[tms] |

907 bool AlignmentTrack::combineTracks(trackParam_t& trcL, const trackParam_t& trcU)
908 {
909  // Assign to trcL the combined tracks (Kalman update of trcL by trcU)
910  // The trcL and trcU MUST be defined at same X,Alpha
911  //
912  // Update equations: tracks described by vectors vL and vU and coviriances CL and CU resp.
913  // then the gain matrix K = CL*(CL+CU)^-1
914  // Updated vector and its covariance:
915  // CL' = CL - K*CL
916  // vL' = vL + K(vU-vL)
917  //
918  if (Abs(trcL.getX() - trcU.getX()) > TinyDist || Abs(trcL.getAlpha() - trcU.getAlpha()) > TinyDist) {
919  LOG(error) << "Tracks must be defined at same reference X and Alpha";
920  trcL.print();
921  trcU.print();
922  return false;
923  }
924  //
925  // const covMat_t& covU = trcU.getCov();
926  // const covMat_t& covL = trcL.getCov();
927  //
928  int mtSize = getFieldON() ? kNKinParBON : kNKinParBOFF;
929  TMatrixD matCL(mtSize, mtSize), matCLplCU(mtSize, mtSize);
930  TVectorD vl(mtSize), vUmnvL(mtSize);
931  //
932  // trcL.print();
933  // trcU.print();
934  //
935  for (int i = mtSize; i--;) {
936  vUmnvL[i] = trcU.getParam(i) - trcL.getParam(i); // y = residual of 2 tracks
937  vl[i] = trcL.getParam(i);
938  for (int j = i + 1; j--;) {
939  int indIJ = ((i * (i + 1)) >> 1) + j; // position of IJ cov element in the trackParam_t covariance array
940  matCL(i, j) = matCL(j, i) = trcL.getCovarElem(i, j);
941  matCLplCU(i, j) = matCLplCU(j, i) = trcL.getCovarElem(i, j) + trcU.getCovarElem(i, j);
942  }
943  }
944  matCLplCU.Invert(); // S^-1 = (Cl + Cu)^-1
945  if (!matCLplCU.IsValid()) {
946 #if DEBUG > 3
947  LOG(error) << "Failed to invert summed cov.matrix of cosmic track";
948  matCLplCU.print();
949 #endif
950  return false; // inversion failed
951  }
952  TMatrixD matK(matCL, TMatrixD::kMult, matCLplCU); // gain K = Cl*(Cl+Cu)^-1
953  TMatrixD matKdotCL(matK, TMatrixD::kMult, matCL); // K*Cl
954  TVectorD vlUp = matK * vUmnvL; // K*(vl - vu)
955  for (int i = mtSize; i--;) {
956  trcL.updateParam(vlUp[i], i); // updated param: vL' = vL + K(vU-vL)
957  for (int j = i + 1; j--;) {
958  trcL.updateCov(-matKdotCL(i, j), i, j);
959  } // updated covariance: Cl' = Cl - K*Cl
960  }
961  //
962  // update chi2
963  double chi2 = 0;
964  for (int i = mtSize; i--;) {
965  for (int j = mtSize; j--;) {
966  chi2 += matCLplCU(i, j) * vUmnvL[i] * vUmnvL[j];
967  }
968  }
969  mChi2 += chi2;
970  //
971  // printf("Combined: Chi2Tot:%.2f ChiUp:%.2f ChiDn:%.2f ChiCmb:%.2f\n",mChi2,mChi2CosmUp,mChi2CosmDn, chi2);
972 
973  return true;
974 }
