58  void run(ProcessingContext& pc) final
59  {
60  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
61  const auto startTime = HighResClock::now();
62  const auto firstTForbit = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->firstTForbit;
63  const auto tfCounter = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->tfCounter;
64  const auto runNumber = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->runNumber;
65 
66  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}}; // TODO: Change to SAC when changed in DD
67  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
68  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
69  // ---| extract hardware information to do the processing |---
70  const auto feeId = (FEEIDType)dh->subSpecification;
71  const auto link = rdh_utils::getLink(feeId);
72  const uint32_t cruID = rdh_utils::getCRU(feeId);
73  const auto endPoint = rdh_utils::getEndPoint(feeId);
74 
75  // only select SACs
76  // ToDo: cleanup once SACs will be propagated not as RAWDATA, but SAC.
77  if (link != rdh_utils::SACLinkID) {
78  continue;
79  }
80  if (mDebugLevel & (uint32_t)DebugFlags::HBFInfo) {
81  LOGP(info, "SAC Processing firstTForbit {:9}, tfCounter {:5}, run {:6}, feeId {:6} ({:3}/{}/{:2})", firstTForbit, tfCounter, runNumber, feeId, cruID, endPoint, link);
82  }
83 
84  // ---| data loop |---
85  const gsl::span<const char> raw = pc.inputs().get<gsl::span<char>>(ref);
86  RawParser parser(raw.data(), raw.size());
87  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
88  const auto size = it.size();
89  if (size == 0) {
90  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
91  if (!rdhPtr) {
92  throw std::runtime_error("could not get RDH from packet");
93  }
94  // TODO: should only be done once for the first packet
95  if ((mDecoder.getReferenceTime() < 0) && (rdhPtr->packetCounter == 0)) {
96  const double referenceTime = o2::base::GRPGeomHelper::instance().getOrbitResetTimeMS() + firstTForbit * o2::constants::lhc::LHCOrbitMUS * 0.001;
97  LOGP(info, "setting time stamp reset reference to: {}, at tfCounter: {}, firstTForbit: {}", referenceTime, tfCounter, firstTForbit);
98  mDecoder.setReferenceTime(referenceTime); // TODO set proper time
99  }
100  continue;
101  }
102  auto data = (const char*)it.data();
103  mDecoder.process(data, size);
104  }
105  }
106 
107  mDecoder.runDecoding();
108 
109  if (mDebugLevel & (uint32_t)sac::Decoder::DebugFlags::TimingInfo) {
110  auto endTime = HighResClock::now();
111  auto elapsed_seconds = std::chrono::duration_cast<std::chrono::duration<double>>(endTime - startTime);
112  LOGP(info, "Time spent for TF {}, firstTForbit {}: {} s", tfCounter, firstTForbit, elapsed_seconds.count());
113  }
114  }
