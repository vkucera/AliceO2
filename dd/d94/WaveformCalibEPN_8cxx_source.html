24 int WaveformCalibEPN::init()
25 {
26  if (mConfig == nullptr) {
27  LOG(fatal) << "o2::zdc::WaveformCalibEPN: missing configuration object";
28  return -1;
29  }
30 
31  auto* cfg = mConfig;
32  if (mVerbosity > DbgZero) {
33  mConfig->print();
34  }
35 
36  // Inspect reconstruction parameters
37  o2::zdc::CalibParamZDC& opt = const_cast<o2::zdc::CalibParamZDC&>(CalibParamZDC::Instance());
38  opt.print();
39 
40  if (opt.debug_output > 0) {
41  setSaveDebugHistos();
42  }
43 
44  mQueue.configure(cfg);
45  if (mVerbosity > DbgZero) {
46  mQueue.printConf();
47  }
48 
49  // number of bins
50  mNBin = cfg->nbun * TSN;
51  mFirst = cfg->ibeg;
52  mLast = cfg->iend;
53  mData.setN(cfg->nbun);
54  mData.mPeak = mQueue.mPeak;
55  mInitDone = true;
56  return 0;
57 }

60 int WaveformCalibEPN::process(const gsl::span<const o2::zdc::BCRecData>& RecBC,
61  const gsl::span<const o2::zdc::ZDCEnergy>& Energy,
62  const gsl::span<const o2::zdc::ZDCTDCData>& TDCData,
63  const gsl::span<const uint16_t>& Info,
64  const gsl::span<const o2::zdc::ZDCWaveform>& wave)
65 {
66  if (!mInitDone) {
67  init();
68  }
69  o2::zdc::RecEventFlat ev;
70  ev.init(RecBC, Energy, TDCData, Info);
71  auto nen = ev.getEntries();
72  std::vector<o2::InteractionRecord> ir;
73  while (int ientry = ev.next()) {
74  uint32_t mask = mQueue.append(ev);
75 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
76  LOGF(info, "WaveformCalibEPN::%s mask=0x%04x %s %s %s %s %s %s %s %s %s %s", __func__, mask,
77  (mask & 0x001) ? ChannelNames[TDCSignal[0]] : " ",
78  (mask & 0x002) ? ChannelNames[TDCSignal[1]] : " ",
79  (mask & 0x004) ? ChannelNames[TDCSignal[2]] : " ",
80  (mask & 0x008) ? ChannelNames[TDCSignal[3]] : " ",
81  (mask & 0x010) ? ChannelNames[TDCSignal[4]] : " ",
82  (mask & 0x020) ? ChannelNames[TDCSignal[5]] : " ",
83  (mask & 0x040) ? ChannelNames[TDCSignal[6]] : " ",
84  (mask & 0x080) ? ChannelNames[TDCSignal[7]] : " ",
85  (mask & 0x100) ? ChannelNames[TDCSignal[8]] : " ",
86  (mask & 0x200) ? ChannelNames[TDCSignal[9]] : " ");
87 #endif
88  if (mask != 0) {
89 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
90  ev.print();
91  ev.printDecodedMessages();
92  mQueue.print();
93 #endif
94  // Analyze signals that refer to the TDC channels that satisfy condition
95  for (int isig = 0; isig < NChannels; isig++) {
96  int itdc = SignalTDC[isig];
97  if ((mask & (0x1 << itdc)) != 0) {
98  // Check which channels have consecutive data
99  mQueue.addData(isig, wave, mData);
100  }
101  }
102  }
103  }
104  return 0;
105 }

108 int WaveformCalibEPN::endOfRun()
109 {
110  if (mVerbosity > DbgZero) {
111  LOGF(info, "WaveformCalibEPN::endOfRun ts (%llu:%llu)", mData.mCTimeBeg, mData.mCTimeEnd);
112  for (int is = 0; is < NChannels; is++) {
113  if (mData.getEntries(is) > 0) {
114  LOGF(info, "Waveform %2d %s with %10d events and cuts AMP:(%g:%g) TDC:(%g:%g) Valid:[%d:%d:%d]", is, ChannelNames[is].data(),
115  mData.getEntries(is), mConfig->cutLow[is], mConfig->cutHigh[is],
116  mConfig->cutTimeLow[is], mConfig->cutTimeHigh[is],
117  mData.getFirstValid(is), mData.mPeak, mData.getLastValid(is));
118  }
119  }
120  }
121  if (mSaveDebugHistos) {
122  saveDebugHistos();
123  }
124  return 0;
125 }
