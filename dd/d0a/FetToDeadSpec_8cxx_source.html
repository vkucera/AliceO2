57  void run(of::ProcessingContext& pc)
58  {
59  std::array<gsl::span<const ColumnData>, 3> data;
60  std::array<gsl::span<const ROFRecord>, 3> dataRof;
61 
62  std::vector<of::InputSpec> filter = {
63  {"check_data", of::ConcreteDataTypeMatcher{header::gDataOriginMID, "DATA"}, of::Lifetime::Timeframe},
64  {"check_rof", of::ConcreteDataTypeMatcher{header::gDataOriginMID, "DATAROF"}, of::Lifetime::Timeframe},
65  };
66 
67  for (auto const& inputRef : of::InputRecordWalker(pc.inputs(), filter)) {
68  auto const* dh = framework::DataRefUtils::getHeader<o2::header::DataHeader*>(inputRef);
69  auto subSpecIdx = static_cast<size_t>(dh->subSpecification);
70  if (of::DataRefUtils::match(inputRef, "mid_data")) {
71  data[subSpecIdx] = pc.inputs().get<gsl::span<o2::mid::ColumnData>>(inputRef);
72  } else if (of::DataRefUtils::match(inputRef, "mid_data_rof")) {
73  dataRof[subSpecIdx] = pc.inputs().get<gsl::span<o2::mid::ROFRecord>>(inputRef);
74  }
75  }
76 
77  std::vector<ColumnData> noisyChannels;
78  noisyChannels.insert(noisyChannels.end(), data[1].begin(), data[1].end());
79 
80  std::vector<ROFRecord> noisyChannelsRof;
81  noisyChannelsRof.insert(noisyChannelsRof.end(), dataRof[1].begin(), dataRof[1].end());
82 
83  auto deadChannels = getDeadChannels(data[2], dataRof[2], data[0], dataRof[0]);
84 
85  std::vector<ROFRecord> deadChannelsRof;
86  deadChannelsRof.insert(deadChannelsRof.end(), dataRof[2].begin(), dataRof[2].end());
87 
88  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "NOISE", 0}, noisyChannels);
89  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "NOISEROF", 0}, noisyChannelsRof);
90  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DEAD", 0}, deadChannels);
91  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DEADROF", 0}, deadChannelsRof);
92  }
93 
94  private:
95  FetToDead mFetToDead{};
96 
97  std::vector<ColumnData> getDeadChannels(gsl::span<const ColumnData> fetData, gsl::span<const ROFRecord> fetDataRof, gsl::span<const ColumnData> selfTrigData, gsl::span<const ROFRecord> selfTrigDataRof)
98  {
99  int64_t maxDiff = 1;
100  int64_t minDiff = -maxDiff;
101 
102  ColumnDataHandler handler;
103  std::vector<ColumnData> deadChannels;
104  // The FET data can be split into different BCs.
105  // Try to merge the expected FET data with the data in the close BCs
106  // which are probably badly tagged FET data
107  auto auxRofIt = selfTrigDataRof.begin();
108  // Loop on FET ROF
109  for (auto& rof : fetDataRof) {
110  handler.clear();
111  auto eventFetData = fetData.subspan(rof.firstEntry, rof.nEntries);
112  handler.merge(eventFetData);
113  for (; auxRofIt != selfTrigDataRof.end(); ++auxRofIt) {
114  auto bcDiff = auxRofIt->interactionRecord.differenceInBC(rof.interactionRecord);
115  if (bcDiff > maxDiff) {
116  // ROFs are time ordered. If the difference is larger than the maximum difference for merging,
117  // it means that the auxRofIt is in the future.
118  // We break and compare it to the next rof.
119  break;
120  } else if (bcDiff >= minDiff) {
121  // With the previous condition, this implies minDiff <= bcDiff <= maxDiff
122  auto auxFet = selfTrigData.subspan(auxRofIt->firstEntry, auxRofIt->nEntries);
123  handler.merge(auxFet);
124  }
125  // If bcDiff is < minDiff, it means that the auxRofIt is too much in the past
126  // So we do nothing, i.e. we move to the next auxRofIt in the for loop
127  }
128  auto eventDeadChannels = mFetToDead.process(handler.getMerged());
129  deadChannels.insert(deadChannels.end(), eventDeadChannels.begin(), eventDeadChannels.end());
130  }
131  return deadChannels;
132  }
133 };

135 of::DataProcessorSpec getFetToDeadSpec(const FEEIdConfig& feeIdConfig, const CrateMasks& crateMasks)
136 {
137  std::vector<of::InputSpec> inputSpecs;
138  inputSpecs.emplace_back("mid_data", of::ConcreteDataTypeMatcher(header::gDataOriginMID, "DATA"), of::Lifetime::Timeframe);
139  inputSpecs.emplace_back("mid_data_rof", of::ConcreteDataTypeMatcher(header::gDataOriginMID, "DATAROF"), of::Lifetime::Timeframe);
140 
141  std::vector<of::OutputSpec> outputSpecs;
142  outputSpecs.emplace_back(header::gDataOriginMID, "NOISE", 0);
143  outputSpecs.emplace_back(header::gDataOriginMID, "NOISEROF", 0);
144  outputSpecs.emplace_back(header::gDataOriginMID, "DEAD", 0);
145  outputSpecs.emplace_back(header::gDataOriginMID, "DEADROF", 0);
146 
147  return of::DataProcessorSpec{
148  "MIDFetToDead",
149  {inputSpecs},
150  {outputSpecs},
151  of::AlgorithmSpec{of::adaptFromTask<o2::mid::FetToDeadDeviceDPL>(feeIdConfig, crateMasks)},
152  of::Options{}};
