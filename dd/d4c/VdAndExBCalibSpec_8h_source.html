38 class VdAndExBCalibDevice : public o2::framework::Task
39 {
40  public:
41  VdAndExBCalibDevice(std::shared_ptr<o2::base::GRPGeomRequest> req) : mCCDBRequest(req) {}
42  void init(o2::framework::InitContext& ic) final
43  {
44  o2::base::GRPGeomHelper::instance().setRequest(mCCDBRequest);
45  int minEnt = ic.options().get<int>("min-entries");
46  auto slotL = ic.options().get<uint32_t>("sec-per-slot");
47  auto delay = ic.options().get<uint32_t>("max-delay");
48  mCalibrator = std::make_unique<o2::trd::CalibratorVdExB>(minEnt);
49  mCalibrator->setSlotLengthInSeconds(slotL);
50  mCalibrator->setMaxSlotsDelay(delay);
51  }
52 
53  void finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj) final
54  {
55  o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj);
56  }
57 
58  void run(o2::framework::ProcessingContext& pc) final
59  {
60  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
61  auto data = pc.inputs().get<o2::trd::AngularResidHistos>("input");
62  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mCalibrator->getCurrentTFInfo());
63  LOG(info) << "Processing TF " << mCalibrator->getCurrentTFInfo().tfCounter << " with " << data.getNEntries() << " AngularResidHistos entries";
64  mCalibrator->process(data);
65  sendOutput(pc.outputs());
66  }
67 
68  void endOfStream(o2::framework::EndOfStreamContext& ec) final
69  {
70  LOG(info) << "Finalizing calibration";
71  mCalibrator->checkSlotsToFinalize(o2::calibration::INFINITE_TF);
72  sendOutput(ec.outputs());
73  }
74 
75  private:
76  std::unique_ptr<o2::trd::CalibratorVdExB> mCalibrator;
77  std::shared_ptr<o2::base::GRPGeomRequest> mCCDBRequest;
78  //________________________________________________________________
79  void sendOutput(DataAllocator& output)
80  {
81  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
82  // TODO in principle, this routine is generic, can be moved to Utils.h
83 
84  using clbUtils = o2::calibration::Utils;
85  const auto& payloadVec = mCalibrator->getCcdbObjectVector();
86  auto& infoVec = mCalibrator->getCcdbObjectInfoVector(); // use non-const version as we update it
87  assert(payloadVec.size() == infoVec.size());
88 
89  for (uint32_t i = 0; i < payloadVec.size(); i++) {
90  auto& w = infoVec[i];
91  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
92  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
93  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
94 
95  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "VDRIFTEXB", i}, *image.get()); // vector<char>
96  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "VDRIFTEXB", i}, w); // root-serialized
97  }
98  if (payloadVec.size()) {
99  mCalibrator->initOutput(); // reset the outputs once they are already sent
100  }
101  }
102 };

109 DataProcessorSpec getTRDVdAndExBCalibSpec()
110 {
111  using device = o2::calibration::VdAndExBCalibDevice;
112  using clbUtils = o2::calibration::Utils;
113 
114  std::vector<OutputSpec> outputs;
115  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "VDRIFTEXB"}, Lifetime::Sporadic);
116  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "VDRIFTEXB"}, Lifetime::Sporadic);
117  std::vector<InputSpec> inputs{{"input", "TRD", "ANGRESHISTS"}};
118  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
119  true, // GRPECS=true
120  false, // GRPLHCIF
121  false, // GRPMagField
122  false, // askMatLUT
123  o2::base::GRPGeomRequest::None, // geometry
124  inputs);
125  return DataProcessorSpec{
126  "calib-vdexb-calibration",
127  inputs,
128  outputs,
129  AlgorithmSpec{adaptFromTask<device>(ccdbRequest)},
130  Options{
131  {"sec-per-slot", VariantType::UInt32, 900u, {"number of seconds per calibration time slot"}},
132  {"max-delay", VariantType::UInt32, 2u, {"number of slots in past to consider"}},
133  {"min-entries", VariantType::Int, 40'000, {"minimum number of entries to fit single time slot"}}}}; // around 3 entries per bin per chamber
134 }
