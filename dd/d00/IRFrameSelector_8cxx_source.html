24 long IRFrameSelector::check(const o2::dataformats::IRFrame& fr)
25 {
26  long ans = -1;
27 
28  // find entry which overlaps or above fr
29  auto fullcheck = [&fr, this]() -> long {
30  auto lower = std::lower_bound(this->mFrames.begin(), this->mFrames.end(), fr);
31  if (lower == this->mFrames.end()) {
32  this->mLastBoundID = this->mFrames.size();
33  } else {
34  this->mLastBoundID = std::distance(this->mFrames.begin(), lower);
35  return fr.isOutside(*lower) == o2::dataformats::IRFrame::Inside ? this->mLastBoundID : -1;
36  }
37  return -1;
38  };
39 
40  if (mLastBoundID < 0) { // 1st call, do full search
41  ans = fullcheck();
42  } else { // we assume that the new query is in the vicinity of the previous one
43  int steps = 0;
44  constexpr int MaxSteps = 3;
45  if (fr.getMin() >= mLastIRFrameChecked.getMin()) { // check entries including and following mLastBoundID
46  for (; mLastBoundID < (long)mFrames.size(); mLastBoundID++) {
47  auto res = fr.isOutside(mFrames[mLastBoundID]);
48  if (res == o2::dataformats::IRFrame::Above) {
49  break; // no point in checking further
50  } else if (res == o2::dataformats::IRFrame::Inside) {
51  ans = mLastBoundID;
52  break;
53  }
54  if (++steps == MaxSteps) {
55  ans = fullcheck();
56  break;
57  }
58  }
59  } else { // check entries preceding mLastBoundID
60  if (mLastBoundID >= (long)mFrames.size()) {
61  mLastBoundID = (long)mFrames.size() - 1;
62  }
63  for (; mLastBoundID >= 0; mLastBoundID--) {
64  auto res = fr.isOutside(mFrames[mLastBoundID]);
65  if (res == o2::dataformats::IRFrame::Below) {
66  mLastBoundID++; // no point in checking further
67  break;
68  } else if (res == o2::dataformats::IRFrame::Inside) {
69  ans = mLastBoundID;
70  break;
71  }
72  if (++steps == MaxSteps) {
73  ans = fullcheck();
74  break;
75  }
76  }
77  if (mLastBoundID < 0) {
78  mLastBoundID = 0;
79  }
80  }
81  }
82  mLastIRFrameChecked = fr;
83  return ans;
84 }

86 size_t IRFrameSelector::loadIRFrames(const std::string& fname)
87 {
88  // read IRFrames to filter from the file
89  std::unique_ptr<TFile> tfl(TFile::Open(fname.c_str()));
90  if (!tfl || tfl->IsZombie()) {
91  LOGP(fatal, "Cannot open file {}", fname);
92  }
93  auto klst = gDirectory->GetListOfKeys();
94  TIter nextkey(klst);
95  TKey* key;
96  std::string clVec{TClass::GetClass("std::vector<o2::dataformats::IRFrame>")->GetName()};
97  bool done = false;
98  while ((key = (TKey*)nextkey())) {
99  std::string kcl(key->GetClassName());
100  if (kcl == clVec) {
101  auto* v = (std::vector<o2::dataformats::IRFrame>*)tfl->GetObjectUnchecked(key->GetName());
102  if (!v) {
103  LOGP(fatal, "Failed to extract vector {} from {}", key->GetName(), fname);
104  }
105  mOwnList.insert(mOwnList.end(), v->begin(), v->end());
106  LOGP(info, "Loaded {} IRFrames from vector {} of {}", mOwnList.size(), key->GetName(), fname);
107  done = true;
108  break;
109  } else if (kcl == "TTree") {
110  std::unique_ptr<TTree> tr((TTree*)tfl->Get(key->GetName()));
111  if (!tr->GetBranch("IRFrames")) {
112  LOGP(fatal, "Did not find branch IRFrames in the tree {}", key->GetName());
113  }
114  std::vector<o2::dataformats::IRFrame> tmpv, *tpmvPtr = &tmpv;
115  tr->SetBranchAddress("IRFrames", &tpmvPtr);
116  for (int i = 0; i < (int)tr->GetEntries(); i++) {
117  tr->GetEntry(i);
118  mOwnList.insert(mOwnList.end(), tmpv.begin(), tmpv.end());
119  }
120  done = true;
121  LOGP(info, "Loaded {} IRFrames from tree {} of {}", mOwnList.size(), key->GetName(), fname);
122  break;
123  }
124  }
125  if (!true) {
126  LOGP(fatal, "Did not find neither tree nor vector of IRFrames in {}", fname);
127  }
128  setSelectedIRFrames(mOwnList);
129  return mOwnList.size();
130 }
