71 void EventManager::displayCurrentEvent()
72 {
73  const auto multiView = MultiView::getInstance();
74  const auto dataSource = getDataSource();
75  if (dataSource->getEventCount() > 0) {
76  if (!vizSettings.firstEvent) {
77  saveVisualisationSettings();
78  }
79 
80  multiView->destroyAllEvents();
81  int no = dataSource->getCurrentEvent();
82 
83  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
84  dataTypeLists[i] = new TEveElementList(gDataTypeNames[i].c_str());
85  }
86 
87  VisualisationEvent event; // collect calorimeters in one drawing step
88  auto displayList = dataSource->getVisualisationList(no, EventManagerFrame::getInstance().getMinTimeFrameSliderValue(), EventManagerFrame::getInstance().getMaxTimeFrameSliderValue(), EventManagerFrame::MaxRange);
89  for (auto it = displayList.begin(); it != displayList.end(); ++it) {
90  if (it->second == EVisualisationGroup::EMC || it->second == EVisualisationGroup::PHS) {
91  displayCalorimeters(it->first, gVisualisationGroupName[it->second]);
92  } else {
93  displayVisualisationEvent(it->first, gVisualisationGroupName[it->second]);
94  }
95  }
96 
97  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
98  multiView->registerElement(dataTypeLists[i]);
99  }
100 
101  if (vizSettings.firstEvent) {
102  saveVisualisationSettings();
103  vizSettings.firstEvent = false;
104  } else {
105  restoreVisualisationSettings();
106  }
107 
108  multiView->getAnnotationTop()->SetText(TString::Format("Run %d\n%s", dataSource->getRunNumber(), dataSource->getCollisionTime().c_str()));
109  auto detectors = detectors::DetID::getNames(dataSource->getDetectorsMask());
110  multiView->getAnnotationBottom()->SetText(TString::Format("TFOrbit: %d\nDetectors: %s", dataSource->getFirstTForbit(), detectors.c_str()));
111  }
112  multiView->redraw3D();
113 }

188 void EventManager::displayVisualisationEvent(VisualisationEvent& event, const std::string& detectorName)
189 {
190  double eta = 0.1;
191  size_t trackCount = event.getTrackCount();
192  LOG(info) << "displayVisualisationEvent: " << trackCount << " detector: " << detectorName;
193  // tracks
194  auto* list = new TEveTrackList(detectorName.c_str());
195  list->IncDenyDestroy();
196  // clusters
197  size_t clusterCount = 0;
198  auto* point_list = new TEvePointSet(detectorName.c_str());
199  point_list->IncDenyDestroy(); // don't delete if zero parent
200  point_list->SetMarkerColor(kBlue);
201 
202  for (size_t i = 0; i < trackCount; ++i) {
203  VisualisationTrack track = event.getTrack(i);
204  TEveRecTrackD t;
205  t.fSign = track.getCharge() > 0 ? 1 : -1;
206  auto* vistrack = new TEveTrack(&t, &TEveTrackPropagator::fgDefault);
207  vistrack->SetLineColor(kMagenta);
208  vistrack->SetName(track.getGIDAsString().c_str());
209  size_t pointCount = track.getPointCount();
210  vistrack->Reset(pointCount);
211 
212  int points = 0;
213  for (size_t j = 0; j < pointCount; ++j) {
214  auto point = track.getPoint(j);
215  if (point[2] > eta || point[2] < -1 * eta) {
216  vistrack->SetNextPoint(point[0], point[1], point[2]);
217  points++;
218  }
219  }
220  if (points > 0) {
221  list->AddElement(vistrack);
222  }
223 
224  // clusters connected with track
225  for (size_t i = 0; i < track.getClusterCount(); ++i) {
226  VisualisationCluster cluster = track.getCluster(i);
227  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
228  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
229  clusterCount++;
230  }
231  }
232  }
233 
234  if (trackCount != 0) {
235  dataTypeLists[EVisualisationDataType::Tracks]->AddElement(list);
236  }
237 
238  // global clusters (with no connection information)
239  for (size_t i = 0; i < event.getClusterCount(); ++i) {
240  VisualisationCluster cluster = event.getCluster(i);
241  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
242  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
243  clusterCount++;
244  }
245  }
246 
247  if (clusterCount != 0) {
248  dataTypeLists[EVisualisationDataType::Clusters]->AddElement(point_list);
249  }
250 
251  LOG(info) << "tracks: " << trackCount << " detector: " << detectorName << ":" << dataTypeLists[EVisualisationDataType::Tracks]->NumChildren();
252  LOG(info) << "clusters: " << clusterCount << " detector: " << detectorName << ":" << dataTypeLists[EVisualisationDataType::Clusters]->NumChildren();
253 }

255 void EventManager::displayCalorimeters(VisualisationEvent& event, const std::string& detectorName)
256 {
257  int size = event.getCaloCount();
258  if (size > 0) {
259  const std::string detector = detectorName == "EMC" ? "emcal" : "phos";
260  struct CaloInfo {
261  std::string name;
262  std::string configColor;
263  int defaultColor;
264  std::string configSizeEta;
265  float sizeEta;
266  std::string configSizePhi;
267  float sizePhi;
268  std::string configNoise;
269  float defaultNoise;
270  std::string configTransparency;
271  int defaultTransparency;
272  std::string configBarrelRadius;
273  int defaultBarrelRadius;
274  double scale;
275  };
276  TEnv settings;
277  ConfigurationManager::getInstance().getConfig(settings);
278 
279  // TODO: calculate values based on info available in O2
280  static const std::unordered_map<o2::dataformats::GlobalTrackID::Source, CaloInfo> caloInfos =
281  {
282  {o2::dataformats::GlobalTrackID::EMC, {"emcal", "emcal.tower.color", kYellow, "emcal.tower.size.eta", 0.0143, "emcal.tower.size.phi", 0.0143, "emcal.tower.noise", 0, "emcal.tower.transparency", 101, "emcal.barel.radius", 500, settings.GetValue("emcal.tower.scale", 1.0)}},
283  {o2::dataformats::GlobalTrackID::PHS, {"phos", "phos.tower.color", kYellow, "phos.tower.size.eta", 0.0046, "phos.tower.size.eta", 0.00478, "phos.tower.noise", 200, "phos.tower.transparency", 101, "phos.barel.radius", 550, settings.GetValue("phos.tower.scale", 1.0)}},
284  };
285 
286  auto data = new TEveCaloDataVec(1);
287  data->IncDenyDestroy();
288  auto key = detectorName == "EMC" ? o2::dataformats::GlobalTrackID::EMC : o2::dataformats::GlobalTrackID::PHS;
289 
290  const CaloInfo& info = caloInfos.at(key);
291  data->RefSliceInfo(0).Setup(info.name.c_str(),
292  settings.GetValue(info.configNoise.c_str(), info.defaultNoise),
293  settings.GetValue(info.configColor.c_str(), info.defaultColor),
294  settings.GetValue(info.configTransparency.c_str(), info.defaultTransparency));
295 
296  const auto dEta = settings.GetValue(info.configSizeEta.c_str(), info.sizeEta) / 2.0;
297  const auto dPhi = settings.GetValue(info.configSizePhi.c_str(), info.sizePhi) / 2.0;
298  const float barrelRadius = settings.GetValue(info.configBarrelRadius.c_str(), info.defaultBarrelRadius);
299 
300  for (const auto& calo : event.getCalorimetersSpan()) {
301  data->AddTower(calo.getEta() - dEta, calo.getEta() + dEta, calo.getPhi() - dPhi, calo.getPhi() + dPhi);
302  data->FillSlice(0, info.scale * calo.getEnergy());
303  }
304 
305  data->DataChanged();
306  data->SetAxisFromBins();
307 
308  auto calo3d = new TEveCalo3D(data);
309  calo3d->SetName(detectorName.c_str());
310 
311  calo3d->SetBarrelRadius(barrelRadius);
312  calo3d->SetEndCapPos(barrelRadius);
313  calo3d->SetRnrFrame(false, false); // do not draw barrel grid
314 
315  dataTypeLists[EVisualisationDataType::Calorimeters]->AddElement(calo3d);
316  }
317 }

322
