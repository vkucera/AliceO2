40 static void BM_EventMixingTraditional(benchmark::State& state)
41 {
42  // Seed with a real random value, if available
43  std::default_random_engine e1(1234567891);
44  std::uniform_real_distribution<float> uniform_dist(0.f, 1.f);
45  std::uniform_real_distribution<float> uniform_dist_x(-0.065f, 0.073f);
46  std::uniform_real_distribution<float> uniform_dist_y(-0.320f, 0.360f);
47  std::uniform_int_distribution<int> uniform_dist_int(0, 5);
48 
49  std::vector<double> xBins{VARIABLE_WIDTH, -0.064, -0.062, -0.060, 0.066, 0.068, 0.070, 0.072};
50  std::vector<double> yBins{VARIABLE_WIDTH, -0.320, -0.301, -0.300, 0.330, 0.340, 0.350, 0.360};
51  using BinningType = BinningPolicy<o2::aod::collision::PosX, o2::aod::collision::PosY>;
52  BinningType binningOnPositions{{xBins, yBins}, true}; // true is for 'ignore overflows' (true by default)
53 
54  TableBuilder colBuilder, trackBuilder;
55  auto rowWriterCol = colBuilder.cursor<o2::aod::Collisions>();
56  for (auto i = 0; i < state.range(0); ++i) {
57  float x = uniform_dist_x(e1);
58  float y = uniform_dist_y(e1);
59  rowWriterCol(0, uniform_dist_int(e1),
60  x, y, uniform_dist(e1),
61  uniform_dist(e1), uniform_dist(e1), uniform_dist(e1),
62  uniform_dist(e1), uniform_dist(e1), uniform_dist(e1),
63  uniform_dist_int(e1), uniform_dist(e1),
64  uniform_dist_int(e1),
65  uniform_dist(e1), uniform_dist(e1));
66  }
67  auto tableCol = colBuilder.finalize();
68  o2::aod::Collisions collisions{tableCol};
69  std::uniform_int_distribution<int> uniform_dist_col_ind(0, collisions.size());
70 
71  auto rowWriterTrack = trackBuilder.cursor<o2::aod::StoredTracks>();
72  for (auto i = 0; i < numTracksPerEvent * state.range(0); ++i) {
73  rowWriterTrack(0, uniform_dist_col_ind(e1), 0,
74  uniform_dist(e1), uniform_dist(e1), uniform_dist(e1),
75  uniform_dist(e1), uniform_dist(e1), uniform_dist(e1),
76  uniform_dist(e1));
77  }
78  auto tableTrack = trackBuilder.finalize();
79  o2::aod::StoredTracks tracks{tableTrack};
80 
81  int64_t count = 0;
82  int64_t colCount = 0;
83  int nBinsTot = (xBins.size() - 2) * (yBins.size() - 2);
84 
85  for (auto _ : state) {
86  count = 0;
87  colCount = 0;
88  int n = state.range(0);
89  std::vector<std::list<o2::aod::Collisions::iterator>> mixingBufferVector;
90  for (int i = 0; i < nBinsTot; i++) {
91  mixingBufferVector.push_back(std::list<o2::aod::Collisions::iterator>());
92  }
93  for (auto& col1 : collisions) {
94  int bin = binningOnPositions.getBin({col1.posX(), col1.posY()});
95  if (bin == -1) {
96  continue;
97  }
98 
99  auto& mixingBuffer = mixingBufferVector[bin];
100 
101  if (mixingBuffer.size() > 0) {
102  auto tracks1 = tracks.sliceByCached(o2::aod::track::collisionId, col1.globalIndex());
103  for (auto& col2 : mixingBuffer) {
104  auto tracks2 = tracks.sliceByCached(o2::aod::track::collisionId, col2.globalIndex());
105  for (auto& [t1, t2] : combinations(CombinationsFullIndexPolicy(tracks1, tracks2))) {
106  count++;
107  }
108  colCount++;
109  }
110  if (mixingBuffer.size() >= numEventsToMix) {
111  mixingBuffer.pop_back();
112  }
113  }
114  mixingBuffer.push_front(col1);
115  }
116 
117  benchmark::DoNotOptimize(count);
118  benchmark::DoNotOptimize(colCount);
119  }
120  state.counters["Mixed track pairs"] = count;
121  state.counters["Mixed collision pairs"] = colCount;
122  state.SetBytesProcessed(state.iterations() * sizeof(float) * count);
123 }
