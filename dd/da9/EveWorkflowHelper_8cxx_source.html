55 void EveWorkflowHelper::selectTracks(const CalibObjectsConst* calib,
56  GID::mask_t maskCl, GID::mask_t maskTrk, GID::mask_t maskMatch)
57 {
58  std::vector<Bracket> itsROFBrackets;
59 
60  if (mEnabledFilters.test(Filter::ITSROF)) {
61  const auto irFrames = getRecoContainer().getIRFramesITS();
62 
63  static int BCDiffErrCount = 0;
64  constexpr int MAXBCDiffErrCount = 5;
65 
66  auto bcDiffToTFTimeMUS = [startIR = getRecoContainer().startIR](const o2::InteractionRecord& ir) {
67  auto bcd = ir.differenceInBC(startIR);
68  if (uint64_t(bcd) > o2::constants::lhc::LHCMaxBunches * 256 && BCDiffErrCount < MAXBCDiffErrCount) {
69  LOGP(alarm, "ATTENTION: wrong bunches diff. {} for current IR {} wrt 1st TF orbit {}", bcd, ir, startIR);
70  BCDiffErrCount++;
71  }
72  return bcd * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
73  };
74 
75  for (const auto& irFrame : irFrames) {
76  itsROFBrackets.emplace_back(bcDiffToTFTimeMUS(irFrame.getMin()), bcDiffToTFTimeMUS(irFrame.getMax()));
77  }
78  }
79 
80  auto correctTrackTime = [this](auto& _tr, float t0, float terr) {
81  if constexpr (isTPCTrack<decltype(_tr)>()) {
82  // unconstrained TPC track, with t0 = TrackTPC.getTime0+0.5*(DeltaFwd-DeltaBwd) and terr = 0.5*(DeltaFwd+DeltaBwd) in TimeBins
83  t0 *= this->mTPCBin2MUS;
84  terr *= this->mTPCBin2MUS;
85  } else if constexpr (isITSTrack<decltype(_tr)>()) {
86  t0 += 0.5f * this->mITSROFrameLengthMUS; // ITS time is supplied in \mus as beginning of ROF
87  terr *= this->mITSROFrameLengthMUS; // error is supplied as a half-ROF duration, convert to \mus
88  } else if constexpr (isMFTTrack<decltype(_tr)>()) { // Same for MFT
89  t0 += 0.5f * this->mMFTROFrameLengthMUS;
90  terr *= this->mMFTROFrameLengthMUS;
91  } else if constexpr (isGlobalFwdTrack<decltype(_tr)>()) {
92  t0 = _tr.getTimeMUS().getTimeStamp();
93  terr = _tr.getTimeMUS().getTimeStampError() * mPVParams->nSigmaTimeTrack; // gaussian errors must be scaled by requested n-sigma
94  } else {
95  terr *= mPVParams->nSigmaTimeTrack; // gaussian errors must be scaled by requested n-sigma
96  }
97  // for all other tracks the time is in \mus with gaussian error
98  terr += mPVParams->timeMarginTrackTime;
99 
100  return Bracket{t0 - terr, t0 + terr};
101  };
102 
103  auto isInsideITSROF = [&itsROFBrackets](const Bracket& br) {
104  for (const auto& ir : itsROFBrackets) {
105  const auto overlap = ir.getOverlap(br);
106 
107  if (overlap.isValid()) {
108  return true;
109  }
110  }
111 
112  return false;
113  };
114 
115  auto flagTime = [](float time, GID::Src_t src) {
116  auto flag = static_cast<int>(time) + TIME_OFFSET;
117 
118  // if it's a tracklet, give it a lower priority in time sort by setting the highest bit
119  if (src <= GID::MID) {
120  flag |= (1 << 31);
121  }
122 
123  return flag;
124  };
125 
126  auto creator = [maskTrk, this, &correctTrackTime, &isInsideITSROF, &flagTime](auto& trk, GID gid, float time, float terr) {
127  const auto src = gid.getSource();
128  mTotalTracks[src]++;
129 
130  if (!maskTrk[src]) {
131  return true;
132  }
133 
134  auto bracket = correctTrackTime(trk, time, terr);
135 
136  if (mEnabledFilters.test(Filter::TimeBracket) && mTimeBracket.getOverlap(bracket).isInvalid()) {
137  return true;
138  }
139 
140  if (mEnabledFilters.test(Filter::ITSROF) && !isInsideITSROF(bracket)) {
141  return true;
142  }
143 
144  mGIDTrackTime[gid] = flagTime(bracket.mean(), src);
145 
146  // If the mode is disabled,
147  // add every track to a symbolic "zero" primary vertex
148  if (!mPrimaryVertexMode) {
149  mPrimaryVertexGIDs[0].push_back(gid);
150  }
151 
152  return true;
153  };
154 
155  this->mRecoCont.createTracksVariadic(creator);
156 
157  if (mPrimaryVertexMode) {
158  const auto trackIndex = mRecoCont.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
159  const auto vtxRefs = mRecoCont.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
160  mTotalPrimaryVertices = vtxRefs.size() - 1; // The last entry is for unassigned tracks, ignore them
161 
162  for (std::size_t iv = 0; iv < mTotalPrimaryVertices; iv++) {
163  const auto& vtref = vtxRefs[iv];
164  int it = vtref.getFirstEntry(), itLim = it + vtref.getEntries();
165  for (; it < itLim; it++) {
166  const auto tvid = trackIndex[it];
167 
168  if (!mRecoCont.isTrackSourceLoaded(tvid.getSource())) {
169  continue;
170  }
171 
172  // TODO: fix TPC tracks?
173 
174  // If a track was not rejected, associate it with its primary vertex
175  if (mGIDTrackTime.find(tvid) != mGIDTrackTime.end()) {
176  mPrimaryVertexGIDs[iv].push_back(tvid);
177  }
178  }
179  }
180  }
181 }

183 void EveWorkflowHelper::draw(std::size_t primaryVertexIdx, bool sortTracks)
184 {
185  auto unflagTime = [](unsigned int time) {
186  return static_cast<float>(static_cast<int>(time & ~(1 << 31)) - TIME_OFFSET);
187  };
188 
189  auto& tracks = mPrimaryVertexGIDs.at(primaryVertexIdx);
190 
191  if (sortTracks) {
192  std::sort(tracks.begin(), tracks.end(),
193  [&](const GID& a, const GID& b) {
194  return mGIDTrackTime.at(a) > mGIDTrackTime.at(b);
195  });
196  }
197 
198  this->drawPHOS();
199  this->drawEMCAL();
200 
201  auto trackCount = tracks.size();
202 
203  if (mEnabledFilters.test(Filter::TotalNTracks) && trackCount >= mMaxNTracks) {
204  trackCount = mMaxNTracks;
205  }
206 
207  for (size_t it = 0; it < trackCount; it++) {
208  const auto& gid = tracks[it];
209  auto tim = unflagTime(mGIDTrackTime.at(gid));
210  mTotalAcceptedTracks.insert(gid);
211  // LOG(info) << "EveWorkflowHelper::draw " << gid.asString();
212  switch (gid.getSource()) {
213  case GID::ITS:
214  drawITS(gid, tim);
215  break;
216  case GID::TPC:
217  drawTPC(gid, tim);
218  break;
219  case GID::MFT:
220  drawMFT(gid, tim);
221  break;
222  case GID::MCH:
223  drawMCH(gid, tim);
224  break;
225  case GID::MID:
226  drawMID(gid, tim);
227  break;
228  case GID::ITSTPC:
229  drawITSTPC(gid, tim);
230  break;
231  case GID::TPCTOF:
232  drawTPCTOF(gid, tim);
233  break;
234  case GID::TPCTRD:
235  drawTPCTRD(gid, tim);
236  break;
237  case GID::ITSTPCTRD:
238  drawITSTPCTRD(gid, tim);
239  break;
240  case GID::ITSTPCTOF:
241  drawITSTPCTOF(gid, tim);
242  break;
243  case GID::TPCTRDTOF:
244  drawTPCTRDTOF(gid, tim);
245  break;
246  case GID::ITSTPCTRDTOF:
247  drawITSTPCTRDTOF(gid, tim);
248  break;
249  default:
250  LOG(info) << "Track type " << gid.getSource() << " not handled";
251  }
252  }

275 std::vector<PNT> EveWorkflowHelper::getTrackPoints(const o2::track::TrackPar& trc, float minR, float maxR, float maxStep, float minZ, float maxZ)
276 {
277  // adjust minR according to real track start from track starting point
278  auto maxR2 = maxR * maxR;
279  float rMin = std::sqrt(trc.getX() * trc.getX() + trc.getY() * trc.getY());
280  if (rMin > minR) {
281  minR = rMin;
282  }
283  // prepare space points from the track param
284  std::vector<PNT> pnts;
285  int nSteps = std::max(2, int((maxR - minR) / maxStep));
286  const auto prop = o2::base::Propagator::Instance();
287  float xMin = trc.getX(), xMax = maxR * maxR - trc.getY() * trc.getY();
288  if (xMax > 0) {
289  xMax = std::sqrt(xMax);
290  }
291 
292  float dx = (xMax - xMin) / nSteps;
293  auto tp = trc;
294  float dxmin = std::abs(xMin - tp.getX()), dxmax = std::abs(xMax - tp.getX());
295 
296  if (dxmin > dxmax) { // start from closest end
297  std::swap(xMin, xMax);
298  dx = -dx;
299  }
300  if (!prop->propagateTo(tp, xMin, false, 0.99, maxStep, o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
301  return pnts;
302  }
303  auto xyz = tp.getXYZGlo();
304  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
305  for (int is = 0; is < nSteps; is++) {
306  if (!prop->propagateTo(tp, tp.getX() + dx, false, 0.99, 999., o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
307  return pnts;
308  }
309  xyz = tp.getXYZGlo();
310  if (xyz.Z() < minZ) {
311  return pnts;
312  }
313  if (xyz.Z() > maxZ) {
314  return pnts;
315  }
316  if (xyz.X() * xyz.X() + xyz.Y() * xyz.Y() > maxR2) {
317  return pnts;
318  }
319  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
320  }
321  return pnts;
322 }

324 void EveWorkflowHelper::addTrackToEvent(const o2::track::TrackPar& tr, GID gid, float trackTime, float dz, GID::Source source, float maxStep)
325 {
326  if (source == GID::NSources) {
327  source = (o2::dataformats::GlobalTrackID::Source)gid.getSource();
328  }
329  auto vTrack = mEvent.addTrack({.time = trackTime,
330  .charge = tr.getCharge(),
331  .PID = tr.getPID(),
332  .startXYZ = {tr.getX(), tr.getY(), tr.getZ()},
333  .phi = tr.getPhi(),
334  .theta = tr.getTheta(),
335  .eta = tr.getEta(),
336  .gid = gid.asString(),
337  .source = source});
338 
339  const auto it = propagationRanges.find(source);
340 
341  const bool rangeNotFound = (it == propagationRanges.cend());
342  if (rangeNotFound) {
343  LOGF(error, "Track source %s has no defined propagation ranges");
344  return;
345  }
346 
347  const auto& prange = it->second;
348 
349  auto pnts = getTrackPoints(tr, prange.minR, prange.maxR, maxStep, prange.minZ, prange.maxZ);
350 
351  for (size_t ip = 0; ip < pnts.size(); ip++) {
352  vTrack->addPolyPoint(pnts[ip][0], pnts[ip][1], pnts[ip][2] + dz);
353  }
354 }
