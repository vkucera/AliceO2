25 bool Decoder::process(const char* data, size_t size)
26 {
27  const auto startTime = HighResClock::now();
28  assert(size == sizeof(sac::packet));
29  auto& sac = *(sac::packet*)data;
30  const auto instance = sac.getInstance();
31  if (instance >= Instances) {
32  return true;
33  }
34 
35  if (mDebugLevel & (uint32_t)DebugFlags::PacketInfo) {
36  printPacketInfo(sac);
37  }
38 
39  const auto packetInstance = sac.header.pktCount;
40  const auto packetFE = sac.data.pktNumber;
41  const bool isOK = sac.check();
42  const auto dataWords = sac.getDataWords();
43  const auto feIndex = sac.getFEIndex();
44  auto& lastPacketInstance = mPktCountInstance[instance];
45  auto& lastPacketFE = mPktCountFEs[feIndex];
46 
47  // check packet counters are increasing by one
48  //
49  if (lastPacketInstance && (packetInstance != (lastPacketInstance + 1))) {
50  LOGP(error, "Packet for instance {} missing, last packet {}, this packet {}", instance, lastPacketInstance, packetInstance);
51  }
52 
53  if (lastPacketFE && (packetFE != (lastPacketFE + 1))) {
54  LOGP(error, "Packet for frontend {} missing, last packet {}, this packet {}", feIndex, lastPacketFE, packetFE);
55  }
56  lastPacketInstance = packetInstance;
57  lastPacketFE = packetFE;
58 
59  if (isOK) {
60  auto& dataStrings = mDataStrings[feIndex];
61  dataStrings.insert(dataStrings.end(), dataWords.begin(), dataWords.end());
62  } else {
63  LOGP(error, "Problem in SAC data found, header check: {}, data check: {}", sac.header.check(), sac.data.check());
64  }
65 
66  if (mDebugLevel & (uint32_t)DebugFlags::TimingInfo) {
67  auto endTime = HighResClock::now();
68  auto elapsed_seconds = std::chrono::duration_cast<std::chrono::duration<double>>(endTime - startTime);
69  LOGP(info, "Time to process data of size {}: {} s", size, elapsed_seconds.count());
70  }
71 
72  ++mCollectedDataPackets;
73  return isOK;
74 }

76 bool Decoder::decodeChannels(DecodedDataFE& sacs, size_t& carry, int feid)
77 {
78  const auto& data = mDataStrings[feid];
79  const size_t dataSize = data.size();
80  const size_t next = std::min(size_t(8 * 8), dataSize - carry);
81  const size_t start = carry;
82  while (carry < dataSize) {
83  if (carry + 8 >= dataSize) {
84  return false;
85  }
86  if (data[carry] >= '0' && data[carry] <= '7') {
87  const uint32_t channel = data[carry] - '0';
88  ++carry;
89  uint32_t value = 0;
90  for (int i = 0; i < 6; ++i) {
91  const auto c = data[carry];
92  uint32_t nibble = 0;
93  if ((c >= '0') && (c <= '9')) {
94  nibble = c - '0';
95  } else if ((c >= 'A') && (c <= 'F')) {
96  nibble = c - 'A' + 10;
97  }
98  value <<= 4;
99  value |= (nibble & 0xF);
100  ++carry;
101  }
102  int32_t valueSigned = value & 0x00FFFFFF;
103  // negative value?
104  if ((valueSigned >> 23) & 1) {
105  valueSigned |= 0xff000000;
106  }
107  sacs.currents[channel] = valueSigned;
108 
109  if (data[carry] != '\n') {
110  LOGP(warning, "Problem decoding data value for FE {}, channel {} at position {} / {}, dump: {}\n",
111  feid, channel, carry, dataSize, std::string_view(&data[start], next));
112  }
113  ++carry;
114  } else {
115  return true;
116  }
117  }
118  return true;
119 }

121 void Decoder::decode(int feid)
122 {
123  const auto startTime = HighResClock::now();
124  auto& data = mDataStrings[feid];
125  DecodedDataFE decdata;
126  DecodedDataFE decAdditional;
127  bool aligned{false};
128 
129  size_t carry = 0;
130  size_t deletePosition = 0;
131  const size_t dataSize = data.size();
132 
133  // fmt::print("================ Processing feid {:2} with size {} =================\n", feid, data.size());
134  while (carry < dataSize) {
135  if (!aligned) {
136  while (data[carry] != '\n') {
137  if (carry >= dataSize) {
138  break;
139  }
140  ++carry;
141  }
142  ++carry;
143  aligned = true;
144  }
145  // fmt::print("Checking position {} / {}, {}\n", carry, dataSize, std::string_view(&data[carry], std::min(size_t(20), dataSize - carry)));
146 
147  if (data[carry] >= '0' && data[carry] <= '7') {
148  if (!decodeChannels(decdata, carry, feid)) {
149  break;
150  }
151  } else if (data[carry] == 'S') {
152  if (carry + 11 >= dataSize) {
153  break;
154  }
155  const auto streamStart = carry;
156  // time stamp comes after channel data
157  ++carry;
158  std::string_view vd(&data[carry], 8);
159  std::stringstream str(vd.data());
160  str.flags(std::ios_base::hex);
161  uint32_t timeStamp;
162  str >> timeStamp;
163  decdata.timeStamp = timeStamp;
164  decAdditional.timeStamp = timeStamp;
165 
166  carry += 8;
167  if (data[carry] != '\n' || data[carry + 1] != 's') {
168  LOGP(warning, "Problem decoding time stamp for FE ({}) at position {} / {}, dump: {}\n",
169  feid, carry - 8, dataSize, std::string_view(&data[carry - 8], std::min(size_t(20), dataSize - 8 - carry)));
170  break; // TODO: makes sense?
171  } else {
172  deletePosition = carry; // keep \ns to align in next iteration
173  carry += 2;
174 
175  if (mDebugStream && (mDebugLevel & (uint32_t)DebugFlags::StreamSingleFE)) {
176  (*mDebugStream) << "d"
177  << "data=" << decdata
178  << "feid=" << feid
179  << "tscount=" << mTSCountFEs[feid]
180  << "\n";
181  }
182  ++mTSCountFEs[feid].first;
183 
184  // copy decoded data to output
185  const auto refTime = timeStamp / SampleDistance;
186  auto& currentsTime = mDecodedData.data;
187  const auto nSamples = currentsTime.size();
188  const auto firstRefTime = (nSamples > 0) ? currentsTime[0].time : 0;
189  auto& refCount = mTSCountFEs[feid].second;
190  if ((refCount == 0) && (refTime > 0)) {
191  LOGP(info, "Skipping initial data packet {} with time stamp {} for FE {}", mTSCountFEs[feid].first, timeStamp, feid);
192  } else {
193  if (refCount != refTime) {
194  LOGP(warning, "Unexpected time stamp in FEid {}. Count {} != TS {} ({}), dump: {}", feid, refCount, refTime, timeStamp, std::string_view(&data[streamStart], std::min(size_t(20), dataSize - streamStart - carry)));
195  }
196  ++refCount;
197  if (refTime < firstRefTime) {
198  currentsTime.insert(currentsTime.begin(), firstRefTime - refTime, DataPoint());
199  } else if (nSamples < refTime - firstRefTime + 1) {
200  currentsTime.resize(refTime - firstRefTime + 1);
201  }
202  auto& currentData = currentsTime[refTime - firstRefTime];
203  currentData.time = refTime;
204  auto& currents = currentData.currents;
205  std::copy(decdata.currents.begin(), decdata.currents.end(), &currents[0] + feid * ChannelsPerFE);
206  }
207  }
208 
209  decdata.reset();
210  } else if (const auto pos = mDecodeAdditional.find(data[carry]); (pos != std::string::npos) && mDebugStream) {
211  // in case of debug stream output, decode additionally configured data streams
212  const auto streamStart = carry;
213  const char streamType = data[carry];
214  const char endMarker = streamType + 32;
215  ++carry;
216 
217  if (!decodeChannels(decAdditional, carry, feid)) {
218  break;
219  }
220 
221  if (data[carry] != endMarker) {
222  LOGP(warning, "Problem decoding additional stream '{}' values for FE ({}) at position {} / {}, dump: {}",
223  streamType, feid, carry, dataSize, std::string_view(&data[streamStart], std::min(size_t(20), dataSize - streamStart - carry)));
224  } else {
225  const char treeName[2] = {streamType, '\0'};
226  (*mDebugStream) << treeName
227  << "data=" << decAdditional
228  << "feid=" << feid
229  << "\n";
230  }
231 
232  decAdditional.reset();
233  ++carry;
234  } else if (AllowedAdditionalStreams.find(data[carry]) != std::string_view::npos) {
235  // skip stream if not configured or no debug stream
236  const char streamType = data[carry];
237  const char endMarker = streamType + 32;
238  while (data[carry] != endMarker) {
239  if (carry >= dataSize) {
240  break;
241  }
242  ++carry;
243  }
244  ++carry;
245  } else if (data[carry] >= 'a' && data[carry] <= 'z') {
246  LOGP(info, "Skipping {}", data[carry]);
247  ++carry;
248  decdata.reset();
249  } else {
250  LOGP(error, "Can't interpret position {} / {}, {}, stopping decoding\n", carry, dataSize, std::string_view(&data[carry - 8], std::min(size_t(20), dataSize - 8 - carry)));
251  break;
252  }
253  }
254 
255  // Remove already decoded data
256  data.erase(data.begin(), data.begin() + deletePosition);
257  // LOGP(info, "removing {} characters from stream. Old size {}, new size {}", deletePosition, dataSize, data.size());
258 
259  if (mDebugLevel & (uint32_t)DebugFlags::TimingInfo) {
260  auto endTime = HighResClock::now();
261  auto elapsed_seconds = std::chrono::duration_cast<std::chrono::duration<double>>(endTime - startTime);
262  LOGP(info, "Time to decode feid {}: {} s", feid, elapsed_seconds.count());
263  }
264 }

267 {
268  const auto startTime = HighResClock::now();
269 
270  for (size_t feid = 0; feid < NumberFEs; ++feid) {
271  decode(feid);
272  }
273 
274  if (mDebugLevel & (uint32_t)DebugFlags::TimingInfo) {
275  auto endTime = HighResClock::now();
