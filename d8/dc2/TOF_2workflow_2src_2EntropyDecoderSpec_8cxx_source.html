47 void EntropyDecoderSpec::run(ProcessingContext& pc)
48 {
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51  o2::ctf::CTFIOSize iosize;
52 
53  mCTFCoder.updateTimeDependentParams(pc);
54  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
55 
56  auto& digitheader = pc.outputs().make<DigitHeader>(OutputRef{"digitheader"});
57  auto& digits = pc.outputs().make<std::vector<Digit>>(OutputRef{"digits"});
58  auto& row = pc.outputs().make<std::vector<ReadoutWindowData>>(OutputRef{"row"});
59  auto& patterns = pc.outputs().make<std::vector<uint8_t>>(OutputRef{"patterns"});
60  // auto& diagnostic = pc.outputs().make<o2::tof::Diagnostic>(OutputRef{"diafreq"});
61 
62  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
63  if (buff.size()) {
64  const auto ctfImage = o2::tof::CTF::getImage(buff.data());
65  iosize = mCTFCoder.decode(ctfImage, row, digits, patterns);
66  }
67 
68  // fill diagnostic frequencies
69  mFiller.clearCounts();
70  for (auto digit : digits) {
71  mFiller.addCount(digit.getChannel());
72  }
73  mFiller.setReadoutWindowData(row, patterns);
74  mFiller.fillDiagnosticFrequency();
75  auto diagnostic = mFiller.getDiagnosticFrequency();
76  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(pc.inputs().getFirstValid(true))->creation;
77  diagnostic.setTimeStamp(creationTime / 1000);
78  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIAFREQ", 0, Lifetime::Timeframe}, diagnostic);
79  pc.outputs().snapshot({"ctfrep", 0}, iosize);
80  mTimer.Stop();
81  LOG(info) << "Decoded " << digits.size() << " digits in " << row.size() << " ROF, (" << iosize.asString() << ") in " << mTimer.CpuTime() - cput << " s";
82 }

91 {
92  std::vector<OutputSpec> outputs{
93  OutputSpec{{"digitheader"}, o2::header::gDataOriginTOF, "DIGITHEADER", 0, Lifetime::Timeframe},
94  OutputSpec{{"digits"}, o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe},
95  OutputSpec{{"row"}, o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe},
96  OutputSpec{{"patterns"}, o2::header::gDataOriginTOF, "PATTERNS", 0, Lifetime::Timeframe},
97  OutputSpec{{"diafreq"}, o2::header::gDataOriginTOF, "DIAFREQ", 0, Lifetime::Timeframe},
98  OutputSpec{{"ctfrep"}, o2::header::gDataOriginTOF, "CTFDECREP", 0, Lifetime::Timeframe}};
99 
100  std::vector<InputSpec> inputs;
101  inputs.emplace_back("ctf", "TOF", "CTFDATA", sspec, Lifetime::Timeframe);
102  inputs.emplace_back("ctfdict", "TOF", "CTFDICT", 0, Lifetime::Condition, ccdbParamSpec("TOF/Calib/CTFDictionary"));
103 
104  return DataProcessorSpec{
105  "tof-entropy-decoder",
106  inputs,
107  outputs,
108  AlgorithmSpec{adaptFromTask<EntropyDecoderSpec>(verbosity)},
109  Options{{"ctf-dict", VariantType::String, "ccdb", {"CTF dictionary: empty or ccdb=CCDB, none=no external dictionary otherwise: local filename"}}}};
110 }
