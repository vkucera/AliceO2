60 void AlignableDetectorITS::defineVolumes()
61 {
62  // define ITS volumes
63  //
64  auto geom = o2::its::GeometryTGeo::Instance();
65 
66  AlignableVolume *volITS = nullptr, *volLr = nullptr, *volHB = nullptr, *volSt = nullptr, *volHSt = nullptr, *volMod = nullptr;
67  AlignableSensorITS* sens = nullptr;
68  //
69  std::unordered_map<std::string, AlignableVolume*> sym2vol;
70  addVolume(volITS = new AlignableVolume(geom->composeSymNameITS(), getDetLabel(), mController));
71  sym2vol[volITS->getSymName()] = volITS;
72  //
73  int nonSensCnt = 0;
74  for (int ilr = 0; ilr < geom->getNumberOfLayers(); ilr++) {
75  for (int ihb = 0; ihb < geom->getNumberOfHalfBarrels(); ihb++) {
76  addVolume(volLr = new AlignableVolume(geom->composeSymNameHalfBarrel(ilr, ihb), getNonSensLabel(nonSensCnt++), mController));
77  sym2vol[volLr->getSymName()] = volLr;
78  volLr->setParent(volITS);
79  int nstavesHB = geom->getNumberOfStaves(ilr) / 2;
80  for (int ist = 0; ist < nstavesHB; ist++) {
81  addVolume(volSt = new AlignableVolume(geom->composeSymNameStave(ilr, ihb, ist), getNonSensLabel(nonSensCnt++), mController));
82  sym2vol[volSt->getSymName()] = volSt;
83  volSt->setParent(volLr);
84  for (int ihst = 0; ihst < geom->getNumberOfHalfStaves(ilr); ihst++) {
85  addVolume(volHSt = new AlignableVolume(geom->composeSymNameHalfStave(ilr, ihb, ist, ihst), getNonSensLabel(nonSensCnt++), mController));
86  sym2vol[volHSt->getSymName()] = volHSt;
87  volHSt->setParent(volSt);
88  for (int imd = 0; imd < geom->getNumberOfModules(ilr); imd++) {
89  addVolume(volMod = new AlignableVolume(geom->composeSymNameModule(ilr, ihb, ist, ihst, imd), getNonSensLabel(nonSensCnt++), mController));
90  sym2vol[volMod->getSymName()] = volMod;
91  volMod->setParent(volHSt);
92  } // module
93  } //halfstave
94  } // stave
95  } // layer halfBarrel
96  } // layer
97 
98  for (int ich = 0; ich < geom->getNumberOfChips(); ich++) {
99  int chID = o2::base::GeometryManager::getSensID(mDetID, ich);
100  if (ich != chID) {
101  throw std::runtime_error(fmt::format("mismatch between counter {} and composed {} chip IDs", ich, chID));
102  }
103  addVolume(sens = new AlignableSensorITS(o2::base::GeometryManager::getSymbolicName(mDetID, ich), chID, getSensLabel(chID), mController));
104  int lay = 0, hba, sta = 0, ssta = 0, modd = 0, chip = 0;
105  geom->getChipId(chID, lay, hba, sta, ssta, modd, chip);
106  AlignableVolume* parVol = sym2vol[modd < 0 ? geom->composeSymNameStave(lay, hba, sta) : geom->composeSymNameModule(lay, hba, sta, ssta, modd)];
107  if (!parVol) {
108  throw std::runtime_error(fmt::format("did not find parent for chip {}", chID));
109  }
110  sens->setParent(parVol);
111  }
112  //
113 }

116 int AlignableDetectorITS::processPoints(GIndex gid, bool inv)
117 {
118  // Extract the points corresponding to this detector, recalibrate/realign them to the
119  // level of the "starting point" for the alignment/calibration session.
120  // If inv==true, the track propagates in direction of decreasing tracking X
121  // (i.e. upper leg of cosmic track)
122  //
123  auto algTrack = mController->getAlgTrack();
124  auto recoData = mController->getRecoContainer();
125  const auto tracks = recoData->getITSTracks();
126  if (tracks.empty()) {
127  return -1; // source not loaded?
128  }
129  const auto& track = tracks[gid.getIndex()];
130  const auto& clusIdx = recoData->getITSTracksClusterRefs();
131  // do we want to apply some cuts?
132  int clEntry = track.getFirstClusterEntry();
133  mNPoints = 0;
134  mFirstPoint = algTrack->getNPoints();
135  for (int icl = track.getNumberOfClusters(); icl--;) {
136  const auto& clus = mITSClustersArray[clusIdx[clEntry++]];
137  auto* sensor = getSensor(clus.getSensorID());
138  auto& pnt = algTrack->addDetectorPoint();
139 
140  if (!getUseErrorParam()) {
141  const auto* sysE = sensor->getAddError(); // additional syst error
142  pnt.setYZErrTracking(clus.getSigmaY2() + sysE[0] * sysE[0], clus.getSigmaYZ(), clus.getSigmaZ2() + sysE[1] * sysE[1]);
143  } else { // errors will be calculated just before using the point in the fit, using track info
144  pnt.setYZErrTracking(0., 0., 0.);
145  pnt.setNeedUpdateFromTrack();
146  }
147  pnt.setXYZTracking(clus.getX(), clus.getY(), clus.getZ());
148  pnt.setSensor(sensor);
149  pnt.setAlphaSens(sensor->getAlpTracking());
150  pnt.setXSens(sensor->getXTracking());
151  pnt.setDetID(mDetID);
152  pnt.setSID(sensor->getSID());
153  //
154  pnt.setContainsMeasurement();
155  pnt.init();
156  mNPoints++;
157  }
158  return track.getNumberOfClusters(); // RS
159 }
