92  void run(o2::framework::ProcessingContext& pc) final
93  {
94  if (EMCALCalibParams::Instance().enableTimeProfiling) {
95  timeMeas[0] = std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::high_resolution_clock::now().time_since_epoch()).count();
96  }
97 
98  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
99  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mTimeCalibrator->getCurrentTFInfo());
100 
101  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get(getCellBinding()).header)->startTime;
102 
103  auto data = pc.inputs().get<gsl::span<o2::emcal::Cell>>(getCellBinding());
104 
105  auto InputTriggerRecord = pc.inputs().get<gsl::span<o2::emcal::TriggerRecord>>(getCellTriggerRecordBinding());
106  LOG(debug) << "[EMCALCalibrator - run] Received " << InputTriggerRecord.size() << " Trigger Records, running calibration ...";
107 
108  LOG(debug) << "Processing TF " << tfcounter << " with " << data.size() << " cells";
109 
110  // call process for every event in the trigger record to ensure correct event counting for the calibration.
111  for (const auto& trg : InputTriggerRecord) {
112  if (!trg.getNumberOfObjects()) {
113  continue;
114  }
115  gsl::span<const o2::emcal::Cell> eventData(data.data() + trg.getFirstEntry(), trg.getNumberOfObjects());
116 
117  // fast calibration
118  if (EMCALCalibParams::Instance().enableFastCalib) {
119  LOG(debug) << "fast calib not yet available!";
120  // normal calibration procedure
121  } else {
122  if (isBadChannelCalib) {
123  mBadChannelCalibrator->process(eventData);
124  } else {
125  mTimeCalibrator->process(eventData);
126  }
127  }
128  }
129  if (isBadChannelCalib) {
130  sendOutput<o2::emcal::BadChannelMap>(pc.outputs());
131  } else {
132  sendOutput<o2::emcal::TimeCalibrationParams>(pc.outputs());
133  }
134  if (EMCALCalibParams::Instance().enableTimeProfiling) {
135  timeMeas[1] = std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::high_resolution_clock::now().time_since_epoch()).count();
136  LOG(info) << "end of run function. Time: " << timeMeas[1] - timeMeas[0] << " [ns] for " << InputTriggerRecord.size() << " events";
137  }
138  }

163  template <typename DataOutput>
164  void sendOutput(DataAllocator& output)
165  {
166  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
167  // TODO in principle, this routine is generic, can be moved to Utils.h
168  using clbUtils = o2::calibration::Utils;
169 
170  // we need this to be a vector of bad channel maps or time params, we will probably need to create this?
171  std::vector<DataOutput> payloadVec;
172  std::vector<o2::ccdb::CcdbObjectInfo> infoVec;
173  if constexpr (std::is_same<DataOutput, o2::emcal::TimeCalibrationParams>::value) {
174  payloadVec = mTimeCalibrator->getOutputVector();
175  infoVec = mTimeCalibrator->getInfoVector();
176  } else {
177  payloadVec = mBadChannelCalibrator->getOutputVector();
178  infoVec = mBadChannelCalibrator->getInfoVector();
179  }
180  // use non-const version as we update it
181  assert(payloadVec.size() == infoVec.size());
182  for (uint32_t i = 0; i < payloadVec.size(); i++) {
183  auto& w = infoVec[i];
184  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
185  if constexpr (std::is_same<DataOutput, o2::emcal::TimeCalibrationParams>::value) {
186  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
187  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
188  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "EMC_TIMECALIB", i}, *image.get()); // vector<char>
189  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "EMC_TIMECALIB", i}, w); // root-serialized
190  } else {
191  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
192  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
193  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "EMC_BADCHANNELS", i}, *image.get()); // vector<char>
194  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "EMC_BADCHANNELS", i}, w); // root-serialized
195  }
196  }
197  if (payloadVec.size()) {
198  if constexpr (std::is_same<DataOutput, o2::emcal::TimeCalibrationParams>::value) {
199  mTimeCalibrator->initOutput(); // reset the outputs once they are already sent
200  } else {
201  mBadChannelCalibrator->initOutput(); // reset the outputs once they are already sent
202  }
203  }
204  }
