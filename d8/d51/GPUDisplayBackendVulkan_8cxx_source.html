53 static int checkValidationLayerSupport(const std::vector<const char*>& validationLayers)
54 {
55  std::vector<vk::LayerProperties> availableLayers = vk::enumerateInstanceLayerProperties();
56  for (const char* layerName : validationLayers) {
57  bool layerFound = false;
58 
59  for (const auto& layerProperties : availableLayers) {
60  if (strcmp(layerName, layerProperties.layerName) == 0) {
61  layerFound = true;
62  break;
63  }
64  }
65 
66  if (!layerFound) {
67  return 1;
68  }
69  }
70  return 0;

272 double GPUDisplayBackendVulkan::checkDevice(vk::PhysicalDevice device, const std::vector<const char*>& reqDeviceExtensions)
273 {
274  double score = -1.;
275  vk::PhysicalDeviceProperties deviceProperties = device.getProperties();
276  vk::PhysicalDeviceFeatures deviceFeatures = device.getFeatures();
277  vk::PhysicalDeviceMemoryProperties memoryProperties = device.getMemoryProperties();
278  if (!deviceFeatures.geometryShader || !deviceFeatures.wideLines || !deviceFeatures.largePoints) {
279  return (-1);
280  }
281 
282  std::vector<vk::QueueFamilyProperties> queueFamilies = device.getQueueFamilyProperties();
283  bool found = false;
284  for (unsigned int i = 0; i < queueFamilies.size(); i++) {
285  if (!(queueFamilies[i].queueFlags & vk::QueueFlagBits::eGraphics)) {
286  return (-1);
287  }
288  vk::Bool32 presentSupport = device.getSurfaceSupportKHR(i, mSurface);
289  if (!presentSupport) {
290  return (-1);
291  }
292  mGraphicsFamily = i;
293  found = true;
294  break;
295  }
296  if (!found) {
297  GPUInfo("%s ignored due to missing queue properties", deviceProperties.deviceName.data());
298  return (-1);
299  }
300 
301  std::vector<vk::ExtensionProperties> availableExtensions = device.enumerateDeviceExtensionProperties(nullptr);
302  unsigned int extensionsFound = 0;
303  for (unsigned int i = 0; i < reqDeviceExtensions.size(); i++) {
304  for (unsigned int j = 0; j < availableExtensions.size(); j++) {
305  if (strcmp(reqDeviceExtensions[i], availableExtensions[j].extensionName) == 0) {
306  extensionsFound++;
307  break;
308  }
309  }
310  }
311  if (extensionsFound < reqDeviceExtensions.size()) {
312  GPUInfo("%s ignored due to missing extensions", deviceProperties.deviceName.data());
313  return (-1);
314  }
315 
316  updateSwapChainDetails(device);
317  if (mSwapChainDetails.formats.empty() || mSwapChainDetails.presentModes.empty()) {
318  GPUInfo("%s ignored due to incompatible swap chain", deviceProperties.deviceName.data());
319  return (-1);
320  }
321 
322  score = 1;
323  if (deviceProperties.deviceType == vk::PhysicalDeviceType::eDiscreteGpu) {
324  score += 1e12;
325  } else if (deviceProperties.deviceType == vk::PhysicalDeviceType::eIntegratedGpu) {
326  score += 1e11;
327  }
328 
329  for (unsigned int i = 0; i < memoryProperties.memoryHeapCount; i++) {
330  if (memoryProperties.memoryHeaps[i].flags & vk::MemoryHeapFlagBits::eDeviceLocal) {
331  score += memoryProperties.memoryHeaps[i].size;
332  }
333  }
334 
335  return score;
336 }

338 void GPUDisplayBackendVulkan::createDevice()
339 {
340  vk::ApplicationInfo appInfo{};
341  appInfo.pApplicationName = "GPU CA Standalone display";
342  appInfo.applicationVersion = VK_MAKE_VERSION(1, 0, 0);
343  appInfo.pEngineName = "GPU CI Standalone Engine";
344  appInfo.engineVersion = VK_MAKE_VERSION(1, 0, 0);
345  appInfo.apiVersion = VK_API_VERSION_1_0;
346 
347  vk::InstanceCreateInfo instanceCreateInfo;
348  instanceCreateInfo.pApplicationInfo = &appInfo;
349 
350  const char** frontendExtensions;
351  uint32_t frontendExtensionCount = mDisplay->frontend()->getReqVulkanExtensions(frontendExtensions);
352  std::vector<const char*> reqInstanceExtensions(frontendExtensions, frontendExtensions + frontendExtensionCount);
353 
354  const std::vector<const char*> reqValidationLayers = {
355  "VK_LAYER_KHRONOS_validation"};
356  auto debugCallback = [](VkDebugUtilsMessageSeverityFlagBitsEXT messageSeverity, VkDebugUtilsMessageTypeFlagsEXT messageType, const VkDebugUtilsMessengerCallbackDataEXT* pCallbackData, void* pUserData) -> VkBool32 {
357  static bool throwOnError = getenv("GPUCA_VULKAN_VALIDATION_THROW") && atoi(getenv("GPUCA_VULKAN_VALIDATION_THROW"));
358  switch (messageSeverity) {
359  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
360  // GPUInfo("%s", pCallbackData->pMessage);
361  break;
362  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
363  GPUWarning("%s", pCallbackData->pMessage);
364  if (throwOnError) {
365  throw std::logic_error("break_on_validation_warning");
366  }
367  break;
368  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
369  GPUError("%s", pCallbackData->pMessage);
370  if (throwOnError) {
371  throw std::logic_error("break_on_validation_error");
372  }
373  break;
374  case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
375  default:
376  GPUInfo("%s", pCallbackData->pMessage);
377  break;
378  }
379  return false;
380  };
381  if (mEnableValidationLayers) {
382  if (checkValidationLayerSupport(reqValidationLayers)) {
383  throw std::runtime_error("Requested validation layer support not available");
384  }
385  reqInstanceExtensions.push_back(VK_EXT_DEBUG_UTILS_EXTENSION_NAME);
386  instanceCreateInfo.enabledLayerCount = static_cast<uint32_t>(reqValidationLayers.size());
387  instanceCreateInfo.ppEnabledLayerNames = reqValidationLayers.data();
388 
389  } else {
390  instanceCreateInfo.enabledLayerCount = 0;
391  }
392 
393  instanceCreateInfo.enabledExtensionCount = static_cast<uint32_t>(reqInstanceExtensions.size());
394  instanceCreateInfo.ppEnabledExtensionNames = reqInstanceExtensions.data();
395 
396  mInstance = vk::createInstance(instanceCreateInfo, nullptr);
397  mDLD = {mInstance, mDL.getProcAddress<PFN_vkGetInstanceProcAddr>("vkGetInstanceProcAddr")};
398 
399  if (mEnableValidationLayers) {
400  vk::DebugUtilsMessengerCreateInfoEXT createInfo{};
401  createInfo.messageSeverity = vk::DebugUtilsMessageSeverityFlagBitsEXT::eVerbose | vk::DebugUtilsMessageSeverityFlagBitsEXT::eWarning | vk::DebugUtilsMessageSeverityFlagBitsEXT::eError;
402  createInfo.messageType = vk::DebugUtilsMessageTypeFlagBitsEXT::eGeneral | vk::DebugUtilsMessageTypeFlagBitsEXT::eValidation | vk::DebugUtilsMessageTypeFlagBitsEXT::ePerformance;
403  createInfo.pfnUserCallback = debugCallback;
404  createInfo.pUserData = nullptr;
405  mDebugMessenger = mInstance.createDebugUtilsMessengerEXT(createInfo, nullptr, mDLD);
406  }
407  std::vector<vk::ExtensionProperties> extensions = vk::enumerateInstanceExtensionProperties(nullptr);
408  if (mDisplay->param()->par.debugLevel >= 3) {
409  std::cout << "available instance extensions: " << extensions.size() << "\n";
410  for (const auto& extension : extensions) {
411  std::cout << '\t' << extension.extensionName << '\n';
412  }
413  }
414 
415  if (mDisplay->frontend()->getVulkanSurface(&mInstance, &mSurface)) {
416  throw std::runtime_error("Frontend does not provide Vulkan surface");
417  }
418 
419  const std::vector<const char*> reqDeviceExtensions = {
420  VK_KHR_SWAPCHAIN_EXTENSION_NAME};
421 
422  mPhysicalDevice = VkPhysicalDevice(VK_NULL_HANDLE);
423  std::vector<vk::PhysicalDevice> devices = mInstance.enumeratePhysicalDevices();
424  if (devices.size() == 0) {
425  throw std::runtime_error("No Vulkan device present!");
426  }
427  double bestScore = -1.;
428  for (unsigned int i = 0; i < devices.size(); i++) {
429  double score = checkDevice(devices[i], reqDeviceExtensions);
430  if (mDisplay->param()->par.debugLevel >= 2) {
431  vk::PhysicalDeviceProperties deviceProperties = devices[i].getProperties();
432  GPUInfo("Available Vulkan device %d: %s - Score %f", i, deviceProperties.deviceName.data(), score);
433  }
434  if (score > bestScore && score > 0) {
435  mPhysicalDevice = devices[i];
436  bestScore = score;
437  }
438  }
439 
440  if (!mPhysicalDevice) {
441  throw std::runtime_error("All available Vulkan devices unsuited");
442  }
443 
444  updateSwapChainDetails(mPhysicalDevice);
445  vk::PhysicalDeviceProperties deviceProperties = mPhysicalDevice.getProperties();
446  vk::PhysicalDeviceFeatures deviceFeatures = mPhysicalDevice.getFeatures();
447  vk::FormatProperties depth32FormatProperties = mPhysicalDevice.getFormatProperties(vk::Format::eD32Sfloat);
448  vk::FormatProperties depth64FormatProperties = mPhysicalDevice.getFormatProperties(vk::Format::eD32SfloatS8Uint);
449  vk::FormatProperties formatProperties = mPhysicalDevice.getFormatProperties(mSurfaceFormat.format);
450  GPUInfo("Using physical Vulkan device %s", deviceProperties.deviceName.data());
451  mMaxMSAAsupported = getMaxUsableSampleCount(deviceProperties);
452  mZSupported = (bool)(depth32FormatProperties.optimalTilingFeatures & vk::FormatFeatureFlagBits::eDepthStencilAttachment);
453  mStencilSupported = (bool)(depth64FormatProperties.optimalTilingFeatures & vk::FormatFeatureFlagBits::eDepthStencilAttachment);
454  mCubicFilterSupported = (bool)(formatProperties.optimalTilingFeatures & vk::FormatFeatureFlagBits::eSampledImageFilterCubicEXT);
455  bool mailboxSupported = std::find(mSwapChainDetails.presentModes.begin(), mSwapChainDetails.presentModes.end(), vk::PresentModeKHR::eMailbox) != mSwapChainDetails.presentModes.end();
456  if (mDisplay->param()->par.debugLevel >= 2) {
457  GPUInfo("Max MSAA: %d, 32 bit Z buffer %d, 32 bit Z buffer + stencil buffer %d, Cubic Filtering %d, Mailbox present mode %d\n", (int)mMaxMSAAsupported, (int)mZSupported, (int)mStencilSupported, (int)mCubicFilterSupported, (int)mailboxSupported);
458  }
459 
460  vk::DeviceQueueCreateInfo queueCreateInfo{};
461  queueCreateInfo.queueFamilyIndex = mGraphicsFamily;
462  queueCreateInfo.queueCount = 1;
463  float queuePriority = 1.0f;
464  queueCreateInfo.pQueuePriorities = &queuePriority;
465  vk::DeviceCreateInfo deviceCreateInfo{};
466  deviceCreateInfo.pQueueCreateInfos = &queueCreateInfo;
467  deviceCreateInfo.queueCreateInfoCount = 1;
468  deviceCreateInfo.pEnabledFeatures = &deviceFeatures;
469  deviceCreateInfo.enabledExtensionCount = static_cast<uint32_t>(reqDeviceExtensions.size());
470  deviceCreateInfo.ppEnabledExtensionNames = reqDeviceExtensions.data();
471  deviceCreateInfo.enabledLayerCount = instanceCreateInfo.enabledLayerCount;
472  deviceCreateInfo.ppEnabledLayerNames = instanceCreateInfo.ppEnabledLayerNames;
473  mDevice = mPhysicalDevice.createDevice(deviceCreateInfo, nullptr);
474  mGraphicsQueue = mDevice.getQueue(mGraphicsFamily, 0);
475 
476  vk::CommandPoolCreateInfo poolInfo{};
477  poolInfo.flags = vk::CommandPoolCreateFlagBits::eResetCommandBuffer;
478  poolInfo.queueFamilyIndex = mGraphicsFamily;
479  mCommandPool = mDevice.createCommandPool(poolInfo, nullptr);
480 }
