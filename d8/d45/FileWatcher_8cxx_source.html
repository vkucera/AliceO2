116 bool FileWatcher::refresh()
117 {
118  string previous = this->currentItem();
119  LOG(info) << "previous:" << previous;
120  LOG(info) << "currentFile:" << this->mCurrentFile;
121 
122  this->mFiles = DirectoryLoader::load(this->mDataFolder, "_"); // already sorted according part staring with marker
123  if (this->mCurrentFile != mEndGuard) {
124  if (this->mFiles.empty()) {
125  this->mCurrentFile = mEndGuard; // list empty - stick to last element
126  } else if (this->mCurrentFile < *(this->mFiles.begin())) {
127  this->mCurrentFile = mLowGuard; // lower then first => go to first
128  } else {
129  auto it = std::find(mFiles.begin(), mFiles.end(), this->mCurrentFile);
130  if (it == this->mFiles.end()) {
131  this->mCurrentFile = mEndGuard; // not on the list -> go to last element
132  }
133  }
134  }
135  //for (auto it = this->mFiles.begin(); it != this->mFiles.end(); ++it) {
136  // LOG(info) << *it;
137  //}
138  this->mFiles.push_front(mLowGuard);
139  this->mFiles.push_back(mEndGuard);
140 
141  LOG(info) << "this->mFiles.size() = " << this->mFiles.size();
142  LOG(info) << "this->mCurrentFile = " << this->mCurrentFile;
143  LOG(info) << "current:" << this->currentItem();
144  return previous != this->currentItem();
145 }
