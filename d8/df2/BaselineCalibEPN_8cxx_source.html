43 int BaselineCalibEPN::process(const gsl::span<const o2::zdc::OrbitData>& orbitdata)
44 {
45  if (!mInitDone) {
46  init();
47  }
48  for (auto& myorbit : orbitdata) {
49  for (int ich = 0; ich < NChannels; ich++) {
50  auto myped = float(myorbit.data[ich]) * mModuleConfig->baselineFactor;
51  if (myped >= ADCMin && myped <= ADCMax) {
52  mData.addEntry(ich, myorbit.data[ich]);
53  }
54  }
55  }
56  return 0;
57 }
