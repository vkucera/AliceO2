53 {
54  const auto map = (typeMap == 0) ? std::make_unique<CalPad>(*mGainMap) : std::make_unique<CalPad>(*mSigmaMap);
55  if (!map) {
56  LOGP(error, "Map not set");

71  type ? IDCDrawHelper::drawSide(drawFun, sector.side(), zAxisTitle, filename, minZ, maxZ) : IDCDrawHelper::drawSector(drawFun, 0, Mapper::NREGIONS, sector, zAxisTitle, filename, minZ, maxZ);

182 void CalibPadGainTracksBase::finalize(const int minEntries, const float minRelgain, const float maxRelgain, const float low, const float high)
183 {
184  for (int roc = 0; roc < ROC::MaxROC; ++roc) {
185  const auto padsInRoc = ROC(roc).isIROC() ? Mapper::getPadsInIROC() : Mapper::getPadsInOROC();
186  for (int pad = 0; pad < padsInRoc; ++pad) {
187  const auto& histo = mPadHistosDet->getCalArray(roc).getData()[pad];
188  unsigned int entries = histo.getEntries();
189  const auto stat = histo.getStatisticsData(low, high);
190  const auto cog = std::clamp(static_cast<float>(stat.mCOG), minRelgain, maxRelgain);
191 
192  // subtract underflow and overflow entries to check if only the valid entries are > 0
193  if (histo.isUnderflowSet()) {
194  entries -= histo.getBinContent(0);
195  }
196 
197  if (histo.isOverflowSet()) {
198  const unsigned int binOverflow = histo.getNBins() + histo.isUnderflowSet();
199  entries -= histo.getBinContent(binOverflow);
200  }
201 
202  if (entries >= minEntries) {
203  mGainMap->getCalArray(roc).getData()[pad] = cog;
204  mSigmaMap->getCalArray(roc).getData()[pad] = stat.mStdDev;
205  } else {
206  mGainMap->getCalArray(roc).getData()[pad] = 1;
207  mSigmaMap->getCalArray(roc).getData()[pad] = 0;
208  }
209  }
210  }
211  normalizeGain(*mGainMap.get());
212 }

222 std::vector<int> CalibPadGainTracksBase::getNPadsForNormalization(const bool iroc) const
223 {
224  if (mNormType == NormType::stack) {
225  // normalize per stack
226  const std::vector<int> nPads = iroc ? std::vector<int>{Mapper::getPadsInIROC()} : std::vector<int>{Mapper::getPadsInOROC1(), Mapper::getPadsInOROC2(), Mapper::getPadsInOROC3()};
227  return nPads;
228  } else if (mNormType == NormType::region) {
229  // normalize per region
230  const std::vector<int> nPads = iroc ? std::vector<int>{Mapper::PADSPERREGION[0], Mapper::PADSPERREGION[1], Mapper::PADSPERREGION[2], Mapper::PADSPERREGION[3]} : std::vector<int>{Mapper::PADSPERREGION[4], Mapper::PADSPERREGION[5], Mapper::PADSPERREGION[6], Mapper::PADSPERREGION[7], Mapper::PADSPERREGION[8], Mapper::PADSPERREGION[9]};
231  return nPads;
232  } else {
233  return std::vector<int>();
234  }
235 }

237 void CalibPadGainTracksBase::normalize(std::vector<float>& data, const std::vector<int>& nPads)
238 {
239  int padStart = 0;
240  for (const auto pads : nPads) {
241  auto median = TMath::Median(pads, data.data() + padStart);
242  std::for_each(data.data() + padStart, data.data() + padStart + pads, [median](auto& val) { val /= (val > 0) ? median : 1; });
243  padStart += pads;
244  }
245 }
