47 void EntropyDecoderSpec::run(ProcessingContext& pc)
48 {
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51  o2::ctf::CTFIOSize iosize;
52  updateTimeDependentParams(pc);
53 
54  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
55  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
56  // const auto ctfImage = o2::itsmft::CTF::getImage(buff.data());
57 
58  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(OutputRef{"ROframes"});
59  if (mGetDigits) {
60  auto& digits = pc.outputs().make<std::vector<o2::itsmft::Digit>>(OutputRef{"Digits"});
61  if (buff.size()) {
62  iosize = mCTFCoder.decode(o2::itsmft::CTF::getImage(buff.data()), rofs, digits, mNoiseMap, mPattIdConverter);
63  }
64  mTimer.Stop();
65  LOG(info) << "Decoded " << digits.size() << " digits in " << rofs.size() << " RO frames, (" << iosize.asString() << ") in " << mTimer.CpuTime() - cput << " s";
66  } else {
67  auto& compcl = pc.outputs().make<std::vector<o2::itsmft::CompClusterExt>>(OutputRef{"compClusters"});
68  auto& patterns = pc.outputs().make<std::vector<unsigned char>>(OutputRef{"patterns"});
69  if (buff.size()) {
70  iosize = mCTFCoder.decode(o2::itsmft::CTF::getImage(buff.data()), rofs, compcl, patterns, mNoiseMap, mPattIdConverter);
71  }
72  mTimer.Stop();
73  LOG(info) << "Decoded " << compcl.size() << " clusters in " << rofs.size() << " RO frames, (" << iosize.asString() << ") in " << mTimer.CpuTime() - cput << " s";
74  }
75  pc.outputs().snapshot({"ctfrep", 0}, iosize);
76 }

95 void EntropyDecoderSpec::finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj)
96 {
97  if (matcher == ConcreteDataMatcher(mOrigin, "NOISEMAP", 0)) {
98  mNoiseMap = (o2::itsmft::NoiseMap*)obj;
99  LOG(info) << mOrigin.as<std::string>() << " noise map updated";
100  return;
101  }
102  if (matcher == ConcreteDataMatcher(mOrigin, "CLUSDICT", 0)) {
103  LOG(info) << mOrigin.as<std::string>() << " cluster dictionary updated" << (!mUseClusterDictionary ? " but its using is disabled" : "");
104  mPattIdConverter.setDictionary((const TopologyDictionary*)obj);
105  return;
106  }
107  if (mCTFCoder.finaliseCCDB<CTF>(matcher, obj)) {
108  return;
109  }
110 }

112 DataProcessorSpec getEntropyDecoderSpec(o2::header::DataOrigin orig, int verbosity, bool getDigits, unsigned int sspec)
113 {
114  std::vector<OutputSpec> outputs;
115  if (getDigits) {
116  outputs.emplace_back(OutputSpec{{"Digits"}, orig, "DIGITS", 0, Lifetime::Timeframe});
117  outputs.emplace_back(OutputSpec{{"ROframes"}, orig, "DIGITSROF", 0, Lifetime::Timeframe});
118  } else {
119  outputs.emplace_back(OutputSpec{{"compClusters"}, orig, "COMPCLUSTERS", 0, Lifetime::Timeframe});
120  outputs.emplace_back(OutputSpec{{"ROframes"}, orig, "CLUSTERSROF", 0, Lifetime::Timeframe});
121  outputs.emplace_back(OutputSpec{{"patterns"}, orig, "PATTERNS", 0, Lifetime::Timeframe});
122  }
123  outputs.emplace_back(OutputSpec{{"ctfrep"}, orig, "CTFDECREP", 0, Lifetime::Timeframe});
124 
125  std::vector<InputSpec> inputs;
126  inputs.emplace_back("ctf", orig, "CTFDATA", sspec, Lifetime::Timeframe);
127  inputs.emplace_back("noise", orig, "NOISEMAP", 0, Lifetime::Condition, ccdbParamSpec(fmt::format("{}/Calib/NoiseMap", orig.as<std::string>())));
128  inputs.emplace_back("cldict", orig, "CLUSDICT", 0, Lifetime::Condition, ccdbParamSpec(fmt::format("{}/Calib/ClusterDictionary", orig.as<std::string>())));
129  inputs.emplace_back("ctfdict", orig, "CTFDICT", 0, Lifetime::Condition, ccdbParamSpec(fmt::format("{}/Calib/CTFDictionary", orig.as<std::string>())));
130 
131  return DataProcessorSpec{
132  EntropyDecoderSpec::getName(orig),
133  inputs,
134  outputs,
135  AlgorithmSpec{adaptFromTask<EntropyDecoderSpec>(orig, verbosity, getDigits)},
136  Options{
137  {"ctf-dict", VariantType::String, "ccdb", {"CTF dictionary: empty or ccdb=CCDB, none=no external dictionary otherwise: local filename"}},
138  {"mask-noise", VariantType::Bool, false, {"apply noise mask to digits or clusters (involves reclusterization)"}},
139  {"ignore-cluster-dictionary", VariantType::Bool, false, {"do not use cluster dictionary, always store explicit patterns"}}}};
140 }
