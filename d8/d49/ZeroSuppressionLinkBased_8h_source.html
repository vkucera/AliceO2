93 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2>
94 struct Data {
95  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
96  static constexpr uint32_t ChannelsPerHalfWord = ChannelsPerWord / 2;
97  static constexpr uint32_t DataBitSize = DataBitSizeT;
98  static constexpr uint32_t SignificantBits = SignificantBitsT;
99  static constexpr uint64_t BitMask = ((uint64_t(1) << DataBitSize) - 1);
100  static constexpr float FloatConversion = 1.f / float(1 << SignificantBits);
101 
102  uint64_t adcValues[2]{};
103 
105  void setADCValue(uint32_t pos, uint64_t value)
106  {
107  const uint32_t word = pos / ChannelsPerHalfWord;
108  const uint32_t posInWord = pos % ChannelsPerHalfWord;
109 
110  const uint64_t set = (value & BitMask) << (posInWord * DataBitSize);
111  const uint64_t mask = (0xFFFFFFFFFFFFFFFF ^ (BitMask << (posInWord * DataBitSize)));
112 
113  adcValues[word] &= mask;
114  adcValues[word] |= set;
115  }
116 
118  void setADCValueFloat(uint32_t pos, float value)
119  {
120  setADCValue(pos, uint64_t((value + 0.5f * FloatConversion) / FloatConversion));
121  }
122 
124  uint32_t getADCValue(uint32_t pos) const
125  {
126  const uint32_t word = pos / ChannelsPerHalfWord;
127  const uint32_t posInWord = pos % ChannelsPerHalfWord;
128 
129  return (adcValues[word] >> (posInWord * DataBitSize)) & BitMask;
130  }
131 
133  float getADCValueFloat(uint32_t pos) const
134  {
135  return float(getADCValue(pos)) * FloatConversion;
136  }
137 
139  void reset()
140  {
141  adcValues[0] = 0;
142  adcValues[1] = 0;
143  }
144 };

187 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2, bool HasHeaderT = true>
188 struct Container {
189  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
190 
191  ContainerT<DataBitSizeT, SignificantBitsT, HasHeaderT> cont;
192 
194  uint32_t getADCValue(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValue(word % ChannelsPerWord); }
195 
197  float getADCValueFloat(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValueFloat(word % ChannelsPerWord); }
198 
200  void setADCValue(uint32_t word, uint64_t value) { cont.data[word / ChannelsPerWord].setADCValue(word % ChannelsPerWord, value); }
201 
203  void setADCValueFloat(uint32_t word, float value) { cont.data[word / ChannelsPerWord].setADCValueFloat(word % ChannelsPerWord, value); }
204 
206  void reset()
207  {
208  for (int i = 0; i < cont.getDataWords(); ++i) {
209  cont.data[i].reset();
210  }
211  }
212 
213  uint32_t getBunchCrossing() const { return cont.getBunchCrossing(); }
214  uint32_t getDataWords() const { return cont.getDataWords(); }
215  uint32_t getTotalWords() const { return cont.getTotalWords(); }
216  std::bitset<80> getChannelBits() const { return cont.getChannelBits(); }
217 
219  size_t getTotalSizeBytes() const { return getTotalWords() * DataWordSizeBytes; }
220 
222  Container* next() const
223  {
224  return (Container*)((const char*)this + getTotalWords() * DataWordSizeBytes);
225  }
