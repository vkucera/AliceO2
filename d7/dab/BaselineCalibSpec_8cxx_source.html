101 void BaselineCalibSpec::sendOutput(o2::framework::DataAllocator& output)
102 {
103  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
104  // TODO in principle, this routine is generic, can be moved to Utils.h
105  using clbUtils = o2::calibration::Utils;
106  const auto& payload = mWorker.getParamUpd();
107  auto& info = mWorker.getCcdbObjectInfo();
108  auto image = o2::ccdb::CcdbApi::createObjectImage<BaselineParam>(&payload, &info);
109  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
110  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
111  if (mVerbosity > DbgMinimal) {
112  payload.print();
113  }
114  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "ZDCBaselinecalib", 0}, *image.get()); // vector<char>
115  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "ZDCBaselinecalib", 0}, info); // root-serialized

120 framework::DataProcessorSpec getBaselineCalibSpec()
121 {
122  using device = o2::zdc::BaselineCalibSpec;
123  using clbUtils = o2::calibration::Utils;
124 
125  std::vector<InputSpec> inputs;
126  inputs.emplace_back("basecalibdata", "ZDC", "BASECALIBDATA", 0, Lifetime::Timeframe);
127  inputs.emplace_back("calibconfig", "ZDC", "BASECALIBCONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathBaselineCalibConfig.data()));
128  inputs.emplace_back("moduleconfig", "ZDC", "MODULECONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathConfigModule.data()));
129 
130  std::vector<OutputSpec> outputs;
131  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ZDCBaselinecalib"}, Lifetime::Sporadic);
132  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ZDCBaselinecalib"}, Lifetime::Sporadic);
133 
134  return DataProcessorSpec{
135  "zdc-calib-baseline",
136  inputs,
137  outputs,
138  AlgorithmSpec{adaptFromTask<device>()},
139  Options{{"verbosity-level", o2::framework::VariantType::Int, 1, {"Verbosity level"}}}};
140 }
