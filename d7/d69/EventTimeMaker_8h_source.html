62  template <typename trackType,
63  bool (*trackFilter)(const trackType&)>
64  void removeBias(const trackType& track,
65  int& nTrackIndex /* index of the track to remove the bias */,
66  float& eventTimeValue,
67  float& eventTimeError,
68  const unsigned short& minimumMultiplicity = 2) const
69  {
70  eventTimeValue = mEventTime;
71  eventTimeError = mEventTimeError;
72  if (!trackFilter(track)) { // Check if the track was usable for the event time
73  nTrackIndex++;
74  return;
75  }
76  if (mEventTimeMultiplicity <= minimumMultiplicity && mWeights[nTrackIndex] > 1E-6f) { // Check if a track was used for the event time and if the multiplicity is low
77  eventTimeValue = 0.f;
78  eventTimeError = mDiamondSpread * 33.356409f; // move from diamond (cm) to spread on event time (ps)
79  LOG(debug) << mEventTimeMultiplicity << " <= " << minimumMultiplicity << " and " << mWeights[nTrackIndex] << ": track was used, setting " << eventTimeValue << " " << eventTimeError;
80  nTrackIndex++;
81  return;
82  }
83  // Remove the bias
84  float sumw = 1.f / eventTimeError / eventTimeError;
85  LOG(debug) << "sumw " << sumw;
86  eventTimeValue *= sumw;
87  eventTimeValue -= mWeights[nTrackIndex] * mTrackTimes[nTrackIndex];
88  sumw -= mWeights[nTrackIndex];
89  eventTimeValue /= sumw;
90  eventTimeError = sqrt(1.f / sumw);
91  nTrackIndex++;
92  }
93 
94  void print()
95  {
96  LOG(info) << "eventTimeContainer " << mEventTime << " +- " << mEventTimeError << " sum of weights " << mSumOfWeights << " tracks used " << mEventTimeMultiplicity;
97  }
98 };

147 template <typename trackTypeContainer,
148  typename trackType,
149  bool (*trackFilter)(const trackType&)>
150 eventTimeContainer evTimeMaker(const trackTypeContainer& tracks,
151  const float& diamond = 6.0 /* spread of primary verdex in cm */,
152  bool isFast = false)
153 {
154  static std::vector<eventTimeTrack> trkWork;
155  trkWork.clear();
156  static std::vector<int> trkIndex; // indexes of working tracks in the track original array
157  trkIndex.clear();
158 
159  static float expt[3], expsigma[3];
160 
161  static eventTimeContainer result = {0, 0, diamond};
162 
163  // reset info
164  result.reset();
165 
166  for (auto track : tracks) { // Loop on tracks
167  if (trackFilter(track)) { // Select tracks good for T0 computation
168  expt[0] = track.tofExpSignalPi();
169  expt[1] = track.tofExpSignalKa();
170  expt[2] = track.tofExpSignalPr();
171  expsigma[0] = track.tofExpSigmaPi();
172  expsigma[1] = track.tofExpSigmaKa();
173  expsigma[2] = track.tofExpSigmaPr();
174  trkWork.emplace_back(track.tofSignal(), expt, expsigma);
175  trkIndex.push_back(result.mWeights.size());
176  }
177  result.mWeights.push_back(0.);
178  result.mTrackTimes.push_back(0.);
179  }
180  if (!isFast) {
181  computeEvTime(trkWork, trkIndex, result);
182  } else {
183  computeEvTimeFast(trkWork, trkIndex, result);
184  }
185  return result;
186 }
