110 void WaveformCalibSpec::sendOutput(o2::framework::DataAllocator& output)
111 {
112  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
113  // TODO in principle, this routine is generic, can be moved to Utils.h
114  using clbUtils = o2::calibration::Utils;
115  const auto& data = mWorker.getData();
116  WaveformCalibParam payload;
117  payload.assign(data);
118  auto& info = mWorker.getCcdbObjectInfo();
119  auto image = o2::ccdb::CcdbApi::createObjectImage<WaveformCalibParam>(&payload, &info);
120  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
121  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
122  if (mVerbosity > DbgZero) {
123  payload.print();
124  }
125  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "ZDCWaveformCalib", 0}, *image.get()); // vector<char>
126  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "ZDCWaveformCalib", 0}, info); // root-serialized
127  // TODO: reset the outputs once they are already sent (is it necessary?)
128  // mWorker.init();
129 }

131 framework::DataProcessorSpec getWaveformCalibSpec()
132 {
133  using device = o2::zdc::WaveformCalibSpec;
134  using clbUtils = o2::calibration::Utils;
135 
136  std::vector<InputSpec> inputs;
137  inputs.emplace_back("waveformcalibdata", "ZDC", "WAVECALIBDATA", 0, Lifetime::Timeframe);
138  inputs.emplace_back("wavecalibconfig", "ZDC", "WAVECALIBCONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathWaveformCalibConfig.data()));
139 
140  std::vector<OutputSpec> outputs;
141  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ZDCWaveformCalib"}, Lifetime::Sporadic);
142  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ZDCWaveformCalib"}, Lifetime::Sporadic);
143 
144  return DataProcessorSpec{
145  "zdc-waveformcalib",
146  inputs,
147  outputs,
148  AlgorithmSpec{adaptFromTask<device>()},
149  Options{{"verbosity-level", o2::framework::VariantType::Int, 1, {"Verbosity level"}}}};
150 }
