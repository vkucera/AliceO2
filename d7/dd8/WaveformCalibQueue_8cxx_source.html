20 void WaveformCalibQueue::configure(const WaveformCalibConfig* cfg)
21 {
22  mCfg = cfg;
23  int ifirst = mCfg->getFirst();
24  int ilast = mCfg->getLast();
25  if (ifirst > 0 || ilast < 0 || ilast < ifirst) {
26  LOGF(fatal, "WaveformCalibQueue configure error with ifirst=%d ilast=%d", ifirst, ilast);
27  }
28  mFirst = ifirst;
29  mLast = ilast;
30  mN = ilast - ifirst + 1;
31  mPk = -mFirst;
32  mPeak = peak(mPk);
33  mNP = mN * NIS;
34  for (int32_t isig = 0; isig < NChannels; isig++) {
35  mTimeLow[isig] = std::nearbyint(cfg->cutTimeLow[SignalTDC[isig]] / FTDCVal);
36  mTimeHigh[isig] = std::nearbyint(cfg->cutTimeHigh[SignalTDC[isig]] / FTDCVal);
37  }
38 }

40 // appends an event to the queue with the request that there are at most
41 // mN consecutive bunches
42 // The TDC conditions are checked and returned in output
43 uint32_t WaveformCalibQueue::append(RecEventFlat& ev)
44 {
45  auto& toadd = ev.ir;
46  // If queue is empty insert event
47  if (mIR.size() == 0) {
48  appendEv(ev);
49  return 0;
50  }
51  // Check last element
52  auto& last = mIR.back();
53  // If BC are not consecutive, clear queue
54  if (toadd.differenceInBC(last) > 1) {
55  clear();
56  }
57  // If queue is not empty and is too long remove first element
58  while (mIR.size() >= mN) {
59  pop();
60  }
61  // If BC are consecutive or cleared queue append element
62  appendEv(ev);
63  if (mIR.size() == mN) {
64  uint32_t mask = 0;
65  for (int32_t itdc = 0; itdc < NTDCChannels; itdc++) {
66  // Check which channels satisfy the condition on TDC
67  bool tdccond = true;
68  for (int i = 0; i < mN; i++) {
69  int n = mNTDC[itdc].at(i);
70  if (i == mPk) {
71  if (n != 1) {
72  tdccond = false;
73  break;
74  } else {
75  auto tdca = mTDCA[itdc].at(i);
76  auto tdcv = mTDCP[itdc].at(i);
77  if (tdca < mCfg->cutLow[itdc] || tdca > mCfg->cutHigh[itdc] || tdcv < mCfg->cutTimeLow[itdc] || tdcv > mCfg->cutTimeHigh[itdc]) {
78  tdccond = false;
79  break;
80  }
81  }
82  } else {
83  if (n != 0) {
84  tdccond = false;
85  break;
86  }
87  }
88  }
89  if (tdccond) {
90  mask = mask | (0x1 << itdc);
91  }
92  }
93  return mask;
94  } else {
95  return 0;
96  }
97 }

99 void WaveformCalibQueue::appendEv(RecEventFlat& ev)
100 {
101 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
102  LOGF(info, "WaveformCalibQueue::%s %u.%04u", __func__, ev.ir.orbit, ev.ir.bc);
103 #endif
104  mIR.push_back(ev.ir);
105  mEntry.push_back(ev.getNextEntry() - 1);
106 
107  auto& curb = ev.getCurB();
108  int firstw, nw;
109  curb.getRefW(firstw, nw);
110  mFirstW.push_back(firstw);
111  mNW.push_back(nw);
112 
113  // Note: pile-up messages are computed only for the 10 TDCs
114  for (int isig = 0; isig < NChannels; isig++) {
115  mHasInfos[isig].push_back(false);
116  }
117  if (ev.getNInfo() > 0) {
118  // Need clean data (no messages)
119  auto& decodedInfo = ev.getDecodedInfo();
120  for (uint16_t info : decodedInfo) {
121  uint8_t ch = (info >> 10) & 0x1f;
122  uint16_t code = info & 0x03ff;
123  auto& last = mHasInfos[ch].back();
124  last = true;
125  }
126  // if (mVerbosity > DbgMinimal) {
127  // ev.print();
128  // }
129  }
130  // TDC channels are used to select reconstructed waveforms
131  for (int32_t itdc = 0; itdc < NTDCChannels; itdc++) {
132  int ich = o2::zdc::TDCSignal[itdc];
133  int nhit = ev.NtdcV(itdc);
134  if (ev.NtdcA(itdc) != nhit) {
135  LOGF(error, "Mismatch in TDC %d data Val=%d Amp=%d\n", itdc, ev.NtdcV(itdc), ev.NtdcA(ich));
136  mNTDC[itdc].push_back(0);
137  mTDCA[itdc].push_back(0);
138  mTDCP[itdc].push_back(0);
139  } else if (nhit == 0) {
140  mNTDC[itdc].push_back(0);
141  mTDCA[itdc].push_back(0);
142  mTDCP[itdc].push_back(0);
143  } else {
144  // Store single TDC entry
145  mNTDC[itdc].push_back(nhit);
146  mTDCA[itdc].push_back(ev.tdcA(itdc, 0));
147  mTDCP[itdc].push_back(ev.tdcV(itdc, 0));
148  }
149  }

152 int WaveformCalibQueue::hasData(int isig, const gsl::span<const o2::zdc::ZDCWaveform>& wave)
153 {
154  int ipk = -1;
155  int ipkb = -1;
156  float min = std::numeric_limits<float>::infinity();
157  for (int ib = 0; ib < mN; ib++) {
158  int ifound = false;
159 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
160  LOG(info) << "WaveformCalibQueue::" << __func__ << " mNW[" << ib << "] = " << mNW[ib] << " mFirstW = " << mFirstW[ib];
161 #endif
162  for (int iw = 0; iw < mNW[ib]; iw++) {
163  auto& mywave = wave[iw + mFirstW[ib]];
164  if (mywave.ch() == isig) {
165  ifound = true;
166  for (int ip = 0; ip < NIS; ip++) {
167  if (mywave.inter[ip] < min) {
168  ipkb = ib;
169  ipk = ip;
170  min = mywave.inter[ip];
171  }
172  }
173  }
174  }
175  // Need to have consecutive data for all bunches
176  if (!ifound) {
177  return -1;
178  }
179  }
180  if (ipkb != mPk) {
181  return -1;
182  } else {
183  int ipos = NTimeBinsPerBC * TSN * ipkb + ipk;
184 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
185  LOG(info) << "WaveformCalibConfig::" << __func__ << " isig = " << isig << " ipkb " << ipkb << " ipk " << ipk << " min " << min;
186 #endif
187  return ipos;
188  }
189 }

191 // Checks if waveform has available data and adds it to summary data
192 // a compensation of the time jitter
193 int WaveformCalibQueue::addData(int isig, const gsl::span<const o2::zdc::ZDCWaveform>& wave, WaveformCalibData& data)
194 {
195  int ipkb = -1; // Bunch where peak is found
196  int ipk = -1; // peak position within bunch
197  float min = std::numeric_limits<float>::infinity();
198  float max = -std::numeric_limits<float>::infinity();
199  bool hasInfos = false;
200  for (int ib = 0; ib < mN; ib++) {
201  bool ifound = false;
202 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
203  LOG(info) << "mNW[" << ib << "] = " << mNW[ib] << " mFirstW = " << mFirstW[ib];
204 #endif
205  if (mHasInfos[isig][ib] || mHasInfos[TDCSignal[SignalTDC[isig]]][ib]) {
206 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
207  LOG(info) << "isig=" << isig << " ib=" << ib << " tdcid=" << SignalTDC[isig] << " tdc_sig=" << TDCSignal[SignalTDC[isig]] << " " << mHasInfos[isig][ib] << " " << mHasInfos[TDCSignal[SignalTDC[isig]]][ib];
208 #endif
209  hasInfos = true;
210  }
211  for (int iw = 0; iw < mNW[ib]; iw++) {
212  // Signal shouldn't have info messages. We check also corresponding TDC signal for pile-up information
213  // TODO: relax this condition to avoid to check pedestal messages since pedestal is subtracted
214  // when converting waveform calibration object into SimConfig object
215  auto& mywave = wave[iw + mFirstW[ib]];
216  if (mywave.ch() == isig) {
217  ifound = true;
218  for (int ip = 0; ip < NIS; ip++) {
219  if (mywave.inter[ip] < min) {
220  ipkb = ib;
221  ipk = ip;
222  min = mywave.inter[ip];
223  }
224  if (mywave.inter[ip] > max) {
225  max = mywave.inter[ip];
226  }
227  }
228  }
229  }
230 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
231  LOG(info) << "WaveformCalibQueue::" << __func__ << " isig=" << isig << " mNW[" << ib << "] = " << mNW[ib] << " mFirstW = " << mFirstW[ib]
232  << " ifound=" << ifound << " hasInfos=" << hasInfos;
233 #endif
234  // Need to have consecutive data for all bunches
235  if (!ifound || hasInfos) {
236  return -1;
237  }
238  }
239  if (ipkb != mPk) {
240  return -1;
241  } else {
242  int ppos = NIS * ipkb + ipk;
243  int itdc = SignalTDC[isig];
244  if (isig != TDCSignal[itdc]) {
245  // Additional checks for towers
246  float amp = max - min;
247  if (amp < mCfg->cutLow[isig] || amp > mCfg->cutHigh[isig]) {
248  // No warning messages for amplitude cuts on towers
249  return -1;
250  }
251  if ((ppos - mPeak) < mTimeLow[itdc] || (ppos - mPeak) > mTimeHigh[itdc]) {
252  // Put a warning message for a signal out of time
253  LOGF(warning, "%d.%04d Signal %2d peak position %d-%d=%d is outside allowed range [%d:%d]", mIR[mPk].orbit, mIR[mPk].bc, isig, ppos, mPeak, ppos - mPeak, mTimeLow[isig], mTimeHigh[isig]);
254  return -1;
255  }
256  }
257  int ipos = mPeak - ppos;
258  data.addEntry(isig);
259  // Restrict validity range because of signal jitter
260  data.setFirstValid(isig, ipos);
261  // We know that points are consecutive
262  for (int ib = 0; ib < mN; ib++) {
263  for (int iw = 0; iw < mNW[ib]; iw++) {
264  auto& mywave = wave[iw + mFirstW[ib]];
265  if (mywave.ch() == isig) {
266  for (int ip = 0; ip < NIS; ip++) {
267  if (ipos >= 0 && ipos < mNP) {
268  // We don't use incapsulation because this section is called too many times
269  data.mWave[isig].mData[ipos] += mywave.inter[ip];
270  }
271  ipos++;
272  }
273  }
274  }
275  }
276  ipos--;
277  // Restrict validity range because of signal jitter
278  data.setLastValid(isig, ipos);
279 #ifdef O2_ZDC_WAVEFORMCALIB_DEBUG
280  LOG(info) << "WaveformCalibConfig::" << __func__ << " isig = " << isig << " ipkb " << ipkb << " ipk " << ipk << " min " << min << " range=[" << data.getFirstValid(isig) << ":" << ppos << ":" << data.getLastValid(isig) << "]";
281 #endif
282  return ipos;
283  }
284 }
