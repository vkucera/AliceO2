85 TH2Poly* o2::tpc::IDCDrawHelper::drawSide(const IDCDraw& idc, const o2::tpc::Side side, const std::string zAxisTitle)
86 {
87  const auto coords = o2::tpc::painter::getPadCoordinatesSector();
88  TH2Poly* poly = o2::tpc::painter::makeSideHist(side);
89  poly->SetContour(255);
90  poly->SetTitle(nullptr);
91  poly->GetXaxis()->SetTitleOffset(1.2f);
92  poly->GetYaxis()->SetTitleOffset(1.3f);
93  poly->GetZaxis()->SetTitleOffset(1.4f);
94  poly->GetZaxis()->SetTitle(zAxisTitle.data());
95  poly->GetZaxis()->SetMaxDigits(3); // force exponential axis
96  poly->SetStats(0);
97 
98  unsigned int sectorStart = (side == Side::A) ? 0 : o2::tpc::SECTORSPERSIDE;
99  unsigned int sectorEnd = (side == Side::A) ? o2::tpc::SECTORSPERSIDE : Mapper::NSECTORS;
100  for (unsigned int sector = sectorStart; sector < sectorEnd; ++sector) {
101  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
102  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
103  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
104  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
105  const float angDeg = 10.f + sector * 20;
106  auto coordinate = coords[padNum];
107  coordinate.rotate(angDeg);
108  const float yPos = static_cast<float>(coordinate.yVals[0] + coordinate.yVals[1] + coordinate.yVals[2] + coordinate.yVals[3]) / 4;
109  const float xPos = static_cast<float>(coordinate.xVals[0] + coordinate.xVals[1] + coordinate.xVals[2] + coordinate.xVals[3]) / 4;
110  const auto padTmp = (side == Side::A) ? ipad : (Mapper::PADSPERROW[region][irow] - ipad); // C-Side is mirrored
111  poly->Fill(xPos, yPos, idc.getIDC(sector, region, irow, padTmp));
112  }
113  }
114  }
115  }
116 
117  return poly;
118 }
