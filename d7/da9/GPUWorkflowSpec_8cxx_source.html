321 void GPURecoWorkflowSpec::run(ProcessingContext& pc)
322 {
323  constexpr static size_t NSectors = Sector::MAXSECTOR;
324  constexpr static size_t NEndpoints = o2::gpu::GPUTrackingInOutZS::NENDPOINTS;
325 
326  if (mReadyToQuit) {
327  return;
328  }
329  auto cput = mTimer->CpuTime();
330  auto realt = mTimer->RealTime();
331  mTimer->Start(false);
332  std::vector<gsl::span<const char>> inputs;
333 
334  const CompressedClustersFlat* pCompClustersFlat = nullptr;
335  size_t compClustersFlatDummyMemory[(sizeof(CompressedClustersFlat) + sizeof(size_t) - 1) / sizeof(size_t)];
336  CompressedClustersFlat& compClustersFlatDummy = reinterpret_cast<CompressedClustersFlat&>(compClustersFlatDummyMemory);
337  CompressedClusters compClustersDummy;
338  o2::gpu::GPUTrackingInOutZS tpcZS;
339  std::vector<const void*> tpcZSmetaPointers[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
340  std::vector<unsigned int> tpcZSmetaSizes[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
341  const void** tpcZSmetaPointers2[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
342  const unsigned int* tpcZSmetaSizes2[GPUTrackingInOutZS::NSLICES][GPUTrackingInOutZS::NENDPOINTS];
343  std::array<unsigned int, NEndpoints * NSectors> tpcZSonTheFlySizes;
344  gsl::span<const ZeroSuppressedContainer8kb> inputZS;
345 
346  bool getWorkflowTPCInput_clusters = false, getWorkflowTPCInput_mc = false, getWorkflowTPCInput_digits = false;
347 
348  // unsigned int totalZSPages = 0;
349  if (mSpecConfig.processMC) {
350  getWorkflowTPCInput_mc = true;
351  }
352  if (!mSpecConfig.decompressTPC && !mSpecConfig.caClusterer) {
353  getWorkflowTPCInput_clusters = true;
354  }
355  if (!mSpecConfig.decompressTPC && mSpecConfig.caClusterer && ((!mSpecConfig.zsOnTheFly || mSpecConfig.processMC) && !mSpecConfig.zsDecoder)) {
356  getWorkflowTPCInput_digits = true;
357  }
358 
359  if (mSpecConfig.zsOnTheFly || mSpecConfig.zsDecoder) {
360  for (unsigned int i = 0; i < GPUTrackingInOutZS::NSLICES; i++) {
361  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
362  tpcZSmetaPointers[i][j].clear();
363  tpcZSmetaSizes[i][j].clear();
364  }
365  }
366  }
367  if (mSpecConfig.zsOnTheFly) {
368  tpcZSonTheFlySizes = {0};
369  // tpcZSonTheFlySizes: #zs pages per endpoint:
370  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "ZSSIZES"}, Lifetime::Timeframe}};
371  bool recv = false, recvsizes = false;
372  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
373  if (recvsizes) {
374  throw std::runtime_error("Received multiple ZSSIZES data");
375  }
376  tpcZSonTheFlySizes = pc.inputs().get<std::array<unsigned int, NEndpoints * NSectors>>(ref);
377  recvsizes = true;
378  }
379  // zs pages
380  std::vector<InputSpec> filter2 = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "TPCZS"}, Lifetime::Timeframe}};
381  for (auto const& ref : InputRecordWalker(pc.inputs(), filter2)) {
382  if (recv) {
383  throw std::runtime_error("Received multiple TPCZS data");
384  }
385  inputZS = pc.inputs().get<gsl::span<ZeroSuppressedContainer8kb>>(ref);
386  recv = true;
387  }
388  if (!recv || !recvsizes) {
389  throw std::runtime_error("TPC ZS on the fly data not received");
390  }
391 
392  unsigned int offset = 0;
393  for (unsigned int i = 0; i < NSectors; i++) {
394  unsigned int pageSector = 0;
395  for (unsigned int j = 0; j < NEndpoints; j++) {
396  pageSector += tpcZSonTheFlySizes[i * NEndpoints + j];
397  offset += tpcZSonTheFlySizes[i * NEndpoints + j];
398  }
399  if (mVerbosity >= 1) {
400  LOG(info) << "GOT ZS on the fly pages FOR SECTOR " << i << " -> pages: " << pageSector;
401  }
402  }
403  }
404  if (mSpecConfig.zsDecoder) {
405  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}};
406  auto isSameRdh = [](const char* left, const char* right) -> bool {
407  return o2::raw::RDHUtils::getFEEID(left) == o2::raw::RDHUtils::getFEEID(right) && o2::raw::RDHUtils::getDetectorField(left) == o2::raw::RDHUtils::getDetectorField(right);
408  };
409  auto insertPages = [&tpcZSmetaPointers, &tpcZSmetaSizes](const char* ptr, size_t count, uint32_t subSpec) -> void {
410  if (subSpec == 0xdeadbeef) {
411  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
412  static int contDeadBeef = 0;
413  if (++contDeadBeef <= maxWarn) {
414  LOGP(alarm, "Found input [TPC/RAWDATA/0xdeadbeef] assuming no payload for all links in this TF{}", contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
415  }
416  return;
417  }
418  int rawcru = rdh_utils::getCRU(ptr);
419  int rawendpoint = rdh_utils::getEndPoint(ptr);
420  if (o2::raw::RDHUtils::getDetectorField(ptr) == 2) {
421  tpcZSmetaPointers[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(ptr);
422  tpcZSmetaSizes[rawcru / 10][(rawcru % 10) * 2 + rawendpoint].emplace_back(count);
423  }
424  };
425  // the sequencer processes all inputs matching the filter and finds sequences of consecutive
426  // raw pages based on the matcher predicate, and calls the inserter for each sequence
427  DPLRawPageSequencer(pc.inputs(), filter)(isSameRdh, insertPages);
428 
429  int totalCount = 0;
430  for (unsigned int i = 0; i < GPUTrackingInOutZS::NSLICES; i++) {
431  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
432  tpcZSmetaPointers2[i][j] = tpcZSmetaPointers[i][j].data();
433  tpcZSmetaSizes2[i][j] = tpcZSmetaSizes[i][j].data();
434  tpcZS.slice[i].zsPtr[j] = tpcZSmetaPointers2[i][j];
435  tpcZS.slice[i].nZSPtr[j] = tpcZSmetaSizes2[i][j];
436  tpcZS.slice[i].count[j] = tpcZSmetaPointers[i][j].size();
437  totalCount += tpcZSmetaPointers[i][j].size();
438  }
439  }
440  } else if (mSpecConfig.decompressTPC) {
441  if (mSpecConfig.decompressTPCFromROOT) {
442  compClustersDummy = *pc.inputs().get<CompressedClustersROOT*>("input");
443  compClustersFlatDummy.setForward(&compClustersDummy);
444  pCompClustersFlat = &compClustersFlatDummy;
445  } else {
446  pCompClustersFlat = pc.inputs().get<CompressedClustersFlat*>("input").get();
447  }
448  } else if (!mSpecConfig.zsOnTheFly) {
449  if (mVerbosity) {
450  LOGF(info, "running tracking for sector(s) 0x%09x", mTPCSectorMask);
451  }
452  }
453 
454  const auto& inputsClustersDigits = getWorkflowTPCInput(pc, mVerbosity, getWorkflowTPCInput_mc, getWorkflowTPCInput_clusters, mTPCSectorMask, getWorkflowTPCInput_digits);
455  GPUTrackingInOutPointers ptrs;
456 
457  o2::globaltracking::RecoContainer inputTracksTRD;
458  decltype(o2::trd::getRecoInputContainer(pc, &ptrs, &inputTracksTRD)) trdInputContainer;
459  if (mSpecConfig.readTRDtracklets) {
460  o2::globaltracking::DataRequest dataRequestTRD;
461  dataRequestTRD.requestTracks(o2::dataformats::GlobalTrackID::getSourcesMask(o2::dataformats::GlobalTrackID::NONE), false);
462  inputTracksTRD.collectData(pc, dataRequestTRD);
463  trdInputContainer = std::move(o2::trd::getRecoInputContainer(pc, &ptrs, &inputTracksTRD));
464  }
465 
466  void* ptrEp[NSectors * NEndpoints] = {};
467  bool doInputDigits = false, doInputDigitsMC = false;
468  if (mSpecConfig.decompressTPC) {
