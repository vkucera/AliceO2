116 VisualisationEvent::VisualisationEvent(const VisualisationEvent& source, EVisualisationGroup filter, float minTime, float maxTime)
117 {
118  for (auto it = source.mTracks.begin(); it != source.mTracks.end(); ++it) {
119  if (it->getTime() < minTime) {
120  continue;
121  }
122  if (it->getTime() > maxTime) {
123  continue;
124  }
125  if (VisualisationEvent::mVis.contains[it->getSource()][filter]) {
126  this->mTracks.push_back(*it);
127  }
128  }
129  for (auto it = source.mClusters.begin(); it != source.mClusters.end(); ++it) {
130  if (VisualisationEvent::mVis.contains[it->getSource()][filter]) {
131  this->mClusters.push_back(*it);
132  }
133  }
134  for (auto it = source.mCalo.begin(); it != source.mCalo.end(); ++it) {
135  if (VisualisationEvent::mVis.contains[it->getSource()][filter]) {
136  this->mCalo.push_back(*it);
137  }
138  }
139 }
