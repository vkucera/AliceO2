58 int GPUDisplayBackend::InitBackend()
59 {
60  int retVal = InitBackendA();
61  if (retVal) {
62  return retVal;
63  }
64  if (mDisplay->cfg().noFreetype) {
65  return retVal;
66  }
67  std::string fontName = mDisplay->cfg().font;
68 #ifdef GPUCA_BUILD_EVENT_DISPLAY_FONTCONFIG
69  FcInit();
70  FcConfig* config = FcInitLoadConfigAndFonts();
71  FcPattern* pat = FcNameParse((const FcChar8*)fontName.c_str());
72  FcConfigSubstitute(config, pat, FcMatchPattern);
73  FcDefaultSubstitute(pat);
74  FcResult result;
75  FcPattern* font = FcFontMatch(config, pat, &result);
76  if (font && result == 0) {
77  FcChar8* file = nullptr;
78  if (FcPatternGetString(font, FC_FILE, 0, &file) == FcResultMatch) {
79  fontName = (char*)file;
80  }
81  } else {
82  GPUError("Coult not find font for pattern %s", fontName.c_str());
83  }
84  FcPatternDestroy(font);
85  FcPatternDestroy(pat);
86  FcConfigDestroy(config);
87  FcFini();
88 #endif // GPUCA_BUILD_EVENT_DISPLAY_FONTCONFIG
89 
90 #ifdef GPUCA_BUILD_EVENT_DISPLAY_FREETYPE
91  FT_Library ft;
92  FT_Face face;
93  if (FT_Init_FreeType(&ft)) {
94  GPUError("Error initializing freetype");
95  return 0;
96  }
97  if (FT_New_Face(ft, fontName.c_str(), 0, &face)) {
98  GPUError("Error loading freetypoe font");
99  return 0;
100  }
101 
102  int fontSize = mDisplay->cfg().fontSize;
103  mDisplay->drawTextFontSize() = fontSize;
104  if (smoothFont()) {
105  fontSize *= 4; // Font size scaled by 4, can be downsampled
106  }
107  FT_Set_Pixel_Sizes(face, 0, fontSize);
108 
109  for (unsigned int i = 0; i < 128; i++) {
110  if (FT_Load_Char(face, i, FT_LOAD_RENDER)) {
111  GPUError("Error loading freetype symbol");
112  return 0;
113  }
114  const auto& glyph = face->glyph;
115  addFontSymbol(i, glyph->bitmap.width, glyph->bitmap.rows, glyph->bitmap_left, glyph->bitmap_top, glyph->advance.x, glyph->bitmap.buffer);
116  }
117  initializeTextDrawing();
118  FT_Done_Face(face);
119  FT_Done_FreeType(ft);
120  mFreetypeInitialized = true;
121 #endif // GPUCA_BUILD_EVENT_DISPLAY_FREETYPE
122  return retVal;
123 }

137 void GPUDisplayBackend::fillIndirectCmdBuffer()
138 {
139  mCmdBuffer.clear();
140  mIndirectSliceOffset.resize(GPUCA_NSLICES);
141  // TODO: Check if this can be parallelized
142  for (int iSlice = 0; iSlice < GPUCA_NSLICES; iSlice++) {
143  mIndirectSliceOffset[iSlice] = mCmdBuffer.size();
144  for (unsigned int k = 0; k < mDisplay->vertexBufferStart()[iSlice].size(); k++) {
145  mCmdBuffer.emplace_back(mDisplay->vertexBufferCount()[iSlice][k], 1, mDisplay->vertexBufferStart()[iSlice][k], 0);
146  }
147  }
148 }

150 float GPUDisplayBackend::getDownsampleFactor(bool screenshot)
151 {
152  float factor = 1.0f;
153  int fsaa = mDisplay->cfgR().drawQualityDownsampleFSAA;
154  int screenshotScale = mDisplay->cfgR().screenshotScaleFactor;
155  if (fsaa) {
156  factor *= fsaa;
157  }
158  if (screenshotScale && screenshot) {
159  factor *= screenshotScale;
160  }
161  return factor;
162 }
