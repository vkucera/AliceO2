63  void init(o2::framework::InitContext& ic) final
64  {
65  // set up ADC value filling
66  if (ic.options().get<bool>("write-debug")) {
67  mDebugStream = std::make_unique<o2::utils::TreeStreamRedirector>("idc_vector_debug.root", "recreate");
68  }
69  auto pedestalFile = ic.options().get<std::string>("pedestal-url");
70  if (pedestalFile.length()) {
71  if (pedestalFile.find("ccdb") != std::string::npos) {
72  if (pedestalFile.find("-default") != std::string::npos) {
73  pedestalFile = o2::base::NameConf::getCCDBServer();
74  }
75  LOGP(info, "Loading pedestals from ccdb: {}", pedestalFile);
76  auto& cdb = o2::ccdb::BasicCCDBManager::instance();
77  cdb.setURL(pedestalFile);
78  if (cdb.isHostReachable()) {
79  auto pedestalNoise = cdb.get<std::unordered_map<std::string, CalPad>>("TPC/Calib/PedestalNoise");
80  try {
81  if (!pedestalNoise) {
82  throw std::runtime_error("Couldn't retrieve PedestaNoise map");
83  }
84  mPedestal = std::make_unique<CalPad>(pedestalNoise->at("Pedestals"));
85  } catch (const std::exception& e) {
86  LOGP(fatal, "could not load pedestals from {} ({}), required for IDC processing", pedestalFile, e.what());
87  }
88  } else {
89  LOGP(fatal, "ccdb access to {} requested, but host is not reachable. Cannot load pedestals, required for IDC processing", pedestalFile);
90  }
91  } else {
92  LOGP(info, "Loading pedestals from file: {}", pedestalFile);
93  auto calPads = utils::readCalPads(pedestalFile, "Pedestals");
94  if (calPads.size() != 1) {
95  LOGP(fatal, "Pedestal could not be loaded from file {}, required for IDC processing", pedestalFile);
96  } else {
97  mPedestal.reset(calPads[0]);
98  }
99  }
100  } else {
101  LOGP(error, "No pedestal file set, IDCs will be without pedestal subtraction!");
102  }
103 
104  initIDC();
105  }

107  void run(o2::framework::ProcessingContext& pc) final
108  {
109  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}}; // TODO: Change to IDC when changed in DD
110  const auto& mapper = Mapper::instance();
111 
112  uint32_t heartbeatOrbit = 0;
113  uint32_t heartbeatBC = 0;
114  uint32_t tfCounter = 0;
115  bool first = true;
116 
117  CalPad* pedestals = mPedestal.get();
118 
119  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
120  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
121  tfCounter = dh->tfCounter;
122 
123  // ---| data loop |---
124  const gsl::span<const char> raw = pc.inputs().get<gsl::span<char>>(ref);
125  o2::framework::RawParser parser(raw.data(), raw.size());
126  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
127  const auto size = it.size();
128  // skip empty packages (HBF open)
129  if (size == 0) {
130  continue;
131  }
132 
133  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
134  if (!rdhPtr) {
135  throw std::runtime_error("could not get RDH from packet");
136  }
137 
138  // ---| extract hardware information to do the processing |---
139  const auto feeId = (FEEIDType)RDHUtils::getFEEID(*rdhPtr);
140  const auto link = rdh_utils::getLink(feeId);
141  const uint32_t cruID = rdh_utils::getCRU(feeId);
142  const auto endPoint = rdh_utils::getEndPoint(feeId);
143  const auto detField = RDHUtils::getDetectorField(*rdhPtr);
144 
145  // only select IDCs
146  // ToDo: cleanup once IDCs will be propagated not as RAWDATA, but IDC.
147  if ((detField != (decltype(detField))RawDataType::IDC) || (link != rdh_utils::IDCLinkID)) {
148  continue;
149  }
150  LOGP(info, "IDC Processing firstTForbit {:9}, tfCounter {:5}, run {:6}, feeId {:6} ({:3}/{}/{:2})", dh->firstTForbit, dh->tfCounter, dh->runNumber, feeId, cruID, endPoint, link);
151 
152  if (std::find(mCRUs.begin(), mCRUs.end(), cruID) == mCRUs.end()) {
153  LOGP(error, "IDC CRU {:3} not configured in CRUs, skipping", cruID);
154  continue;
155  }
156 
157  const CRU cru(cruID);
158  const int sector = cru.sector();
159  const auto& partInfo = mapper.getPartitionInfo(cru.partition());
160  const int fecLinkOffsetCRU = (partInfo.getNumberOfFECs() + 1) / 2;
161  const int fecSectorOffset = partInfo.getSectorFECOffset();
162  const GlobalPadNumber regionPadOffset = Mapper::GLOBALPADOFFSET[cru.region()];
163  const GlobalPadNumber numberPads = Mapper::PADSPERREGION[cru.region()];
164  int sampaOnFEC{}, channelOnSAMPA{};
165  auto& idcVec = mIDCvectors[cruID];
166  auto& infoVec = mIDCInfos[cruID];
167 
168  assert(size == sizeof(idc::Container));
169  auto data = it.data();
170  auto& idcs = *((idc::Container*)(data));
171  const uint32_t orbit = idcs.header.heartbeatOrbit;
172  const uint32_t bc = idcs.header.heartbeatBC;
173  // LOGP(info, "IDC Procssing orbit/BC: {:9}/{:4}", orbit, bc);
174 
175  auto infoIt = std::find(infoVec.begin(), infoVec.end(), orbit);
176  if (!infoVec.size()) {
177  infoVec.emplace_back(orbit, bc);
178  infoIt = infoVec.end() - 1;
179  } else if (infoIt == infoVec.end()) {
180  auto& lastInfo = infoVec.back();
181  if ((orbit - lastInfo.heartbeatOrbit) != mNOrbitsIDC) {
182  LOGP(error, "received packet with invalid jump in idc orbit ({} - {} == {} != {})", orbit, lastInfo.heartbeatOrbit, orbit - lastInfo.heartbeatOrbit, mNOrbitsIDC);
183  }
184  infoVec.emplace_back(orbit, bc);
185  infoIt = infoVec.end() - 1;
186  }
187 
188  // check if end poit was already processed
189  auto& lastInfo = *infoIt;
190  if (lastInfo.wasEPseen(endPoint)) {
191  LOGP(info, "Already received another data packet for CRU {}, ep {}, orbit {}, bc {}", cruID, endPoint, orbit, bc);
192  continue;
193  }
194 
195  lastInfo.setEPseen(endPoint);
196  // idc value offset in present time frame
197  const size_t idcOffset = std::distance(infoVec.begin(), infoIt);
198 
199  // TODO: for debugging, remove later
200  // LOGP(info, "processing IDCs for CRU {}, ep {}, feeId {:6} ({:3}/{}/{:2}), detField: {}, orbit {}, bc {}, idcOffset {}, idcVec size {}, epSeen {:02b}", cruID, endPoint, feeId, cruID, endPoint, link, detField, orbit, bc, idcOffset, idcVec.size(), lastInfo.epSeen);
201 
202  const float norm = 1. / float(mTimeStampsPerIntegrationInterval);
203  for (uint32_t iLink = 0; iLink < idc::Links; ++iLink) {
204  if (!idcs.hasLink(iLink)) {
205  continue;
206  }
207  const int fecInSector = iLink + endPoint * fecLinkOffsetCRU + fecSectorOffset;
208 
209  for (uint32_t iChannel = 0; iChannel < idc::Channels; ++iChannel) {
210  auto val = idcs.getChannelValueFloat(iLink, iChannel);
211  Mapper::getSampaAndChannelOnFEC(cruID, iChannel, sampaOnFEC, channelOnSAMPA);
212  const GlobalPadNumber padInSector = mapper.globalPadNumber(fecInSector, sampaOnFEC, channelOnSAMPA);
213  if (pedestals) {
214  val -= pedestals->getValue(sector, padInSector) * mTimeStampsPerIntegrationInterval;
215  val *= norm;
216  }
217  const GlobalPadNumber padInRegion = padInSector - regionPadOffset;
218  const GlobalPadNumber vectorPosition = padInRegion + idcOffset * numberPads;
219  // TODO: for debugging, remove later
220  // auto rawVal = idcs.getChannelValue(iLink, iChannel);
221  // auto rawValF = idcs.getChannelValueFloat(iLink, iChannel);
222  // LOGP(info, "filling channel {}, link {}, fecLinkOffsetCRU {:2}, fecSectorOffset {:3}, fecInSector {:3}, idcVec[{} ({})] = {} ({} / {})", iChannel, iLink, fecLinkOffsetCRU, fecSectorOffset, fecInSector, vectorPosition, padInRegion, val, rawVal, rawValF);
223  idcVec[vectorPosition] = val;
224  }
225  }
226  }
227  }
228 
229  if (mDebugStream) {
230  writeDebugOutput(tfCounter);
231  }
232  snapshotIDCs(pc.outputs());
233  }

248  private:
250  struct IDCInfo {
251  IDCInfo() = default;
252  IDCInfo(const IDCInfo&) = default;
253  IDCInfo(uint32_t orbit, uint16_t bc) : heartbeatOrbit(orbit), heartbeatBC(bc) {}
254 
255  uint32_t heartbeatOrbit{0};
256  uint16_t heartbeatBC{0};
257  uint16_t epSeen{0};
258 
259  bool operator==(const uint32_t orbit) const { return (heartbeatOrbit == orbit); }
260  bool operator==(const IDCInfo& inf) const { return (inf.heartbeatOrbit == heartbeatOrbit) && (inf.heartbeatBC == heartbeatBC) && (inf.epSeen == epSeen); }
261  void setEPseen(uint32_t ep) { epSeen |= uint16_t(1 << ep); }
262  bool wasEPseen(uint32_t ep) const { return epSeen & uint16_t(1 << ep); }
263  bool matches(uint32_t orbit, int16_t bc) const { return ((heartbeatOrbit == orbit) && (heartbeatBC == bc)); }
264  bool hasBothEPs() const { return epSeen == 3; }
265  };

277  void snapshotIDCs(DataAllocator& output)
278  {
279  LOGP(info, "snapshotIDCs");
280 
281  // check integrety of data between CRUs
282  size_t orbitsInTF = 0;
283  std::vector<IDCInfo> const* infVecComp = nullptr;
284  std::vector<uint64_t> orbitBCInfo;
285 
286  for (const auto& [cru, infVec] : mIDCInfos) {
287 
288  for (const auto& inf : infVec) {
289  if (!inf.hasBothEPs()) {
290  LOGP(fatal, "IDC CRU {:3}: data missing at ({:8}, {:4}) for one or both end points {:02b}", cru, inf.heartbeatOrbit, inf.heartbeatBC, inf.epSeen);
291  }
292  }
293 
294  if (!infVecComp) {
295  infVecComp = &infVec;
296  orbitsInTF = infVec.size();
297  std::for_each(infVec.begin(), infVec.end(), [&orbitBCInfo](const auto& inf) { orbitBCInfo.emplace_back((uint64_t(inf.heartbeatOrbit) << 32) + uint64_t(inf.heartbeatBC)); });
298  continue;
299  }
300 
301  if (orbitsInTF != infVec.size()) {
302  LOGP(fatal, "IDC CRU {:3}: unequal number of IDC values {} != {}", cru, orbitsInTF, infVec.size());
303  }
304 
305  if (!std::equal(infVecComp->begin(), infVecComp->end(), infVec.begin())) {
306  LOGP(fatal, "IDC CRU {:3}: mismatch in orbits");
307  }
308  }
309 
310  // send data
311  for (auto& [cru, idcVec] : mIDCvectors) {
312  idcVec.resize(Mapper::PADSPERREGION[CRU(cru).region()] * orbitsInTF);
313  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
314  LOGP(info, "Sending IDCs for CRU {} of size {}", cru, idcVec.size());
315  output.snapshot(Output{gDataOriginTPC, "IDCVECTOR", subSpec}, idcVec);
316  output.snapshot(Output{gDataOriginTPC, "IDCORBITS", subSpec}, orbitBCInfo);
317  }
318 
319  // clear output
320  initIDC();
321  }

339  void writeDebugOutput(uint32_t tfCounter)
340  {
341  const auto& mapper = Mapper::instance();
342 
343  mDebugStream->GetFile()->cd();
344  auto& stream = (*mDebugStream) << "idcs";
345  uint32_t seen = 0;
346  static uint32_t firstOrbit = std::numeric_limits<uint32_t>::max();
347 
348  for (auto cru : mCRUs) {
349  if (mIDCInfos.find(cru) == mIDCInfos.end()) {
350  continue;
351  }
352  auto& infos = mIDCInfos[cru];
353  auto& idcVec = mIDCvectors[cru];
354 
355  for (int i = 0; i < infos.size(); ++i) {
356  auto& info = infos[i];
357 
358  if (firstOrbit == std::numeric_limits<uint32_t>::max()) {
359  firstOrbit = info.heartbeatOrbit;
360  }
361  auto idcFirst = idcVec.begin() + i * Mapper::PADSPERREGION[cru % Mapper::NREGIONS];
362  auto idcLast = idcFirst + Mapper::PADSPERREGION[cru % Mapper::NREGIONS];
363  std::vector<float> idcs(idcFirst, idcLast);
364  std::vector<short> cpad(idcs.size());
365  std::vector<short> row(idcs.size());
366  for (int ipad = 0; ipad < idcs.size(); ++ipad) {
367  const auto& padPos = mapper.padPos(ipad + Mapper::GLOBALPADOFFSET[cru % Mapper::NREGIONS]);
368  row[ipad] = (short)padPos.getRow();
369  const short pads = (short)mapper.getNumberOfPadsInRowSector(row[ipad]);
370  cpad[ipad] = (short)padPos.getPad() - pads / 2;
371  }
372  auto idcSort = idcs;
373  std::sort(idcSort.begin(), idcSort.end());
374  const auto idcSize = idcSort.size();
375  float median = idcSize % 2 ? idcSort[idcSize / 2] : (idcSort[idcSize / 2] + idcSort[idcSize / 2 - 1]) / 2.f;
376  // outlier removal
377  auto itEnd = idcSort.end();
378  while (std::abs(*(itEnd - 1) - median) > 40) {
379  --itEnd;
380  }
381 
382  float mean = 0;
383  const auto nForMean = std::distance(idcSort.begin(), itEnd);
384  if (nForMean > 0) {
385  mean = std::accumulate(idcSort.begin(), itEnd, 0.f) / float(nForMean);
386  }
387  uint32_t outliers = uint32_t(idcSort.size() - nForMean);
388 
389  stream << "cru=" << cru
390  << "entry=" << i
391  << "epSeen=" << info.epSeen
392  << "tfCounter=" << tfCounter
393  << "firstOrbit=" << firstOrbit
394  << "orbit=" << info.heartbeatOrbit
395  << "bc=" << info.heartbeatBC
396  << "idcs=" << idcs
397  << "cpad=" << cpad
398  << "row=" << row
399  << "outliers=" << outliers
400  << "idc_mean=" << mean
401  << "idc_median=" << median
402  << "\n";
403  }
404  }
405  }
