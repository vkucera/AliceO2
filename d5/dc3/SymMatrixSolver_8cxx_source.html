32 void SymMatrixSolver::solve()
33 {
34  // Upper Triangulization
35  for (int i = 0; i < mN; i++) {
36  double* rowI = &mA[i * mShift];
37  double* rowIb = &mA[i * mShift + mN];
38  double c = (fabs(rowI[i]) > 1.e-10) ? 1. / rowI[i] : 0.;
39  double* rowJ = rowI + mShift;
40  for (int j = i + 1; j < mN; j++, rowJ += mShift) { // row j
41  if (rowI[j] != 0.) {
42  double aij = c * rowI[j]; // A[i][j] / A[i][i]
43  for (int k = j; k < mShift; k++) {
44  rowJ[k] -= aij * rowI[k]; // A[j][k] -= A[i][k]/A[i][i]*A[j][i]
45  }
46  rowI[j] = aij; // A[i][j] /= A[i][i]
47  }
48  }
49  for (int k = 0; k < mM; k++) {
50  rowIb[k] *= c;
51  }
52  }
53  // Diagonalization
54  for (int i = mN - 1; i >= 0; i--) {
55  double* rowIb = &mA[i * mShift + mN];
56  double* rowJb = rowIb - mShift;
57  for (int j = i - 1; j >= 0; j--, rowJb -= mShift) { // row j
58  double aji = mA[j * mShift + i];
59  if (aji != 0.) {
60  for (int k = 0; k < mM; k++) {
61  rowJb[k] -= aji * rowIb[k];
62  }
63  }
64  }
65  }
66 }
