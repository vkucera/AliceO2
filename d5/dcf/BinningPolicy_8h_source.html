24 template <typename C, typename... Cs>
25 struct BinningPolicy {
26  BinningPolicy(std::array<std::vector<double>, sizeof...(Cs) + 1> bins, bool ignoreOverflows = true) : mBins(bins), mIgnoreOverflows(ignoreOverflows)
27  {
28  static_assert(sizeof...(Cs) < 3, "No default binning for more than 3 columns, you need to implement a binning class yourself");
29  for (int i = 0; i < sizeof...(Cs) + 1; i++) {
30  expandConstantBinning(bins[i], i);
31  }
32  }
33 
34  int getBin(std::tuple<typename C::type, typename Cs::type...> const& data) const
35  {
36  unsigned int i = 2, j = 2, k = 2;
37  if (this->mIgnoreOverflows) {
38  // underflow
39  if (std::get<0>(data) < this->mBins[0][1]) { // xBins[0] is a dummy VARIABLE_WIDTH
40  return -1;
41  }
42  if constexpr (sizeof...(Cs) > 0) {
43  if (std::get<1>(data) < this->mBins[1][1]) { // this->mBins[1][0] is a dummy VARIABLE_WIDTH
44  return -1;
45  }
46  }
47  if constexpr (sizeof...(Cs) > 1) {
48  if (std::get<2>(data) < this->mBins[2][1]) { // this->mBins[2][0] is a dummy VARIABLE_WIDTH
49  return -1;
50  }
51  }
52  } else {
53  i = 1;
54  j = 1;
55  k = 1;
56  }
57 
58  for (; i < this->mBins[0].size(); i++) {
59  if (std::get<0>(data) < this->mBins[0][i]) {
60 
61  if constexpr (sizeof...(Cs) > 0) {
62  for (; j < this->mBins[1].size(); j++) {
63  if (std::get<1>(data) < this->mBins[1][j]) {
64 
65  if constexpr (sizeof...(Cs) > 1) {
66  for (; k < this->mBins[2].size(); k++) {
67  if (std::get<2>(data) < this->mBins[2][k]) {
68  return getBinAt(i, j, k);
69  }
70  }
71  if (this->mIgnoreOverflows) {
72  return -1;
73  }
74  }
75 
76  // overflow for this->mBins[2] only
77  return getBinAt(i, j, k);
78  }
79  }
80 
81  if (this->mIgnoreOverflows) {
82  return -1;
83  }
84 
85  // overflow for this->mBins[1] only
86  if constexpr (sizeof...(Cs) > 1) {
87  for (k = 2; k < this->mBins[2].size(); k++) {
88  if (std::get<2>(data) < this->mBins[2][k]) {
89  return getBinAt(i, j, k);
90  }
91  }
92  }
93  }
94 
95  // overflow for this->mBins[2] and this->mBins[1]
96  return getBinAt(i, j, k);
97  }
98  }
99 
100  if (this->mIgnoreOverflows) {
101  // overflow
102  return -1;
103  }
104 
105  // overflow for this->mBins[0] only
106  if constexpr (sizeof...(Cs) > 0) {
107  for (j = 2; j < this->mBins[1].size(); j++) {
108  if (std::get<1>(data) < this->mBins[1][j]) {
109 
110  if constexpr (sizeof...(Cs) > 1) {
111  for (k = 2; k < this->mBins[2].size(); k++) {
112  if (std::get<2>(data) < this->mBins[2][k]) {
113  return getBinAt(i, j, k);
114  }
115  }
116  }
117 
118  // overflow for this->mBins[0] and this->mBins[2]
119  return getBinAt(i, j, k);
120  }
121  }
122  }
123 
124  // overflow for this->mBins[0] and this->mBins[1]
125  if constexpr (sizeof...(Cs) > 1) {
126  for (k = 2; k < this->mBins[2].size(); k++) {
127  if (std::get<2>(data) < this->mBins[2][k]) {
128  return getBinAt(i, j, k);
129  }
130  }
131  }
132 
133  // overflow for all bins
134  return getBinAt(i, j, k);
135  }
136 
137  // Note: Overflow / underflow bin -1 is not included
138  int getXBinsCount() const
139  {
140  return this->mBins[0].size() - 1 - getOverflowShift();
141  }
142 
143  // Note: Overflow / underflow bin -1 is not included
144  int getYBinsCount() const
145  {
146  if constexpr (sizeof...(Cs) == 0) {
147  return 0;
148  }
149  return this->mBins[1].size() - 1 - getOverflowShift();
150  }
151 
152  // Note: Overflow / underflow bin -1 is not included
153  int getZBinsCount() const
154  {
155  if constexpr (sizeof...(Cs) < 2) {
156  return 0;
157  }
158  return this->mBins[2].size() - 1 - getOverflowShift();
159  }
160 
161  // Note: Overflow / underflow bin -1 is not included
162  int getAllBinsCount() const
163  {
164  if constexpr (sizeof...(Cs) == 0) {
165  return getXBinsCount();
166  }
167  if constexpr (sizeof...(Cs) == 1) {
168  return getXBinsCount() * getYBinsCount();
169  }
170  if constexpr (sizeof...(Cs) == 2) {
171  return getXBinsCount() * getYBinsCount() * getZBinsCount();
172  }
173  return -1;
174  }
175 
176  using persistent_columns_t = framework::selected_pack<o2::soa::is_persistent_t, C, Cs...>;
177 
178  private:
179  // We substract 1 to account for VARIABLE_WIDTH in the bins vector
180  // We substract second 1 if we omit values below minima (underflow, mapped to -1)
181  // Otherwise we add 1 and we get the number of bins including those below and over the outer edges
182  int getBinAt(unsigned int iRaw, unsigned int jRaw, unsigned int kRaw) const
183  {
184  int shiftBinsWithoutOverflow = getOverflowShift();
185  unsigned int i = iRaw - 1 - shiftBinsWithoutOverflow;
186  unsigned int j = jRaw - 1 - shiftBinsWithoutOverflow;
187  unsigned int k = kRaw - 1 - shiftBinsWithoutOverflow;
188  auto xBinsCount = getXBinsCount();
189  if constexpr (sizeof...(Cs) == 0) {
190  return i;
191  } else if constexpr (sizeof...(Cs) == 1) {
192  return i + j * xBinsCount;
193  } else if constexpr (sizeof...(Cs) == 2) {
194  return i + j * xBinsCount + k * xBinsCount * (this->mBins[1].size() - 1 - shiftBinsWithoutOverflow);
195  } else {
196  return -1;
197  }
198  }
199 
200  int getOverflowShift() const
201  {
202  return mIgnoreOverflows ? 1 : -1;
203  }
204 
205  void expandConstantBinning(std::vector<double> const& bins, int ind)
206  {
207  if (bins[0] != VARIABLE_WIDTH) {
208  int nBins = static_cast<int>(bins[0]);
209  this->mBins[ind].clear();
210  this->mBins[ind].resize(nBins + 2);
211  this->mBins[ind][0] = VARIABLE_WIDTH;
212  for (int i = 0; i <= nBins; i++) {
213  this->mBins[ind][i + 1] = bins[1] + i * (bins[2] - bins[1]) / nBins;
214  }
215  }
216  }
217 
218  std::array<std::vector<double>, sizeof...(Cs) + 1> mBins;
219  bool mIgnoreOverflows;
220 };
