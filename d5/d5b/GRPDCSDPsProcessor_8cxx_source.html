38 int GRPDCSDPsProcessor::process(const gsl::span<const DPCOM> dps)
39 {
40 
41  // first we check which DPs are missing - if some are, it means that
42  // the delta map was sent
43  if (mVerbose) {
44  LOG(info) << "\n\n\nProcessing new TF\n-----------------";
45  }
46  if (!mFirstTimeSet) {
47  mFirstTime = mStartValidity;
48  mFirstTimeSet = true;
49  }
50  mMagFieldHelper.updated = false;
51 
52  mUpdateLHCIFInfo = false; // by default, we do not foresee a new entry in the CCDB for the LHCIF DPs
53 
54  // now we process all DPs, one by one
55  for (const auto& it : dps) {
56  processDP(it);
57  mPids[it.id] = true;
58  }
59 
60  if (isMagFieldUpdated()) {
61  updateMagFieldCCDB();
62  }
63  mCallSlice++;
64  return 0;
65 }

68 int GRPDCSDPsProcessor::processDP(const DPCOM& dpcom)
69 {
70 
71  // processing single DP
72 
73  auto& dpid = dpcom.id;
74  const auto& type = dpid.get_type();
75  auto& val = dpcom.data;
76  if (mVerbose) {
77  if (type == DPVAL_DOUBLE) {
78  LOG(info);
79  LOG(info) << "Processing DP = " << dpcom << ", with double value = " << o2::dcs::getValue<double>(dpcom);
80  } else if (type == DPVAL_BOOL) {
81  LOG(info);
82  LOG(info) << "Processing DP = " << dpcom << ", with bool value = " << o2::dcs::getValue<bool>(dpcom);
83  } else if (type == DPVAL_STRING) {
84  LOG(info);
85  LOG(info) << "Processing DP = " << dpcom << ", with string value = " << o2::dcs::getValue<std::string>(dpcom);
86  }
87  }
88  auto flags = val.get_flags();
89  if (processFlags(flags, dpid.get_alias()) == 0) {
90  // now I need to access the correct element
91  std::string aliasStr(dpid.get_alias());
92  // LOG(info) << "alias 0 = " << aliasStr;
93  // B-field DPs
94  if (aliasStr == "L3Current") {
95  mMagFieldHelper.updateCurL3(static_cast<float>(o2::dcs::getValue<double>(dpcom)));
96  } else if (aliasStr == "DipoleCurrent") {
97  mMagFieldHelper.updateCurDip(static_cast<float>(o2::dcs::getValue<double>(dpcom)));
98  } else if (aliasStr == "L3Polarity") {
99  mMagFieldHelper.updateSignL3(o2::dcs::getValue<bool>(dpcom)); // true is negative
100  } else if (aliasStr == "DipolePolarity") {
101  mMagFieldHelper.updateSignDip(o2::dcs::getValue<bool>(dpcom)); // true is negative
102  } else if (processEnvVar(dpcom)) { // environment variables
103  } else if (processCollimators(dpcom)) {
104  } else {
105  processLHCIFDPs(dpcom);
106  }
107  }
108  return 0;
109 }

142 bool GRPDCSDPsProcessor::processPairD(const DPCOM& dpcom, const std::string& alias, std::unordered_map<std::string, std::vector<std::pair<uint64_t, double>>>& mapToUpdate)
143 {
144 
145  // function to process Data Points that is stored in a pair
146 
147  auto& vect = mapToUpdate[alias]; // we also create at this point the vector in the map if it did not exist yet
148 
149  auto& dpcomdata = dpcom.data;
150  auto& dpid = dpcom.id;
151  std::string aliasStr(dpid.get_alias());
152 
153  if (aliasStr == alias) {
154  if (mVerbose) {
155  LOG(info) << "It matches the requested string " << alias << ", let's check if the value needs to be updated";
156  }
157  updateVector(dpid, mapToUpdate[alias], aliasStr, dpcomdata.get_epoch_time(), o2::dcs::getValue<double>(dpcom));
158  return true;
159  }
160  return false;
161 }

164 bool GRPDCSDPsProcessor::processPairS(const DPCOM& dpcom, const std::string& alias, std::pair<uint64_t, std::string>& p, bool& flag)
165 {
166 
167  // function to process string Data Points that is stored in a pair
168 
169  auto& dpcomdata = dpcom.data;
170  auto& dpid = dpcom.id;
171  std::string aliasStr(dpid.get_alias());
172 
173  if (aliasStr == alias) {
174  auto val = o2::dcs::getValue<std::string>(dpcom);
175  if (mVerbose) {
176  LOG(info) << "It matches the requested string " << alias << ", let's check if the value needs to be updated";
177  LOG(info) << "old value = " << p.second << ", new value = " << val << ", call " << mCallSlice;
178  }
179  if (mCallSlice == 0 || (val != p.second)) {
180  p.first = dpcomdata.get_epoch_time();
181  p.second = val;
182  flag = true;
183  if (mVerbose) {
184  LOG(info) << "Updating value " << alias << " with timestamp " << p.first;
185  }
186  }
187  return true;
188  }
189  return false;
190 }

194 bool GRPDCSDPsProcessor::compareToLatest(std::pair<uint64_t, double>& p, double val)
195 {
196 
197  // check if the content of the pair should be updated
198 
199  if (mVerbose) {
200  LOG(info) << "old value = " << p.second << ", new value = " << val << ", absolute difference = " << std::abs(p.second - val) << " call " << mCallSlice;
201  }
202  if (mCallSlice == 0 || (std::abs(p.second - val) > 0.5e-7 * (std::abs(p.second) + std::abs(val)))) {
203  if (mVerbose) {
204  LOG(info) << "value will be updated";
205  }
206  return true;
207  }
208  if (mVerbose) {
209  LOG(info) << "value will not be updated";
210  }
211  return false;
212 }

216 bool GRPDCSDPsProcessor::processLHCIFDPs(const DPCOM& dpcom)
217 {
218 
219  // function to process the remaining LHCIF DPs
220 
221  auto& dpcomdata = dpcom.data;
222  auto& dpid = dpcom.id;
223  std::string aliasStr(dpid.get_alias());
224  if (mVerbose) {
225  LOG(info) << "Processing LHCIF DP " << aliasStr;
226  }
227  const auto& type = dpid.get_type();
228  double val = -99999999;
229  if (type == DPVAL_DOUBLE || type == RAW_DOUBLE) {
230  val = o2::dcs::getValue<double>(dpcom);
231  }
232 
233  for (int ibeam = 1; ibeam <= 2; ++ibeam) {
234  if (aliasStr.find(fmt::format("{}{}", "LHC_IntensityBeam", ibeam)) != string::npos) {
235  updateVector(dpid, mLHCInfo.mIntensityBeam[ibeam - 1], aliasStr, dpcomdata.get_epoch_time(), val);
236  return true;
237  }
238  }
239 
240  if (aliasStr.find("BPTX") != string::npos) {
241  if (aliasStr == "BPTX_deltaT_B1_B2") {
242  updateVector(dpid, mLHCInfo.mBPTXdeltaT, aliasStr, dpcomdata.get_epoch_time(), val);
243  return true;
244  }
245  if (aliasStr == "BPTX_deltaTRMS_B1_B2") {
246  updateVector(dpid, mLHCInfo.mBPTXdeltaTRMS, aliasStr, dpcomdata.get_epoch_time(), val);
247  return true;
248  }
249  for (int ibeam = 1; ibeam <= 2; ++ibeam) {
250  if (aliasStr == fmt::format("{}{}", "BPTX_Phase_B", ibeam)) {
251  updateVector(dpid, mLHCInfo.mBPTXPhase[ibeam - 1], aliasStr, dpcomdata.get_epoch_time(), val);
252  return true;
253  } else if (aliasStr == fmt::format("{}{}", "BPTX_PhaseRMS_B", ibeam)) {
254  updateVector(dpid, mLHCInfo.mBPTXPhaseRMS[ibeam - 1], aliasStr, dpcomdata.get_epoch_time(), val);
255  return true;
256  } else if (aliasStr == fmt::format("{}{}", "BPTX_Phase_Shift_B", ibeam)) {
257  updateVector(dpid, mLHCInfo.mBPTXPhaseShift[ibeam - 1], aliasStr, dpcomdata.get_epoch_time(), val);
258  return true;
259  }
260  }
261  return true;
262  }
263  if (aliasStr == "ALI_Lumi_Total_Inst") {
264  updateVector(dpid, mLHCInfo.mInstLumi, aliasStr, dpcomdata.get_epoch_time(), val);
265  return true;
266  }
267 
268  for (int ibkg = 1; ibkg <= 3; ++ibkg) {
269  if (aliasStr.find(fmt::format("{}{}", "ALI_Background", ibkg)) != string::npos) {
270  updateVector(dpid, mLHCInfo.mBackground[ibkg - 1], aliasStr, dpcomdata.get_epoch_time(), val);
271  return true;
272  }
273  }
274 
275  if (processPairS(dpcom, "ALI_Lumi_Source_Name", mLHCInfo.mLumiSource, mUpdateLHCIFInfo)) {
276  return true;
277  }
278  if (processPairS(dpcom, "MACHINE_MODE", mLHCInfo.mMachineMode, mUpdateLHCIFInfo)) {
279  return true;
280  }
281  if (processPairS(dpcom, "BEAM_MODE", mLHCInfo.mBeamMode, mUpdateLHCIFInfo)) {
282  return true;
283  }
284 
285  // LOG(error) << "DP " << aliasStr << " not known for GRP, please check";
286  return false;
287 }

291 void GRPDCSDPsProcessor::updateMagFieldCCDB()
292 {
293  // we need to update a CCDB for the B field --> let's prepare the CCDBInfo
294  if (mVerbose) {
295  LOG(info) << "At least one DP related to B field changed --> we will update CCDB with startTime " << mStartValidity;
296  }
297  if (mMagFieldHelper.isSet != (0x1 << 4) - 1) {
298  LOG(alarm) << "Magnetic field was updated but not all fields were set: no FBI was seen?";
299  mMagFieldHelper.updated = false;
300  return;
301  }
302  mMagField.setL3Current(mMagFieldHelper.negL3 ? -mMagFieldHelper.curL3 : mMagFieldHelper.curL3);
303  mMagField.setDipoleCurrent(mMagFieldHelper.negDip ? -mMagFieldHelper.curDip : mMagFieldHelper.curDip);
304  std::map<std::string, std::string> md;
305  md["responsible"] = "Chiara Zampolli";
306  o2::calibration::Utils::prepareCCDBobjectInfo(mMagField, mccdbMagFieldInfo, "GLO/Config/GRPMagField", md, mStartValidity, mStartValidity + o2::ccdb::CcdbObjectInfo::MONTH);
307 }

373 void GRPDCSDPsProcessor::updateVector(const DPID& dpid, std::vector<std::pair<uint64_t, double>>& vect, std::string alias, uint64_t timestamp, double val)
374 {
375  printVectorInfo(vect, 0);
376  bool updateFlag = false;
377 
378  if (mPids[dpid] == false) { // let's remove the first value when it is the leftover from the previous processing, since we now have a newer one
379  if (mVerbose) {
380  LOG(info) << "We will clear the existing vector, since it is the very first time we receive values for it and we have a dummy one, or the only value present is from the previous processing, so it is old";
381  }
382  vect.clear(); // won't hurt if the vector is empty as at the very beginning of the processing
383  updateFlag = true;
384  } else { // we are accumulating entries in the vector already
385  if (mVerbose) {
386  LOG(info) << "We will just update the existing vector without clearing it";
387  }
388  updateFlag = compareToLatest(vect.back(), val);
389  }
390 
391  // add new value if needed
392  if (updateFlag) {
393  vect.emplace_back(timestamp, val);
394  if (mVerbose) {
395  LOG(info) << "Adding value " << val << " with timestamp " << timestamp << " to vector for DP " << alias;
396  }
397  }
398  printVectorInfo(vect, 1);
399 }
