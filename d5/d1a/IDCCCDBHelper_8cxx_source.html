362 template <typename DataT>
363 void o2::tpc::IDCCCDBHelper<DataT>::dumpToTree(int integrationIntervals, const char* outFileName) const
364 {
365  const Mapper& mapper = Mapper::instance();
366  o2::utils::TreeStreamRedirector pcstream(outFileName, "RECREATE");
367  pcstream.GetFile()->cd();
368 
369  if (integrationIntervals <= 0) {
370  integrationIntervals = std::min(getNIntegrationIntervalsIDCDelta(Side::A), getNIntegrationIntervalsIDCDelta(Side::C));
371  }
372 
373  for (unsigned int integrationInterval = 0; integrationInterval < integrationIntervals; ++integrationInterval) {
374  const unsigned int nIDCsSector = Mapper::getPadsInSector() * Mapper::NSECTORS;
375  std::vector<int> vRow(nIDCsSector);
376  std::vector<int> vPad(nIDCsSector);
377  std::vector<float> vXPos(nIDCsSector);
378  std::vector<float> vYPos(nIDCsSector);
379  std::vector<float> vGlobalXPos(nIDCsSector);
380  std::vector<float> vGlobalYPos(nIDCsSector);
381  std::vector<float> idcs(nIDCsSector);
382  std::vector<float> idcsZero(nIDCsSector);
383  std::vector<float> idcsDelta(nIDCsSector);
384  std::vector<unsigned int> sectorv(nIDCsSector);
385 
386  unsigned int index = 0;
387  for (unsigned int sector = 0; sector < Mapper::NSECTORS; ++sector) {
388  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
389  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
390  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
391  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
392  const auto padTmp = (sector < SECTORSPERSIDE) ? ipad : (Mapper::PADSPERROW[region][irow] - ipad - 1); // C-Side is mirrored
393  const auto& padPosLocal = mapper.padPos(padNum);
394  vRow[index] = padPosLocal.getRow();
395  vPad[index] = padPosLocal.getPad();
396  vXPos[index] = mapper.getPadCentre(padPosLocal).X();
397  vYPos[index] = mapper.getPadCentre(padPosLocal).Y();
398  const GlobalPosition2D globalPos = mapper.LocalToGlobal(LocalPosition2D(vXPos[index], vYPos[index]), Sector(sector));
399  vGlobalXPos[index] = globalPos.X();
400  vGlobalYPos[index] = globalPos.Y();
401  idcs[index] = getIDCVal(sector, region, irow, padTmp, integrationInterval);
402  idcsZero[index] = getIDCZeroVal(sector, region, irow, padTmp);
403  idcsDelta[index] = getIDCDeltaVal(sector, region, irow, padTmp, integrationInterval);
404  sectorv[index++] = sector;
405  }
406  }
407  }
408  }
409  float idcOneA = getIDCOneVal(Side::A, integrationInterval);
410  float idcOneC = getIDCOneVal(Side::C, integrationInterval);
411 
412  pcstream << "tree"
413  << "integrationInterval=" << integrationInterval
414  << "IDC.=" << idcs
415  << "IDC0.=" << idcsZero
416  << "IDC1A=" << idcOneA
417  << "IDC1C=" << idcOneC
418  << "IDCDelta.=" << idcsDelta
419  << "pad.=" << vPad
420  << "row.=" << vRow
421  << "lx.=" << vXPos
422  << "ly.=" << vYPos
423  << "gx.=" << vGlobalXPos
424  << "gy.=" << vGlobalYPos
425  << "sector.=" << sectorv
426  << "\n";
427  }
428  pcstream.Close();
429 }
