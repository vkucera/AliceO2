105 void GPUDisplayBackendOpenGL::createFB(GLfb& fb, bool tex, bool withDepth, bool msaa, unsigned int width, unsigned int height)
106 {
107  fb.tex = tex;
108  fb.depth = withDepth;
109  fb.msaa = msaa;
110  GLint drawFboId = 0, readFboId = 0;
111  glGetIntegerv(GL_DRAW_FRAMEBUFFER_BINDING, &drawFboId);
112  glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &readFboId);
113  CHKERR(glGenFramebuffers(1, &fb.fb_id));
114  CHKERR(glBindFramebuffer(GL_FRAMEBUFFER, fb.fb_id));
115 
116  auto createFB_texture = [&](GLuint& id, GLenum storage, GLenum attachment) {
117  GLenum textureType = fb.msaa ? GL_TEXTURE_2D_MULTISAMPLE : GL_TEXTURE_2D;
118  CHKERR(glGenTextures(1, &id));
119  CHKERR(glBindTexture(textureType, id));
120  if (fb.msaa) {
121  CHKERR(glTexImage2DMultisample(GL_TEXTURE_2D_MULTISAMPLE, mDisplay->cfgR().drawQualityMSAA, storage, width, height, false));
122  } else {
123  CHKERR(glTexImage2D(GL_TEXTURE_2D, 0, storage, width, height, 0, storage, GL_UNSIGNED_BYTE, nullptr));
124  CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR));
125  CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR));
126  }
127  CHKERR(glFramebufferTexture2D(GL_FRAMEBUFFER, attachment, textureType, id, 0));
128  };
129 
130  auto createFB_renderbuffer = [&](GLuint& id, GLenum storage, GLenum attachment) {
131  CHKERR(glGenRenderbuffers(1, &id));
132  CHKERR(glBindRenderbuffer(GL_RENDERBUFFER, id));
133  if (fb.msaa) {
134  CHKERR(glRenderbufferStorageMultisample(GL_RENDERBUFFER, mDisplay->cfgR().drawQualityMSAA, storage, width, height));
135  } else {
136  CHKERR(glRenderbufferStorage(GL_RENDERBUFFER, storage, width, height));
137  }
138  CHKERR(glFramebufferRenderbuffer(GL_FRAMEBUFFER, attachment, GL_RENDERBUFFER, id));
139  };
140 
141  if (tex) {
142  createFB_texture(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
143  } else {
144  createFB_renderbuffer(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
145  }
146 
147  if (withDepth) {
148  if (tex && fb.msaa) {
149  createFB_texture(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
150  } else {
151  createFB_renderbuffer(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
152  }
153  }
154 
155  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
156  if (status != GL_FRAMEBUFFER_COMPLETE) {
157  GPUError("Error creating framebuffer (tex %d) - incomplete (%d)", (int)tex, status);
158  exit(1);
159  }
160  CHKERR(glBindFramebuffer(GL_DRAW_FRAMEBUFFER, drawFboId));
161  CHKERR(glBindFramebuffer(GL_READ_FRAMEBUFFER, readFboId));
162  fb.created = true;
163 }

183 unsigned int GPUDisplayBackendOpenGL::drawVertices(const vboList& v, const drawType tt)
184 {
185  static constexpr GLenum types[3] = {GL_POINTS, GL_LINES, GL_LINE_STRIP};
186  GLenum t = types[tt];
187  auto first = std::get<0>(v);
188  auto count = std::get<1>(v);
189  auto iSlice = std::get<2>(v);
190  if (count == 0) {
191  return 0;
192  }
193 
194  if (mDisplay->useMultiVBO()) {
195  if (mDisplay->cfgR().openGLCore) {
196  CHKERR(glBindVertexArray(mVertexArray));
197  }
198  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[iSlice]));
199 #ifndef GPUCA_DISPLAY_OPENGL_CORE
200  if (!mDisplay->cfgR().openGLCore) {
201  CHKERR(glVertexPointer(3, GL_FLOAT, 0, nullptr));
202  } else
203 #endif
204  {
205  CHKERR(glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr));
206  glEnableVertexAttribArray(0);
207  }
208  }
209 
210  if (mDisplay->cfgR().useGLIndirectDraw) {
211  CHKERR(glMultiDrawArraysIndirect(t, (void*)(size_t)((mIndirectSliceOffset[iSlice] + first) * sizeof(DrawArraysIndirectCommand)), count, 0));
212  } else if (OPENGL_EMULATE_MULTI_DRAW) {
213  for (unsigned int k = 0; k < count; k++) {
214  CHKERR(glDrawArrays(t, mDisplay->vertexBufferStart()[iSlice][first + k], mDisplay->vertexBufferCount()[iSlice][first + k]));
215  }
216  } else {
217  static_assert(sizeof(GLsizei) == sizeof(*mDisplay->vertexBufferCount()[iSlice].data()), "Invalid counter size does not match GLsizei");
218  CHKERR(glMultiDrawArrays(t, mDisplay->vertexBufferStart()[iSlice].data() + first, ((const GLsizei*)mDisplay->vertexBufferCount()[iSlice].data()) + first, count));
219  }
220  return count;
221 }

272 static int checkShaderStatus(unsigned int shader)
273 {
274  int status, loglen;
275  glGetShaderiv(shader, GL_COMPILE_STATUS, &status);
276  if (!status) {
277  printf("failed to compile shader\n");
278  glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &loglen);
279  std::vector<char> buf(loglen + 1);
280  glGetShaderInfoLog(shader, loglen, nullptr, buf.data());
281  buf[loglen] = 0;
282  printf("%s\n", buf.data());
283  return 1;
284  }
285  return 0;

288 int GPUDisplayBackendOpenGL::InitBackendA()
289 {
290  int glVersion[2] = {0, 0};
291  glGetIntegerv(GL_MAJOR_VERSION, &glVersion[0]);
292  glGetIntegerv(GL_MINOR_VERSION, &glVersion[1]);
293  if (glVersion[0] < GPUDisplayFrontend::GL_MIN_VERSION_MAJOR || (glVersion[0] == GPUDisplayFrontend::GL_MIN_VERSION_MAJOR && glVersion[1] < GPUDisplayFrontend::GL_MIN_VERSION_MINOR)) {
294  GPUError("Unsupported OpenGL runtime %d.%d < %d.%d", glVersion[0], glVersion[1], GPUDisplayFrontend::GL_MIN_VERSION_MAJOR, GPUDisplayFrontend::GL_MIN_VERSION_MINOR);
295  return (1);
296  }
297  mVBOId.resize(GPUCA_NSLICES);
298  CHKERR(glCreateBuffers(mVBOId.size(), mVBOId.data()));
299  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0]));
300  CHKERR(glGenBuffers(1, &mIndirectId));
301  CHKERR(glBindBuffer(GL_DRAW_INDIRECT_BUFFER, mIndirectId));
302 #ifndef GPUCA_DISPLAY_OPENGL_CORE
303  CHKERR(glShadeModel(GL_SMOOTH)); // Enable Smooth Shading
304 #endif
305  setDepthBuffer();
306  setQuality();
307  CHKERR(mVertexShader = glCreateShader(GL_VERTEX_SHADER));
308  CHKERR(mFragmentShader = glCreateShader(GL_FRAGMENT_SHADER));
309  CHKERR(mVertexShaderTexture = glCreateShader(GL_VERTEX_SHADER));
310  CHKERR(mFragmentShaderTexture = glCreateShader(GL_FRAGMENT_SHADER));
311  CHKERR(mFragmentShaderText = glCreateShader(GL_FRAGMENT_SHADER));
312 #if defined(GL_VERSION_4_6) && GL_VERSION_4_6 == 1 && defined(GPUCA_BUILD_EVENT_DISPLAY_VULKAN)
313  if (getenv("USE_SPIRV_SHADERS") && atoi(getenv("USE_SPIRV_SHADERS"))) {
314  CHKERR(glShaderBinary(1, &mVertexShader, GL_SHADER_BINARY_FORMAT_SPIR_V_ARB, _binary_shaders_shaders_vertex_vert_spv_start, _binary_shaders_shaders_vertex_vert_spv_len));
315  CHKERR(glSpecializeShader(mVertexShader, "main", 0, 0, 0));
316  CHKERR(glShaderBinary(1, &mFragmentShader, GL_SHADER_BINARY_FORMAT_SPIR_V_ARB, _binary_shaders_shaders_fragmentUniform_frag_spv_start, _binary_shaders_shaders_fragmentUniform_frag_spv_len));
317  CHKERR(glSpecializeShader(mFragmentShader, "main", 0, 0, 0));
318  GPUInfo("Using SPIR-V shaders");
319  mSPIRVShaders = true;
320  } else
321 #endif
322  {
323  CHKERR(glShaderSource(mVertexShader, 1, &GPUDisplayShaders::vertexShader, nullptr));
324  CHKERR(glCompileShader(mVertexShader));
325  CHKERR(glShaderSource(mFragmentShader, 1, &GPUDisplayShaders::fragmentShader, nullptr));
326  CHKERR(glCompileShader(mFragmentShader));
327  }
328  CHKERR(glShaderSource(mVertexShaderTexture, 1, &GPUDisplayShaders::vertexShaderTexture, nullptr));
329  CHKERR(glCompileShader(mVertexShaderTexture));
330  CHKERR(glShaderSource(mFragmentShaderTexture, 1, &GPUDisplayShaders::fragmentShaderTexture, nullptr));
331  CHKERR(glCompileShader(mFragmentShaderTexture));
332  CHKERR(glShaderSource(mFragmentShaderText, 1, &GPUDisplayShaders::fragmentShaderText, nullptr));
333  CHKERR(glCompileShader(mFragmentShaderText));
334  if (checkShaderStatus(mVertexShader) || checkShaderStatus(mFragmentShader) || checkShaderStatus(mVertexShaderTexture) || checkShaderStatus(mFragmentShaderTexture) || checkShaderStatus(mFragmentShaderText)) {
335  return 1;
336  }
337  CHKERR(mShaderProgram = glCreateProgram());
338  CHKERR(glAttachShader(mShaderProgram, mVertexShader));
339  CHKERR(glAttachShader(mShaderProgram, mFragmentShader));
340  CHKERR(glLinkProgram(mShaderProgram));
341  if (mSPIRVShaders) {
342  CHKERR(glGenBuffers(1, &mSPIRVModelViewBuffer));
343  CHKERR(glGenBuffers(1, &mSPIRVColorBuffer));
344  } else {
345  CHKERR(mModelViewProjId = glGetUniformLocation(mShaderProgram, "ModelViewProj"));
346  CHKERR(mColorId = glGetUniformLocation(mShaderProgram, "color"));
347  }
348  CHKERR(mShaderProgramText = glCreateProgram());
349  CHKERR(glAttachShader(mShaderProgramText, mVertexShaderTexture));
350  CHKERR(glAttachShader(mShaderProgramText, mFragmentShaderText));
351  CHKERR(glLinkProgram(mShaderProgramText));
352  CHKERR(mModelViewProjIdText = glGetUniformLocation(mShaderProgramText, "projection"));
353  CHKERR(mColorIdText = glGetUniformLocation(mShaderProgramText, "textColor"));
354  CHKERR(mShaderProgramTexture = glCreateProgram());
355  CHKERR(glAttachShader(mShaderProgramTexture, mVertexShaderTexture));
356  CHKERR(glAttachShader(mShaderProgramTexture, mFragmentShaderTexture));
357  CHKERR(glLinkProgram(mShaderProgramTexture));
358  CHKERR(mModelViewProjIdTexture = glGetUniformLocation(mShaderProgramTexture, "projection"));
359  CHKERR(mAlphaIdTexture = glGetUniformLocation(mShaderProgramTexture, "alpha"));
360  CHKERR(glGenVertexArrays(1, &mVertexArray));
361 
362  CHKERR(glGenVertexArrays(1, &VAO_texture));
363  CHKERR(glGenBuffers(1, &VBO_texture));
364  CHKERR(glBindVertexArray(VAO_texture));
365  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, VBO_texture));
366  CHKERR(glBufferData(GL_ARRAY_BUFFER, sizeof(float) * 6 * 4, nullptr, GL_DYNAMIC_DRAW));
367  CHKERR(glEnableVertexAttribArray(0));
368  CHKERR(glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4 * sizeof(float), nullptr));
369  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, 0));
370 
371  CHKERR(glBindVertexArray(0));
372  return (0); // Initialization Went OK
373 }

442 void GPUDisplayBackendOpenGL::prepareDraw(const hmm_mat4& proj, const hmm_mat4& view, bool requestScreenshot, bool toMixBuffer, float includeMixImage)
443 {
444  if (mDisplay->updateRenderPipeline() || mDownsampleFactor != getDownsampleFactor(requestScreenshot)) {
445  updateRenderer(requestScreenshot);
446  }
447  if (mOffscreenBufferMSAA.created) {
448  setFrameBuffer(mOffscreenBufferMSAA.fb_id);
449  } else if (toMixBuffer) {
450  setFrameBuffer(mMixBuffer.fb_id);
451  } else if (mOffscreenBuffer.created) {
452  setFrameBuffer(mOffscreenBuffer.fb_id);
453  } else {
454  setFrameBuffer(0);
455  }
456  clearScreen();
457 #ifndef GPUCA_DISPLAY_OPENGL_CORE
458  if (mDisplay->cfgL().smoothPoints && !mDisplay->cfgR().openGLCore) {
459  CHKERR(glEnable(GL_POINT_SMOOTH));
460  } else {
461  CHKERR(glDisable(GL_POINT_SMOOTH));
462  }
463  if (mDisplay->cfgL().smoothLines && !mDisplay->cfgR().openGLCore) {
464  CHKERR(glEnable(GL_LINE_SMOOTH));
465  } else {
466  CHKERR(glDisable(GL_LINE_SMOOTH));
467  }
468 #endif
469  CHKERR(glEnable(GL_BLEND));
470  CHKERR(glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA));
471 
472  glViewport(0, 0, mRenderWidth, mRenderHeight);
473 #ifndef GPUCA_DISPLAY_OPENGL_CORE
474  if (!mDisplay->cfgR().openGLCore) {
475  CHKERR(glEnableClientState(GL_VERTEX_ARRAY));
476  CHKERR(glVertexPointer(3, GL_FLOAT, 0, nullptr));
477  } else
478 #endif
479  {
480  CHKERR(glBindVertexArray(mVertexArray));
481  CHKERR(glUseProgram(mShaderProgram));
482  CHKERR(glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr));
483  CHKERR(glEnableVertexAttribArray(0));
484  }
485 #ifndef GPUCA_DISPLAY_OPENGL_CORE
486  if (!mDisplay->cfgR().openGLCore) {
487  CHKERR(glMatrixMode(GL_PROJECTION));
488  CHKERR(glLoadMatrixf(&proj.Elements[0][0]));
489  CHKERR(glMatrixMode(GL_MODELVIEW));
490  CHKERR(glLoadMatrixf(&view.Elements[0][0]));
491  } else
492 #endif
493  {
494  const hmm_mat4 modelViewProj = proj * view;
495  if (mSPIRVShaders) {
496  glBindBuffer(GL_UNIFORM_BUFFER, mSPIRVModelViewBuffer);
497  glBufferData(GL_UNIFORM_BUFFER, sizeof(modelViewProj), &modelViewProj, GL_STATIC_DRAW);
498  glBindBufferBase(GL_UNIFORM_BUFFER, 0, mSPIRVModelViewBuffer);
499  } else {
500  CHKERR(glUniformMatrix4fv(mModelViewProjId, 1, GL_FALSE, &modelViewProj.Elements[0][0]));
501  }
502  }
503 }

505 void GPUDisplayBackendOpenGL::finishDraw(bool doScreenshot, bool toMixBuffer, float includeMixImage)
506 {
507 #ifndef GPUCA_DISPLAY_OPENGL_CORE

692 void GPUDisplayBackendOpenGL::OpenGLPrint(const char* s, float x, float y, float* color, float scale)
693 {
694  if (!mFreetypeInitialized || mDisplay->drawTextInCompatMode()) {
695  return;
696  }
697  CHKERR(glUniform4fv(mColorIdText, 1, color));
698 
699  if (smoothFont()) {
700  scale *= 0.25f; // Font size is 48 to have nice bitmap, scale to size 12
701  }
702  for (const char* c = s; *c; c++) {
703  if ((int)*c > (int)mFontSymbols.size()) {
704  GPUError("Trying to draw unsupported symbol: %d > %d\n", (int)*c, (int)mFontSymbols.size());
705  continue;
706  }
707  FontSymbolOpenGL sym = mFontSymbols[*c];
708  if (sym.size[0] && sym.size[1]) {
709  float xpos = x + sym.offset[0] * scale;
710  float ypos = y - (sym.size[1] - sym.offset[1]) * scale;
711  float w = sym.size[0] * scale;
712  float h = sym.size[1] * scale;
713  float vertices[6][4] = {
714  {xpos, ypos + h, 0.0f, 0.0f},
715  {xpos, ypos, 0.0f, 1.0f},
716  {xpos + w, ypos, 1.0f, 1.0f},
717  {xpos, ypos + h, 0.0f, 0.0f},
718  {xpos + w, ypos, 1.0f, 1.0f},
719  {xpos + w, ypos + h, 1.0f, 0.0f}};
720  CHKERR(glBindTexture(GL_TEXTURE_2D, sym.texId));
721  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, VBO_text));
722  CHKERR(glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices));
723  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, 0));
724  CHKERR(glDrawArrays(GL_TRIANGLES, 0, 6));
725  }
726  x += (sym.advance >> 6) * scale; // shift is in 1/64th of a pixel
727  }
728  if (!mDisplay->useMultiVBO()) {
729  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0])); // If we don't use multiple buffers, we keep the default buffer bound
730  }
731 }
