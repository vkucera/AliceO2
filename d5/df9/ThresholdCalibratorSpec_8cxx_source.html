28 // Define error function for ROOT fitting
29 double erf(double* xx, double* par)
30 {
31  return (N_INJ / 2) * TMath::Erf((xx[0] - par[0]) / (sqrt(2) * par[1])) + (N_INJ / 2);
32 }
33 
34 // ITHR erf is reversed
35 double erf_ithr(double* xx, double* par)
36 {
37  return (N_INJ / 2) * (1 - TMath::Erf((xx[0] - par[0]) / (sqrt(2) * par[1])));
38 }

187 // Returns upper / lower limits for threshold determination.
188 // data is the number of trigger counts per charge injected;
189 // x is the array of charge injected values;
190 // NPoints is the length of both arrays.
191 bool ITSThresholdCalibrator::findUpperLower(
192  const unsigned short int* data, const short int* x, const short int& NPoints,
193  short int& lower, short int& upper, bool flip)
194 {
195  // Initialize (or re-initialize) upper and lower
196  upper = -1;
197  lower = -1;
198 
199  if (flip) { // ITHR case. lower is at large mX[i], upper is at small mX[i]
200 
201  for (int i = 0; i < NPoints; i++) {
202  if (data[i] == 0) {
203  upper = i;
204  break;
205  }
206  }
207 
208  if (upper == -1) {
209  return false;
210  }
211  for (int i = upper; i > 0; i--) {
212  if (data[i] >= N_INJ) {
213  lower = i;
214  break;
215  }
216  }
217 
218  } else { // not flipped
219 
220  for (int i = 0; i < NPoints; i++) {
221  if (data[i] >= N_INJ) {
222  upper = i;
223  break;
224  }
225  }
226 
227  if (upper == -1) {
228  return false;
229  }
230  for (int i = upper; i > 0; i--) {
231  if (data[i] == 0) {
232  lower = i;
233  break;
234  }
235  }
236  }
237 
238  // If search was successful, return central x value
239  if ((lower == -1) || (upper < lower)) {
240  return false;
241  }
242  return true;
243 }

246 // Main findThreshold function which calls one of the three methods
247 bool ITSThresholdCalibrator::findThreshold(
248  const unsigned short int* data, const short int* x, const short int& NPoints,
249  float& thresh, float& noise)
250 {
251  bool success = false;
252 
253  switch (this->mFitType) {
254  case DERIVATIVE: // Derivative method
255  success = this->findThresholdDerivative(data, x, NPoints, thresh, noise);
256  break;
257 
258  case FIT: // Fit method
259  success = this->findThresholdFit(data, x, NPoints, thresh, noise);
260  break;
261 
262  case HITCOUNTING: // Hit-counting method
263  success = this->findThresholdHitcounting(data, x, NPoints, thresh);
264  // noise = 0;
265  break;
266  }
267 
268  return success;
269 }

272 // Use ROOT to find the threshold and noise via S-curve fit
273 // data is the number of trigger counts per charge injected;
274 // x is the array of charge injected values;
275 // NPoints is the length of both arrays.
276 // thresh, noise, chi2 pointers are updated with results from the fit
277 bool ITSThresholdCalibrator::findThresholdFit(
278  const unsigned short int* data, const short int* x, const short int& NPoints,
279  float& thresh, float& noise)
280 {
281  // Find lower & upper values of the S-curve region
282  short int lower, upper;
283  bool flip = (this->mScanType == 'I');
284  if (!this->findUpperLower(data, x, NPoints, lower, upper, flip) || lower == upper) {
285  if (this->mVerboseOutput) {
286  LOG(warning) << "Start-finding unsuccessful: (lower, upper) = ("
287  << lower << ", " << upper << ")";
288  }
289  return false;
290  }
291  float start = (this->mX[upper] + this->mX[lower]) / 2;
292 
293  if (start < 0) {
294  if (this->mVerboseOutput) {
295  LOG(warning) << "Start-finding unsuccessful: Start = " << start;
296  }
297  return false;
298  }
299 
300  for (int i = 0; i < NPoints; i++) {
301  this->mFitHist->SetBinContent(i + 1, data[i]);
302  }
303 
304  // Initialize starting parameters
305  this->mFitFunction->SetParameter(0, start);
306  this->mFitFunction->SetParameter(1, 8);
307 
308  this->mFitHist->Fit("mFitFunction", "QL");
309 
310  noise = this->mFitFunction->GetParameter(1);
311  thresh = this->mFitFunction->GetParameter(0);
312  float chi2 = this->mFitFunction->GetChisquare() / this->mFitFunction->GetNDF();
313 
314  // Clean up histogram for next time it is used
315  this->mFitHist->Reset();
316 
317  return (chi2 < 5);
318 }

321 // Use ROOT to find the threshold and noise via derivative method
322 // data is the number of trigger counts per charge injected;
323 // x is the array of charge injected values;
324 // NPoints is the length of both arrays.
325 bool ITSThresholdCalibrator::findThresholdDerivative(
326  const unsigned short int* data, const short int* x, const short int& NPoints,
327  float& thresh, float& noise)
328 {
329  // Find lower & upper values of the S-curve region
330  short int lower, upper;
331  bool flip = (this->mScanType == 'I');
332  if (!this->findUpperLower(data, x, NPoints, lower, upper, flip) || lower == upper) {
333  if (this->mVerboseOutput) {
334  LOG(warning) << "Start-finding unsuccessful: (lower, upper) = ("
335  << lower << ", " << upper << ")";
336  }
337  return false;
338  }
339 
340  int deriv_size = upper - lower;
341  float deriv[deriv_size];
342  float xfx = 0, fx = 0;
343 
344  // Fill array with derivatives
345  for (int i = lower; i < upper; i++) {
346  deriv[i - lower] = (data[i + 1] - data[i]) / float(this->mX[i + 1] - mX[i]);
347  if (deriv[i - lower] < 0) {
348  deriv[i - lower] = 0.;
349  }
350  xfx += this->mX[i] * deriv[i - lower];
351  fx += deriv[i - lower];
352  }
353 
354  thresh = xfx / fx;
355  float stddev = 0;
356  for (int i = lower; i < upper; i++) {
357  stddev += std::pow(this->mX[i] - thresh, 2) * deriv[i - lower];
358  }
359 
360  stddev /= fx;
361  noise = std::sqrt(stddev);
362 
363  return true;
364 }

367 // Use ROOT to find the threshold and noise via derivative method
368 // data is the number of trigger counts per charge injected;
369 // x is the array of charge injected values;
370 // NPoints is the length of both arrays.
371 bool ITSThresholdCalibrator::findThresholdHitcounting(
372  const unsigned short int* data, const short int* x, const short int& NPoints, float& thresh)
373 {
374  unsigned short int numberOfHits = 0;
375  bool is50 = false;
376  for (unsigned short int i = 0; i < NPoints; i++) {
377  numberOfHits += data[i];
378  if (!is50 && data[i] == N_INJ) {
379  is50 = true;
380  }
381  }
382 
383  // If not enough counts return a failure
384  // if (numberOfHits < N_INJ) { return false; }
385  if (!is50) {
386  if (this->mVerboseOutput) {
387  LOG(warning) << "Calculation unsuccessful: too few hits. Skipping this pixel";
388  }
389  return false;
390  }
391 
392  if (this->mScanType == 'T') {
393  thresh = this->mX[*(this->N_RANGE) - 1] - numberOfHits / float(N_INJ);
394  } else if (this->mScanType == 'V') {
395  thresh = (this->mX[*(this->N_RANGE) - 1] * N_INJ - numberOfHits) / float(N_INJ);
396  } else if (this->mScanType == 'I') {
397  thresh = (numberOfHits + N_INJ * this->mX[0]) / float(N_INJ);
398  } else {
399  LOG(error) << "Unexpected runtype encountered in findThresholdHitcounting()";
400  return false;
401  }
402 
403  return true;
404 }

407 // Run threshold extraction on completed row and update memory
408 void ITSThresholdCalibrator::extractThresholdRow(const short int& chipID, const short int& row)
409 {
410  if (this->mScanType == 'D' || this->mScanType == 'A') {
411  // Loop over all columns (pixels) in the row
412  for (short int col_i = 0; col_i < this->N_COL; col_i++) {
413  vChipid[col_i] = chipID;
414  vRow[col_i] = row;
415  vThreshold[col_i] = this->mPixelHits[chipID][row][col_i][0];
416  if (vThreshold[col_i] > N_INJ) {
417  this->mNoisyPixID[chipID].push_back(col_i * 1000 + row);
418  }
419  }
420  } else {
421 
422 #ifdef WITH_OPENMP
423  omp_set_num_threads(mNThreads);
424 #pragma omp parallel for schedule(dynamic)
425 #endif
426  // Loop over all columns (pixels) in the row
427  for (short int col_i = 0; col_i < this->N_COL; col_i++) {
428 
429  // Do the threshold fit
430  float thresh = 0., noise = 0.;
431  bool success = false;
432  success = this->findThreshold(&(this->mPixelHits[chipID][row][col_i][0]),
433  this->mX, *(this->N_RANGE), thresh, noise);
434 
435  vChipid[col_i] = chipID;
436  vRow[col_i] = row;
437  vThreshold[col_i] = this->mScanType == 'T' ? (short int)(thresh * 10.) : (short int)(thresh);
438  vNoise[col_i] = (unsigned char)(noise * 10.); // always factor 10 also for ITHR/VCASN to not have all zeros
439  vSuccess[col_i] = success;
440  }
441  }
442 
443  // Saves threshold information to internal memory
444  this->saveThreshold();
445 }

448 void ITSThresholdCalibrator::saveThreshold()
449 {
450  // In the case of a full threshold scan, write to TTree
451  if (this->mScanType == 'T' || this->mScanType == 'D' || this->mScanType == 'A') {
452  this->mThresholdTree->Fill();
453  }
454 
455  if (this->mScanType != 'D' && this->mScanType != 'A') {
456  // Save info in a map for later averaging
457  int sumT = 0, sumSqT = 0, sumN = 0, sumSqN = 0;
458  int countSuccess = 0;
459  for (int i = 0; i < this->N_COL; i++) {
460  if (vSuccess[i]) {
461  sumT += vThreshold[i];
462  sumN += (int)vNoise[i];
463  sumSqT += (vThreshold[i]) * (vThreshold[i]);
464  sumSqN += ((int)vNoise[i]) * ((int)vNoise[i]);
465  countSuccess++;
466  }
467  }
468  short int chipID = vChipid[0];
469  std::array<int, 5> dataSum{{sumT, sumSqT, sumN, sumSqN, countSuccess}};
470  if (!(this->mThresholds.count(chipID))) {
471  this->mThresholds[chipID] = dataSum;
472  } else {
473  std::array<int, 5> dataAll{{this->mThresholds[chipID][0] + dataSum[0], this->mThresholds[chipID][1] + dataSum[1], this->mThresholds[chipID][2] + dataSum[2], this->mThresholds[chipID][3] + dataSum[3], this->mThresholds[chipID][4] + dataSum[4]}};
474  this->mThresholds[chipID] = dataAll;
475  }
476  }
477 
478  return;
479 }

552 // Set the run_type for this run
553 // Initialize the memory needed for this specific type of run
554 void ITSThresholdCalibrator::setRunType(const short int& runtype)
555 {
556 
557  // Save run type info for future evaluation
558  this->mRunType = runtype;
559 
560  if (runtype == THR_SCAN) {
561  // full_threshold-scan -- just extract thresholds for each pixel and write to TTree
562  // 512 rows per chip
563  this->mScanType = 'T';
564  this->initThresholdTree();
565  this->N_RANGE = &(this->N_CHARGE);
566  this->mMin = 0;
567  this->mMax = 50;
568  this->mCheckExactRow = true;
569 
570  } else if (runtype == THR_SCAN_SHORT || runtype == THR_SCAN_SHORT_100HZ ||
571  runtype == THR_SCAN_SHORT_200HZ) {
572  // threshold_scan_short -- just extract thresholds for each pixel and write to TTree
573  // 10 rows per chip
574  this->mScanType = 'T';
575  this->initThresholdTree();
576  this->N_RANGE = &(this->N_CHARGE);
577  this->mMin = 0;
578  this->mMax = 50;
579  this->mCheckExactRow = true;
580 
581  } else if (runtype == VCASN150 || runtype == VCASN100 || runtype == VCASN100_100HZ) {
582  // VCASN tuning for different target thresholds
583  // Store average VCASN for each chip into CCDB
584  // 4 rows per chip
585  this->N_RANGE = &(this->N_VCASN);
586  this->mScanType = 'V';
587  this->mMin = 30;
588  this->mMax = 80;
589  this->mCheckExactRow = true;
590 
591  } else if (runtype == ITHR150 || runtype == ITHR100 || runtype == ITHR100_100HZ) {
592  // ITHR tuning -- average ITHR per chip
593  // S-curve is backwards from VCASN case, otherwise same
594  // 4 rows per chip
595  this->N_RANGE = &(this->N_ITHR);
596  this->mScanType = 'I';
597  this->mMin = 30;
598  this->mMax = 100;
599  this->mCheckExactRow = true;
600 
601  } else if (runtype == DIGITAL_SCAN || runtype == DIGITAL_SCAN_100HZ) {
602  // Digital scan -- only storing one value per chip, no fit needed
603  this->mScanType = 'D';
604  this->initThresholdTree();
605  this->N_RANGE = &(this->N_DIGANA);
606  this->mFitType = NO_FIT;
607  this->mMin = 0;
608  this->mMax = 0;
609  this->mCheckExactRow = false;
610 
611  } else if (runtype == ANALOGUE_SCAN) {
612  // Analogue scan -- only storing one value per chip, no fit needed
613  this->mScanType = 'A';
614  this->initThresholdTree();
615  this->N_RANGE = &(this->N_DIGANA);
616  this->mFitType = NO_FIT;
617  this->mScanType = 'A';
618  this->mMin = 0;
619  this->mMax = 0;
620  this->mCheckExactRow = false;
621 
622  } else {
623  // No other run type recognized by this workflow
624  LOG(error) << "runtype " << runtype << " not recognized by threshold scan workflow.";
625  throw runtype;
626  }
627 
628  this->mX = new short int[*(this->N_RANGE)];
629  for (short int i = this->mMin; i <= this->mMax; i++) {
630  this->mX[i - this->mMin] = i;
631  }
632

683 // Main running function
684 // Get info from previous stf decoder workflow, then loop over readout frames
685 // (ROFs) to count hits and extract thresholds
686 void ITSThresholdCalibrator::run(ProcessingContext& pc)
687 {
688  updateTimeDependentParams(pc);
689 
690  // Calibration vector
691  const auto calibs = pc.inputs().get<gsl::span<o2::itsmft::GBTCalibData>>("calib");
692  const auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
693  const auto ROFs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("digitsROF");
694 
695  // Store some lengths for convenient looping
696  const unsigned int nROF = (unsigned int)ROFs.size();
697 
698  // Loop over readout frames (usually only 1, sometimes 2)
699  for (unsigned int iROF = 0; iROF < nROF; iROF++) {
700 
701  unsigned int rofIndex = ROFs[iROF].getFirstEntry();
702  unsigned int rofNEntries = ROFs[iROF].getNEntries();
703 
704  // Find the correct charge, row, cw counter values for this ROF
705  short int charge = -1;
706  short int row = -1;
707  short int cwcnt = -1;
708  for (short int iRU = 0; iRU < this->N_RU; iRU++) {
709  const auto& calib = calibs[iROF * this->N_RU + iRU];
710  if (calib.calibUserField != 0) {
711  if (charge >= 0) {
712  LOG(warning) << "More than one charge detected!";
713  }
714 
715  if (this->mRunType == -1) {
716  short int runtype = ((short int)(calib.calibUserField >> 24)) & 0xff;
717  mConfDBv = ((short int)(calib.calibUserField >> 32)) & 0xffff; // confDB version
718  LOG(info) << "confDB version in use: " << mConfDBv;
719  this->setRunType(runtype);
720  }
721  this->mRunTypeUp = ((short int)(calib.calibUserField >> 24)) & 0xff;
722  // Divide calibration word (24-bit) by 2^16 to get the first 8 bits
723  if (this->mScanType == 'T') {
724  // For threshold scan have to subtract from 170 to get charge value
725  charge = (short int)(170 - (calib.calibUserField >> 16) & 0xff);
726  } else if (this->mScanType == 'D' || this->mScanType == 'A') { // Digital scan
727  charge = 0;
728  } else { // VCASN or ITHR tuning
729  charge = (short int)((calib.calibUserField >> 16) & 0xff);
730  }
731 
732  // Last 16 bits should be the row (only uses up to 9 bits)
733  row = (short int)(calib.calibUserField & 0xffff);
734  // cw counter
735  cwcnt = (short int)(calib.calibCounter);
736 
737  break;
738  }
739  }
740  if (charge > this->mMax || charge < this->mMin) {
741  if (this->mVerboseOutput) {
742  LOG(warning) << "CW missing - charge value " << charge << " out of range for min " << this->mMin
743  << " and max " << this->mMax << " (range: " << *(this->N_RANGE) << ")";
744  }
745  } else {
746  std::vector<short int> mChips;
747  std::map<short int, bool> mChipsForbRows;
748  // loop to retrieve list of chips and start tagging bad dcols if the hits does not come from this row
749  for (unsigned int idig = rofIndex; idig < rofIndex + rofNEntries; idig++) { // gets chipid
750  auto& d = digits[idig];
751  short int chipID = (short int)d.getChipIndex();
752 
753  if ((chipID % mChipModBase) != mChipModSel) {
754  continue;
755  }
756  if (d.getRow() != row) {
757  if (this->mVerboseOutput) {
758  LOG(info) << "iROF: " << iROF << " ChipID " << chipID << ": current row is " << d.getRow() << " (col = " << d.getColumn() << ") but the one in CW is " << row;
759  }
760  }
761  if (std::find(mChips.begin(), mChips.end(), chipID) != mChips.end()) {
762  continue;
763  }
764  mChips.push_back(chipID);
765  }
766  // loop to allocate memory only for allowed rows
767  for (auto& chipID : mChips) {
768  if (mForbiddenRows.count(chipID)) {
769  for (int iforb = mForbiddenRows[chipID].size() - 1; iforb >= 0; iforb--) {
770  if (mForbiddenRows[chipID][iforb] == row) {
771  mChipsForbRows[chipID] = true;
772  break;
773  }
774  }
775  }
776  if (mChipsForbRows[chipID]) {
777  continue;
778  }
779  if (!this->mPixelHits.count(chipID)) {
780  if (mScanType == 'D' || mScanType == 'A') { // for digital and analog scan initialize the full matrix for each chipID
781  for (int irow = 0; irow < 512; irow++) {
782  this->mPixelHits[chipID][irow] = std::vector<std::vector<unsigned short int>>(this->N_COL, std::vector<unsigned short int>(*(this->N_RANGE), 0));
783  }
784  } else {
785  this->mPixelHits[chipID][row] = std::vector<std::vector<unsigned short int>>(this->N_COL, std::vector<unsigned short int>(*(this->N_RANGE), 0));
786  }
787  } else if (!this->mPixelHits[chipID].count(row)) { // allocate memory for chip = chipID or for a row of this chipID
788  this->mPixelHits[chipID][row] = std::vector<std::vector<unsigned short int>>(this->N_COL, std::vector<unsigned short int>(*(this->N_RANGE), 0));
789  }
790  }
791 
792  // loop to count hits from digits
793  short int chgPoint = charge - this->mMin;
794  for (unsigned int idig = rofIndex; idig < rofIndex + rofNEntries; idig++) {
795  auto& d = digits[idig];
796  short int chipID = (short int)d.getChipIndex();
797  short int col = (short int)d.getColumn();
798 
799  if ((chipID % mChipModBase) != mChipModSel) {
800  continue;
801  }
802 
803  if (!mChipsForbRows[chipID] && (!mCheckExactRow || d.getRow() == row)) { // row has NOT to be forbidden and we ignore hits coming from other rows (potential masking issue on chip)
804  // Increment the number of counts for this pixel
805  this->mPixelHits[chipID][d.getRow()][col][chgPoint]++;
806  }
807  }
808  // check collected chips in previous loop on digits
809  for (auto& chipID : mChips) {
810  // count the zeros
811  if (!this->mRunTypeUp) {
812  this->mRunTypeChip[chipID]++;
813  }
814  // check forbidden rows
815  if (mChipsForbRows[chipID]) {
816  continue;
817  }
818 
819  if (mScanType != 'D' && mScanType != 'A' && this->isScanFinished(chipID, row, cwcnt)) { // for D and A we do it at the end in finalize()
820  this->extractAndUpdate(chipID, row);
821  // remove entry for this row whose scan is completed
822  mPixelHits[chipID].erase(row);
823  mForbiddenRows[chipID].push_back(row); // due to the loose cut in isScanFinished, extra hits may come for this deleted row. In this way the row is ignored afterwards
824  }
825  }
826 
827  for (auto& chipID : mChips) {
828  if (mRunTypeChip[chipID] == N_INJ) {
829  this->addDatabaseEntry(chipID, "", 0, 0, 0, 0, 0, true); // output for QC (mainly)
830  if (mCheckEos) {
831  mRunTypeChip[chipID] = 0; // to avoid to re-write the chip in the DCSConfigObject
832  }
833  }
834  }
835  } // if (charge)
836  } // for (ROFs)
837 
838  if (!(this->mCheckEos) && !(this->mRunTypeUp)) {
839  this->finalize(nullptr);
840  LOG(info) << "Shipping all outputs to aggregator (no endOfStream will be used!)";
841  pc.outputs().snapshot(Output{"ITS", "TSTR", (unsigned int)mChipModSel}, this->mTuning);
842  pc.outputs().snapshot(Output{"ITS", "RUNT", (unsigned int)mChipModSel}, this->mRunType);
843  pc.outputs().snapshot(Output{"ITS", "SCANT", (unsigned int)mChipModSel}, this->mScanType);
844  pc.outputs().snapshot(Output{"ITS", "FITT", (unsigned int)mChipModSel}, this->mFitType);
845  pc.outputs().snapshot(Output{"ITS", "CONFDBV", (unsigned int)mChipModSel}, this->mConfDBv);
846  pc.outputs().snapshot(Output{"ITS", "QCSTR", (unsigned int)mChipModSel}, this->mChipDoneQc);
847  // reset the DCSconfigObject_t before next ship out
848  mTuning.clear();
849  mChipDoneQc.clear();
850  } else if (mCheckEos) {
851  pc.outputs().snapshot(Output{"ITS", "TSTR", (unsigned int)mChipModSel}, this->mTuning); // dummy here
852  pc.outputs().snapshot(Output{"ITS", "RUNT", (unsigned int)mChipModSel}, this->mRunType);
853  pc.outputs().snapshot(Output{"ITS", "SCANT", (unsigned int)mChipModSel}, this->mScanType);
854  pc.outputs().snapshot(Output{"ITS", "FITT", (unsigned int)mChipModSel}, this->mFitType);
855  pc.outputs().snapshot(Output{"ITS", "CONFDBV", (unsigned int)mChipModSel}, this->mConfDBv);
856  pc.outputs().snapshot(Output{"ITS", "QCSTR", (unsigned int)mChipModSel}, this->mChipDoneQc);
857  mChipDoneQc.clear();
858  }
859 
860  return;
861 }

874 // Calculate the average threshold given a vector of threshold objects
875 void ITSThresholdCalibrator::findAverage(const std::array<int, 5>& data, float& avgT, float& rmsT, float& avgN, float& rmsN)
876 {
877  avgT = (!data[4]) ? 0. : (float)data[0] / (float)data[4];
878  rmsT = (!data[4]) ? 0. : std::sqrt((float)data[1] / (float)data[4] - avgT * avgT);
879  avgN = (!data[4]) ? 0. : (float)data[2] / (float)data[4];
880  rmsN = (!data[4]) ? 0. : std::sqrt((float)data[3] / (float)data[4] - avgN * avgN);
881  return;
882 }

1016  }
1017  }

1025  ++it;
1026  continue;
1027  }
1028  float avgT, rmsT, avgN, rmsN;

1032  if (!this->mCheckEos) {
1033  this->mRunTypeChip[it->first] = 0; // so that this chip will never appear again in the DCSconfigObject_t
1034  it = this->mThresholds.erase(it);
1035  } else {
1036  ++it;
1037  }
1038  }
1039 
1040  } else if (this->mScanType == 'T') {
1041  // Loop over each chip and calculate avg and rms

1043  auto it = this->mThresholds.cbegin();
