65 template <typename VEC>
66 o2::ctf::CTFIOSize CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
67 {
68  using MD = o2::ctf::Metadata::OptStore;
69  // what to do which each field: see o2::ctd::Metadata explanation
70  constexpr MD optField[CTF::getNBlocks()] = {
71  MD::EENCODE, // BLC_bcInc
72  MD::EENCODE, // BLC_orbitInc
73  MD::EENCODE, // BLC_nChan
74  MD::EENCODE, // BLC_idChan
75  MD::EENCODE, // BLC_cfdTime
76  MD::EENCODE, // BLC_qtcAmpl
77  // extra slot was added in the end
78  MD::EENCODE, // BLC_trigger
79  MD::EENCODE // BLC_qtcChain
80  };
81  CompressedDigits cd;
82  if (mExtHeader.isValidDictTimeStamp()) {
83  if (mExtHeader.minorVersion == 0 && mExtHeader.majorVersion == 1) {
84  compress<1, 0>(cd, digitVec, channelVec);
85  } else {
86  compress<1, 1>(cd, digitVec, channelVec);
87  }
88  } else {
89  compress<1, 1>(cd, digitVec, channelVec);
90  }
91  // book output size with some margin
92  auto szIni = estimateCompressedSize(cd);
93  buff.resize(szIni);
94 
95  auto ec = CTF::create(buff);
96  using ECB = CTF::base;
97 
98  ec->setHeader(cd.header);
99  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
100  ec->getANSHeader().majorVersion = 0;
101  ec->getANSHeader().minorVersion = 1;
102  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
103  o2::ctf::CTFIOSize iosize;
104 #define ENCODEFV0(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
105  // clang-format off
106  iosize += ENCODEFV0(cd.bcInc, CTF::BLC_bcInc, 0);
107  iosize += ENCODEFV0(cd.orbitInc, CTF::BLC_orbitInc, 0);
108  iosize += ENCODEFV0(cd.nChan, CTF::BLC_nChan, 0);
109  iosize += ENCODEFV0(cd.idChan , CTF::BLC_idChan, 0);
110  iosize += ENCODEFV0(cd.cfdTime, CTF::BLC_cfdTime, 0);
111  iosize += ENCODEFV0(cd.qtcAmpl, CTF::BLC_qtcAmpl, 0);
112  // extra slot was added in the end
113  iosize += ENCODEFV0(cd.trigger, CTF::BLC_trigger, 0);
114  iosize += ENCODEFV0(cd.qtcChain, CTF::BLC_qtcChain, 0);
115  // clang-format on
116  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
117  finaliseCTFOutput<CTF>(buff);
118  iosize.rawIn = sizeof(Digit) * digitVec.size() + sizeof(ChannelData) * channelVec.size();
119  return iosize;
120 }

123 template <typename VDIG, typename VCHAN>
124 o2::ctf::CTFIOSize CTFCoder::decode(const CTF::base& ec, VDIG& digitVec, VCHAN& channelVec)
125 {
126  CompressedDigits cd;
127  cd.header = ec.getHeader();
128  const auto& hd = static_cast<const o2::ctf::CTFDictHeader&>(cd.header);
129  checkDictVersion(hd);
130  ec.print(getPrefix(), mVerbosity);
131  o2::ctf::CTFIOSize iosize;
132 #define DECODEFV0(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
133  // clang-format off
134  iosize += DECODEFV0(cd.bcInc, CTF::BLC_bcInc);
135  iosize += DECODEFV0(cd.orbitInc, CTF::BLC_orbitInc);
136  iosize += DECODEFV0(cd.nChan, CTF::BLC_nChan);
137  iosize += DECODEFV0(cd.idChan, CTF::BLC_idChan);
138  iosize += DECODEFV0(cd.cfdTime, CTF::BLC_cfdTime);
139  iosize += DECODEFV0(cd.qtcAmpl, CTF::BLC_qtcAmpl);
140  // extra slot was added in the end
141  iosize += DECODEFV0(cd.trigger, CTF::BLC_trigger);
142  iosize += DECODEFV0(cd.qtcChain, CTF::BLC_qtcChain);
143  // triggers and qtcChain were added later, in old data they are absent:
144  if (cd.trigger.empty()) {
145  cd.trigger.resize(cd.header.nTriggers);
146  }
147  if (cd.qtcChain.empty()) {
148  cd.qtcChain.resize(cd.cfdTime.size());
149  }
150  // clang-format on
151  //
152  if (hd.minorVersion == 0 && hd.majorVersion == 1) {
153  decompress<1, 0>(cd, digitVec, channelVec);
154  } else {
155  decompress<1, 1>(cd, digitVec, channelVec);
156  }
157  iosize.rawIn = sizeof(Digit) * digitVec.size() + sizeof(ChannelData) * channelVec.size();
158  return iosize;
159 }

162 template <int MAJOR_VERSION, int MINOR_VERSION, typename VDIG, typename VCHAN>
163 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
164 {
165  digitVec.clear();
166  channelVec.clear();
167  digitVec.reserve(cd.header.nTriggers);
168  channelVec.reserve(cd.idChan.size());
169 
170  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
171  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
172 
173  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
174  // restore ROFRecord
175  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
176  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
177  ir.orbit += cd.orbitInc[idig];
178  } else {
179  ir.bc += cd.bcInc[idig];
180  }
181  const auto& params = FV0DigParam::Instance();
182  int triggerGate = params.mTime_trg_gate;
183  firstEntry = channelVec.size();
184  uint8_t chID = 0;
185  int8_t nChanA = 0, nChanC = 0;
186  int32_t amplA = 0, amplC = Triggers::DEFAULT_AMP;
187  int16_t timeA = 0, timeC = Triggers::DEFAULT_TIME;
188  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
189  auto icc = channelVec.size();
190  if constexpr (MINOR_VERSION == 0 && MAJOR_VERSION == 1) {
191  // Old decoding procedure, mostly for Pilot Beam in October 2021
192  chID += cd.idChan[icc];
193  } else {
194  // New decoding procedure, w/o sorted ChID requriment
195  chID = cd.idChan[icc];
196  }
197  const auto& chan = channelVec.emplace_back(chID, cd.cfdTime[icc], cd.qtcAmpl[icc], cd.qtcChain[icc]);
198  if (std::abs(chan.CFDTime) < triggerGate) {
199  amplA += chan.QTCAmpl;
200  timeA += chan.CFDTime;
201  nChanA++;
202  }
203  }
204  if (nChanA) {
205  timeA /= nChanA;
206  amplA *= 0.125;
207  } else {
208  timeA = Triggers::DEFAULT_TIME;
209  amplA = Triggers::DEFAULT_AMP;
210  }
211  Triggers trig;
212  trig.setTriggers(cd.trigger[idig], nChanA, nChanC, amplA, amplC, timeA, timeC);
213  digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig, idig);
214  }

217 template <int MAJOR_VERSION, int MINOR_VERSION>
218 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
219 {
220  // convert digits/channel to their compressed version
221  cd.clear();
222  if (!digitVec.size()) {
223  return;
224  }
225  const auto& dig0 = digitVec[0];
226  cd.header.det = mDet;
227  cd.header.nTriggers = digitVec.size();
228  cd.header.firstOrbit = dig0.getOrbit();
229  cd.header.firstBC = dig0.getBC();
230  cd.header.triggerGate = FV0DigParam::Instance().mTime_trg_gate;
231 
232  cd.trigger.resize(cd.header.nTriggers);
233  cd.bcInc.resize(cd.header.nTriggers);
234  cd.orbitInc.resize(cd.header.nTriggers);
235  cd.nChan.resize(cd.header.nTriggers);
236 
237  cd.idChan.resize(channelVec.size());
238  cd.qtcChain.resize(channelVec.size());
239  cd.cfdTime.resize(channelVec.size());
240  cd.qtcAmpl.resize(channelVec.size());
241 
242  uint16_t prevBC = cd.header.firstBC;
243  uint32_t prevOrbit = cd.header.firstOrbit;
244  uint32_t ccount = 0;
245  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
246  const auto& digit = digitVec[idig];
247  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
248 
249  // fill trigger info
250  cd.trigger[idig] = digit.getTriggers().getTriggersignals();
251  if (prevOrbit == digit.getOrbit()) {
252  cd.bcInc[idig] = digit.getBC() - prevBC;
253  cd.orbitInc[idig] = 0;
254  } else {
255  cd.bcInc[idig] = digit.getBC();
256  cd.orbitInc[idig] = digit.getOrbit() - prevOrbit;
257  }
258  prevBC = digit.getBC();
259  prevOrbit = digit.getOrbit();
260  // fill channels info
261  cd.nChan[idig] = chanels.size();
262  if (!cd.nChan[idig]) {
263  LOG(debug) << "Digits with no channels";
264  continue;
265  }
266  uint8_t prevChan = 0;
267  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
268  if constexpr (MINOR_VERSION == 0 && MAJOR_VERSION == 1) {
269  cd.idChan[ccount] = chanels[ic].ChId - prevChan; // Old method, lets keep it for a while
270  } else {
271  cd.idChan[ccount] = chanels[ic].ChId;
272  }
273  cd.qtcChain[ccount] = chanels[ic].ChainQTC;
274  cd.cfdTime[ccount] = chanels[ic].CFDTime;
275  cd.qtcAmpl[ccount] = chanels[ic].QTCAmpl;
276  prevChan = chanels[ic].ChId;
277  ccount++;
278  }
279  }
280 }
