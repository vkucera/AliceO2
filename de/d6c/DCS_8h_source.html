60 template <typename T>
61 struct DataPointVector {
62  using DPType = DataPoint<T>;
63  uint32_t sensorNumber{};
64  std::vector<DPType> data;
65 
66  void fill(const TimeStampType time, const T& value) { data.emplace_back(DPType{time, value}); }
67 
68  void fill(const DPType& dataPoint) { data.emplace_back(dataPoint); }
69 
70  void sort() { std::sort(data.begin(), data.end()); }
71 
72  void sortAndClean()
73  {
74  sort();
75  data.erase(
76  std::unique(data.begin(), data.end(),
77  [](const auto& dp1, const auto& dp2) {
78  return dp1.time == dp2.time;
79  }),
80  data.end());
81  data.shrink_to_fit();
82  }
83 
84  void clear() { data.clear(); }
85 
89  const T& getValueForTime(const TimeStampType timeStamp) const
90  {
91  const auto i = std::upper_bound(data.begin(), data.end(), DPType{timeStamp, {}});
92  return (i == data.begin()) ? (*i).value : (*(i - 1)).value;
93  };
94 
95  ClassDefNV(DataPointVector, 1);
96 };

235  static int getSector(std::string_view sensor)
236  {
237  const auto sideOffset = (sensor[SidePos] == 'A') ? 0 : SECTORSPERSIDE;
238  const auto sector = std::atoi(sensor.substr(SectorPos, 2).data());
239  return sector + sideOffset;
240  }

243  {
244  if (sensor[ROCPos] == 'I') {
245  return GEMstack::IROCgem;

247  const auto orocType = int(sensor[ROCPos + 1] - '0');
248  return static_cast<GEMstack>(orocType);

252  void fillUI(std::string_view sensor, const TimeStampType time, const DataType value)
253  {
254  const int sector = getSector(sensor);
255  const auto stack = getStack(sensor);
256  const auto rocOffset = int(stack != GEMstack::IROCgem);
257  const auto gem = int(sensor[GEMPos + rocOffset] - '0');
258  const bool isTop = sensor[ElectrodePos + rocOffset] == 'T';
259  // the counting is GEM1 top, bottom, GEM2 top, bottom, ...
260  const int electrode = 2 * (gem - 1) + !isTop;
261  const StackID stackID{sector, stack};
262  const int index = stackID.getIndex() * 2 * GEMSPERSTACK + electrode;
263 
264  const auto type = sensor.back();
265  // LOGP(info, "Fill type: {}, index: {} (sec: {}, stack: {}, gem: {}, elec: {}), time: {}, value: {}", type, index, sector, stack, gem, electrode, time, value);
266  if (type == 'I') {
267  currents[index].fill(time, value);
268  } else if (type == 'U') {
269  voltages[index].fill(time, value);
270  }
271  }

304 struct Gas {
305  static constexpr size_t SensorPos = 4;
306  static constexpr size_t TypePosGC = 7;
307  static constexpr size_t TypePosAn = 15;
308 
309  RawDPsF neon{};
310  RawDPsF co2{};
311  RawDPsF n2{};
312  RawDPsF argon{};
313  RawDPsF h2o{};
314  RawDPsF o2{};
315  RawDPsF h2oSensor{};
316  RawDPsF o2Sensor{};
317 
318  void fill(std::string_view sensor, const TimeStampType time, const DataType value)
319  {
320  if (sensor[SensorPos] == 'G') { // check if from GC
321  switch (sensor[TypePosGC]) {
322  case 'N': {
323  if (sensor[TypePosGC + 1] == 'E') {
324  neon.fill(time, value);
325  } else {
326  n2.fill(time, value);
327  }
328  break;
329  }
330  case 'A':
331  argon.fill(time, value);
332  break;
333  case 'C':
334  co2.fill(time, value);
335  break;
336  case 'O':
337  o2.fill(time, value);
338  break;
339  case 'W':
340  h2o.fill(time, value);
341  break;
342  default:
343  LOGP(warning, "Unknown gas sensor {}", sensor);
344  break;
345  }
346  } else { // otherwise dedicated sensor
347  switch (sensor[TypePosAn]) {
348  case 'H':
349  h2oSensor.fill(time, value);
350  break;
351  case 'O':
352  o2Sensor.fill(time, value);
353  break;
354  default:
355  LOGP(warning, "Unknown gas sensor {}", sensor);
356  break;
357  }
358  }
359  };
