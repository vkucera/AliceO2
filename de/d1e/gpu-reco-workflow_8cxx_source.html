107 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
108 {
109  WorkflowSpec specs;
110  std::vector<int> tpcSectors(o2::tpc::Sector::MAXSECTOR);
111  std::iota(tpcSectors.begin(), tpcSectors.end(), 0);
112 
113  auto inputType = cfgc.options().get<std::string>("input-type");
114  bool doMC = !cfgc.options().get<bool>("disable-mc");
115 
116  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
117  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
118  o2::conf::ConfigurableParam::writeINI("o2gpurecoworkflow_configuration.ini");
119 
120  std::vector<ioType> outputTypes, inputTypes;
121  try {
122  outputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("output-type"), [](std::string const& token) { return OutputMap.at(token); });
123  inputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("input-type"), [](std::string const& token) { return InputMap.at(token); });
124  } catch (std::out_of_range&) {
125  throw std::invalid_argument("invalid input / output type");
126  }
127 
128  auto isEnabled = [](auto& list, ioType type) {
129  return std::find(list.begin(), list.end(), type) != list.end();
130  };
131 
132  GPURecoWorkflowSpec::Config cfg;
133  cfg.decompressTPC = isEnabled(inputTypes, ioType::CompClustCTF);
134  cfg.decompressTPCFromROOT = isEnabled(inputTypes, ioType::CompClustROOT);
135  cfg.zsDecoder = isEnabled(inputTypes, ioType::ZSRaw);
136  cfg.zsOnTheFly = isEnabled(inputTypes, ioType::ZSRawOTF);
137  cfg.caClusterer = cfg.zsDecoder || cfg.zsOnTheFly || isEnabled(inputTypes, ioType::Digits);
138  cfg.outputTracks = isEnabled(outputTypes, ioType::Tracks);
139  cfg.outputCompClusters = isEnabled(outputTypes, ioType::CompClustROOT);
140  cfg.outputCompClustersFlat = isEnabled(outputTypes, ioType::CompClustCTF);
141  cfg.outputCAClusters = isEnabled(outputTypes, ioType::Clusters);
142  cfg.outputQA = isEnabled(outputTypes, ioType::QA);
143  cfg.outputSharedClusterMap = (cfg.outputCAClusters || cfg.caClusterer || isEnabled(inputTypes, ioType::Clusters)) && cfg.outputTracks && !isEnabled(outputTypes, ioType::NoSharedMap);
144  cfg.processMC = doMC;
145  cfg.sendClustersPerSector = isEnabled(outputTypes, ioType::SendClustersPerSector);
146  cfg.askDISTSTF = !cfgc.options().get<bool>("ignore-dist-stf");
147  cfg.readTRDtracklets = isEnabled(inputTypes, ioType::TRDTracklets);
148  cfg.runTRDTracking = isEnabled(outputTypes, ioType::TRDTracks);
149 
150  std::shared_ptr<GPURecoWorkflowSpec> task = std::make_shared<GPURecoWorkflowSpec>(&gPolicyData, cfg, tpcSectors, gTpcSectorMask);
151  specs.emplace_back(DataProcessorSpec{
152  "gpu-reconstruction",
153  task->inputs(),
154  task->outputs(),
155  AlgorithmSpec{adoptTask<GPURecoWorkflowSpec>(task)}});
156 
157  if (!cfgc.options().get<bool>("ignore-dist-stf")) {
158  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask("none");
159  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask("TPC");
160  o2::globaltracking::InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, doMC);
161  }
162 
163  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
164  o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
165 
166  return specs;
167 }
