90 void DigitRecoSpec::finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj)
91 {
92  if (matcher == ConcreteDataMatcher("ZDC", "MODULECONFIG", 0)) {
93  auto* config = (const o2::zdc::ModuleConfig*)obj;
94  if (mVerbosity > DbgZero) {
95  config->print();
96  }
97  mWorker.setModuleConfig(config);
98  }
99  if (matcher == ConcreteDataMatcher("ZDC", "RECOCONFIG", 0)) {
100  // Configuration parameters for ZDC reconstruction
101  auto* config = (const o2::zdc::RecoConfigZDC*)obj;
102  if (mVerbosity > DbgZero) {
103  config->print();
104  }
105  mWorker.setRecoConfigZDC(config);
106  }
107  if (matcher == ConcreteDataMatcher("ZDC", "TDCCALIB", 0)) {
108  // TDC centering
109  auto* config = (const o2::zdc::ZDCTDCParam*)obj;
110  if (mVerbosity > DbgZero) {
111  config->print();
112  }
113  mWorker.setTDCParam(config);
114  }
115  if (matcher == ConcreteDataMatcher("ZDC", "TDCCORR", 0)) {
116  // TDC correction parameters
117  auto* config = (const o2::zdc::ZDCTDCCorr*)obj;
118  if (mVerbosity > DbgZero) {
119  config->print();
120  }
121  mWorker.setTDCCorr(config);
122  }
123  if (matcher == ConcreteDataMatcher("ZDC", "ADCCALIB", 0)) {
124  // Energy calibration
125  auto* config = (const o2::zdc::ZDCEnergyParam*)obj;
126  if (mVerbosity > DbgZero) {
127  config->print();
128  }
129  mWorker.setEnergyParam(config);
130  }
131  if (matcher == ConcreteDataMatcher("ZDC", "TOWERCALIB", 0)) {
132  // Tower intercalibration
133  auto* config = (const o2::zdc::ZDCTowerParam*)obj;
134  if (mVerbosity > DbgZero) {
135  config->print();
136  }
137  mWorker.setTowerParam(config);
138  }
139  if (matcher == ConcreteDataMatcher("ZDC", "BASECALIB", 0)) {
140  // Average pedestals
141  auto* config = (const o2::zdc::BaselineParam*)obj;
142  if (mVerbosity > DbgZero) {
143  config->print();
144  }
145  mWorker.setBaselineParam(config);
146  }

149 void DigitRecoSpec::run(ProcessingContext& pc)
150 {
151  if (!mInitialized) {
152  mInitialized = true;
153  updateTimeDependentParams(pc);
154  if (mDebugOut) {
155  mWorker.setDebugOutput();
156  }
157  mWorker.setVerbosity(mVerbosity);
158  mWorker.init();
159  }
160  auto cput = mTimer.CpuTime();
161  mTimer.Start(false);
162  auto bcdata = pc.inputs().get<gsl::span<o2::zdc::BCData>>("trig");
163  auto chans = pc.inputs().get<gsl::span<o2::zdc::ChannelData>>("chan");
164  auto peds = pc.inputs().get<gsl::span<o2::zdc::OrbitData>>("peds");
165 
166  mWorker.process(peds, bcdata, chans);
167  const std::vector<o2::zdc::RecEventAux>& recAux = mWorker.getReco();
168 
169  // Transfer wafeform
170  bool fullinter = mWorker.getFullInterpolation();
171  RecEvent recEvent;
172  LOGF(info, "BC processed during reconstruction %d%s", recAux.size(), (fullinter ? " FullInterpolation" : ""));
173  uint32_t nte = 0, ntt = 0, nti = 0, ntw = 0;
174  for (auto reca : recAux) {
175  bool toAddBC = true;
176  int32_t ne = reca.ezdc.size();
177  int32_t nt = 0;
178  // Store TDC hits
179  for (int32_t it = 0; it < o2::zdc::NTDCChannels; it++) {
180  for (int32_t ih = 0; ih < reca.ntdc[it]; ih++) {
181  if (toAddBC) {
182  recEvent.addBC(reca);
183  toAddBC = false;
184  }
185  nt++;
186  recEvent.addTDC(it, reca.TDCVal[it][ih], reca.TDCAmp[it][ih], reca.isBeg[it], reca.isEnd[it]);
187  }
188  }
189  // Add waveform information
190  if (fullinter) {
191  for (int32_t isig = 0; isig < o2::zdc::NChannels; isig++) {
192  if (reca.inter[isig].size() == NIS) {
193  if (toAddBC) {
194  recEvent.addBC(reca);
195  toAddBC = false;
196  }
197  recEvent.addWaveform(isig, reca.inter[isig]);
198  ntw++;
199  }
200  }
201  // Limit the number of waveforms in output message
202  if (mMaxWave > 0 && ntw >= mMaxWave) {
203  LOG(warning) << "Maximum number of output waveforms per TF reached: " << mMaxWave;
204  break;
205  }
206  }
207  if (ne > 0) {
208  if (toAddBC) {
209  recEvent.addBC(reca);
210  toAddBC = false;
211  }
212  std::map<uint8_t, float>::iterator it;
213  for (it = reca.ezdc.begin(); it != reca.ezdc.end(); it++) {
214  recEvent.addEnergy(it->first, it->second);
215  }
216  }
217  nte += ne;
218  ntt += nt;
219  if (mVerbosity > 1 && (nt > 0 || ne > 0)) {
220  printf("Orbit %9u bc %4u ntdc %2d ne %2d channels=0x%08x\n", reca.ir.orbit, reca.ir.bc, nt, ne, reca.channels);
221  }
222  // Event information
223  nti += recEvent.addInfos(reca);
224  }
225  LOG(info) << "Reconstructed " << ntt << " signal TDCs and " << nte << " ZDC energies and "
226  << nti << " info messages in " << recEvent.mRecBC.size() << "/" << recAux.size() << " b.c. and "
227  << ntw << " waveform chunks";
228  // TODO: rate information for all channels
229  // TODO: summary of reconstruction to be collected by DQM?
230  pc.outputs().snapshot(Output{"ZDC", "BCREC", 0, Lifetime::Timeframe}, recEvent.mRecBC);
231  pc.outputs().snapshot(Output{"ZDC", "ENERGY", 0, Lifetime::Timeframe}, recEvent.mEnergy);
232  pc.outputs().snapshot(Output{"ZDC", "TDCDATA", 0, Lifetime::Timeframe}, recEvent.mTDCData);
233  pc.outputs().snapshot(Output{"ZDC", "INFO", 0, Lifetime::Timeframe}, recEvent.mInfo);
234  pc.outputs().snapshot(Output{"ZDC", "WAVE", 0, Lifetime::Timeframe}, recEvent.mWaveform);
235  mTimer.Stop();
236 }

245 framework::DataProcessorSpec getDigitRecoSpec(const int verbosity = 0, const bool enableDebugOut = true,
246  const bool enableZDCTDCCorr = true, const bool enableZDCEnergyParam = true, const bool enableZDCTowerParam = true, const bool enableBaselineParam = true)
247 {
248  std::vector<InputSpec> inputs;
249  inputs.emplace_back("trig", "ZDC", "DIGITSBC", 0, Lifetime::Timeframe);
250  inputs.emplace_back("chan", "ZDC", "DIGITSCH", 0, Lifetime::Timeframe);
251  inputs.emplace_back("peds", "ZDC", "DIGITSPD", 0, Lifetime::Timeframe);
252  inputs.emplace_back("moduleconfig", "ZDC", "MODULECONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathConfigModule.data()));
253  inputs.emplace_back("recoconfig", "ZDC", "RECOCONFIG", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathRecoConfigZDC.data()));
254  inputs.emplace_back("tdccalib", "ZDC", "TDCCALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathTDCCalib.data()));
255  if (enableZDCTDCCorr) {
256  inputs.emplace_back("tdccorr", "ZDC", "TDCCORR", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathTDCCorr.data()));
257  } else {
258  LOG(warning) << "ZDCTDCCorr has been disabled - no correction is applied";
259  }
260  if (enableZDCEnergyParam) {
261  inputs.emplace_back("adccalib", "ZDC", "ADCCALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathEnergyCalib.data()));
262  } else {
263  LOG(warning) << "ZDCEnergyParam has been disabled - no energy calibration is applied";
264  }
265  if (enableZDCTowerParam) {
266  inputs.emplace_back("towercalib", "ZDC", "TOWERCALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathTowerCalib.data()));
267  } else {
268  LOG(warning) << "ZDCTowerParam has been disabled - no tower intercalibration";
269  }
270  if (enableBaselineParam) {
271  inputs.emplace_back("basecalib", "ZDC", "BASECALIB", 0, Lifetime::Condition, o2::framework::ccdbParamSpec(o2::zdc::CCDBPathBaselineCalib.data()));
272  } else {
273  LOG(warning) << "BaselineParam has been disabled - no fallback in case orbit pedestals are missing";
274  }
275 
276  std::vector<OutputSpec> outputs;
277  outputs.emplace_back("ZDC", "BCREC", 0, Lifetime::Timeframe);
278  outputs.emplace_back("ZDC", "ENERGY", 0, Lifetime::Timeframe);
279  outputs.emplace_back("ZDC", "TDCDATA", 0, Lifetime::Timeframe);
280  outputs.emplace_back("ZDC", "INFO", 0, Lifetime::Timeframe);
281  outputs.emplace_back("ZDC", "WAVE", 0, Lifetime::Timeframe);
282 
283  return DataProcessorSpec{
284  "zdc-digi-reco",
285  inputs,
286  outputs,
287  AlgorithmSpec{adaptFromTask<DigitRecoSpec>(verbosity, enableDebugOut, enableZDCTDCCorr, enableZDCEnergyParam, enableZDCTowerParam, enableBaselineParam)},
288  o2::framework::Options{{"max-wave", o2::framework::VariantType::Int, 0, {"Maximum number of waveforms per TF in output"}}}};
289 }
