62 int GPUDisplayFrontendWayland::GetKey(unsigned int key, unsigned int state)
63 {
64  int retVal = 0;
65  if (mXKBkeymap) {
66  xkb_keysym_t sym = xkb_state_key_get_one_sym(mXKBstate, key + 8);
67  if (sym == 65453) {
68  return ('-');
69  } else if (sym == 65451) {
70  return ('+');
71  } else if (sym == XKB_KEY_Shift_L || sym == XKB_KEY_Shift_R) {
72  return (KEY_SHIFT);
73  } else if (sym == XKB_KEY_Alt_L) {
74  return (KEY_ALT);
75  } else if (sym == XKB_KEY_ISO_Level3_Shift || sym == XKB_KEY_Alt_R) {
76  return (KEY_RALT);
77  } else if (sym == XKB_KEY_Control_L) {
78  return (KEY_CTRL);
79  } else if (sym == XKB_KEY_Control_R) {
80  return (KEY_RCTRL);
81  } else if (sym == XKB_KEY_Up) {
82  return (KEY_UP);
83  } else if (sym == XKB_KEY_Down) {
84  return (KEY_DOWN);
85  } else if (sym == XKB_KEY_Left) {
86  return (KEY_LEFT);
87  } else if (sym == XKB_KEY_Right) {
88  return (KEY_RIGHT);
89  } else if (sym == XKB_KEY_Page_Up) {
90  return (KEY_PAGEUP);
91  } else if (sym == XKB_KEY_Page_Down) {
92  return (KEY_PAGEDOWN);
93  } else if (sym == XKB_KEY_Escape) {
94  return (KEY_ESCAPE);
95  } else if (sym == XKB_KEY_Return) {
96  return (KEY_ENTER);
97  } else if (sym == XKB_KEY_End) {
98  return (KEY_END);
99  } else if (sym == XKB_KEY_Home) {
100  return (KEY_HOME);
101  } else if (sym == XKB_KEY_Insert) {
102  return (KEY_INSERT);
103  } else if (sym == XKB_KEY_F1) {
104  return (KEY_F1);
105  } else if (sym == XKB_KEY_F2) {
106  return (KEY_F2);
107  } else if (sym == XKB_KEY_F3) {
108  return (KEY_F3);
109  } else if (sym == XKB_KEY_F4) {
110  return (KEY_F4);
111  } else if (sym == XKB_KEY_F5) {
112  return (KEY_F5);
113  } else if (sym == XKB_KEY_F6) {
114  return (KEY_F6);
115  } else if (sym == XKB_KEY_F7) {
116  return (KEY_F7);
117  } else if (sym == XKB_KEY_F8) {
118  return (KEY_F8);
119  } else if (sym == XKB_KEY_F9) {
120  return (KEY_F9);
121  } else if (sym == XKB_KEY_F10) {
122  return (KEY_F10);
123  } else if (sym == XKB_KEY_F11) {
124  return (KEY_F11);
125  } else if (sym == XKB_KEY_F12) {
126  return (KEY_F12);
127  } else if (sym == 32) {
128  return (KEY_SPACE);
129  } else if (sym > 255) {
130  return (0);
131  } else {
132  retVal = xkb_keysym_to_utf32(sym);
133  }
134  }
135  if (retVal > 255) {
136  return (0);
137  }
138  return retVal;
139 }

141 void GPUDisplayFrontendWayland::createBuffer(unsigned int width, unsigned int height)
142 {
143  const unsigned int stride = width * 4;
144  const unsigned int size = stride * height;
145  if (ftruncate(mFd, size) < 0) {
146  throw std::runtime_error("Error setting waysland shm file size");
147  }
148  void* shm_data = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, mFd, 0);
149  if (shm_data == MAP_FAILED) {
150  throw std::runtime_error("wayland mmap failed");
151  }
152  memset(shm_data, 0, size);
153  munmap(shm_data, size);
154 
155  mPool = wl_shm_create_pool(mShm, mFd, size);
156  mBuffer = wl_shm_pool_create_buffer(mPool, 0, width, height, stride, WL_SHM_FORMAT_XRGB8888);
157  mDisplayWidth = width;
158  mDisplayHeight = height;
159  wl_surface_attach(mSurface, mBuffer, 0, 0);
160  wl_surface_commit(mSurface);
161 }

163 void GPUDisplayFrontendWayland::recreateBuffer(unsigned int width, unsigned int height)
164 {

166  wl_surface_commit(mSurface);
167  wl_buffer_destroy(mBuffer);
168  wl_shm_pool_destroy(mPool);
169  createBuffer(width, height);
