39 GRPGeomRequest::GRPGeomRequest(bool orbitResetTime, bool GRPECS, bool GRPLHCIF, bool GRPMagField, bool askMatLUT, GeomRequest geom, std::vector<o2::framework::InputSpec>& inputs)
40  : askGRPECS(GRPECS), askGRPLHCIF(GRPLHCIF), askGRPMagField(GRPMagField), askMatLUT(askMatLUT), askTime(orbitResetTime)
41 {
42  if (geom == Aligned) {
43  askGeomAlign = true;
44  addInput({"geomAlp", "GLO", "GEOMALIGN", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/GeometryAligned")}, inputs);
45  } else if (geom == Ideal || geom == Alignments) {
46  askGeomIdeal = true;
47  addInput({"geomIdeal", "GLO", "GEOMIDEAL", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/Geometry")}, inputs);
48  }
49  if (geom == Alignments) {
50  askAlignments = true;
51  for (auto id = DetID::First; id <= DetID::Last; id++) {
52  std::string binding = fmt::format("align{}", DetID::getName(id));
53  addInput({binding, DetID::getDataOrigin(id), "ALIGNMENT", 0, Lifetime::Condition, ccdbParamSpec(fmt::format("{}/Calib/Align", DetID::getName(id)))}, inputs);
54  }
55  }
56  if (askMatLUT) {
57  addInput({"matLUT", "GLO", "MATLUT", 0, Lifetime::Condition, ccdbParamSpec("GLO/Param/MatLUT")}, inputs);
58  }
59  if (askTime) {
60  addInput({"orbitReset", "CTP", "ORBITRESET", 0, Lifetime::Condition, ccdbParamSpec("CTP/Calib/OrbitReset")}, inputs);
61  }
62  if (askGRPECS) {
63  addInput({"grpecs", "GLO", "GRPECS", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/GRPECS")}, inputs);
64  }
65  if (askGRPLHCIF) {
66  addInput({"grplhcif", "GLO", "GRPLHCIF", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/GRPLHCIF")}, inputs);
67  }
68  if (askGRPMagField) {
69  addInput({"grpfield", "GLO", "GRPMAGFIELD", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/GRPMagField")}, inputs);
70  }
71 }

90 void GRPGeomHelper::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
91 {
92  if (mRequest->askGRPMagField && matcher == ConcreteDataMatcher("GLO", "GRPMAGFIELD", 0)) {
93  bool needInit = mGRPMagField == nullptr;
94  mGRPMagField = (o2::parameters::GRPMagField*)obj;
95  LOG(info) << "GRP MagField object updated";
96  if (needInit) {
97  o2::base::Propagator::initFieldFromGRP(mGRPMagField);
98  } else {
99  auto field = TGeoGlobalMagField::Instance()->GetField();
100  if (field->InheritsFrom("o2::field::MagneticField")) {
101  ((o2::field::MagneticField*)field)->rescaleField(mGRPMagField->getL3Current(), mGRPMagField->getDipoleCurrent(), mGRPMagField->getFieldUniformity());
102  }
103  o2::base::Propagator::Instance(false)->updateField();
104  }
105  return;
106  }
107  if (mRequest->askGRPECS && matcher == ConcreteDataMatcher("GLO", "GRPECS", 0)) {
108  mGRPECS = (o2::parameters::GRPECSObject*)obj;
109  LOG(info) << "GRP ECS object updated";
110  return;
111  }
112  if (mRequest->askGRPLHCIF && matcher == ConcreteDataMatcher("GLO", "GRPLHCIF", 0)) {
113  mGRPLHCIF = (o2::parameters::GRPLHCIFData*)obj;
114  LOG(info) << "GRP LHCIF object updated";
115  return;
116  }
117  if (mRequest->askTime && matcher == ConcreteDataMatcher("CTP", "ORBITRESET", 0)) {
118  mOrbitResetTimeMS = (*(std::vector<Long64_t>*)obj)[0] / 1000;
119  LOG(info) << "orbit reset time updated to " << mOrbitResetTimeMS;
120  return;
121  }
122  if (mRequest->askMatLUT && matcher == ConcreteDataMatcher("GLO", "MATLUT", 0)) {
123  LOG(info) << "material LUT updated";
124  mMatLUT = o2::base::MatLayerCylSet::rectifyPtrFromFile((o2::base::MatLayerCylSet*)obj);
125  o2::base::Propagator::Instance(false)->setMatLUT(mMatLUT);
126  return;
127  }
128  if (mRequest->askGeomAlign && matcher == ConcreteDataMatcher("GLO", "GEOMALIGN", 0)) {
129  LOG(info) << "aligned geometry updated";
130  return;
131  }
132  if (mRequest->askGeomIdeal && matcher == ConcreteDataMatcher("GLO", "GEOMIDEAL", 0)) {
133  LOG(info) << "ideal geometry updated";
134  return;
135  }
136  constexpr o2::header::DataDescription algDesc{"ALIGNMENT"};
137  if (mRequest->askAlignments && matcher.description == algDesc) {
138  for (auto id = DetID::First; id <= DetID::Last; id++) {
139  if (matcher.origin == DetID::getDataOrigin(id)) {
140  LOG(info) << DetID::getName(id) << " alignment updated";
141  mAlignments[id] = (std::vector<o2::detectors::AlignParam>*)obj;
142  break;
143  }
144  }
145  return;
146  }
147 }

149 void GRPGeomHelper::checkUpdates(ProcessingContext& pc) const
150 {
151  // request input just to trigger finaliseCCDB if there was an update
152  if (mRequest->askGRPMagField) {
153  pc.inputs().get<o2::parameters::GRPMagField*>("grpfield");
154  }
155  if (mRequest->askGRPLHCIF) {
156  pc.inputs().get<o2::parameters::GRPLHCIFData*>("grplhcif");
157  }
158  if (mRequest->askGRPECS) {
159  pc.inputs().get<o2::parameters::GRPECSObject*>("grpecs");
160  }
161  if (mRequest->askTime) {
162  pc.inputs().get<std::vector<Long64_t>*>("orbitReset");
163  }
164  if (mRequest->askMatLUT) {
165  pc.inputs().get<o2::base::MatLayerCylSet*>("matLUT");
166  }
167  if (mRequest->askGeomAlign) {
168  pc.inputs().get<TGeoManager*>("geomAlp");
169  } else if (mRequest->askGeomIdeal) {
170  pc.inputs().get<TGeoManager*>("geomIdeal");
171  }
172  if (mRequest->askAlignments) {
173  for (auto id = DetID::First; id <= DetID::Last; id++) {
174  std::string binding = fmt::format("align{}", DetID::getName(id));
175  pc.inputs().get<std::vector<o2::detectors::AlignParam>*>(binding);
176  }
177  }
178 }
