74 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
75 {
76  WorkflowSpec specs;
77 
78  // Update the (declared) parameters if changed from the command line
79  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
80  // write the configuration used for the workflow
81  o2::conf::ConfigurableParam::writeINI("o2match-tof-workflow_configuration.ini");
82 
83  auto useMC = !configcontext.options().get<bool>("disable-mc");
84  auto disableRootIn = configcontext.options().get<bool>("disable-root-input");
85  auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
86  auto useFIT = configcontext.options().get<bool>("use-fit");
87  auto useCCDB = configcontext.options().get<bool>("use-ccdb");
88  auto strict = configcontext.options().get<bool>("strict-matching");
89  auto diagnostic = configcontext.options().get<bool>("enable-dia");
90 
91  bool writematching = 0;
92  bool writecalib = 0;
93  auto outputType = configcontext.options().get<std::string>("output-type");
94  if (outputType.rfind("matching-info") < outputType.size()) {
95  writematching = 1;
96  }
97  if (outputType.rfind("calib-info") < outputType.size()) {
98  writecalib = 1;
99  }
100 
101  if (!writecalib) {
102  useFIT = false;
103  }
104 
105  LOG(debug) << "TOF MATCHER WORKFLOW configuration";
106  LOG(debug) << "TOF track inputs = " << configcontext.options().get<std::string>("track-sources");
107  LOG(debug) << "TOF output = " << outputType;
108  LOG(debug) << "TOF disable-mc = " << configcontext.options().get<std::string>("disable-mc");
109  LOG(debug) << "TOF use-ccdb = " << useCCDB;
110  LOG(debug) << "TOF use-fit = " << useFIT;
111  LOG(debug) << "TOF disable-root-input = " << disableRootIn;
112  LOG(debug) << "TOF disable-root-output = " << disableRootOut;
113  LOG(debug) << "TOF matching in strict mode = " << strict;
114 
115  //GID::mask_t alowedSources = GID::getSourcesMask("TPC,ITS-TPC");
116  GID::mask_t alowedSources = GID::getSourcesMask("TPC,ITS-TPC,TPC-TRD,ITS-TPC-TRD");
117 
118  GID::mask_t src = alowedSources & GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
119  if (strict && (src & ~GID::getSourcesMask("TPC,TPC-TRD")).any()) {
120  LOGP(warning, "In strict matching mode only TPC and TPC-TRD sources allowed, {} asked, redefining", GID::getSourcesNames(src));
121  src &= GID::getSourcesMask("TPC,TPC-TRD");
122  }
123  GID::mask_t mcmaskcl;
124  GID::mask_t nonemask = GID::getSourcesMask(GID::NONE);
125  GID::mask_t clustermask = GID::getSourcesMask("TOF");
126  if (useFIT) {
127  clustermask |= GID::getSourceMask(GID::FT0);
128  }
129 
130  if (useMC) {
131  mcmaskcl |= GID::getSourceMask(GID::TOF);
132  }
133  // pass strict flag to fetch eventual TPC-TRD input with correct subspec
134  WorkflowSpec inputspecs;
135  o2::globaltracking::InputHelper::addInputSpecs(configcontext, inputspecs, clustermask, nonemask, src, useMC, mcmaskcl, GID::getSourcesMask(GID::ALL), strict);
136  if (configcontext.options().get<bool>("combine-devices")) {
137  std::vector<DataProcessorSpec> unmerged;
138  specs.push_back(specCombiner("TOF-readers", inputspecs, unmerged));
139  if (unmerged.size() > 0) {
140  LOG(fatal) << "Unexpected DPL device merge problem";
141  }
142  } else {
143  for (auto& s : inputspecs) {
144  specs.push_back(s);
145  }
146  }
147 
148  specs.emplace_back(o2::globaltracking::getTOFMatcherSpec(src, useMC, useFIT, false, strict)); // doTPCrefit not yet supported (need to load TPC clusters?)
149 
150  // initialize collision context
151  if (gSystem->AccessPathName("collisioncontext.root")) {
152  LOG(info) << "collisioncontext.root not available, let's skip it (cosmics?) ";
153  } else {
154  auto mcReader = std::make_unique<o2::steer::MCKinematicsReader>("collisioncontext.root");
155  auto context = mcReader->getDigitizationContext();
156  if (context) {
157  auto bcf = context->getBunchFilling();
158  std::bitset<3564> isInBC = bcf.getBCPattern();
159  for (unsigned int i = 0; i < isInBC.size(); i++) {
160  if (isInBC.test(i)) {
161  o2::tof::Utils::addInteractionBC(i, true);
162  }
163  }
164  }
165  }
166 
167  if (!disableRootOut) {
168  std::vector<DataProcessorSpec> writers;
169  if (writematching) {
170  if (GID::includesSource(GID::TPC, src)) { // matching to TPC was requested
171  writers.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_tpc.root", true, (int)o2::dataformats::MatchInfoTOFReco::TrackType::TPC, strict));
172  }
173  if (GID::includesSource(GID::ITSTPC, src)) { // matching to ITS-TPC was requested, there is not strict mode in this case
174  writers.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_itstpc.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPC, false));
175  }
176  if (GID::includesSource(GID::TPCTRD, src)) { // matching to TPC-TRD was requested, there is not strict mode in this case
177  writers.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_tpctrd.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::TPCTRD, strict));
178  }
179  if (GID::includesSource(GID::ITSTPCTRD, src)) { // matching to ITS-TPC-TRD was requested, there is not strict mode in this case
180  writers.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_itstpctrd.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPCTRD, false));
181  }
182  }
183  if (writecalib) {
184  writers.emplace_back(o2::tof::getTOFCalibWriterSpec("o2calib_tof.root", 0, diagnostic));
185  }
186 
187  if (configcontext.options().get<bool>("combine-devices")) {
188  std::vector<DataProcessorSpec> unmerged;
189  specs.push_back(specCombiner("TOF-writers", writers, unmerged));
190  if (unmerged.size() > 0) {
191  LOG(fatal) << "Unexpected DPL device merge problem";
192  }
193  } else {
194  for (auto& s : writers) {
195  specs.push_back(s);
196  }
197  }
198  }
199 
200  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
201  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
202 
203  return std::move(specs);
204 }
