36 class CalibdEdxDevice : public Task
37 {
38  public:
39  void init(framework::InitContext& ic) final
40  {
41  const auto minEntriesSector = ic.options().get<int>("min-entries-sector");
42  const auto minEntries1D = ic.options().get<int>("min-entries-1d");
43  const auto minEntries2D = ic.options().get<int>("min-entries-2d");
44  const auto fitPasses = ic.options().get<int>("fit-passes");
45  const auto fitThreshold = ic.options().get<float>("fit-threshold");
46 
47  const auto dEdxBins = ic.options().get<int>("dedxbins");
48  const auto mindEdx = ic.options().get<float>("min-dedx");
49  const auto maxdEdx = ic.options().get<float>("max-dedx");
50  const auto angularBins = ic.options().get<int>("angularbins");
51  const auto fitSnp = ic.options().get<bool>("fit-snp");
52 
53  mDumpToFile = ic.options().get<bool>("file-dump");
54  auto field = ic.options().get<float>("field");
55 
56  if (field <= -10.f) {
57  const auto inputGRP = o2::base::NameConf::getGRPFileName();
58  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
59  if (grp != nullptr) {
60  field = 5.00668f * grp->getL3Current() / 30000.;
61  LOGP(info, "Using GRP file to set the magnetic field to {} kG", field);
62  }
63  }
64 
65  mCalib = std::make_unique<CalibdEdx>(dEdxBins, mindEdx, maxdEdx, angularBins, fitSnp);
66  mCalib->setApplyCuts(false);
67  mCalib->setSectorFitThreshold(minEntriesSector);
68  mCalib->set1DFitThreshold(minEntries1D);
69  mCalib->set2DFitThreshold(minEntries2D);
70  mCalib->setField(field);
71  mCalib->setElectronCut(fitThreshold, fitPasses);
72  }
73 
74  void run(ProcessingContext& pc) final
75  {
76  const auto tfcounter = o2::header::get<DataProcessingHeader*>(pc.inputs().get("tracks").header)->startTime;
77  const auto tracks = pc.inputs().get<gsl::span<TrackTPC>>("tracks");
78 
79  LOGP(info, "Processing TF {} with {} tracks", tfcounter, tracks.size());
80  mRunNumber = processing_helpers::getRunNumber(pc);
81  mCalib->fill(tracks);
82  }
83 
84  void endOfStream(EndOfStreamContext& eos) final
85  {
86  LOGP(info, "Finalizing calibration");
87  mCalib->finalize();
88  mCalib->print();
89  sendOutput(eos.outputs());
90 
91  if (mDumpToFile) {
92  mCalib->getCalib().writeToFile("calibdEdx.root");
93  }
94  }
95 
96  private:
97  void sendOutput(DataAllocator& output)
98  {
99  using clbUtils = o2::calibration::Utils;
100  const auto& corr = mCalib->getCalib();
101 
102  o2::ccdb::CcdbObjectInfo info;
103 
104  const auto now = std::chrono::system_clock::now();
105  const long timeStart = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
106  const long timeEnd = o2::ccdb::CcdbObjectInfo::INFINITE_TIMESTAMP;
107 
108  info.setPath("TPC/Calib/dEdx");
109  info.setStartValidityTimestamp(timeStart);
110  info.setEndValidityTimestamp(timeEnd);
111 
112  auto md = info.getMetaData();
113  md["runNumber"] = std::to_string(mRunNumber);
114  info.setMetaData(md);
115 
116  auto image = o2::ccdb::CcdbApi::createObjectImage(&corr, &info);
117 
118  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", info.getPath(), info.getFileName(), image->size(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
119  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx", 0}, *image.get());
120  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx", 0}, info);
121  }
122 
123  bool mDumpToFile{};
124  uint64_t mRunNumber{0};
125  std::unique_ptr<CalibdEdx> mCalib;
126 };

128 DataProcessorSpec getCalibdEdxSpec()
129 {
130  std::vector<OutputSpec> outputs;
131  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx"}, Lifetime::Sporadic);
132  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx"}, Lifetime::Sporadic);
133 
134  return DataProcessorSpec{
135  "tpc-calib-dEdx",
136  Inputs{
137  InputSpec{"tracks", "TPC", "MIPS"},
138  },
139  outputs,
140  adaptFromTask<CalibdEdxDevice>(),
141  Options{
142  {"min-entries-sector", VariantType::Int, 1000, {"min entries per GEM stack to enable sector by sector correction. Below this value we only perform one fit per ROC type (IROC, OROC1, ...; no side nor sector information)."}},
143  {"min-entries-1d", VariantType::Int, 10000, {"minimum entries per stack to fit 1D correction"}},
144  {"min-entries-2d", VariantType::Int, 50000, {"minimum entries per stack to fit 2D correction"}},
145  {"fit-passes", VariantType::Int, 3, {"number of fit iterations"}},
146  {"fit-threshold", VariantType::Float, 0.2f, {"dEdx width around the MIP peak used in the fit"}},
147 
148  {"dedxbins", VariantType::Int, 60, {"number of dEdx bins"}},
149  {"min-dedx", VariantType::Float, 20.0f, {"minimum value for dEdx histograms"}},
150  {"max-dedx", VariantType::Float, 90.0f, {"maximum value for dEdx histograms"}},
151  {"angularbins", VariantType::Int, 36, {"number of angular bins: Tgl and Snp"}},
152  {"fit-snp", VariantType::Bool, false, {"enable Snp correction"}},
153 
154  {"field", VariantType::Float, -100.f, {"magnetic field"}},
155  {"file-dump", VariantType::Bool, false, {"directly dump calibration to file"}}}};
156 }
