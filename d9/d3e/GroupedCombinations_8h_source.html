23 // Create an instance of a tuple interleaved from given tuples
24 template <typename... T1s, typename... T2s, std::size_t... Is>
25 auto interleaveTuplesImpl(std::tuple<T1s...>& t1, std::tuple<T2s...>& t2, std::index_sequence<Is...>)

27  return std::tuple_cat(std::make_tuple(std::get<Is>(t1), std::get<Is>(t2))...);

39 template <typename G, typename A>
40 expressions::BindingNode getMatchingIndexNode()
41 {
42  using external_index_columns_pack = typename A::external_index_columns_t;
43  using selected_indices_t = selected_pack_multicondition<is_index_to_g_t, pack<G>, external_index_columns_pack>;
44  static_assert(pack_size(selected_indices_t{}) == 1, "No matching index column from associated to grouping");
45  using index_column_t = pack_head_t<selected_indices_t>;
46  auto label = std::string(index_column_t::mLabel);
47  return expressions::BindingNode{label, typeid(typename index_column_t::column_t).hash_code(), expressions::selectArrowType<typename index_column_t::type>()};
48 }

54  struct GroupedIterator : public GroupingPolicy {
55  public:

57  using value_type = GroupedIteratorType;
58  using pointer = GroupedIteratorType*;

61  GroupedIterator(const GroupingPolicy& groupingPolicy) : GroupingPolicy(groupingPolicy), mIndexColumns{getMatchingIndexNode<G, As>()...} {}
62  template <typename... T2s>
63  GroupedIterator(const GroupingPolicy& groupingPolicy, const G& grouping, std::tuple<T2s...>& associated) : GroupingPolicy(groupingPolicy), mGrouping{std::make_shared<G>(std::vector{grouping.asArrowTable()})}, mAssociated{std::make_shared<std::tuple<As...>>(std::make_tuple(std::get<As>(associated)...))}, mIndexColumns{getMatchingIndexNode<G, As>()...}
64  {
65  if constexpr (soa::is_soa_filtered_t<std::decay_t<G>>::value) {
66  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()}, grouping.getSelectedRows());
67  } else {
68  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()});
69  }
70  setMultipleGroupingTables<sizeof...(As)>(grouping);
71  if (!this->mIsEnd) {
72  setCurrentGroupedCombination();
73  }
74  }

80  template <typename... T2s>
81  void setTables(const G& grouping, std::tuple<T2s...>& associated)
82  {

84  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()}, grouping.getSelectedRows());

95  template <std::size_t N, typename T, typename... Args>
96  void setMultipleGroupingTables(const T& param, const Args&... args)
97  {
98  if constexpr (N == 1) {

101  setMultipleGroupingTables<N - 1>(param, param, args...);

105  void moveToEnd()
106  {
107  GroupingPolicy::moveToEnd();

110  // prefix increment
111  GroupedIterator& operator++()
112  {
113  if (!this->mIsEnd) {
114  this->addOne();
115  setCurrentGroupedCombination();
116  }
117  return *this;
118  }
119  // postfix increment
120  GroupedIterator operator++(int /*unused*/)
121  {
122  GroupedIterator copy(*this);
123  operator++();
124  return copy;
125  }
126  // return reference
127  reference operator*()
128  {
129  return *mCurrentGrouped;
130  }
131  bool operator==(const GroupedIterator& rh)
132  {
133  return (this->mIsEnd && rh.mIsEnd) || (this->mCurrent == rh.mCurrent);
134  }
135  bool operator!=(const GroupedIterator& rh)
136  {
137  return !(*this == rh);
138  }
139 
140  private:
141  std::tuple<As...> getAssociatedTables()
142  {
143  return doGetAssociatedTables(std::make_index_sequence<sizeof...(As)>());
144  }
145 
146  template <std::size_t... Is>
147  std::tuple<As...> doGetAssociatedTables(std::index_sequence<Is...>)
148  {
149  return std::make_tuple(getAssociatedTable<Is>()...);
150  }
151 
152  template <std::size_t I>
153  auto getAssociatedTable()
154  {
155  uint64_t ind = *std::get<0>(std::get<I>(this->mCurrent).getIndices());
156  return std::get<I>(*mAssociated).sliceByCached(mIndexColumns[I], ind);
157  }
158 
159  void setCurrentGroupedCombination()
160  {
161  if (!this->mIsEnd) {
162  auto& currentGrouping = GroupingPolicy::mCurrent;
163  mSlices.emplace(getAssociatedTables());
164  mCurrentGrouped.emplace(interleaveTuples(currentGrouping, *mSlices));
165  }
166  }
167 
168  std::array<expressions::BindingNode, sizeof...(As)> mIndexColumns;
169  std::shared_ptr<G> mGrouping;
170  std::shared_ptr<std::tuple<As...>> mAssociated;
171  std::optional<std::tuple<As...>> mSlices;
172  std::optional<GroupedIteratorType> mCurrentGrouped;
173  };
