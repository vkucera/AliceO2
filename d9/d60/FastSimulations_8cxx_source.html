43 void NeuralFastSimulation::setInputOutputData()
44 {
45  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
46  mInputNames.push_back(mSession.GetInputName(i, mAllocator));
47  }
48  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
49  mInputShapes.emplace_back(mSession.GetInputTypeInfo(i).GetTensorTypeAndShapeInfo().GetShape());
50  }
51  for (size_t i = 0; i < mSession.GetOutputCount(); ++i) {
52  mOutputNames.push_back(mSession.GetOutputName(i, mAllocator));
53  }
54 
55  // Prevent negative values from being passed as tensor shape
56  // Which is no problem in python implementation of ONNX where -1 means that
57  // shape has to be figured out by library. In C++ this is illegal
58  for (auto& shape : mInputShapes) {
59  for (auto& elem : shape) {
60  if (elem < 0) {
61  elem = mBatchSize;
62  }
63  }
64  }
65 }

129 std::optional<std::pair<std::vector<float>, std::vector<float>>> o2::zdc::fastsim::loadScales(const std::string& path)
130 {
131  std::fstream file(path, std::fstream::in);
132  if (!file.is_open()) {
133  return std::nullopt;
134  }
135 
136  auto means = parse_block(file, "#means");
137  if (means.empty()) {
138  return std::nullopt;
139  }
140 
141  auto scales = parse_block(file, "#scales");
142  if (scales.empty()) {
143  return std::nullopt;
144  }
145 
146  if (means.size() != scales.size()) {
147  return std::nullopt;
148  }
149  return std::make_pair(std::move(means), std::move(scales));
150 }
