33 void CalibPadGainTracks::processTracks(const int nMaxTracks)
34 {
35  if (!mPropagateTrack && !mFastTransform) {
36  mFastTransform = TPCFastTransformHelperO2::instance()->create(0);
37  }
38 
39  std::unique_ptr<o2::gpu::GPUO2InterfaceRefit> refit;
40  if (!mPropagateTrack) {
41  mBufVec.resize(mClusterIndex->nClustersTotal);
42  o2::gpu::GPUO2InterfaceRefit::fillSharedClustersMap(mClusterIndex, *mTracks, mTPCTrackClIdxVecInput->data(), mBufVec.data());
43  mClusterShMapTPC = mBufVec.data();
44  refit = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(mClusterIndex, mFastTransform.get(), mField, mTPCTrackClIdxVecInput->data(), mClusterShMapTPC);
45  }
46 
47  const size_t loopEnd = (nMaxTracks < 0) ? mTracks->size() : ((nMaxTracks > mTracks->size()) ? mTracks->size() : size_t(nMaxTracks));
48 
49  if (loopEnd < mTracks->size()) {
50  // draw random tracks
51  std::vector<size_t> ind(mTracks->size());
52  std::iota(ind.begin(), ind.end(), 0);
53  std::minstd_rand rng(std::time(nullptr));
54  std::shuffle(ind.begin(), ind.end(), rng);
55  for (size_t i = 0; i < loopEnd; ++i) {
56  processTrack((*mTracks)[ind[i]], refit.get());
57  }
58  } else {
59  for (const auto& trk : *mTracks) {
60  processTrack(trk, refit.get());
61  }
62  }
63 }

65 void CalibPadGainTracks::processTrack(o2::tpc::TrackTPC track, o2::gpu::GPUO2InterfaceRefit* refit)
66 {
67  // make momentum cut
68  const float mom = track.getP();
69  const int nClusters = track.getNClusterReferences();
70  if (mom < mMomMin || mom > mMomMax || std::abs(track.getEta()) > mEtaMax || nClusters < mMinClusters) {
71  return;
72  }
73 
74  // clearing memory
75  for (auto& buffer : mDEdxBuffer) {
76  buffer.clear();
77  }
78  mClTrk.clear();
79 
80  for (int iCl = 0; iCl < nClusters; iCl++) { // loop over cluster
81  const o2::tpc::ClusterNative& cl = track.getCluster(*mTPCTrackClIdxVecInput, iCl, *mClusterIndex);
82 
83  const auto flagsCl = cl.getFlags();
84  if ((flagsCl & ClusterNative::flagSingle) == ClusterNative::flagSingle) {
85  continue;
86  }
87 
88  unsigned char sectorIndex = 0;
89  unsigned char rowIndex = 0;
90  unsigned int clusterIndexNumb = 0;
91 
92  // this function sets sectorIndex, rowIndex, clusterIndexNumb
93  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
94  const float xPosition = Mapper::instance().getPadCentre(PadPos(rowIndex, 0)).X();
95  if (!mPropagateTrack) {
96  refit->setTrackReferenceX(xPosition);
97  }
98  const bool check = mPropagateTrack ? track.propagateTo(xPosition, mField) : ((refit->RefitTrackAsGPU(track, false, true) < 0) ? false : true); // propagate this track to the plane X=xk (cm) in the field "b" (kG)
99 
100  if (!check || std::isnan(track.getParam(1))) {
101  continue;
102  }
103 
104  const int region = Mapper::REGION[rowIndex];
105  const float charge = (mChargeType == ChargeType::Max) ? cl.qMax : cl.qTot;
106  const float effectiveLength = mCalibTrackTopologyPol ? getTrackTopologyCorrectionPol(track, cl, region, charge) : getTrackTopologyCorrection(track, region);
107 
108  const unsigned char pad = static_cast<unsigned char>(cl.getPad() + 0.5f); // the left side of the pad is defined at e.g. 3.5 and the right side at 4.5
109  const float gain = mGainMapRef ? mGainMapRef->getValue(sectorIndex, rowIndex, pad) : 1;
110  const float chargeNorm = charge / (effectiveLength * gain);
111 
112  if (mMode == dedxTracking) {
113  const auto& dEdx = track.getdEdx();
114  float dedx = -1;
115  const CRU cru(Sector(sectorIndex), region);
116 
117  if (mDedxRegion == stack) {
118  const auto stack = cru.gemStack();
119  if (stack == GEMstack::IROCgem && dEdx.NHitsIROC > mMinClusters) {
120  dedx = getdEdxIROC(dEdx);
121  } else if (stack == GEMstack::OROC1gem && dEdx.NHitsOROC1 > mMinClusters) {
122  dedx = getdEdxOROC1(dEdx);
123  } else if (stack == GEMstack::OROC2gem && dEdx.NHitsOROC2 > mMinClusters) {
124  dedx = getdEdxOROC2(dEdx);
125  } else if (stack == GEMstack::OROC3gem && dEdx.NHitsOROC3 > mMinClusters) {
126  dedx = getdEdxOROC3(dEdx);
127  }
128  } else if (mDedxRegion == chamber) {
129  if (cru.isIROC() && dEdx.NHitsIROC > mMinClusters) {
130  dedx = getdEdxIROC(dEdx);
131  } else {
132  int count = 0;
133  if (dEdx.NHitsOROC1 > mMinClusters) {
134  dedx += getdEdxOROC1(dEdx);
135  ++count;
136  }
137  if (dEdx.NHitsOROC2 > mMinClusters) {
138  dedx += getdEdxOROC2(dEdx);
139  ++count;
140  }
141  if (dEdx.NHitsOROC3 > mMinClusters) {
142  dedx += getdEdxOROC3(dEdx);
143  ++count;
144  }
145  if (count > 0) {
146  dedx /= count;
147  }
148  }
149  } else if (mDedxRegion == sector) {
150  dedx = getdEdxTPC(dEdx);
151  }
152 
153  if (dedx <= 0) {
154  continue;
155  }
156 
157  const float fillVal = chargeNorm / dedx;
158  int index = Mapper::GLOBALPADOFFSET[region] + Mapper::OFFSETCRUGLOBAL[rowIndex] + pad;
159  if (cru.isOROC()) {
160  index -= Mapper::getPadsInIROC();
161  }
162 
163  fillPadByPadHistogram(cru.roc().getRoc(), index, fillVal);
164  }
165 
166  if (mMode == dedxTrack) {
167  const int indexBuffer = getdEdxBufferIndex(region);
168 
169  const bool isEdge = (flagsCl & ClusterNative::flagEdge) == ClusterNative::flagEdge;
170  const int isSectorCentre = std::abs(static_cast<int>(pad) - static_cast<int>(Mapper::PADSPERROW[region][rowIndex] / 2)); // do not use clusters at the centre of a sector for dE/dx calculation
171  const int nPadsSector = 1;
172 
173  if (!isEdge && (isSectorCentre > nPadsSector)) {
174  mDEdxBuffer[indexBuffer].emplace_back(chargeNorm);
175  }
176 
177  mClTrk.emplace_back(std::make_tuple(sectorIndex, rowIndex, pad, chargeNorm)); // fill with dummy dedx value
178  }
179  }
180 
181  if (mMode == dedxTrack) {
182  getTruncMean();
183 
184  // set the dEdx
185  for (auto& x : mClTrk) {
186  const unsigned char globRow = std::get<1>(x);
187  const int region = Mapper::REGION[globRow];
188  const int indexBuffer = getdEdxBufferIndex(region);
189 
190  const float dedxTmp = mDedxTmp[indexBuffer];
191  if (dedxTmp <= 0) {
192  continue;
193  }
194 
195  const unsigned char pad = std::get<2>(x);
196  int index = Mapper::GLOBALPADOFFSET[region] + Mapper::OFFSETCRUGLOBAL[globRow] + pad;
197  const ROC roc = CRU(Sector(std::get<0>(x)), region).roc();
198  if (roc.isOROC()) {
199  index -= Mapper::getPadsInIROC();
200  }
201 
202  // fill the normalizes charge in pad histogram
203  const float fillVal = std::get<3>(x) / dedxTmp;
204  fillPadByPadHistogram(roc.getRoc(), index, fillVal);
205  }
206  } else {
207  }
208 }

210 void CalibPadGainTracks::getTruncMean(float low, float high)
211 {
212  mDedxTmp.clear();
213  mDedxTmp.reserve(mDEdxBuffer.size());
214  // returns the truncated mean for input vector
215  for (auto& charge : mDEdxBuffer) {
216  const int nClustersUsed = static_cast<int>(charge.size());
217  if (nClustersUsed < mMinClusters) {
218  mDedxTmp.emplace_back(-1);
219  continue;
220  }
221 
222  std::sort(charge.begin(), charge.end()); // sort the vector for performing truncated mean
223 
224  const int startInd = static_cast<int>(low * nClustersUsed);
225  const int endInd = static_cast<int>(high * nClustersUsed);
226 
227  if (endInd <= startInd) {
228  mDedxTmp.emplace_back(-1);
229  continue;
230  }
231 
232  const float dEdx = std::accumulate(charge.begin() + startInd, charge.begin() + endInd, 0.f);
233  const int nClustersTrunc = endInd - startInd; // count number of clusters
234  mDedxTmp.emplace_back(dEdx / nClustersTrunc);
235  }
236 }

238 float CalibPadGainTracks::getTrackTopologyCorrection(const o2::tpc::TrackTPC& track, const unsigned int region) const
239 {
240  const float padLength = Mapper::instance().getPadRegionInfo(region).getPadHeight();
241  const float sinPhi = track.getSnp();
242  const float tgl = track.getTgl();
243  const float snp2 = sinPhi * sinPhi;
244  const float effectiveLength = padLength * std::sqrt((1 + tgl * tgl) / (1 - snp2)); // calculate the trace length of the track over the pad
245  return effectiveLength;
246 }

248 float CalibPadGainTracks::getTrackTopologyCorrectionPol(const o2::tpc::TrackTPC& track, const o2::tpc::ClusterNative& cl, const unsigned int region, const float charge) const
249 {
250  const float trackSnp = track.getSnp();
251  const float maxSnp = mCalibTrackTopologyPol->getMaxSinPhi();
252  float snp = std::abs(trackSnp);
253  if (snp > maxSnp) {
254  snp = maxSnp;
255  }
256 
257  float snp2 = trackSnp * trackSnp;
258  const float sec2 = 1.f / (1.f - snp2);
259  const float trackTgl = track.getTgl();
260  const float tgl2 = trackTgl * trackTgl;
261  float tanTheta = std::sqrt(tgl2 * sec2);
262  const float maxTanTheta = mCalibTrackTopologyPol->getMaxTanTheta();
263  if (tanTheta > maxTanTheta) {
264  tanTheta = maxTanTheta;
265  }
266 
267  const float z = std::abs(track.getParam(1));
268  const float padTmp = cl.getPad();
269  const float absRelPad = std::abs(padTmp - int(padTmp + 0.5f));
270  const float relTime = cl.getTime() - int(cl.getTime() + 0.5f);
271 
272  const float effectiveLength = (mChargeType == ChargeType::Max) ? mCalibTrackTopologyPol->getCorrectionqMax(region, tanTheta, snp, z, absRelPad, relTime) : mCalibTrackTopologyPol->getCorrectionqTot(region, tanTheta, snp, z, 3.5f /*dummy threshold for now*/, std::clamp(charge, mCalibTrackTopologyPol->getMinqTot(), mCalibTrackTopologyPol->getMaxqTot()));
273  return effectiveLength;
274 }
