57 //_____________________________________________________________________________
58 void AmplitudeSpectrum::fill(float amplitude)
59 {
60  if ((lRange <= amplitude) && (amplitude < rRange)) {
61  int bin = (amplitude - lRange) / nBins;
62  mBinContent[bin]++;
63  mNEntries++;
64  mSumA += amplitude;
65  mSumA2 += amplitude * amplitude;
66  }
67 }
68 //_____________________________________________________________________________
69 void AmplitudeSpectrum::fillBinData(TH1F* h)
70 {
71  for (uint16_t i = 0; i < nBins; i++) {
72  h->SetBinContent(i + 1, float(mBinContent[i]));
73  h->SetBinError(i + 1, sqrt(float(mBinContent[i])));
74  }
75 }

119 void GainCalibrator::configParameters()
120 {
121  auto& cpvParams = o2::cpv::CPVCalibParams::Instance();
122  mMinEvents = cpvParams.gainMinEvents;
123  mMinNChannelsToCalibrate = cpvParams.gainMinNChannelsToCalibrate;
124  mDesiredLandauMPV = cpvParams.gainDesiredLandauMPV;
125  mToleratedChi2PerNDF = cpvParams.gainToleratedChi2PerNDF;
126  mMinAllowedCoeff = cpvParams.gainMinAllowedCoeff;
127  mMaxAllowedCoeff = cpvParams.gainMaxAllowedCoeff;
128  mFitRangeL = cpvParams.gainFitRangeL;
129  mFitRangeR = cpvParams.gainFitRangeR;
130  // adjust fit ranges to descrete binned values
131  if (mFitRangeL < AmplitudeSpectrum::lRange) {
132  mFitRangeL = AmplitudeSpectrum::lRange;
133  }
134  if (mFitRangeR > AmplitudeSpectrum::rRange) {
135  mFitRangeR = AmplitudeSpectrum::rRange;
136  }
137  double binWidth = (AmplitudeSpectrum::rRange - AmplitudeSpectrum::lRange) / AmplitudeSpectrum::nBins;
138  mFitRangeL = AmplitudeSpectrum::lRange + std::floor((mFitRangeL - AmplitudeSpectrum::lRange) / binWidth) * binWidth;
139  mFitRangeR = AmplitudeSpectrum::lRange + std::floor((mFitRangeR - AmplitudeSpectrum::lRange) / binWidth) * binWidth;
140 
141  LOG(info) << "GainCalibrator::configParameters() : Parameters used: ";
142  LOG(info) << "mMinEvents = " << mMinEvents;
143  LOG(info) << "mDesiredLandauMPV = " << mDesiredLandauMPV;
144  LOG(info) << "mToleratedChi2PerNDF = " << mToleratedChi2PerNDF;
145  LOG(info) << "mMinAllowedCoeff = " << mMinAllowedCoeff;
146  LOG(info) << "mMaxAllowedCoeff = " << mMaxAllowedCoeff;
147  LOG(info) << "mFitRangeL = " << mFitRangeL;
148  LOG(info) << "mFitRangeR = " << mFitRangeR;
149 }

160 void GainCalibrator::finalizeSlot(GainTimeSlot& slot)
161 {
162  GainCalibData* calibData = slot.getContainer();
163  LOG(info) << "GainCalibrator::finalizeSlot() : finalizing slot "
164  << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
165  slot.getContainer()->print();
166  CalibParams* newGains = new CalibParams(1);
167 
168  // estimate new gains by fitting amplitude spectra
169  int nChannelsCalibrated = 0, nChannelsTooSmallCoeff = 0, nChannelsTooLargeCoeff = 0;
170  TF1* fLandau = new TF1("fLandau", "landau", mFitRangeL, mFitRangeR);
171  fLandau->SetParLimits(0, 0., 1.E6);
172  fLandau->SetParLimits(1, mDesiredLandauMPV / mMaxAllowedCoeff, mDesiredLandauMPV / mMinAllowedCoeff);
173  fLandau->SetParLimits(1, 0., 1.E3);
174  TH1F h("histoCPVMaxAmplSpectrum", "", AmplitudeSpectrum::nBins, AmplitudeSpectrum::lRange, AmplitudeSpectrum::rRange);
175  // double binWidth = (AmplitudeSpectrum::rRange - AmplitudeSpectrum::lRange) / AmplitudeSpectrum::nBins;
176  // size_t nBinsToFit = (mFitRangeR - mFitRangeL) / binWidth;
177  // uint32_t xMin = (mFitRangeL - AmplitudeSpectrum::lRange) / binWidth;
178  // uint32_t xMax = (mFitRangeR - AmplitudeSpectrum::lRange) / binWidth;
179  int nCalibratedChannels = 0;
180  for (int i = 0; i < Geometry::kNCHANNELS; i++) {
181  // print some info
182  if ((i % 500) == 0) {
183  LOG(info) << "GainCalibrator::finalizeSlot() : checking channel " << i;
184  }
185  if (slot.getContainer()->mAmplitudeSpectra[i].getNEntries() > mMinEvents) { // we are ready to fit
186  h.Reset();
187  slot.getContainer()->mAmplitudeSpectra[i].fillBinData(&h);
188  // set some starting values
189  double mean = slot.getContainer()->mAmplitudeSpectra[i].getMean();
190  double rms = slot.getContainer()->mAmplitudeSpectra[i].getRMS();
191  double startingValue = slot.getContainer()->mAmplitudeSpectra[i].getBinContent()[(int)mean];
192  fLandau->SetParameters(startingValue, mean, rms);
193  auto fitResult = h.Fit(fLandau, "SQL0N", "", mFitRangeL, mFitRangeR);
194  // auto fitResult = o2::math_utils::fit<uint32_t>(nBinsToFit, &(slot.getContainer()->mAmplitudeSpectra[i].getBinContent()[xMin]), xMin, xMax, *fLandau);
195  if (fitResult->Chi2() / fitResult->Ndf() > mToleratedChi2PerNDF) {
196  // in case of bad fit -> do something sofisticated. but what?
197  // continue;
198  LOG(info) << "GainCalibrator::finalizeSlot() : bad chi2/ndf in fit of spectrum in channel " << i;
199  fitResult->Print("V");
200  }
201  // calib coeffs are defined as mDesiredLandauMPV/actualMPV*previousCoeff
202  float coeff = mDesiredLandauMPV / fLandau->GetParameter(1) * mPreviousGains.get()->getGain(i);
203  slot.getContainer()->mAmplitudeSpectra[i].reset();
204  if (mMinAllowedCoeff <= coeff && coeff <= mMaxAllowedCoeff) {
205  newGains->setGain(i, coeff);
206  nChannelsCalibrated++;
207  } else if (mMinAllowedCoeff >= coeff) {
208  newGains->setGain(i, mMinAllowedCoeff);
209  } else {
210  newGains->setGain(i, mMaxAllowedCoeff);
211  }
212  }
213  }
214  delete fLandau;
215  LOG(info) << "GainCalibrator::finalizeSlot() : succesfully calibrated " << nChannelsCalibrated << "channels";
216 
217  // prepare new ccdb entries for sending
218  mGainsVec.push_back(*newGains);
219  // metadata for o2::cpv::CalibParams
220  std::map<std::string, std::string> metaData;
221  auto className = o2::utils::MemFileHelper::getClassName(newGains);
222  auto fileName = o2::ccdb::CcdbApi::generateFileName(className);
223  auto timeStamp = o2::ccdb::getCurrentTimestamp();
224  mCcdbInfoGainsVec.emplace_back("CPV/Calib/Gains", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
225 
226  mPreviousGainCalibData.reset(new GainCalibData(*slot.getContainer())); // Slot is going to be closed and removed after finalisation -> save calib data for next Slot
227  mPreviousGains.reset(newGains); // save new obtained gains as previous ones
228 }

261 bool GainCalibrator::hasEnoughData(const GainTimeSlot& slot) const
262 {
263  if (mCurrentTFInfo.tfCounter % mUpdateTFInterval != 0) {
264  return false; // check once per N TFs if enough statistics
265  }
266  int nChannelsToCalibrate = 0;
267  for (int i = 0; i < Geometry::kNCHANNELS; i++) {
268  if (slot.getContainer()->mAmplitudeSpectra[i].getNEntries() > mMinEvents) {
269  nChannelsToCalibrate++;
270  }
271  }
272  if (nChannelsToCalibrate >= mMinNChannelsToCalibrate) {
273  LOG(info) << "GainCalibrator::hasEnoughtData() : slot "
274  << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " is ready for calibration ("
275  << nChannelsToCalibrate << " are going to be calibrated).";
276  return true;
277  } else {
278  LOG(info) << "GainCalibrator::hasEnoughtData() : slot "
279  << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " is not ready for calibration ("
280  << nChannelsToCalibrate << " channels to be calibrated wich is not enough).";
281  }
282  return false;
283 }
