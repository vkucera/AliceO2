57 void O2GPUDPLDisplaySpec::init(InitContext& ic)
58 {
59  const auto grp = o2::parameters::GRPObject::loadFrom();
60  o2::base::GeometryManager::loadGeometry();
61  o2::base::Propagator::initFieldFromGRP();
62  mConfig.reset(new GPUO2InterfaceConfiguration);
63  mConfig->configGRP.solenoidBz = 5.00668f * grp->getL3Current() / 30000.;
64  mConfig->configGRP.continuousMaxTimeBin = grp->isDetContinuousReadOut(o2::detectors::DetID::TPC) ? -1 : 0; // Number of timebins in timeframe if continuous, 0 otherwise
65  mConfig->ReadConfigurableParam();
66 
67  mFastTransform = std::move(TPCFastTransformHelperO2::instance()->create(0));
68  mConfig->configCalib.fastTransform = mFastTransform.get();
69 
70  auto gm = o2::trd::Geometry::instance();
71  gm->createPadPlaneArray();
72  gm->createClusterMatrixArray();
73  mTrdGeo.reset(new o2::trd::GeometryFlat(*gm));
74  mConfig->configCalib.trdGeometry = mTrdGeo.get();
75 
76  mITSDict = std::make_unique<o2::itsmft::TopologyDictionary>();
77  mConfig->configCalib.itsPatternDict = mITSDict.get();
78 
79  mConfig->configProcessing.runMC = mUseMC;
80 
81  o2::tof::Geo::Init();
82 
83  o2::its::GeometryTGeo::Instance()->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2GRot, o2::math_utils::TransformType::T2G, o2::math_utils::TransformType::L2G, o2::math_utils::TransformType::T2L));
84 
85  mTFSettings.reset(new o2::gpu::GPUSettingsTF);
86  mTFSettings->hasNHBFPerTF = 1;
87  mTFSettings->nHBFPerTF = grp->getNHBFPerTF();
88  mTFSettings->hasRunStartOrbit = 1;
89  mTFSettings->runStartOrbit = grp->getFirstOrbit();
90  mTFSettings->hasSimStartOrbit = 1;
91  auto& hbfu = o2::raw::HBFUtils::Instance();
92  mTFSettings->simStartOrbit = hbfu.getFirstIRofTF(o2::InteractionRecord(0, hbfu.orbitFirstSampled)).orbit;
93  if (mConfig->configGRP.continuousMaxTimeBin == -1) {
94  mConfig->configGRP.continuousMaxTimeBin = (mTFSettings->nHBFPerTF * o2::constants::lhc::LHCMaxBunches + 2 * o2::tpc::constants::LHCBCPERTIMEBIN - 2) / o2::tpc::constants::LHCBCPERTIMEBIN;
95  }
96 
97  mDisplay.reset(new GPUO2InterfaceDisplay(mConfig.get()));
98 }

100 void O2GPUDPLDisplaySpec::run(ProcessingContext& pc)
101 {
102  if (mDisplayShutDown) {
103  return;
104  }
105  if (mUpdateCalib) {
106  mDisplay->UpdateCalib(&mConfig->configCalib);
107  }
108  if (mFirst == false) {
109  if (mDisplay->startDisplay()) {
110  throw std::runtime_error("Error starting event display");
111  }
112  mFirst = true;
113  }
114 
115  o2::globaltracking::RecoContainer recoData;
116  recoData.collectData(pc, *mDataRequest);
117  GPUTrackingInOutPointers ptrs;
118  auto tmpContainer = GPUWorkflowHelper::fillIOPtr(ptrs, recoData, mUseMC, &(mConfig->configCalib), mClMask, mTrkMask, mTrkMask);
119 
120  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getFirstValid(true).header);
121  mTFSettings->tfStartOrbit = dh->firstTForbit;
122  mTFSettings->hasTfStartOrbit = 1;
123  ptrs.settingsTF = mTFSettings.get();
124 
125  if (mDisplay->show(&ptrs)) {
126  mDisplay->endDisplay();
127  mDisplayShutDown = true;
128  }
129 }

149 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
150 {
151  WorkflowSpec specs;
152 
153  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
154 
155  bool useMC = cfgc.options().get<bool>("enable-mc") && !cfgc.options().get<bool>("disable-mc");
156  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks"));
157  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters"));
158  if (!srcTrk.any() && !srcCl.any()) {
159  throw std::runtime_error("No input configured");
160  }
161  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
162  dataRequest->requestTracks(srcTrk, useMC);
163  dataRequest->requestClusters(srcCl, useMC);
164 
165  if (cfgc.options().get<bool>("read-from-files")) {
166  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
167  }
168 
169  specs.emplace_back(DataProcessorSpec{
170  "o2-gpu-display",
171  dataRequest->inputs,
172  {},
173  AlgorithmSpec{adaptFromTask<O2GPUDPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest)}});
174 
175  return specs;
176 }
