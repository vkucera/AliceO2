38 class ResidualAggregatorDevice : public o2::framework::Task
39 {
40  public:
41  ResidualAggregatorDevice(std::shared_ptr<o2::base::GRPGeomRequest> req) : mCCDBRequest(req) {}
42 
43  void init(o2::framework::InitContext& ic) final
44  {
45  o2::base::GRPGeomHelper::instance().setRequest(mCCDBRequest);
46  int minEnt = ic.options().get<int>("min-entries");
47  auto slotLength = ic.options().get<uint32_t>("tf-per-slot");
48  if (slotLength == 0) {
49  slotLength = o2::calibration::INFINITE_TF;
50  }
51  auto updateInterval = ic.options().get<uint32_t>("updateInterval");
52  auto delay = ic.options().get<uint32_t>("max-delay");
53 
54  std::string outputDir = o2::utils::Str::rectifyDirectory(ic.options().get<std::string>("output-dir"));
55  std::string metaFileDir = ic.options().get<std::string>("meta-output-dir");
56  bool storeMetaFile = false;
57  if (metaFileDir != "/dev/null") {
58  metaFileDir = o2::utils::Str::rectifyDirectory(metaFileDir);
59  storeMetaFile = true;
60  }
61  mAggregator = std::make_unique<o2::tpc::ResidualAggregator>(minEnt);
62  mAggregator->setOutputDir(outputDir);
63  if (storeMetaFile) {
64  mAggregator->setMetaFileOutputDir(metaFileDir);
65  }
66  // TODO mAggregator should get an option to set the binning externally (expose TrackResiduals::setBinning methods to user? as command line option?)
67  mAggregator->setSlotLength(slotLength);
68  mAggregator->setMaxSlotsDelay(delay);
69  mAggregator->setCheckIntervalInfiniteSlot(updateInterval);
70  mAggregator->initOutput();
71  }
72 
73  void finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj) final
74  {
75  o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj);
76  }
77 
78  void run(o2::framework::ProcessingContext& pc) final
79  {
80  updateTimeDependentParams(pc);
81 
82  auto data = pc.inputs().get<gsl::span<o2::tpc::TrackResiduals::UnbinnedResid>>("input");
83  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mAggregator->getCurrentTFInfo());
84  LOG(debug) << "Processing TF " << mAggregator->getCurrentTFInfo().tfCounter << " with " << data.size() << " unbinned residuals";
85  mAggregator->process(data);
86  }
87 
88  void endOfStream(o2::framework::EndOfStreamContext& ec) final
89  {
90  LOG(info) << "Finalizing calibration for end of stream";
91  mAggregator->checkSlotsToFinalize(o2::calibration::INFINITE_TF);
92  mAggregator.reset(); // must invoke destructor manually here, otherwise we get a segfault
93  }
94 
95  private:
96  void updateTimeDependentParams(ProcessingContext& pc)
97  {
98  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
99  static bool initOnceDone = false;
100  if (!initOnceDone) {
101  initOnceDone = true;
102  mAggregator->setDataTakingContext(pc.services().get<DataTakingContext>());
103  }
104  }
105  std::unique_ptr<o2::tpc::ResidualAggregator> mAggregator;
106  std::shared_ptr<o2::base::GRPGeomRequest> mCCDBRequest;
107 };

114 DataProcessorSpec getTPCResidualAggregatorSpec()
115 {
116  std::vector<InputSpec> inputs{{"input", "GLO", "UNBINNEDRES"}};
117  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
118  true, // GRPECS=true
119  false, // GRPLHCIF
120  false, // GRPMagField
121  false, // askMatLUT
122  o2::base::GRPGeomRequest::None, // geometry
123  inputs);
124  return DataProcessorSpec{
125  "residual-aggregator",
126  inputs,
127  Outputs{},
128  AlgorithmSpec{adaptFromTask<o2::calibration::ResidualAggregatorDevice>(ccdbRequest)},
129  Options{
130  {"tf-per-slot", VariantType::UInt32, 6'000u, {"number of TFs per calibration time slot (put 0 for infinite slot length)"}},
131  {"updateInterval", VariantType::UInt32, 6'000u, {"update interval in number of TFs in case slot length is infinite"}},
132  {"max-delay", VariantType::UInt32, 10u, {"number of slots in past to consider"}},
133  {"min-entries", VariantType::Int, 0, {"minimum number of entries on average per voxel"}},
134  {"output-dir", VariantType::String, "none", {"Output directory for residuals, must exist"}},
135  {"meta-output-dir", VariantType::String, "/dev/null", {"Residuals metadata output directory, must exist (if not /dev/null)"}}}};
136 }
